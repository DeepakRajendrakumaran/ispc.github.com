00145 /** Utility routine that emits code to initialize a symbol given an
00146     initializer expression.
00147 
00148     @param lvalue    Memory location of storage for the symbol's data
00149     @param symName   Name of symbol (used in error messages)
00150     @param symType   Type of variable being initialized
00151     @param initExpr  Expression for the initializer
00152     @param ctx       FunctionEmitContext to use for generating instructions
00153     @param pos       Source file position of the variable being initialized
00154 */

00156 lInitSymbol(llvm::Value *lvalue, const char *symName, const Type *symType,
00157             Expr *initExpr, FunctionEmitContext *ctx, SourcePos pos) {
00158     if (initExpr == NULL)
00159         // leave it uninitialized
00160         return;
00161 
00162     // If the initializer is a straight up expression that isn't an
00163     // ExprList, then we'll see if we can type convert it to the type of
00164     // the variable.
00165     if (dynamic_cast<ExprList *>(initExpr) == NULL) {
00166         if (lPossiblyResolveFunctionOverloads(initExpr, symType) == false)
00167             return;
00168         initExpr = TypeConvertExpr(initExpr, symType, "initializer");
00169 
00170         if (initExpr != NULL) {
00171             llvm::Value *initializerValue = initExpr->GetValue(ctx);
00172             if (initializerValue != NULL)
00173                 // Bingo; store the value in the variable's storage
00174                 ctx->StoreInst(initializerValue, lvalue);
00175             return;
00176         }
00177     }
00178 
00179     // Atomic types and enums can't be initialized with { ... } initializer
00180     // expressions, so print an error and return if that's what we've got
00181     // here..
00182     if (dynamic_cast<const AtomicType *>(symType) != NULL ||
00183         dynamic_cast<const EnumType *>(symType) != NULL ||
00184         dynamic_cast<const PointerType *>(symType) != NULL) {
00185         ExprList *elist = dynamic_cast<ExprList *>(initExpr);
00186         if (elist != NULL) {
00187             if (elist->exprs.size() == 1)
00188                 lInitSymbol(lvalue, symName, symType, elist->exprs[0], ctx,
00189                             pos);
00190             else
00191                 Error(initExpr->pos, "Expression list initializers can't be used for "
00192                       "variable \"%s\' with type \"%s\".", symName,
00193                       symType->GetString().c_str());
00194         }
00195         return;
00196     }
00197 
00198     const ReferenceType *rt = dynamic_cast<const ReferenceType *>(symType);
00199     if (rt) {
00200         if (!Type::Equal(initExpr->GetType(), rt)) {
00201             Error(initExpr->pos, "Initializer for reference type \"%s\" must have same "
00202                   "reference type itself. \"%s\" is incompatible.", 
00203                   rt->GetString().c_str(), initExpr->GetType()->GetString().c_str());
00204             return;
00205         }
00206 
00207         llvm::Value *initializerValue = initExpr->GetValue(ctx);
00208         if (initializerValue)
00209             ctx->StoreInst(initializerValue, lvalue);
00210         return;
00211     }
00212 
00213     // There are two cases for initializing structs, arrays and vectors;
00214     // either a single initializer may be provided (float foo[3] = 0;), in
00215     // which case all of the elements are initialized to the given value,
00216     // or an initializer list may be provided (float foo[3] = { 1,2,3 }),
00217     // in which case the elements are initialized with the corresponding
00218     // values.
00219     const CollectionType *collectionType = 
00220         dynamic_cast<const CollectionType *>(symType);
00221     if (collectionType != NULL) {
00222         std::string name;
00223         if (dynamic_cast<const StructType *>(symType) != NULL)
00224             name = "struct";
00225         else if (dynamic_cast<const ArrayType *>(symType) != NULL) 
00226             name = "array";
00227         else if (dynamic_cast<const VectorType *>(symType) != NULL) 
00228             name = "vector";
00229         else 
00230             FATAL("Unexpected CollectionType in lInitSymbol()");
00231 
00232         ExprList *exprList = dynamic_cast<ExprList *>(initExpr);
00233         if (exprList != NULL) {
00234             // The { ... } case; make sure we have the same number of
00235             // expressions in the ExprList as we have struct members
00236             int nInits = exprList->exprs.size();
00237             if (nInits != collectionType->GetElementCount()) {
00238                 Error(initExpr->pos, "Initializer for %s \"%s\" requires "
00239                       "%d values; %d provided.", name.c_str(), symName, 
00240                       collectionType->GetElementCount(), nInits);
00241                 return;
00242             }
00243 
00244             // Initialize each element with the corresponding value from
00245             // the ExprList
00246             for (int i = 0; i < nInits; ++i) {
00247                 llvm::Value *ep;
00248                 if (dynamic_cast<const StructType *>(symType) != NULL)
00249                     ep = ctx->AddElementOffset(lvalue, i, NULL, "element");
00250                 else
00251                     ep = ctx->GetElementPtrInst(lvalue, LLVMInt32(0), LLVMInt32(i), 
00252                                                 PointerType::GetUniform(collectionType->GetElementType(i)), 
00253                                                 "gep");
00254 
00255                 lInitSymbol(ep, symName, collectionType->GetElementType(i), 
00256                             exprList->exprs[i], ctx, pos);
00257             }
00258         }
00259         else
00260             Error(initExpr->pos, "Can't assign type \"%s\" to \"%s\".",
00261                   initExpr->GetType()->GetString().c_str(),
00262                   collectionType->GetString().c_str());
00263         return;
00264     }
00265 
00266     FATAL("Unexpected Type in lInitSymbol()");
00267 }

00390     for (unsigned int i = 0; i < vars.size(); ++i) {
00391         if (vars[i].init != NULL) {
00392             vars[i].init = vars[i].init->Optimize();
00393             Expr *init = vars[i].init;
00394 
00395             // If the variable is const-qualified, after we've optimized
00396             // the initializer expression, see if we have a ConstExpr.  If
00397             // so, save it in Symbol::constValue where it can be used in
00398             // optimizing later expressions that have this symbol in them.
00399             // Note that there are cases where the expression may be
00400             // constant but where we don't have a ConstExpr; an example is
00401             // const arrays--the ConstExpr implementation just can't
00402             // represent an array of values.
00403             //
00404             // All this is fine in terms of the code that's generated in
00405             // the end (LLVM's constant folding stuff is good), but it
00406             // means that the ispc compiler's ability to reason about what
00407             // is definitely a compile-time constant for things like
00408             // computing array sizes from non-trivial expressions is
00409             // consequently limited.
00410             Symbol *sym = vars[i].sym;
00411             if (sym->type && sym->type->IsConstType() && init != NULL && 
00412                 dynamic_cast<ExprList *>(init) == NULL &&
00413                 Type::Equal(init->GetType(), sym->type))
00414                 sym->constValue = dynamic_cast<ConstExpr *>(init);
00415         }
00416     }

00659 /** Similar to the Stmt variant of this function, this conservatively
00660     checks to see if it's safe to run the code for the given Expr even if
00661     the mask is 'all off'.
00662  */
00663 static bool
00664 lSafeToRunWithAllLanesOff(Expr *expr) {
00665     if (expr == NULL)
00666         return false;
00667 
00668     UnaryExpr *ue;
00669     if ((ue = dynamic_cast<UnaryExpr *>(expr)) != NULL)
00670         return lSafeToRunWithAllLanesOff(ue->expr);
00671 
00672     BinaryExpr *be;
00673     if ((be = dynamic_cast<BinaryExpr *>(expr)) != NULL)
00674         return (lSafeToRunWithAllLanesOff(be->arg0) &&
00675                 lSafeToRunWithAllLanesOff(be->arg1));
00676 
00677     AssignExpr *ae;
00678     if ((ae = dynamic_cast<AssignExpr *>(expr)) != NULL)
00679         return (lSafeToRunWithAllLanesOff(ae->lvalue) &&
00680                 lSafeToRunWithAllLanesOff(ae->rvalue));
00681 
00682     SelectExpr *se;
00683     if ((se = dynamic_cast<SelectExpr *>(expr)) != NULL)
00684         return (lSafeToRunWithAllLanesOff(se->test) && 
00685                 lSafeToRunWithAllLanesOff(se->expr1) && 
00686                 lSafeToRunWithAllLanesOff(se->expr2));
00687 
00688     ExprList *el;
00689     if ((el = dynamic_cast<ExprList *>(expr)) != NULL) {
00690         for (unsigned int i = 0; i < el->exprs.size(); ++i)
00691             if (!lSafeToRunWithAllLanesOff(el->exprs[i]))
00692                 return false;
00693         return true;
00694     }
00695 
00696     FunctionCallExpr *fce;
00697     if ((fce = dynamic_cast<FunctionCallExpr *>(expr)) != NULL)
00698         // FIXME: If we could somehow determine that the function being
00699         // called was safe (and all of the args Exprs were safe, then it'd
00700         // be nice to be able to return true here.  (Consider a call to
00701         // e.g. floatbits() in the stdlib.)  Unfortunately for now we just
00702         // have to be conservative.
00703         return false;
00704 
00705     IndexExpr *ie;
00706     if ((ie = dynamic_cast<IndexExpr *>(expr)) != NULL) {
00707         // If we can determine at compile time the size of the array/vector
00708         // and if the indices are compile-time constants, then we may be
00709         // able to safely run this under a predicated if statement..
00710         if (ie->baseExpr == NULL)
00711             return false;
00712 
00713         const Type *type = ie->baseExpr->GetType();
00714         ConstExpr *ce = dynamic_cast<ConstExpr *>(ie->index);
00715         if (type == NULL || ce == NULL)
00716             return false;
00717         if (dynamic_cast<const ReferenceType *>(type) != NULL)
00718             type = type->GetReferenceTarget();
00719

00721             dynamic_cast<const PointerType *>(type);
00722         if (pointerType != NULL)
00723             // pointer[offset] -> can't be sure
00724             return false;
00725

00727             dynamic_cast<const SequentialType *>(type);
00728         assert(seqType != NULL);
00729         int nElements = seqType->GetElementCount();
00730         if (nElements == 0)
00731             // Unsized array, so we can't be sure
00732             return false;
00733 
00734         int32_t indices[ISPC_MAX_NVEC];
00735         int count = ce->AsInt32(indices);
00736         for (int i = 0; i < count; ++i)
00737             if (indices[i] < 0 || indices[i] >= nElements)
00738                 return false;
00739 
00740         // All indices are in-bounds
00741         return true;
00742     }
00743 
00744     MemberExpr *me;
00745     if ((me = dynamic_cast<MemberExpr *>(expr)) != NULL)
00746         return lSafeToRunWithAllLanesOff(me->expr);
00747 
00748     if (dynamic_cast<ConstExpr *>(expr) != NULL)
00749         return true;
00750 
00751     TypeCastExpr *tce;
00752     if ((tce = dynamic_cast<TypeCastExpr *>(expr)) != NULL)
00753         return lSafeToRunWithAllLanesOff(tce->expr);
00754 
00755     ReferenceExpr *re;
00756     if ((re = dynamic_cast<ReferenceExpr *>(expr)) != NULL)
00757         return lSafeToRunWithAllLanesOff(re->expr);
00758 
00759     DereferenceExpr *dre;

00761         return lSafeToRunWithAllLanesOff(dre->expr);
00762 
00763     SizeOfExpr *soe;
00764     if ((soe = dynamic_cast<SizeOfExpr *>(expr)) != NULL)
00765         return lSafeToRunWithAllLanesOff(soe->expr);
00766 
00767     AddressOfExpr *aoe;

00769         return lSafeToRunWithAllLanesOff(aoe->expr);
00770 
00771     if (dynamic_cast<SymbolExpr *>(expr) != NULL ||
00772         dynamic_cast<FunctionSymbolExpr *>(expr) != NULL ||
00773         dynamic_cast<SyncExpr *>(expr) != NULL ||

00782 /** Given an arbitrary statement, this function conservatively tests to see
00783     if it's safe to run the code for the statement even if the mask is all
00784     off.  Here we just need to determine which kind of statement we have
00785     and recursively traverse it and/or the expressions inside of it.
00786  */
00787 static bool
00788 lSafeToRunWithAllLanesOff(Stmt *stmt) {
00789     if (stmt == NULL)
00790         return true;
00791 
00792     ExprStmt *es;
00793     if ((es = dynamic_cast<ExprStmt *>(stmt)) != NULL)
00794         return lSafeToRunWithAllLanesOff(es->expr);
00795 
00796     DeclStmt *ds;
00797     if ((ds = dynamic_cast<DeclStmt *>(stmt)) != NULL) {
00798         for (unsigned int i = 0; i < ds->vars.size(); ++i)
00799             if (!lSafeToRunWithAllLanesOff(ds->vars[i].init))
00800                 return false;
00801         return true;
00802     }
00803 
00804     IfStmt *is;
00805     if ((is = dynamic_cast<IfStmt *>(stmt)) != NULL)
00806         return (lSafeToRunWithAllLanesOff(is->test) &&
00807                 lSafeToRunWithAllLanesOff(is->trueStmts) &&
00808                 lSafeToRunWithAllLanesOff(is->falseStmts));
00809 
00810     DoStmt *dos;
00811     if ((dos = dynamic_cast<DoStmt *>(stmt)) != NULL)
00812         return (lSafeToRunWithAllLanesOff(dos->testExpr) &&
00813                 lSafeToRunWithAllLanesOff(dos->bodyStmts));
00814 
00815     ForStmt *fs;
00816     if ((fs = dynamic_cast<ForStmt *>(stmt)) != NULL)
00817         return (lSafeToRunWithAllLanesOff(fs->init) &&
00818                 lSafeToRunWithAllLanesOff(fs->test) &&
00819                 lSafeToRunWithAllLanesOff(fs->step) &&
00820                 lSafeToRunWithAllLanesOff(fs->stmts));
00821 
00822     ForeachStmt *fes;
00823     if ((fes = dynamic_cast<ForeachStmt *>(stmt)) != NULL) {
00824         for (unsigned int i = 0; i < fes->startExprs.size(); ++i)
00825             if (!lSafeToRunWithAllLanesOff(fes->startExprs[i]))
00826                 return false;
00827         for (unsigned int i = 0; i < fes->endExprs.size(); ++i)
00828             if (!lSafeToRunWithAllLanesOff(fes->endExprs[i]))
00829                 return false;
00830         return lSafeToRunWithAllLanesOff(fes->stmts);
00831     }
00832 
00833     if (dynamic_cast<BreakStmt *>(stmt) != NULL ||
00834         dynamic_cast<ContinueStmt *>(stmt) != NULL)
00835         return true;
00836 
00837     ReturnStmt *rs;
00838     if ((rs = dynamic_cast<ReturnStmt *>(stmt)) != NULL)
00839         return lSafeToRunWithAllLanesOff(rs->val);
00840 
00841     StmtList *sl;
00842     if ((sl = dynamic_cast<StmtList *>(stmt)) != NULL) {
00843         const std::vector<Stmt *> &sls = sl->GetStatements();
00844         for (unsigned int i = 0; i < sls.size(); ++i)
00845             if (!lSafeToRunWithAllLanesOff(sls[i]))
00846                 return false;
00847         return true;
00848     }
00849 
00850     PrintStmt *ps;
00851     if ((ps = dynamic_cast<PrintStmt *>(stmt)) != NULL)
00852         return lSafeToRunWithAllLanesOff(ps->values);
00853 
00854     AssertStmt *as;
00855     if ((as = dynamic_cast<AssertStmt *>(stmt)) != NULL)
00856         // While this is fine for varying tests, it's not going to be
00857         // desirable to check an assert on a uniform variable if all of the
00858         // lanes are off.
00859         return false;
00860 
00861     FATAL("Unexpected stmt type in lSafeToRunWithAllLanesOff()");
00862     return false;
00863 }

00866 /** Emit code for an if test that checks the mask and the test values and
00867     tries to be smart about jumping over code that doesn't need to be run.
00868  */
00869 void
00870 IfStmt::emitVaryingIf(FunctionEmitContext *ctx, llvm::Value *ltest) const {
00871     llvm::Value *oldMask = ctx->GetInternalMask();
00872     if (ctx->GetFullMask() == LLVMMaskAllOn) {
00873         // We can tell that the mask is on statically at compile time; just
00874         // emit code for the 'if test with the mask all on' path
00875         llvm::BasicBlock *bDone = ctx->CreateBasicBlock("cif_done");
00876         emitMaskAllOn(ctx, ltest, bDone);
00877         ctx->SetCurrentBasicBlock(bDone);
00878     }
00879     else if (doAllCheck) {
00880         // We can't tell if the mask going into the if is all on at the
00881         // compile time.  Emit code to check for this and then either run
00882         // the code for the 'all on' or the 'mixed' case depending on the
00883         // mask's value at runtime.
00884         llvm::BasicBlock *bAllOn = ctx->CreateBasicBlock("cif_mask_all");
00885         llvm::BasicBlock *bMixedOn = ctx->CreateBasicBlock("cif_mask_mixed");
00886         llvm::BasicBlock *bDone = ctx->CreateBasicBlock("cif_done");
00887 
00888         // Jump to either bAllOn or bMixedOn, depending on the mask's value 
00889         llvm::Value *maskAllQ = ctx->All(ctx->GetFullMask());
00890         ctx->BranchInst(bAllOn, bMixedOn, maskAllQ);
00891 
00892         // Emit code for the 'mask all on' case
00893         ctx->SetCurrentBasicBlock(bAllOn);
00894         emitMaskAllOn(ctx, ltest, bDone);
00895         
00896         // And emit code for the mixed mask case
00897         ctx->SetCurrentBasicBlock(bMixedOn);
00898         emitMaskMixed(ctx, oldMask, ltest, bDone);
00899 
00900         // When done, set the current basic block to the block that the two
00901         // paths above jump to when they're done.
00902         ctx->SetCurrentBasicBlock(bDone);
00903     }
00904     else if (trueStmts != NULL || falseStmts != NULL) {
00905         // If there is nothing that is potentially unsafe to run with all
00906         // lanes off in the true and false statements and if the total
00907         // complexity of those two is relatively simple, then we'll go
00908         // ahead and emit straightline code that runs both sides, updating
00909         // the mask accordingly.  This is useful for efficiently compiling
00910         // things like:
00911         //
00912         // if (foo) x = 0;
00913         // else     ++x;
00914         //
00915         // Where the overhead of checking if any of the program instances wants
00916         // to run one side or the other is more than the actual computation.
00917         // The lSafeToRunWithAllLanesOff() checks to make sure that we don't do this
00918         // for potentially dangerous code like:
00919         //
00920         // if (index < count) array[index] = 0;
00921         //
00922         // where our use of blend for conditional assignments doesn't check
00923         // for the 'all lanes' off case.
00924         if (lSafeToRunWithAllLanesOff(trueStmts) &&
00925             lSafeToRunWithAllLanesOff(falseStmts) &&
00926             (((trueStmts ? trueStmts->EstimateCost() : 0) + 
00927               (falseStmts ? falseStmts->EstimateCost() : 0)) < 
00928              PREDICATE_SAFE_IF_STATEMENT_COST)) {
00929             ctx->StartVaryingIf(oldMask);
00930             emitMaskedTrueAndFalse(ctx, oldMask, ltest);
00931             assert(ctx->GetCurrentBasicBlock());
00932             ctx->EndIf();
00933         }
00934         else {
00935             llvm::BasicBlock *bDone = ctx->CreateBasicBlock("if_done");
00936             emitMaskMixed(ctx, oldMask, ltest, bDone);
00937             ctx->SetCurrentBasicBlock(bDone);
00938         }
00939     }
00940 }

00943 /** Emits code for 'if' tests under the case where we know that the program
00944     mask is all on going into the 'if'.
00945  */
00946 void
00947 IfStmt::emitMaskAllOn(FunctionEmitContext *ctx, llvm::Value *ltest, 
00948                       llvm::BasicBlock *bDone) const {
00949     // We start by explicitly storing "all on" into the mask mask.  Note
00950     // that this doesn't change its actual value, but doing so lets the
00951     // compiler see what's going on so that subsequent optimizations for
00952     // code emitted here can operate with the knowledge that the mask is
00953     // definitely all on (until it modifies the mask itself).
00954     ctx->SetInternalMask(LLVMMaskAllOn);
00955     llvm::Value *oldFunctionMask = ctx->GetFunctionMask();
00956     ctx->SetFunctionMask(LLVMMaskAllOn);
00957 
00958     // First, check the value of the test.  If it's all on, then we jump to
00959     // a basic block that will only have code for the true case.
00960     llvm::BasicBlock *bTestAll = ctx->CreateBasicBlock("cif_test_all");
00961     llvm::BasicBlock *bTestNoneCheck = ctx->CreateBasicBlock("cif_test_none_check");
00962     llvm::Value *testAllQ = ctx->All(ltest);
00963     ctx->BranchInst(bTestAll, bTestNoneCheck, testAllQ);
00964 
00965     // Emit code for the 'test is all true' case
00966     ctx->SetCurrentBasicBlock(bTestAll);
00967     ctx->StartVaryingIf(LLVMMaskAllOn);
00968     lEmitIfStatements(ctx, trueStmts, "if: all on mask, expr all true");
00969     ctx->EndIf();
00970     if (ctx->GetCurrentBasicBlock() != NULL)
00971         // bblock may legitimately be NULL since if there's a return stmt
00972         // or break or continue we can actually jump and end emission since
00973         // we know all of the lanes are following this path...
00974         ctx->BranchInst(bDone);
00975 
00976     // The test isn't all true.  Now emit code to determine if it's all
00977     // false, or has mixed values.
00978     ctx->SetCurrentBasicBlock(bTestNoneCheck);
00979     llvm::BasicBlock *bTestNone = ctx->CreateBasicBlock("cif_test_none");
00980     llvm::BasicBlock *bTestMixed = ctx->CreateBasicBlock("cif_test_mixed");
00981     llvm::Value *testMixedQ = ctx->Any(ltest);
00982     ctx->BranchInst(bTestMixed, bTestNone, testMixedQ);
00983 
00984     // Emit code for the 'test is all false' case
00985     ctx->SetCurrentBasicBlock(bTestNone);
00986     ctx->StartVaryingIf(LLVMMaskAllOn);
00987     lEmitIfStatements(ctx, falseStmts, "if: all on mask, expr all false");
00988     ctx->EndIf();
00989     if (ctx->GetCurrentBasicBlock())
00990         // bblock may be NULL since if there's a return stmt or break or
00991         // continue we can actually jump or whatever and end emission...
00992         ctx->BranchInst(bDone);
00993 
00994     // Finally emit code for the 'mixed true/false' case.  We unavoidably
00995     // need to run both the true and the false statements.
00996     ctx->SetCurrentBasicBlock(bTestMixed);
00997     ctx->StartVaryingIf(LLVMMaskAllOn);
00998     emitMaskedTrueAndFalse(ctx, LLVMMaskAllOn, ltest);
00999     // In this case, return/break/continue isn't allowed to jump and end
01000     // emission.
01001     assert(ctx->GetCurrentBasicBlock());
01002     ctx->EndIf();
01003     ctx->BranchInst(bDone);
01004 
01005     ctx->SetCurrentBasicBlock(bDone);
01006     ctx->SetFunctionMask(oldFunctionMask);
01007 }

01010 /** Emit code for an 'if' test where the lane mask is known to be mixed
01011     on/off going into it.
01012  */
01013 void
01014 IfStmt::emitMaskMixed(FunctionEmitContext *ctx, llvm::Value *oldMask, 
01015                       llvm::Value *ltest, llvm::BasicBlock *bDone) const {
01016     ctx->StartVaryingIf(oldMask);
01017     llvm::BasicBlock *bNext = ctx->CreateBasicBlock("safe_if_after_true");
01018     if (trueStmts != NULL) {
01019         llvm::BasicBlock *bRunTrue = ctx->CreateBasicBlock("safe_if_run_true");
01020         ctx->SetInternalMaskAnd(oldMask, ltest);
01021 
01022         // Do any of the program instances want to run the 'true'
01023         // block?  If not, jump ahead to bNext.
01024         llvm::Value *maskAnyQ = ctx->Any(ctx->GetFullMask());
01025         ctx->BranchInst(bRunTrue, bNext, maskAnyQ);
01026 
01027         // Emit statements for true
01028         ctx->SetCurrentBasicBlock(bRunTrue);
01029         lEmitIfStatements(ctx, trueStmts, "if: expr mixed, true statements");
01030         assert(ctx->GetCurrentBasicBlock()); 
01031         ctx->BranchInst(bNext);
01032         ctx->SetCurrentBasicBlock(bNext);
01033     }
01034     if (falseStmts != NULL) {
01035         llvm::BasicBlock *bRunFalse = ctx->CreateBasicBlock("safe_if_run_false");
01036         bNext = ctx->CreateBasicBlock("safe_if_after_false");
01037         ctx->SetInternalMaskAndNot(oldMask, ltest);
01038 
01039         // Similarly, check to see if any of the instances want to
01040         // run the 'false' block...
01041         llvm::Value *maskAnyQ = ctx->Any(ctx->GetFullMask());
01042         ctx->BranchInst(bRunFalse, bNext, maskAnyQ);
01043 
01044         // Emit code for false
01045         ctx->SetCurrentBasicBlock(bRunFalse);
01046         lEmitIfStatements(ctx, falseStmts, "if: expr mixed, false statements");
01047         assert(ctx->GetCurrentBasicBlock());
01048         ctx->BranchInst(bNext);
01049         ctx->SetCurrentBasicBlock(bNext);
01050     }
01051     ctx->BranchInst(bDone);
01052     ctx->SetCurrentBasicBlock(bDone);
01053     ctx->EndIf();
01054 }

01060 /** Given a statment, walk through it to see if there is a 'break' or
01061     'continue' statement inside if its children, under varying control
01062     flow.  We need to detect this case for loops since what might otherwise
01063     look like a 'uniform' loop needs to have code emitted to do all of the
01064     lane management stuff if this is the case.
01065  */ 
01066 static bool
01067 lHasVaryingBreakOrContinue(Stmt *stmt, bool inVaryingCF = false) {
01068     StmtList *sl;
01069     IfStmt *is;
01070 
01071     if ((sl = dynamic_cast<StmtList *>(stmt)) != NULL) {
01072         // Recurse through the children statements 
01073         const std::vector<Stmt *> &stmts = sl->GetStatements();
01074         for (unsigned int i = 0; i < stmts.size(); ++i)
01075             if (lHasVaryingBreakOrContinue(stmts[i], inVaryingCF))
01076                 return true;
01077     }
01078     else if ((is = dynamic_cast<IfStmt *>(stmt)) != NULL) {
01079         // We've come to an 'if'.  Is the test type varying?  If so, then
01080         // we're under 'varying' control flow when we recurse through the
01081         // true and false statements.
01082         if (is->test != NULL) {
01083             const Type *type = is->test->GetType();
01084             if (type)
01085                 inVaryingCF |= type->IsVaryingType();
01086         }
01087 
01088         if (lHasVaryingBreakOrContinue(is->trueStmts, inVaryingCF) ||
01089             lHasVaryingBreakOrContinue(is->falseStmts, inVaryingCF))
01090             return true;
01091     }
01092     else if (dynamic_cast<BreakStmt *>(stmt) != NULL) {
01093         if (inVaryingCF)
01094             return true;
01095     }
01096     else if (dynamic_cast<ContinueStmt *>(stmt) != NULL) {
01097         if (inVaryingCF)
01098             return true;
01099     }
01100     // Important: note that we don't recurse into do/for loops here but
01101     // just return false.  For the question of whether a given loop needs
01102     // to do mask management stuff, breaks/continues inside nested loops
01103     // inside of them don't matter.
01104     return false;
01105 }

01114 void DoStmt::EmitCode(FunctionEmitContext *ctx) const {
01115     // Check for things that could be NULL due to earlier errors during
01116     // compilation.
01117     if (!ctx->GetCurrentBasicBlock()) 
01118         return;
01119     if (!testExpr || !testExpr->GetType()) 
01120         return;
01121 
01122     bool uniformTest = testExpr->GetType()->IsUniformType();
01123     if (uniformTest && doCoherentCheck)
01124         Warning(pos, "Uniform condition supplied to \"cdo\" statement.");
01125 
01126     llvm::BasicBlock *bloop = ctx->CreateBasicBlock("do_loop");
01127     llvm::BasicBlock *bexit = ctx->CreateBasicBlock("do_exit");
01128     llvm::BasicBlock *btest = ctx->CreateBasicBlock("do_test");
01129 
01130     ctx->StartLoop(bexit, btest, uniformTest);
01131 
01132     // Start by jumping into the loop body
01133     ctx->BranchInst(bloop);
01134 
01135     // And now emit code for the loop body
01136     ctx->SetCurrentBasicBlock(bloop);
01137     ctx->SetLoopMask(ctx->GetInternalMask());
01138     ctx->SetDebugPos(pos);
01139     // FIXME: in the StmtList::EmitCode() method takes starts/stops a new
01140     // scope around the statements in the list.  So if the body is just a
01141     // single statement (and thus not a statement list), we need a new
01142     // scope, but we don't want two scopes in the StmtList case.
01143     if (!dynamic_cast<StmtList *>(bodyStmts))
01144         ctx->StartScope();
01145 
01146     ctx->AddInstrumentationPoint("do loop body");
01147     if (doCoherentCheck && !uniformTest) {
01148         // Check to see if the mask is all on
01149         llvm::BasicBlock *bAllOn = ctx->CreateBasicBlock("do_all_on");
01150         llvm::BasicBlock *bMixed = ctx->CreateBasicBlock("do_mixed");
01151         ctx->BranchIfMaskAll(bAllOn, bMixed);
01152 
01153         // If so, emit code for the 'mask all on' case.  In particular,
01154         // explicitly set the mask to 'all on' (see rationale in
01155         // IfStmt::emitCoherentTests()), and then emit the code for the
01156         // loop body.
01157         ctx->SetCurrentBasicBlock(bAllOn);
01158         ctx->SetInternalMask(LLVMMaskAllOn);
01159         llvm::Value *oldFunctionMask = ctx->GetFunctionMask();
01160         ctx->SetFunctionMask(LLVMMaskAllOn);
01161         if (bodyStmts)
01162             bodyStmts->EmitCode(ctx);
01163         assert(ctx->GetCurrentBasicBlock());
01164         ctx->SetFunctionMask(oldFunctionMask);
01165         ctx->BranchInst(btest);
01166 
01167         // The mask is mixed.  Just emit the code for the loop body.
01168         ctx->SetCurrentBasicBlock(bMixed);
01169         if (bodyStmts)
01170             bodyStmts->EmitCode(ctx);
01171         assert(ctx->GetCurrentBasicBlock());
01172         ctx->BranchInst(btest);
01173     }
01174     else {
01175         // Otherwise just emit the code for the loop body.  The current
01176         // mask is good.
01177         if (bodyStmts)
01178             bodyStmts->EmitCode(ctx);
01179         if (ctx->GetCurrentBasicBlock())
01180             ctx->BranchInst(btest);
01181     }
01182     // End the scope we started above, if needed.
01183     if (!dynamic_cast<StmtList *>(bodyStmts))
01184         ctx->EndScope();
01185 
01186     // Now emit code for the loop test.
01187     ctx->SetCurrentBasicBlock(btest);
01188     // First, emit code to restore the mask value for any lanes that
01189     // executed a 'continue' during the current loop before we go and emit
01190     // the code for the test.  This is only necessary for varying loops;
01191     // 'uniform' loops just jump when they hit a continue statement and
01192     // don't mess with the mask.
01193     if (!uniformTest)
01194         ctx->RestoreContinuedLanes();
01195     llvm::Value *testValue = testExpr->GetValue(ctx);
01196     if (!testValue)
01197         return;
01198 
01199     if (uniformTest)
01200         // For the uniform case, just jump to the top of the loop or the
01201         // exit basic block depending on the value of the test.
01202         ctx->BranchInst(bloop, bexit, testValue);
01203     else {
01204         // For the varying case, update the mask based on the value of the
01205         // test.  If any program instances still want to be running, jump
01206         // to the top of the loop.  Otherwise, jump out.
01207         llvm::Value *mask = ctx->GetInternalMask();
01208         ctx->SetInternalMaskAnd(mask, testValue);
01209         ctx->BranchIfMaskAny(bloop, bexit);
01210     }
01211 
01212     // ...and we're done.  Set things up for subsequent code to be emitted
01213     // in the right basic block.
01214     ctx->SetCurrentBasicBlock(bexit);
01215     ctx->EndLoop();
01216 }

01303 void
01304 ForStmt::EmitCode(FunctionEmitContext *ctx) const {
01305     if (!ctx->GetCurrentBasicBlock()) 
01306         return;
01307 
01308     llvm::BasicBlock *btest = ctx->CreateBasicBlock("for_test");
01309     llvm::BasicBlock *bstep = ctx->CreateBasicBlock("for_step");
01310     llvm::BasicBlock *bloop = ctx->CreateBasicBlock("for_loop");
01311     llvm::BasicBlock *bexit = ctx->CreateBasicBlock("for_exit");
01312 
01313     bool uniformTest = test ? test->GetType()->IsUniformType() :
01314         (!g->opt.disableUniformControlFlow &&
01315          !lHasVaryingBreakOrContinue(stmts));
01316 
01317     ctx->StartLoop(bexit, bstep, uniformTest);
01318     ctx->SetDebugPos(pos);
01319 
01320     // If we have an initiailizer statement, start by emitting the code for
01321     // it and then jump into the loop test code.  (Also start a new scope
01322     // since the initiailizer may be a declaration statement).
01323     if (init) {
01324         assert(dynamic_cast<StmtList *>(init) == NULL);
01325         ctx->StartScope();
01326         init->EmitCode(ctx);
01327     }
01328     ctx->BranchInst(btest);
01329 
01330     // Emit code to get the value of the loop test.  If no test expression
01331     // was provided, just go with a true value.
01332     ctx->SetCurrentBasicBlock(btest);
01333     llvm::Value *ltest = NULL;
01334     if (test) {
01335         ltest = test->GetValue(ctx);
01336         if (!ltest) {
01337             ctx->EndScope();
01338             ctx->EndLoop();
01339             return;
01340         }
01341     }
01342     else
01343         ltest = uniformTest ? LLVMTrue : LLVMBoolVector(true);
01344 
01345     // Now use the test's value.  For a uniform loop, we can either jump to
01346     // the loop body or the loop exit, based on whether it's true or false.
01347     // For a non-uniform loop, we update the mask and jump into the loop if
01348     // any of the mask values are true.
01349     if (uniformTest) {
01350         if (doCoherentCheck)
01351             Warning(pos, "Uniform condition supplied to cfor/cwhile statement.");
01352         assert(ltest->getType() == LLVMTypes::BoolType);
01353         ctx->BranchInst(bloop, bexit, ltest);
01354     }
01355     else {
01356         llvm::Value *mask = ctx->GetInternalMask();
01357         ctx->SetInternalMaskAnd(mask, ltest);
01358         ctx->BranchIfMaskAny(bloop, bexit);
01359     }
01360 
01361     // On to emitting the code for the loop body.
01362     ctx->SetCurrentBasicBlock(bloop);
01363     ctx->SetLoopMask(ctx->GetInternalMask());
01364     ctx->AddInstrumentationPoint("for loop body");
01365     if (!dynamic_cast<StmtList *>(stmts))
01366         ctx->StartScope();
01367 
01368     if (doCoherentCheck && !uniformTest) {
01369         // For 'varying' loops with the coherence check, we start by
01370         // checking to see if the mask is all on, after it has been updated
01371         // based on the value of the test.
01372         llvm::BasicBlock *bAllOn = ctx->CreateBasicBlock("for_all_on");
01373         llvm::BasicBlock *bMixed = ctx->CreateBasicBlock("for_mixed");
01374         ctx->BranchIfMaskAll(bAllOn, bMixed);
01375 
01376         // Emit code for the mask being all on.  Explicitly set the mask to
01377         // be on so that the optimizer can see that it's on (i.e. now that
01378         // the runtime test has passed, make this fact clear for code
01379         // generation at compile time here.)
01380         ctx->SetCurrentBasicBlock(bAllOn);
01381         ctx->SetInternalMask(LLVMMaskAllOn);
01382         llvm::Value *oldFunctionMask = ctx->GetFunctionMask();
01383         ctx->SetFunctionMask(LLVMMaskAllOn);
01384         if (stmts)
01385             stmts->EmitCode(ctx);
01386         assert(ctx->GetCurrentBasicBlock());
01387         ctx->SetFunctionMask(oldFunctionMask);
01388         ctx->BranchInst(bstep);
01389 
01390         // Emit code for the mask being mixed.  We should never run the
01391         // loop with the mask all off, based on the BranchIfMaskAny call
01392         // above.
01393         ctx->SetCurrentBasicBlock(bMixed);
01394         if (stmts)
01395             stmts->EmitCode(ctx);
01396         ctx->BranchInst(bstep);
01397     }
01398     else {
01399         // For both uniform loops and varying loops without the coherence
01400         // check, we know that at least one program instance wants to be
01401         // running the loop, so just emit code for the loop body and jump
01402         // to the loop step code.
01403         if (stmts)
01404             stmts->EmitCode(ctx);
01405         if (ctx->GetCurrentBasicBlock())
01406             ctx->BranchInst(bstep);
01407     }
01408     if (!dynamic_cast<StmtList *>(stmts))
01409         ctx->EndScope();
01410 
01411     // Emit code for the loop step.  First, restore the lane mask of any
01412     // program instances that executed a 'continue' during the previous
01413     // iteration.  Then emit code for the loop step and then jump to the
01414     // test code.
01415     ctx->SetCurrentBasicBlock(bstep);
01416     ctx->RestoreContinuedLanes();
01417     if (step)
01418         step->EmitCode(ctx);
01419     ctx->BranchInst(btest);
01420 
01421     // Set the current emission basic block to the loop exit basic block
01422     ctx->SetCurrentBasicBlock(bexit);
01423     if (init)
01424         ctx->EndScope();
01425     ctx->EndLoop();
01426 }

01610                          const std::vector<Expr *> &se,

01612                          Stmt *s, bool t, SourcePos pos)

01618 /* Given a uniform counter value in the memory location pointed to by
01619    uniformCounterPtr, compute the corresponding set of varying counter
01620    values for use within the loop body.
01621  */
01622 static llvm::Value *
01623 lUpdateVaryingCounter(int dim, int nDims, FunctionEmitContext *ctx, 
01624                       llvm::Value *uniformCounterPtr,
01625                       llvm::Value *varyingCounterPtr,
01626                       const std::vector<int> &spans) {
01627     // Smear the uniform counter value out to be varying
01628     llvm::Value *counter = ctx->LoadInst(uniformCounterPtr);
01629     llvm::Value *smearCounter = 
01630         llvm::UndefValue::get(LLVMTypes::Int32VectorType);
01631     for (int i = 0; i < g->target.vectorWidth; ++i)
01632         smearCounter = 
01633             ctx->InsertInst(smearCounter, counter, i, "smear_counter");
01634 
01635     // Figure out the offsets; this is a little bit tricky.  As an example,
01636     // consider a 2D tiled foreach loop, where we're running 8-wide and
01637     // where the inner dimension has a stride of 4 and the outer dimension
01638     // has a stride of 2.  For the inner dimension, we want the offsets
01639     // (0,1,2,3,0,1,2,3), and for the outer dimension we want
01640     // (0,0,0,0,1,1,1,1).
01641     int32_t delta[ISPC_MAX_NVEC];
01642     for (int i = 0; i < g->target.vectorWidth; ++i) {
01643         int d = i;
01644         // First, account for the effect of any dimensions at deeper
01645         // nesting levels than the current one.
01646         int prevDimSpanCount = 1;
01647         for (int j = dim; j < nDims-1; ++j)
01648             prevDimSpanCount *= spans[j+1];
01649         d /= prevDimSpanCount;
01650 
01651         // And now with what's left, figure out our own offset
01652         delta[i] = d % spans[dim];
01653     }
01654 
01655     // Add the deltas to compute the varying counter values; store the
01656     // result to memory and then return it directly as well.
01657     llvm::Value *varyingCounter = 
01658         ctx->BinaryOperator(llvm::Instruction::Add, smearCounter,
01659                             LLVMInt32Vector(delta), "iter_val");
01660     ctx->StoreInst(varyingCounter, varyingCounterPtr);
01661     return varyingCounter;
01662 }

01665 /** Returns the integer log2 of the given integer. */
01666 static int
01667 lLog2(int i) {
01668     int ret = 0;
01669     while (i != 0) {
01670         ++ret;
01671         i >>= 1;
01672     }
01673     return ret-1;
01674 }
01675 
01676 
01677 /* Figure out how many elements to process in each dimension for each time
01678    through a foreach loop.  The untiled case is easy; all of the outer
01679    dimensions up until the innermost one have a span of 1, and the
01680    innermost one takes the entire vector width.  For the tiled case, we
01681    give wider spans to the innermost dimensions while also trying to
01682    generate relatively square domains.
01683 
01684    This code works recursively from outer dimensions to inner dimensions.
01685  */
01686 static void
01687 lGetSpans(int dimsLeft, int nDims, int itemsLeft, bool isTiled, int *a) {
01688     if (dimsLeft == 0) {
01689         // Nothing left to do but give all of the remaining work to the
01690         // innermost domain.
01691         *a = itemsLeft;
01692         return;
01693     }
01694 
01695     if (isTiled == false || (dimsLeft >= lLog2(itemsLeft)))
01696         // If we're not tiled, or if there are enough dimensions left that
01697         // giving this one any more than a span of one would mean that a
01698         // later dimension would have to have a span of one, give this one
01699         // a span of one to save the available items for later.
01700         *a = 1;
01701     else if (itemsLeft >= 16 && (dimsLeft == 1))
01702         // Special case to have 4x4 domains for the 2D case when running
01703         // 16-wide.
01704         *a = 4;
01705     else
01706         // Otherwise give this dimension a span of two. 
01707         *a = 2;
01708 
01709     lGetSpans(dimsLeft-1, nDims, itemsLeft / *a, isTiled, a+1);
01710 }

01712 
01713 /* Emit code for a foreach statement.  We effectively emit code to run the
01714    set of n-dimensional nested loops corresponding to the dimensionality of
01715    the foreach statement along with the extra logic to deal with mismatches
01716    between the vector width we're compiling to and the number of elements
01717    to process.
01718  */
01719 void
01720 ForeachStmt::EmitCode(FunctionEmitContext *ctx) const {
01721     if (ctx->GetCurrentBasicBlock() == NULL || stmts == NULL) 
01722         return;
01723 
01724     llvm::BasicBlock *bbCheckExtras = ctx->CreateBasicBlock("foreach_check_extras");
01725     llvm::BasicBlock *bbDoExtras = ctx->CreateBasicBlock("foreach_do_extras");
01726     llvm::BasicBlock *bbBody = ctx->CreateBasicBlock("foreach_body");
01727     llvm::BasicBlock *bbExit = ctx->CreateBasicBlock("foreach_exit");
01728 
01729     llvm::Value *oldMask = ctx->GetInternalMask();
01730 
01731     ctx->SetDebugPos(pos);
01732     ctx->StartScope();
01733 
01734     // This should be caught during typechecking
01735     assert(startExprs.size() == dimVariables.size() && 
01736            endExprs.size() == dimVariables.size());
01737     int nDims = (int)dimVariables.size();
01738 
01739     ///////////////////////////////////////////////////////////////////////
01740     // Setup: compute the number of items we have to work on in each
01741     // dimension and a number of derived values.
01742     std::vector<llvm::BasicBlock *> bbReset, bbStep, bbTest;
01743     std::vector<llvm::Value *> startVals, endVals, uniformCounterPtrs;
01744     std::vector<llvm::Value *> nItems, nExtras, alignedEnd;
01745     std::vector<llvm::Value *> extrasMaskPtrs;
01746 
01747     std::vector<int> span(nDims, 0);
01748     lGetSpans(nDims-1, nDims, g->target.vectorWidth, isTiled, &span[0]);
01749 
01750     for (int i = 0; i < nDims; ++i) {
01751         // Basic blocks that we'll fill in later with the looping logic for
01752         // this dimension.
01753         bbReset.push_back(ctx->CreateBasicBlock("foreach_reset"));
01754         bbStep.push_back(ctx->CreateBasicBlock("foreach_step"));
01755         bbTest.push_back(ctx->CreateBasicBlock("foreach_test"));
01756 
01757         // Start and end value for this loop dimension
01758         llvm::Value *sv = startExprs[i]->GetValue(ctx);
01759         llvm::Value *ev = endExprs[i]->GetValue(ctx);
01760         if (sv == NULL || ev == NULL)
01761             return;
01762         startVals.push_back(sv);
01763         endVals.push_back(ev);
01764 
01765         // nItems = endVal - startVal
01766         nItems.push_back(ctx->BinaryOperator(llvm::Instruction::Sub, ev, sv,
01767                                              "nitems"));
01768 
01769         // nExtras = nItems % (span for this dimension)
01770         // This gives us the number of extra elements we need to deal with
01771         // at the end of the loop for this dimension that don't fit cleanly
01772         // into a vector width.
01773         nExtras.push_back(ctx->BinaryOperator(llvm::Instruction::SRem, nItems[i],
01774                                               LLVMInt32(span[i]), "nextras"));
01775 
01776         // alignedEnd = endVal - nExtras
01777         alignedEnd.push_back(ctx->BinaryOperator(llvm::Instruction::Sub, ev,
01778                                                  nExtras[i], "aligned_end"));
01779 
01780         ///////////////////////////////////////////////////////////////////////
01781         // Each dimension has a loop counter that is a uniform value that
01782         // goes from startVal to endVal, in steps of the span for this
01783         // dimension.  Its value is only used internally here for looping
01784         // logic and isn't directly available in the user's program code.
01785         uniformCounterPtrs.push_back(ctx->AllocaInst(LLVMTypes::Int32Type, 
01786                                                      "counter"));
01787         ctx->StoreInst(startVals[i], uniformCounterPtrs[i]);
01788 
01789         // There is also a varying variable that holds the set of index
01790         // values for each dimension in the current loop iteration; this is
01791         // the value that is program-visible.
01792         dimVariables[i]->storagePtr = ctx->AllocaInst(LLVMTypes::Int32VectorType, 
01793                                                   dimVariables[i]->name.c_str());
01794         dimVariables[i]->parentFunction = ctx->GetFunction();
01795         ctx->EmitVariableDebugInfo(dimVariables[i]);
01796 
01797         // Each dimension also maintains a mask that represents which of
01798         // the varying elements in the current iteration should be
01799         // processed.  (i.e. this is used to disable the lanes that have
01800         // out-of-bounds offsets.)
01801         extrasMaskPtrs.push_back(ctx->AllocaInst(LLVMTypes::MaskType, "extras mask"));
01802         ctx->StoreInst(LLVMMaskAllOn, extrasMaskPtrs[i]);
01803     }
01804 
01805     ctx->StartForeach(bbStep[nDims-1]);
01806 
01807     // On to the outermost loop's test
01808     ctx->BranchInst(bbTest[0]);
01809 
01810     ///////////////////////////////////////////////////////////////////////////
01811     // foreach_reset: this code runs when we need to reset the counter for
01812     // a given dimension in preparation for running through its loop again,
01813     // after the enclosing level advances its counter.
01814     for (int i = 0; i < nDims; ++i) {
01815         ctx->SetCurrentBasicBlock(bbReset[i]);
01816         if (i == 0)
01817             ctx->BranchInst(bbExit);
01818         else {
01819             ctx->StoreInst(LLVMMaskAllOn, extrasMaskPtrs[i]);
01820             ctx->StoreInst(startVals[i], uniformCounterPtrs[i]);
01821             ctx->BranchInst(bbStep[i-1]);
01822         }
01823     }
01824 
01825     ///////////////////////////////////////////////////////////////////////////
01826     // foreach_test
01827     std::vector<llvm::Value *> inExtras;
01828     for (int i = 0; i < nDims; ++i) {
01829         ctx->SetCurrentBasicBlock(bbTest[i]);
01830 
01831         llvm::Value *haveExtras = 
01832             ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SGT,
01833                          endVals[i], alignedEnd[i], "have_extras");
01834 
01835         llvm::Value *counter = ctx->LoadInst(uniformCounterPtrs[i], "counter");
01836         llvm::Value *atAlignedEnd = 
01837             ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_EQ,
01838                          counter, alignedEnd[i], "at_aligned_end");
01839         llvm::Value *inEx = 
01840             ctx->BinaryOperator(llvm::Instruction::And, haveExtras,
01841                                 atAlignedEnd, "in_extras");
01842 
01843         if (i == 0)
01844             inExtras.push_back(inEx);
01845         else
01846             inExtras.push_back(ctx->BinaryOperator(llvm::Instruction::Or, inEx,
01847                                                    inExtras[i-1], "in_extras_all"));
01848 
01849         llvm::Value *varyingCounter = 
01850             lUpdateVaryingCounter(i, nDims, ctx, uniformCounterPtrs[i], 
01851                                   dimVariables[i]->storagePtr, span);
01852 
01853         llvm::Value *smearEnd = llvm::UndefValue::get(LLVMTypes::Int32VectorType);
01854         for (int j = 0; j < g->target.vectorWidth; ++j)
01855             smearEnd = ctx->InsertInst(smearEnd, endVals[i], j, "smear_end");
01856         // Do a vector compare of its value to the end value to generate a
01857         // mask for this last bit of work.
01858         llvm::Value *emask = 
01859             ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SLT,
01860                          varyingCounter, smearEnd);
01861         emask = ctx->I1VecToBoolVec(emask);
01862 
01863         if (i == 0)
01864             ctx->StoreInst(emask, extrasMaskPtrs[i]);
01865         else {
01866             // FIXME: at least specialize the innermost loop to not do all
01867             // this mask stuff each time through the test...
01868             llvm::Value *oldMask = ctx->LoadInst(extrasMaskPtrs[i-1]);
01869             llvm::Value *newMask =
01870                 ctx->BinaryOperator(llvm::Instruction::And, oldMask, emask,
01871                                     "extras_mask");
01872             ctx->StoreInst(newMask, extrasMaskPtrs[i]);
01873         }
01874 
01875         llvm::Value *notAtEnd = 
01876             ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SLT,
01877                          counter, endVals[i]);
01878         if (i != nDims-1)
01879             ctx->BranchInst(bbTest[i+1], bbReset[i], notAtEnd);
01880         else
01881             ctx->BranchInst(bbCheckExtras, bbReset[i], notAtEnd);
01882     }
01883 
01884     ///////////////////////////////////////////////////////////////////////////
01885     // foreach_step: increment the uniform counter by the vector width.
01886     // Note that we don't increment the varying counter here as well but
01887     // just generate its value when we need it in the loop body.
01888     for (int i = 0; i < nDims; ++i) {
01889         ctx->SetCurrentBasicBlock(bbStep[i]);
01890         if (i == nDims-1)
01891             ctx->RestoreContinuedLanes();
01892         llvm::Value *counter = ctx->LoadInst(uniformCounterPtrs[i]);
01893         llvm::Value *newCounter =  
01894             ctx->BinaryOperator(llvm::Instruction::Add, counter,
01895                                 LLVMInt32(span[i]), "new_counter");
01896         ctx->StoreInst(newCounter, uniformCounterPtrs[i]);
01897         ctx->BranchInst(bbTest[i]);
01898     }
01899 
01900     ///////////////////////////////////////////////////////////////////////////
01901     // foreach_check_extras: see if we need to deal with any partial
01902     // vector's worth of work that's left.
01903     ctx->SetCurrentBasicBlock(bbCheckExtras);
01904     ctx->AddInstrumentationPoint("foreach loop check extras");
01905     ctx->BranchInst(bbDoExtras, bbBody, inExtras[nDims-1]);
01906 
01907     ///////////////////////////////////////////////////////////////////////////
01908     // foreach_body: do a full vector's worth of work.  We know that all
01909     // lanes will be running here, so we explicitly set the mask to be 'all
01910     // on'.  This ends up being relatively straightforward: just update the
01911     // value of the varying loop counter and have the statements in the
01912     // loop body emit their code.
01913     ctx->SetCurrentBasicBlock(bbBody);
01914     ctx->SetInternalMask(LLVMMaskAllOn);
01915     ctx->AddInstrumentationPoint("foreach loop body");
01916     stmts->EmitCode(ctx);
01917     assert(ctx->GetCurrentBasicBlock() != NULL);
01918     ctx->BranchInst(bbStep[nDims-1]);
01919 
01920     ///////////////////////////////////////////////////////////////////////////
01921     // foreach_doextras: set the mask and have the statements emit their
01922     // code again.  Note that it's generally worthwhile having two copies
01923     // of the statements' code, since the code above is emitted with the
01924     // mask known to be all-on, which in turn leads to more efficient code
01925     // for that case.
01926     ctx->SetCurrentBasicBlock(bbDoExtras);
01927     llvm::Value *mask = ctx->LoadInst(extrasMaskPtrs[nDims-1]);
01928     ctx->SetInternalMask(mask);
01929     stmts->EmitCode(ctx);
01930     ctx->BranchInst(bbStep[nDims-1]);
01931 
01932     ///////////////////////////////////////////////////////////////////////////
01933     // foreach_exit: All done.  Restore the old mask and clean up
01934     ctx->SetCurrentBasicBlock(bbExit);
01935     ctx->SetInternalMask(oldMask);
01936 
01937     ctx->EndForeach();
01938     ctx->EndScope();
01939 }

02188 /* Because the pointers to values that are passed to __do_print() are all
02189    void *s (and because ispc print() formatting strings statements don't
02190    encode types), we pass along a string to __do_print() where the i'th
02191    character encodes the type of the i'th value to be printed.  Needless to
02192    say, the encoding chosen here and the decoding code in __do_print() need
02193    to agree on the below!
02194  */
02195 static char
02196 lEncodeType(const Type *t) {
02197     if (t == AtomicType::UniformBool)   return 'b';
02198     if (t == AtomicType::VaryingBool)   return 'B';
02199     if (t == AtomicType::UniformInt32)  return 'i';
02200     if (t == AtomicType::VaryingInt32)  return 'I';
02201     if (t == AtomicType::UniformUInt32) return 'u';
02202     if (t == AtomicType::VaryingUInt32) return 'U';
02203     if (t == AtomicType::UniformFloat)  return 'f';
02204     if (t == AtomicType::VaryingFloat)  return 'F';
02205     if (t == AtomicType::UniformInt64)  return 'l';
02206     if (t == AtomicType::VaryingInt64)  return 'L';
02207     if (t == AtomicType::UniformUInt64) return 'v';
02208     if (t == AtomicType::VaryingUInt64) return 'V';
02209     if (t == AtomicType::UniformDouble) return 'd';
02210     if (t == AtomicType::VaryingDouble) return 'D';
02211     if (dynamic_cast<const PointerType *>(t) != NULL) {
02212         if (t->IsUniformType())
02213             return 'p';
02214         else
02215             return 'P';
02216     }
02217     else return '\0';
02218 }

02221 /** Given an Expr for a value to be printed, emit the code to evaluate the
02222     expression and store the result to alloca'd memory.  Update the
02223     argTypes string with the type encoding for this expression.
02224  */
02225 static llvm::Value *
02226 lProcessPrintArg(Expr *expr, FunctionEmitContext *ctx, std::string &argTypes) {
02227     const Type *type = expr->GetType();
02228     if (type == NULL)
02229         return NULL;
02230 
02231     if (dynamic_cast<const ReferenceType *>(type) != NULL) {
02232         expr = new DereferenceExpr(expr, expr->pos);
02233         type = expr->GetType();
02234         if (type == NULL)
02235             return NULL;
02236     }
02237 
02238     // Just int8 and int16 types to int32s...
02239     const Type *baseType = type->GetAsNonConstType()->GetAsUniformType();
02240     if (baseType == AtomicType::UniformInt8 ||
02241         baseType == AtomicType::UniformUInt8 ||
02242         baseType == AtomicType::UniformInt16 ||
02243         baseType == AtomicType::UniformUInt16) {
02244         expr = new TypeCastExpr(type->IsUniformType() ? AtomicType::UniformInt32 :
02245                                                         AtomicType::VaryingInt32, 
02246                                 expr, false, expr->pos);
02247         type = expr->GetType();
02248     }
02249         
02250     char t = lEncodeType(type->GetAsNonConstType());
02251     if (t == '\0') {
02252         Error(expr->pos, "Only atomic types are allowed in print statements; "
02253               "type \"%s\" is illegal.", type->GetString().c_str());
02254         return NULL;
02255     }
02256     else {
02257         argTypes.push_back(t);
02258 
02259         LLVM_TYPE_CONST llvm::Type *llvmExprType = type->LLVMType(g->ctx);
02260         llvm::Value *ptr = ctx->AllocaInst(llvmExprType, "print_arg");
02261         llvm::Value *val = expr->GetValue(ctx);
02262         if (!val)
02263             return NULL;
02264         ctx->StoreInst(val, ptr);
02265 
02266         ptr = ctx->BitCastInst(ptr, LLVMTypes::VoidPointerType);
02267         return ptr;
02268     }

02272 /* PrintStmt works closely with the __do_print() function implemented in
02273    the builtins-c.c file.  In particular, the EmitCode() method here needs to
02274    take the arguments passed to it from ispc and generate a valid call to
02275    __do_print() with the information that __do_print() then needs to do the
02276    actual printing work at runtime.
02277  */
02278 void
02279 PrintStmt::EmitCode(FunctionEmitContext *ctx) const {
02280     ctx->SetDebugPos(pos);
02281 
02282     // __do_print takes 5 arguments; we'll get them stored in the args[] array
02283     // in the code emitted below
02284     //
02285     // 1. the format string
02286     // 2. a string encoding the types of the values being printed, 
02287     //    one character per value
02288     // 3. the number of running program instances (i.e. the target's
02289     //    vector width)
02290     // 4. the current lane mask
02291     // 5. a pointer to an array of pointers to the values to be printed
02292     llvm::Value *args[5];
02293     std::string argTypes;
02294 
02295     if (values == NULL) {
02296         LLVM_TYPE_CONST llvm::Type *ptrPtrType = 
02297             llvm::PointerType::get(LLVMTypes::VoidPointerType, 0);
02298         args[4] = llvm::Constant::getNullValue(ptrPtrType);
02299     }
02300     else {
02301         // Get the values passed to the print() statement evaluated and
02302         // stored in memory so that we set up the array of pointers to them
02303         // for the 5th __do_print() argument
02304         ExprList *elist = dynamic_cast<ExprList *>(values);
02305         int nArgs = elist ? elist->exprs.size() : 1;
02306 
02307         // Allocate space for the array of pointers to values to be printed 
02308         LLVM_TYPE_CONST llvm::Type *argPtrArrayType = 
02309             llvm::ArrayType::get(LLVMTypes::VoidPointerType, nArgs);
02310         llvm::Value *argPtrArray = ctx->AllocaInst(argPtrArrayType,
02311                                                    "print_arg_ptrs");
02312         // Store the array pointer as a void **, which is what __do_print()
02313         // expects
02314         args[4] = ctx->BitCastInst(argPtrArray, 
02315                                    llvm::PointerType::get(LLVMTypes::VoidPointerType, 0));
02316 
02317         // Now, for each of the arguments, emit code to evaluate its value
02318         // and store the value into alloca'd storage.  Then store the
02319         // pointer to the alloca'd storage into argPtrArray.
02320         if (elist) {
02321             for (unsigned int i = 0; i < elist->exprs.size(); ++i) {
02322                 Expr *expr = elist->exprs[i];
02323                 if (!expr)
02324                     return;
02325                 llvm::Value *ptr = lProcessPrintArg(expr, ctx, argTypes);
02326                 if (!ptr)
02327                     return;
02328 
02329                 llvm::Value *arrayPtr = ctx->AddElementOffset(argPtrArray, i, NULL);
02330                 ctx->StoreInst(ptr, arrayPtr);
02331             }
02332         }
02333         else {
02334             llvm::Value *ptr = lProcessPrintArg(values, ctx, argTypes);
02335             if (!ptr)
02336                 return;
02337             llvm::Value *arrayPtr = ctx->AddElementOffset(argPtrArray, 0, NULL);
02338             ctx->StoreInst(ptr, arrayPtr);
02339         }
02340     }
02341 
02342     // Now we can emit code to call __do_print()
02343     llvm::Function *printFunc = m->module->getFunction("__do_print");
02344     assert(printFunc);
02345 
02346     llvm::Value *mask = ctx->GetFullMask();
02347     // Set up the rest of the parameters to it
02348     args[0] = ctx->GetStringPtr(format);
02349     args[1] = ctx->GetStringPtr(argTypes);
02350     args[2] = LLVMInt32(g->target.vectorWidth);
02351     args[3] = ctx->LaneMask(mask);
02352     std::vector<llvm::Value *> argVec(&args[0], &args[5]);
02353     ctx->CallInst(printFunc, NULL, argVec, "");
02354 }
