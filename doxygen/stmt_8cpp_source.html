00240 DeclStmt::Optimize() {
00241     for (unsigned int i = 0; i < vars.size(); ++i) {
00242         Expr *init = vars[i].init;
00243         if (init != NULL && dynamic_cast<ExprList *>(init) == NULL) {
00244             // If the variable is const-qualified, after we've optimized
00245             // the initializer expression, see if we have a ConstExpr.  If
00246             // so, save it in Symbol::constValue where it can be used in
00247             // optimizing later expressions that have this symbol in them.
00248             // Note that there are cases where the expression may be
00249             // constant but where we don't have a ConstExpr; an example is
00250             // const arrays--the ConstExpr implementation just can't
00251             // represent an array of values.
00252             //
00253             // All this is fine in terms of the code that's generated in
00254             // the end (LLVM's constant folding stuff is good), but it
00255             // means that the ispc compiler's ability to reason about what
00256             // is definitely a compile-time constant for things like
00257             // computing array sizes from non-trivial expressions is
00258             // consequently limited.
00259             Symbol *sym = vars[i].sym;
00260             if (sym->type && sym->type->IsConstType() && 
00261                 Type::Equal(init->GetType(), sym->type))
00262                 sym->constValue = dynamic_cast<ConstExpr *>(init);
00263         }
00264     }
00265     return this;

00379         llvm::BasicBlock *belse = ctx->CreateBasicBlock("if_else");

00383         // the 'if' test
00384         ctx->BranchInst(bthen, belse, testValue);

00392         // Emit code for the 'false' case
00393         ctx->SetCurrentBasicBlock(belse);

00396             ctx->BranchInst(bexit);

00476 /** Emit code for an if test that checks the mask and the test values and
00477     tries to be smart about jumping over code that doesn't need to be run.
00478  */
00479 void
00480 IfStmt::emitVaryingIf(FunctionEmitContext *ctx, llvm::Value *ltest) const {
00481     llvm::Value *oldMask = ctx->GetInternalMask();
00482     if (ctx->GetFullMask() == LLVMMaskAllOn && 
00483         !g->opt.disableCoherentControlFlow &&
00484         !g->opt.disableMaskAllOnOptimizations) {
00485         // We can tell that the mask is on statically at compile time; just
00486         // emit code for the 'if test with the mask all on' path
00487         llvm::BasicBlock *bDone = ctx->CreateBasicBlock("cif_done");
00488         emitMaskAllOn(ctx, ltest, bDone);
00489         ctx->SetCurrentBasicBlock(bDone);
00490     }
00491     else if (doAllCheck) {
00492         // We can't tell if the mask going into the if is all on at the
00493         // compile time.  Emit code to check for this and then either run
00494         // the code for the 'all on' or the 'mixed' case depending on the
00495         // mask's value at runtime.
00496         llvm::BasicBlock *bAllOn = ctx->CreateBasicBlock("cif_mask_all");
00497         llvm::BasicBlock *bMixedOn = ctx->CreateBasicBlock("cif_mask_mixed");
00498         llvm::BasicBlock *bDone = ctx->CreateBasicBlock("cif_done");
00499 
00500         // Jump to either bAllOn or bMixedOn, depending on the mask's value 
00501         llvm::Value *maskAllQ = ctx->All(ctx->GetFullMask());
00502         ctx->BranchInst(bAllOn, bMixedOn, maskAllQ);
00503 
00504         // Emit code for the 'mask all on' case
00505         ctx->SetCurrentBasicBlock(bAllOn);
00506         emitMaskAllOn(ctx, ltest, bDone);
00507         
00508         // And emit code for the mixed mask case
00509         ctx->SetCurrentBasicBlock(bMixedOn);
00510         emitMaskMixed(ctx, oldMask, ltest, bDone);
00511 
00512         // When done, set the current basic block to the block that the two
00513         // paths above jump to when they're done.
00514         ctx->SetCurrentBasicBlock(bDone);
00515     }
00516     else if (trueStmts != NULL || falseStmts != NULL) {
00517         // If there is nothing that is potentially unsafe to run with all
00518         // lanes off in the true and false statements and if the total
00519         // complexity of those two is relatively simple, then we'll go
00520         // ahead and emit straightline code that runs both sides, updating
00521         // the mask accordingly.  This is useful for efficiently compiling
00522         // things like:
00523         //
00524         // if (foo) x = 0;
00525         // else     ++x;
00526         //
00527         // Where the overhead of checking if any of the program instances wants
00528         // to run one side or the other is more than the actual computation.
00529         // SafeToRunWithMaskAllOff() checks to make sure that we don't do this
00530         // for potentially dangerous code like:
00531         //
00532         // if (index < count) array[index] = 0;
00533         //
00534         // where our use of blend for conditional assignments doesn't check
00535         // for the 'all lanes' off case.
00536         int trueFalseCost = (::EstimateCost(trueStmts) + 
00537                              ::EstimateCost(falseStmts));
00538         bool costIsAcceptable = (trueFalseCost <
00539                                  PREDICATE_SAFE_IF_STATEMENT_COST);
00540 
00541         bool safeToRunWithAllLanesOff = (SafeToRunWithMaskAllOff(trueStmts) &&
00542                                          SafeToRunWithMaskAllOff(falseStmts));
00543 
00544         if (safeToRunWithAllLanesOff &&
00545             (costIsAcceptable || g->opt.disableCoherentControlFlow)) {
00546             ctx->StartVaryingIf(oldMask);
00547             emitMaskedTrueAndFalse(ctx, oldMask, ltest);
00548             Assert(ctx->GetCurrentBasicBlock());
00549             ctx->EndIf();
00550         }
00551         else {
00552             llvm::BasicBlock *bDone = ctx->CreateBasicBlock("if_done");
00553             emitMaskMixed(ctx, oldMask, ltest, bDone);
00554             ctx->SetCurrentBasicBlock(bDone);
00555         }
00556     }
00557 }

00641

00643         // block?  If not, jump ahead to bNext.

00645         ctx->BranchInst(bRunTrue, bNext, maskAnyQ);
00646

00648         ctx->SetCurrentBasicBlock(bRunTrue);

00652         ctx->SetCurrentBasicBlock(bNext);
00653     }

00655         llvm::BasicBlock *bRunFalse = ctx->CreateBasicBlock("safe_if_run_false");

00691 /** Returns true if the given node is an 'if' statement where the test
00692     condition has varying type. */
00693 static bool
00694 lIsVaryingFor(ASTNode *node) {
00695     IfStmt *ifStmt;
00696     if ((ifStmt = dynamic_cast<IfStmt *>(node)) != NULL &&
00697         ifStmt->test != NULL) {
00698         const Type *type = ifStmt->test->GetType();
00699         return (type != NULL && type->IsVaryingType());
00700     }
00701     else
00702         return false;

00706 /** Preorder callback function for checking for varying breaks or
00707     continues. */
00708 static bool
00709 lVaryingBCPreFunc(ASTNode *node, void *d) {
00710     VaryingBCCheckInfo *info = (VaryingBCCheckInfo *)d;
00711 
00712     // We found a break or continue statement; if we're under varying
00713     // control flow, then bingo.
00714     if ((dynamic_cast<BreakStmt *>(node) != NULL ||
00715          dynamic_cast<ContinueStmt *>(node) != NULL) &&
00716         info->varyingControlFlowDepth > 0) {
00717         info->foundVaryingBreakOrContinue = true;
00718         return false;
00719     }
00720 
00721     // Update the count of the nesting depth of varying control flow if
00722     // this is an if statement with a varying condition.
00723     if (lIsVaryingFor(node))
00724         ++info->varyingControlFlowDepth;
00725 
00726     if (dynamic_cast<ForStmt *>(node) != NULL ||
00727         dynamic_cast<DoStmt *>(node) != NULL ||
00728         dynamic_cast<ForeachStmt *>(node) != NULL)
00729         // Don't recurse into these guys, since we don't care about varying
00730         // breaks or continues within them...
00731         return false;
00732     else
00733         return true;
00734 }

00737 /** Postorder callback function for checking for varying breaks or
00738     continues; decrement the varying control flow depth after the node's
00739     children have been processed, if this is a varying if statement. */
00740 static ASTNode *
00741 lVaryingBCPostFunc(ASTNode *node, void *d) {
00742     VaryingBCCheckInfo *info = (VaryingBCCheckInfo *)d;
00743     if (lIsVaryingFor(node))
00744         --info->varyingControlFlowDepth;
00745     return node;
00746 }

00755 static bool
00756 lHasVaryingBreakOrContinue(Stmt *stmt) {
00757     VaryingBCCheckInfo info;

00769 void DoStmt::EmitCode(FunctionEmitContext *ctx) const {
00770     // Check for things that could be NULL due to earlier errors during
00771     // compilation.
00772     if (!ctx->GetCurrentBasicBlock()) 
00773         return;
00774     if (!testExpr || !testExpr->GetType()) 
00775         return;
00776 
00777     bool uniformTest = testExpr->GetType()->IsUniformType();
00778     if (uniformTest && doCoherentCheck)
00779         Warning(testExpr->pos, "Uniform condition supplied to \"cdo\" "
00780                 "statement.");
00781 
00782     llvm::BasicBlock *bloop = ctx->CreateBasicBlock("do_loop");
00783     llvm::BasicBlock *bexit = ctx->CreateBasicBlock("do_exit");
00784     llvm::BasicBlock *btest = ctx->CreateBasicBlock("do_test");
00785 
00786     ctx->StartLoop(bexit, btest, uniformTest);
00787 
00788     // Start by jumping into the loop body
00789     ctx->BranchInst(bloop);
00790 
00791     // And now emit code for the loop body
00792     ctx->SetCurrentBasicBlock(bloop);
00793     ctx->SetLoopMask(ctx->GetInternalMask());
00794     ctx->SetDebugPos(pos);
00795     // FIXME: in the StmtList::EmitCode() method takes starts/stops a new
00796     // scope around the statements in the list.  So if the body is just a
00797     // single statement (and thus not a statement list), we need a new
00798     // scope, but we don't want two scopes in the StmtList case.
00799     if (!dynamic_cast<StmtList *>(bodyStmts))
00800         ctx->StartScope();
00801 
00802     ctx->AddInstrumentationPoint("do loop body");
00803     if (doCoherentCheck && !uniformTest) {
00804         // Check to see if the mask is all on
00805         llvm::BasicBlock *bAllOn = ctx->CreateBasicBlock("do_all_on");
00806         llvm::BasicBlock *bMixed = ctx->CreateBasicBlock("do_mixed");
00807         ctx->BranchIfMaskAll(bAllOn, bMixed);
00808 
00809         // If so, emit code for the 'mask all on' case.  In particular,
00810         // explicitly set the mask to 'all on' (see rationale in
00811         // IfStmt::emitCoherentTests()), and then emit the code for the
00812         // loop body.
00813         ctx->SetCurrentBasicBlock(bAllOn);
00814         if (!g->opt.disableMaskAllOnOptimizations)
00815             ctx->SetInternalMask(LLVMMaskAllOn);
00816         llvm::Value *oldFunctionMask = ctx->GetFunctionMask();
00817         if (!g->opt.disableMaskAllOnOptimizations)
00818             ctx->SetFunctionMask(LLVMMaskAllOn);
00819         if (bodyStmts)
00820             bodyStmts->EmitCode(ctx);
00821         Assert(ctx->GetCurrentBasicBlock());
00822         ctx->SetFunctionMask(oldFunctionMask);
00823         ctx->BranchInst(btest);
00824 
00825         // The mask is mixed.  Just emit the code for the loop body.
00826         ctx->SetCurrentBasicBlock(bMixed);
00827         if (bodyStmts)
00828             bodyStmts->EmitCode(ctx);
00829         Assert(ctx->GetCurrentBasicBlock());
00830         ctx->BranchInst(btest);
00831     }
00832     else {
00833         // Otherwise just emit the code for the loop body.  The current
00834         // mask is good.
00835         if (bodyStmts)
00836             bodyStmts->EmitCode(ctx);
00837         if (ctx->GetCurrentBasicBlock())
00838             ctx->BranchInst(btest);
00839     }
00840     // End the scope we started above, if needed.
00841     if (!dynamic_cast<StmtList *>(bodyStmts))
00842         ctx->EndScope();
00843 
00844     // Now emit code for the loop test.
00845     ctx->SetCurrentBasicBlock(btest);
00846     // First, emit code to restore the mask value for any lanes that
00847     // executed a 'continue' during the current loop before we go and emit
00848     // the code for the test.  This is only necessary for varying loops;
00849     // 'uniform' loops just jump when they hit a continue statement and
00850     // don't mess with the mask.
00851     if (!uniformTest)
00852         ctx->RestoreContinuedLanes();
00853     llvm::Value *testValue = testExpr->GetValue(ctx);
00854     if (!testValue)
00855         return;
00856 
00857     if (uniformTest)
00858         // For the uniform case, just jump to the top of the loop or the
00859         // exit basic block depending on the value of the test.
00860         ctx->BranchInst(bloop, bexit, testValue);
00861     else {
00862         // For the varying case, update the mask based on the value of the
00863         // test.  If any program instances still want to be running, jump
00864         // to the top of the loop.  Otherwise, jump out.
00865         llvm::Value *mask = ctx->GetInternalMask();
00866         ctx->SetInternalMaskAnd(mask, testValue);
00867         ctx->BranchIfMaskAny(bloop, bexit);
00868     }
00869 
00870     // ...and we're done.  Set things up for subsequent code to be emitted
00871     // in the right basic block.
00872     ctx->SetCurrentBasicBlock(bexit);
00873     ctx->EndLoop();
00874 }

00877 Stmt *
00878 DoStmt::TypeCheck() {
00879     const Type *testType;
00880     if (testExpr != NULL && (testType = testExpr->GetType()) != NULL) {
00881         if (!testType->IsNumericType() && !testType->IsBoolType()) {
00882             Error(testExpr->pos, "Type \"%s\" can't be converted to boolean for \"while\" "
00883                   "test in \"do\" loop.", testExpr->GetType()->GetString().c_str());
00884             return NULL;
00885         }
00886 
00887         // Should the test condition for the loop be uniform or varying?
00888         // It can be uniform only if three conditions are met:
00889         //
00890         // - First and foremost, the type of the test condition must be
00891         //   uniform.
00892         //
00893         // - Second, the user must not have set the dis-optimization option
00894         //   that disables uniform flow control.
00895         //
00896         // - Thirdly, and most subtlely, there must not be any break or
00897         //   continue statements inside the loop that are within the scope
00898         //   of a 'varying' if statement.  If there are, then we type cast
00899         //   the test to be 'varying', so that the code generated for the
00900         //   loop includes masking stuff, so that we can track which lanes
00901         //   actually want to be running, accounting for breaks/continues.
00902         //
00903         bool uniformTest = (testType->IsUniformType() &&
00904                             !g->opt.disableUniformControlFlow &&
00905                             !lHasVaryingBreakOrContinue(bodyStmts));
00906         testExpr = new TypeCastExpr(uniformTest ? AtomicType::UniformBool :
00907                                                   AtomicType::VaryingBool,
00908                                     testExpr, testExpr->pos);
00909     }
00910 
00911     return this;
00912 }

00949 void
00950 ForStmt::EmitCode(FunctionEmitContext *ctx) const {
00951     if (!ctx->GetCurrentBasicBlock()) 
00952         return;
00953 
00954     llvm::BasicBlock *btest = ctx->CreateBasicBlock("for_test");
00955     llvm::BasicBlock *bstep = ctx->CreateBasicBlock("for_step");
00956     llvm::BasicBlock *bloop = ctx->CreateBasicBlock("for_loop");
00957     llvm::BasicBlock *bexit = ctx->CreateBasicBlock("for_exit");
00958 
00959     bool uniformTest = test ? test->GetType()->IsUniformType() :
00960         (!g->opt.disableUniformControlFlow &&
00961          !lHasVaryingBreakOrContinue(stmts));
00962 
00963     ctx->StartLoop(bexit, bstep, uniformTest);
00964     ctx->SetDebugPos(pos);
00965 
00966     // If we have an initiailizer statement, start by emitting the code for
00967     // it and then jump into the loop test code.  (Also start a new scope
00968     // since the initiailizer may be a declaration statement).
00969     if (init) {
00970         Assert(dynamic_cast<StmtList *>(init) == NULL);
00971         ctx->StartScope();
00972         init->EmitCode(ctx);
00973     }
00974     ctx->BranchInst(btest);
00975 
00976     // Emit code to get the value of the loop test.  If no test expression
00977     // was provided, just go with a true value.
00978     ctx->SetCurrentBasicBlock(btest);
00979     llvm::Value *ltest = NULL;
00980     if (test) {
00981         ltest = test->GetValue(ctx);
00982         if (!ltest) {
00983             ctx->EndScope();
00984             ctx->EndLoop();
00985             return;
00986         }
00987     }
00988     else
00989         ltest = uniformTest ? LLVMTrue : LLVMBoolVector(true);
00990 
00991     // Now use the test's value.  For a uniform loop, we can either jump to
00992     // the loop body or the loop exit, based on whether it's true or false.
00993     // For a non-uniform loop, we update the mask and jump into the loop if
00994     // any of the mask values are true.
00995     if (uniformTest) {
00996         if (doCoherentCheck)
00997             Warning(test->pos, "Uniform condition supplied to cfor/cwhile "
00998                     "statement.");
00999         Assert(ltest->getType() == LLVMTypes::BoolType);
01000         ctx->BranchInst(bloop, bexit, ltest);
01001     }
01002     else {
01003         llvm::Value *mask = ctx->GetInternalMask();
01004         ctx->SetInternalMaskAnd(mask, ltest);
01005         ctx->BranchIfMaskAny(bloop, bexit);
01006     }
01007 
01008     // On to emitting the code for the loop body.
01009     ctx->SetCurrentBasicBlock(bloop);
01010     ctx->SetLoopMask(ctx->GetInternalMask());
01011     ctx->AddInstrumentationPoint("for loop body");
01012     if (!dynamic_cast<StmtList *>(stmts))
01013         ctx->StartScope();
01014 
01015     if (doCoherentCheck && !uniformTest) {
01016         // For 'varying' loops with the coherence check, we start by
01017         // checking to see if the mask is all on, after it has been updated
01018         // based on the value of the test.
01019         llvm::BasicBlock *bAllOn = ctx->CreateBasicBlock("for_all_on");
01020         llvm::BasicBlock *bMixed = ctx->CreateBasicBlock("for_mixed");
01021         ctx->BranchIfMaskAll(bAllOn, bMixed);
01022 
01023         // Emit code for the mask being all on.  Explicitly set the mask to
01024         // be on so that the optimizer can see that it's on (i.e. now that
01025         // the runtime test has passed, make this fact clear for code
01026         // generation at compile time here.)
01027         ctx->SetCurrentBasicBlock(bAllOn);
01028         if (!g->opt.disableMaskAllOnOptimizations)
01029             ctx->SetInternalMask(LLVMMaskAllOn);
01030         llvm::Value *oldFunctionMask = ctx->GetFunctionMask();
01031         if (!g->opt.disableMaskAllOnOptimizations)
01032             ctx->SetFunctionMask(LLVMMaskAllOn);
01033         if (stmts)
01034             stmts->EmitCode(ctx);
01035         Assert(ctx->GetCurrentBasicBlock());
01036         ctx->SetFunctionMask(oldFunctionMask);
01037         ctx->BranchInst(bstep);
01038 
01039         // Emit code for the mask being mixed.  We should never run the
01040         // loop with the mask all off, based on the BranchIfMaskAny call
01041         // above.
01042         ctx->SetCurrentBasicBlock(bMixed);
01043         if (stmts)
01044             stmts->EmitCode(ctx);
01045         ctx->BranchInst(bstep);
01046     }
01047     else {
01048         // For both uniform loops and varying loops without the coherence
01049         // check, we know that at least one program instance wants to be
01050         // running the loop, so just emit code for the loop body and jump
01051         // to the loop step code.
01052         if (stmts)
01053             stmts->EmitCode(ctx);
01054         if (ctx->GetCurrentBasicBlock())
01055             ctx->BranchInst(bstep);
01056     }
01057     if (!dynamic_cast<StmtList *>(stmts))
01058         ctx->EndScope();
01059 
01060     // Emit code for the loop step.  First, restore the lane mask of any
01061     // program instances that executed a 'continue' during the previous
01062     // iteration.  Then emit code for the loop step and then jump to the
01063     // test code.
01064     ctx->SetCurrentBasicBlock(bstep);
01065     ctx->RestoreContinuedLanes();
01066     if (step)
01067         step->EmitCode(ctx);
01068     ctx->BranchInst(btest);
01069 
01070     // Set the current emission basic block to the loop exit basic block
01071     ctx->SetCurrentBasicBlock(bexit);
01072     if (init)
01073         ctx->EndScope();
01074     ctx->EndLoop();
01075 }

01228 /* Given a uniform counter value in the memory location pointed to by
01229    uniformCounterPtr, compute the corresponding set of varying counter
01230    values for use within the loop body.
01231  */
01232 static llvm::Value *
01233 lUpdateVaryingCounter(int dim, int nDims, FunctionEmitContext *ctx, 
01234                       llvm::Value *uniformCounterPtr,
01235                       llvm::Value *varyingCounterPtr,
01236                       const std::vector<int> &spans) {
01237     // Smear the uniform counter value out to be varying
01238     llvm::Value *counter = ctx->LoadInst(uniformCounterPtr);
01239     llvm::Value *smearCounter = 
01240         llvm::UndefValue::get(LLVMTypes::Int32VectorType);
01241     for (int i = 0; i < g->target.vectorWidth; ++i)
01242         smearCounter = 
01243             ctx->InsertInst(smearCounter, counter, i, "smear_counter");
01244 
01245     // Figure out the offsets; this is a little bit tricky.  As an example,
01246     // consider a 2D tiled foreach loop, where we're running 8-wide and
01247     // where the inner dimension has a stride of 4 and the outer dimension
01248     // has a stride of 2.  For the inner dimension, we want the offsets
01249     // (0,1,2,3,0,1,2,3), and for the outer dimension we want
01250     // (0,0,0,0,1,1,1,1).
01251     int32_t delta[ISPC_MAX_NVEC];
01252     for (int i = 0; i < g->target.vectorWidth; ++i) {
01253         int d = i;
01254         // First, account for the effect of any dimensions at deeper
01255         // nesting levels than the current one.
01256         int prevDimSpanCount = 1;
01257         for (int j = dim; j < nDims-1; ++j)
01258             prevDimSpanCount *= spans[j+1];
01259         d /= prevDimSpanCount;
01260 
01261         // And now with what's left, figure out our own offset
01262         delta[i] = d % spans[dim];
01263     }
01264 
01265     // Add the deltas to compute the varying counter values; store the
01266     // result to memory and then return it directly as well.
01267     llvm::Value *varyingCounter = 
01268         ctx->BinaryOperator(llvm::Instruction::Add, smearCounter,
01269                             LLVMInt32Vector(delta), "iter_val");
01270     ctx->StoreInst(varyingCounter, varyingCounterPtr);
01271     return varyingCounter;
01272 }

01275 /** Returns the integer log2 of the given integer. */
01276 static int
01277 lLog2(int i) {
01278     int ret = 0;
01279     while (i != 0) {
01280         ++ret;
01281         i >>= 1;
01282     }
01283     return ret-1;
01284 }

01287 /* Figure out how many elements to process in each dimension for each time
01288    through a foreach loop.  The untiled case is easy; all of the outer
01289    dimensions up until the innermost one have a span of 1, and the
01290    innermost one takes the entire vector width.  For the tiled case, we
01291    give wider spans to the innermost dimensions while also trying to
01292    generate relatively square domains.
01293 
01294    This code works recursively from outer dimensions to inner dimensions.
01295  */
01296 static void
01297 lGetSpans(int dimsLeft, int nDims, int itemsLeft, bool isTiled, int *a) {
01298     if (dimsLeft == 0) {
01299         // Nothing left to do but give all of the remaining work to the
01300         // innermost domain.
01301         *a = itemsLeft;
01302         return;
01303     }
01304 
01305     if (isTiled == false || (dimsLeft >= lLog2(itemsLeft)))
01306         // If we're not tiled, or if there are enough dimensions left that
01307         // giving this one any more than a span of one would mean that a
01308         // later dimension would have to have a span of one, give this one
01309         // a span of one to save the available items for later.
01310         *a = 1;
01311     else if (itemsLeft >= 16 && (dimsLeft == 1))
01312         // Special case to have 4x4 domains for the 2D case when running
01313         // 16-wide.
01314         *a = 4;
01315     else
01316         // Otherwise give this dimension a span of two. 
01317         *a = 2;
01318 
01319     lGetSpans(dimsLeft-1, nDims, itemsLeft / *a, isTiled, a+1);
01320 }

01323 /* Emit code for a foreach statement.  We effectively emit code to run the
01324    set of n-dimensional nested loops corresponding to the dimensionality of
01325    the foreach statement along with the extra logic to deal with mismatches
01326    between the vector width we're compiling to and the number of elements
01327    to process.
01328  */
01329 void
01330 ForeachStmt::EmitCode(FunctionEmitContext *ctx) const {
01331     if (ctx->GetCurrentBasicBlock() == NULL || stmts == NULL) 
01332         return;
01333 
01334     llvm::BasicBlock *bbFullBody = ctx->CreateBasicBlock("foreach_full_body");
01335     llvm::BasicBlock *bbMaskedBody = ctx->CreateBasicBlock("foreach_masked_body");
01336     llvm::BasicBlock *bbExit = ctx->CreateBasicBlock("foreach_exit");
01337 
01338     llvm::Value *oldMask = ctx->GetInternalMask();
01339     llvm::Value *oldFunctionMask = ctx->GetFunctionMask();
01340 
01341     ctx->SetDebugPos(pos);
01342     ctx->StartScope();
01343 
01344     ctx->SetInternalMask(LLVMMaskAllOn);
01345     ctx->SetFunctionMask(LLVMMaskAllOn);
01346 
01347     // This should be caught during typechecking
01348     Assert(startExprs.size() == dimVariables.size() && 
01349            endExprs.size() == dimVariables.size());
01350     int nDims = (int)dimVariables.size();
01351 
01352     ///////////////////////////////////////////////////////////////////////
01353     // Setup: compute the number of items we have to work on in each
01354     // dimension and a number of derived values.
01355     std::vector<llvm::BasicBlock *> bbReset, bbStep, bbTest;
01356     std::vector<llvm::Value *> startVals, endVals, uniformCounterPtrs;
01357     std::vector<llvm::Value *> nExtras, alignedEnd, extrasMaskPtrs;
01358 
01359     std::vector<int> span(nDims, 0);
01360     lGetSpans(nDims-1, nDims, g->target.vectorWidth, isTiled, &span[0]);
01361 
01362     for (int i = 0; i < nDims; ++i) {
01363         // Basic blocks that we'll fill in later with the looping logic for
01364         // this dimension.
01365         bbReset.push_back(ctx->CreateBasicBlock("foreach_reset"));
01366         if (i < nDims-1)
01367             // stepping for the innermost dimension is handled specially
01368             bbStep.push_back(ctx->CreateBasicBlock("foreach_step"));
01369         bbTest.push_back(ctx->CreateBasicBlock("foreach_test"));
01370 
01371         // Start and end value for this loop dimension
01372         llvm::Value *sv = startExprs[i]->GetValue(ctx);
01373         llvm::Value *ev = endExprs[i]->GetValue(ctx);
01374         if (sv == NULL || ev == NULL)
01375             return;
01376         startVals.push_back(sv);
01377         endVals.push_back(ev);
01378 
01379         // nItems = endVal - startVal
01380         llvm::Value *nItems = 
01381             ctx->BinaryOperator(llvm::Instruction::Sub, ev, sv, "nitems");
01382 
01383         // nExtras = nItems % (span for this dimension)
01384         // This gives us the number of extra elements we need to deal with
01385         // at the end of the loop for this dimension that don't fit cleanly
01386         // into a vector width.
01387         nExtras.push_back(ctx->BinaryOperator(llvm::Instruction::SRem, nItems,
01388                                               LLVMInt32(span[i]), "nextras"));
01389 
01390         // alignedEnd = endVal - nExtras
01391         alignedEnd.push_back(ctx->BinaryOperator(llvm::Instruction::Sub, ev,
01392                                                  nExtras[i], "aligned_end"));
01393 
01394         ///////////////////////////////////////////////////////////////////////
01395         // Each dimension has a loop counter that is a uniform value that
01396         // goes from startVal to endVal, in steps of the span for this
01397         // dimension.  Its value is only used internally here for looping
01398         // logic and isn't directly available in the user's program code.
01399         uniformCounterPtrs.push_back(ctx->AllocaInst(LLVMTypes::Int32Type, 
01400                                                      "counter"));
01401         ctx->StoreInst(startVals[i], uniformCounterPtrs[i]);
01402 
01403         // There is also a varying variable that holds the set of index
01404         // values for each dimension in the current loop iteration; this is
01405         // the value that is program-visible.
01406         dimVariables[i]->storagePtr = 
01407             ctx->AllocaInst(LLVMTypes::Int32VectorType, 
01408                             dimVariables[i]->name.c_str());
01409         dimVariables[i]->parentFunction = ctx->GetFunction();
01410         ctx->EmitVariableDebugInfo(dimVariables[i]);
01411 
01412         // Each dimension also maintains a mask that represents which of
01413         // the varying elements in the current iteration should be
01414         // processed.  (i.e. this is used to disable the lanes that have
01415         // out-of-bounds offsets.)
01416         extrasMaskPtrs.push_back(ctx->AllocaInst(LLVMTypes::MaskType, "extras mask"));
01417         ctx->StoreInst(LLVMMaskAllOn, extrasMaskPtrs[i]);
01418     }
01419 
01420     ctx->StartForeach();
01421 
01422     // On to the outermost loop's test
01423     ctx->BranchInst(bbTest[0]);
01424 
01425     ///////////////////////////////////////////////////////////////////////////
01426     // foreach_reset: this code runs when we need to reset the counter for
01427     // a given dimension in preparation for running through its loop again,
01428     // after the enclosing level advances its counter.
01429     for (int i = 0; i < nDims; ++i) {
01430         ctx->SetCurrentBasicBlock(bbReset[i]);
01431         if (i == 0)
01432             ctx->BranchInst(bbExit);
01433         else {
01434             ctx->StoreInst(LLVMMaskAllOn, extrasMaskPtrs[i]);
01435             ctx->StoreInst(startVals[i], uniformCounterPtrs[i]);
01436             ctx->BranchInst(bbStep[i-1]);
01437         }
01438     }
01439 
01440     ///////////////////////////////////////////////////////////////////////////
01441     // foreach_step: increment the uniform counter by the vector width.
01442     // Note that we don't increment the varying counter here as well but
01443     // just generate its value when we need it in the loop body.  Don't do
01444     // this for the innermost dimension, which has a more complex stepping
01445     // structure..
01446     for (int i = 0; i < nDims-1; ++i) {
01447         ctx->SetCurrentBasicBlock(bbStep[i]);
01448         llvm::Value *counter = ctx->LoadInst(uniformCounterPtrs[i]);
01449         llvm::Value *newCounter =  
01450             ctx->BinaryOperator(llvm::Instruction::Add, counter,
01451                                 LLVMInt32(span[i]), "new_counter");
01452         ctx->StoreInst(newCounter, uniformCounterPtrs[i]);
01453         ctx->BranchInst(bbTest[i]);
01454     }
01455 
01456     ///////////////////////////////////////////////////////////////////////////
01457     // foreach_test (for all dimensions other than the innermost...)
01458     std::vector<llvm::Value *> inExtras;
01459     for (int i = 0; i < nDims-1; ++i) {
01460         ctx->SetCurrentBasicBlock(bbTest[i]);
01461 
01462         llvm::Value *haveExtras = 
01463             ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SGT,
01464                          endVals[i], alignedEnd[i], "have_extras");
01465 
01466         llvm::Value *counter = ctx->LoadInst(uniformCounterPtrs[i], "counter");
01467         llvm::Value *atAlignedEnd = 
01468             ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_EQ,
01469                          counter, alignedEnd[i], "at_aligned_end");
01470         llvm::Value *inEx = 
01471             ctx->BinaryOperator(llvm::Instruction::And, haveExtras,
01472                                 atAlignedEnd, "in_extras");
01473 
01474         if (i == 0)
01475             inExtras.push_back(inEx);
01476         else
01477             inExtras.push_back(ctx->BinaryOperator(llvm::Instruction::Or, inEx,
01478                                                    inExtras[i-1], "in_extras_all"));
01479 
01480         llvm::Value *varyingCounter = 
01481             lUpdateVaryingCounter(i, nDims, ctx, uniformCounterPtrs[i], 
01482                                   dimVariables[i]->storagePtr, span);
01483 
01484         llvm::Value *smearEnd = llvm::UndefValue::get(LLVMTypes::Int32VectorType);
01485         for (int j = 0; j < g->target.vectorWidth; ++j)
01486             smearEnd = ctx->InsertInst(smearEnd, endVals[i], j, "smear_end");
01487         // Do a vector compare of its value to the end value to generate a
01488         // mask for this last bit of work.
01489         llvm::Value *emask = 
01490             ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SLT,
01491                          varyingCounter, smearEnd);
01492         emask = ctx->I1VecToBoolVec(emask);
01493 
01494         if (i == 0)
01495             ctx->StoreInst(emask, extrasMaskPtrs[i]);
01496         else {
01497             llvm::Value *oldMask = ctx->LoadInst(extrasMaskPtrs[i-1]);
01498             llvm::Value *newMask =
01499                 ctx->BinaryOperator(llvm::Instruction::And, oldMask, emask,
01500                                     "extras_mask");
01501             ctx->StoreInst(newMask, extrasMaskPtrs[i]);
01502         }
01503 
01504         llvm::Value *notAtEnd = 
01505             ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SLT,
01506                          counter, endVals[i]);
01507         ctx->BranchInst(bbTest[i+1], bbReset[i], notAtEnd);
01508     }
01509 
01510     ///////////////////////////////////////////////////////////////////////////
01511     // foreach_test (for innermost dimension)
01512     //
01513     // All of the outer dimensions are handled generically--basically as a
01514     // for() loop from the start value to the end value, where at each loop
01515     // test, we compute the mask of active elements for the current
01516     // dimension and then update an overall mask that is the AND
01517     // combination of all of the outer ones.
01518     //
01519     // The innermost loop is handled specially, for performance purposes.
01520     // When starting the innermost dimension, we start by checking once
01521     // whether any of the outer dimensions has set the mask to be
01522     // partially-active or not.  We follow different code paths for these
01523     // two cases, taking advantage of the knowledge that the mask is all
01524     // on, when this is the case.
01525     //
01526     // In each of these code paths, we start with a loop from the starting
01527     // value to the aligned end value for the innermost dimension; we can
01528     // guarantee that the innermost loop will have an "all on" mask (as far
01529     // as its dimension is concerned) for the duration of this loop.  Doing
01530     // so allows us to emit code that assumes the mask is all on (for the
01531     // case where none of the outer dimensions has set the mask to be
01532     // partially on), or allows us to emit code that just uses the mask
01533     // from the outer dimensions directly (for the case where they have).
01534     //
01535     // After this loop, we just need to deal with one vector's worth of
01536     // "ragged extra bits", where the mask used includes the effect of the
01537     // mask for the innermost dimension.
01538     //
01539     // We start out this process by emitting the check that determines
01540     // whether any of the enclosing dimensions is partially active
01541     // (i.e. processing extra elements that don't exactly fit into a
01542     // vector).
01543     llvm::BasicBlock *bbOuterInExtras = 
01544         ctx->CreateBasicBlock("outer_in_extras");
01545     llvm::BasicBlock *bbOuterNotInExtras = 
01546         ctx->CreateBasicBlock("outer_not_in_extras");
01547 
01548     ctx->SetCurrentBasicBlock(bbTest[nDims-1]);
01549     if (inExtras.size())
01550         ctx->BranchInst(bbOuterInExtras, bbOuterNotInExtras,
01551                         inExtras.back());
01552     else
01553         // for a 1D iteration domain, we certainly don't have any enclosing
01554         // dimensions that are processing extra elements.
01555         ctx->BranchInst(bbOuterNotInExtras);
01556 
01557     ///////////////////////////////////////////////////////////////////////////
01558     // One or more outer dimensions in extras, so we need to mask for the loop
01559     // body regardless.  We break this into two cases, roughly:
01560     // for (counter = start; counter < alignedEnd; counter += step) {
01561     //   // mask is all on for inner, so set mask to outer mask
01562     //   // run loop body with mask
01563     // }
01564     // // counter == alignedEnd
01565     // if (counter < end) {
01566     //   // set mask to outermask & (counter+programCounter < end)
01567     //   // run loop body with mask
01568     // }
01569     llvm::BasicBlock *bbAllInnerPartialOuter =
01570         ctx->CreateBasicBlock("all_inner_partial_outer");
01571     llvm::BasicBlock *bbPartial =
01572         ctx->CreateBasicBlock("both_partial");
01573     ctx->SetCurrentBasicBlock(bbOuterInExtras); {
01574         // Update the varying counter value here, since all subsequent
01575         // blocks along this path need it.
01576         lUpdateVaryingCounter(nDims-1, nDims, ctx, uniformCounterPtrs[nDims-1], 
01577                               dimVariables[nDims-1]->storagePtr, span);
01578 
01579         // here we just check to see if counter < alignedEnd
01580         llvm::Value *counter = ctx->LoadInst(uniformCounterPtrs[nDims-1], "counter");
01581         llvm::Value *beforeAlignedEnd = 
01582             ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SLT,
01583                          counter, alignedEnd[nDims-1], "before_aligned_end");
01584         ctx->BranchInst(bbAllInnerPartialOuter, bbPartial, beforeAlignedEnd);
01585     }
01586 
01587     // Below we have a basic block that runs the loop body code for the
01588     // case where the mask is partially but not fully on.  This same block
01589     // runs in multiple cases: both for handling any ragged extra data for
01590     // the innermost dimension but also when outer dimensions have set the
01591     // mask to be partially on. 
01592     //
01593     // The value stored in stepIndexAfterMaskedBodyPtr is used after each
01594     // execution of the body code to determine whether the innermost index
01595     // value should be incremented by the step (we're running the "for"
01596     // loop of full vectors at the innermost dimension, with outer
01597     // dimensions having set the mask to be partially on), or whether we're
01598     // running once for the ragged extra bits at the end of the innermost
01599     // dimension, in which case we're done with the innermost dimension and
01600     // should step the loop counter for the next enclosing dimension
01601     // instead.
01602     llvm::Value *stepIndexAfterMaskedBodyPtr =
01603         ctx->AllocaInst(LLVMTypes::BoolType, "step_index");
01604 
01605     ///////////////////////////////////////////////////////////////////////////
01606     // We're in the inner loop part where the only masking is due to outer
01607     // dimensions but the innermost dimension fits fully into a vector's
01608     // width.  Set the mask and jump to the masked loop body.
01609     ctx->SetCurrentBasicBlock(bbAllInnerPartialOuter); {
01610         llvm::Value *mask;
01611         if (extrasMaskPtrs.size() == 0)
01612             // 1D loop; we shouldn't ever get here anyway
01613             mask = LLVMMaskAllOff;
01614         else
01615             mask = ctx->LoadInst(extrasMaskPtrs.back());
01616         ctx->SetInternalMask(mask);
01617 
01618         ctx->StoreInst(LLVMTrue, stepIndexAfterMaskedBodyPtr);
01619         ctx->BranchInst(bbMaskedBody);
01620     }
01621 
01622     ///////////////////////////////////////////////////////////////////////////
01623     // We need to include the effect of the innermost dimension in the mask
01624     // for the final bits here
01625     ctx->SetCurrentBasicBlock(bbPartial); {
01626         llvm::Value *varyingCounter = 
01627             ctx->LoadInst(dimVariables[nDims-1]->storagePtr);
01628         llvm::Value *smearEnd = llvm::UndefValue::get(LLVMTypes::Int32VectorType);
01629         for (int j = 0; j < g->target.vectorWidth; ++j)
01630             smearEnd = ctx->InsertInst(smearEnd, endVals[nDims-1], j, "smear_end");
01631         llvm::Value *emask = 
01632             ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SLT,
01633                          varyingCounter, smearEnd);
01634         emask = ctx->I1VecToBoolVec(emask);
01635 
01636         if (nDims == 1)
01637             ctx->SetInternalMask(emask);
01638         else {
01639             llvm::Value *oldMask = ctx->LoadInst(extrasMaskPtrs[nDims-2]);
01640             llvm::Value *newMask =
01641                 ctx->BinaryOperator(llvm::Instruction::And, oldMask, emask,
01642                                     "extras_mask");
01643             ctx->SetInternalMask(newMask);
01644         }
01645 
01646         ctx->StoreInst(LLVMFalse, stepIndexAfterMaskedBodyPtr);
01647         ctx->BranchInst(bbMaskedBody);
01648     }
01649 
01650     ///////////////////////////////////////////////////////////////////////////
01651     // None of the outer dimensions is processing extras; along the lines
01652     // of above, we can express this as:
01653     // for (counter = start; counter < alignedEnd; counter += step) {
01654     //   // mask is all on
01655     //   // run loop body with mask all on
01656     // }
01657     // // counter == alignedEnd
01658     // if (counter < end) {
01659     //   // set mask to (counter+programCounter < end)
01660     //   // run loop body with mask
01661     // }
01662     llvm::BasicBlock *bbPartialInnerAllOuter =
01663         ctx->CreateBasicBlock("partial_inner_all_outer");
01664     ctx->SetCurrentBasicBlock(bbOuterNotInExtras); {
01665         llvm::Value *counter = ctx->LoadInst(uniformCounterPtrs[nDims-1], "counter");
01666         llvm::Value *beforeAlignedEnd = 
01667             ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SLT,
01668                          counter, alignedEnd[nDims-1], "before_aligned_end");
01669         ctx->BranchInst(bbFullBody, bbPartialInnerAllOuter,
01670                         beforeAlignedEnd);
01671     }
01672 
01673     ///////////////////////////////////////////////////////////////////////////
01674     // full_body: do a full vector's worth of work.  We know that all
01675     // lanes will be running here, so we explicitly set the mask to be 'all
01676     // on'.  This ends up being relatively straightforward: just update the
01677     // value of the varying loop counter and have the statements in the
01678     // loop body emit their code.
01679     llvm::BasicBlock *bbFullBodyContinue = 
01680         ctx->CreateBasicBlock("foreach_full_continue");
01681     ctx->SetCurrentBasicBlock(bbFullBody); {
01682         ctx->SetInternalMask(LLVMMaskAllOn);
01683         lUpdateVaryingCounter(nDims-1, nDims, ctx, uniformCounterPtrs[nDims-1], 
01684                               dimVariables[nDims-1]->storagePtr, span);
01685         ctx->SetContinueTarget(bbFullBodyContinue);
01686         ctx->AddInstrumentationPoint("foreach loop body (all on)");
01687         stmts->EmitCode(ctx);
01688         Assert(ctx->GetCurrentBasicBlock() != NULL);
01689         ctx->BranchInst(bbFullBodyContinue);
01690     }
01691     ctx->SetCurrentBasicBlock(bbFullBodyContinue); {
01692         ctx->RestoreContinuedLanes();
01693         llvm::Value *counter = ctx->LoadInst(uniformCounterPtrs[nDims-1]);
01694         llvm::Value *newCounter =  
01695             ctx->BinaryOperator(llvm::Instruction::Add, counter,
01696                                 LLVMInt32(span[nDims-1]), "new_counter");
01697         ctx->StoreInst(newCounter, uniformCounterPtrs[nDims-1]);
01698         ctx->BranchInst(bbOuterNotInExtras);
01699     }
01700 
01701     ///////////////////////////////////////////////////////////////////////////
01702     // We're done running blocks with the mask all on; see if the counter is
01703     // less than the end value, in which case we need to run the body one
01704     // more time to get the extra bits.
01705     llvm::BasicBlock *bbSetInnerMask = 
01706         ctx->CreateBasicBlock("partial_inner_only");
01707     ctx->SetCurrentBasicBlock(bbPartialInnerAllOuter); {
01708         llvm::Value *counter = ctx->LoadInst(uniformCounterPtrs[nDims-1], "counter");
01709         llvm::Value *beforeFullEnd = 
01710             ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SLT,
01711                          counter, endVals[nDims-1], "before_full_end");
01712         ctx->BranchInst(bbSetInnerMask, bbReset[nDims-1], beforeFullEnd);
01713     }
01714 
01715     ///////////////////////////////////////////////////////////////////////////
01716     // The outer dimensions are all on, so the mask is just given by the
01717     // mask for the innermost dimension
01718     ctx->SetCurrentBasicBlock(bbSetInnerMask); {
01719         llvm::Value *varyingCounter = 
01720             lUpdateVaryingCounter(nDims-1, nDims, ctx, uniformCounterPtrs[nDims-1], 
01721                                   dimVariables[nDims-1]->storagePtr, span);
01722         llvm::Value *smearEnd = llvm::UndefValue::get(LLVMTypes::Int32VectorType);
01723         for (int j = 0; j < g->target.vectorWidth; ++j)
01724             smearEnd = ctx->InsertInst(smearEnd, endVals[nDims-1], j, "smear_end");
01725         llvm::Value *emask = 
01726             ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SLT,
01727                          varyingCounter, smearEnd);
01728         emask = ctx->I1VecToBoolVec(emask);
01729         ctx->SetInternalMask(emask);
01730 
01731         ctx->StoreInst(LLVMFalse, stepIndexAfterMaskedBodyPtr);
01732         ctx->BranchInst(bbMaskedBody);
01733     }
01734 
01735     ///////////////////////////////////////////////////////////////////////////
01736     // masked_body: set the mask and have the statements emit their
01737     // code again.  Note that it's generally worthwhile having two copies
01738     // of the statements' code, since the code above is emitted with the
01739     // mask known to be all-on, which in turn leads to more efficient code
01740     // for that case.
01741     llvm::BasicBlock *bbStepInnerIndex = 
01742         ctx->CreateBasicBlock("step_inner_index");
01743     llvm::BasicBlock *bbMaskedBodyContinue = 
01744         ctx->CreateBasicBlock("foreach_masked_continue");
01745     ctx->SetCurrentBasicBlock(bbMaskedBody); {
01746         ctx->AddInstrumentationPoint("foreach loop body (masked)");
01747         ctx->SetContinueTarget(bbMaskedBodyContinue);
01748         ctx->DisableGatherScatterWarnings();
01749         stmts->EmitCode(ctx);
01750         ctx->EnableGatherScatterWarnings();
01751         ctx->BranchInst(bbMaskedBodyContinue);
01752     }
01753     ctx->SetCurrentBasicBlock(bbMaskedBodyContinue); {
01754         ctx->RestoreContinuedLanes();
01755         llvm::Value *stepIndex = ctx->LoadInst(stepIndexAfterMaskedBodyPtr);
01756         ctx->BranchInst(bbStepInnerIndex, bbReset[nDims-1], stepIndex);
01757     }
01758 
01759     ///////////////////////////////////////////////////////////////////////////
01760     // step the innermost index, for the case where we're doing the
01761     // innermost for loop over full vectors.
01762     ctx->SetCurrentBasicBlock(bbStepInnerIndex); {
01763         llvm::Value *counter = ctx->LoadInst(uniformCounterPtrs[nDims-1]);
01764         llvm::Value *newCounter =  
01765             ctx->BinaryOperator(llvm::Instruction::Add, counter,
01766                                 LLVMInt32(span[nDims-1]), "new_counter");
01767         ctx->StoreInst(newCounter, uniformCounterPtrs[nDims-1]);
01768         ctx->BranchInst(bbOuterInExtras);
01769     }
01770 
01771     ///////////////////////////////////////////////////////////////////////////
01772     // foreach_exit: All done.  Restore the old mask and clean up
01773     ctx->SetCurrentBasicBlock(bbExit);
01774 
01775     ctx->SetInternalMask(oldMask);
01776     ctx->SetFunctionMask(oldFunctionMask);
01777 
01778     ctx->EndForeach();
01779     ctx->EndScope();
01780 }

01983 /* An instance of this structure is carried along as we traverse the AST
01984    nodes for the statements after a "switch" statement.  We use this
01985    structure to record all of the 'case' and 'default' statements after the
01986    "switch". */
01987 struct SwitchVisitInfo {
01988     SwitchVisitInfo(FunctionEmitContext *c) { 
01989         ctx = c;
01990         defaultBlock = NULL; 
01991         lastBlock = NULL;
01992     }
01993 
01994     FunctionEmitContext *ctx;
01995 
01996     /* Basic block for the code following the "default" label (if any). */
01997     llvm::BasicBlock *defaultBlock;
01998 
01999     /* Map from integer values after "case" labels to the basic blocks that
02000        follow the corresponding "case" label. */
02001     std::vector<std::pair<int, llvm::BasicBlock *> > caseBlocks;
02002 
02003     /* For each basic block for a "case" label or a "default" label,
02004        nextBlock[block] stores the basic block pointer for the next
02005        subsequent "case" or "default" label in the program. */
02006     std::map<llvm::BasicBlock *, llvm::BasicBlock *> nextBlock;
02007 
02008     /* The last basic block created for a "case" or "default" label; when
02009        we create the basic block for the next one, we'll use this to update
02010        the nextBlock map<> above. */
02011     llvm::BasicBlock *lastBlock;
02012 };

02014 
02015 static bool
02016 lSwitchASTPreVisit(ASTNode *node, void *d) {
02017     if (dynamic_cast<SwitchStmt *>(node) != NULL)
02018         // don't continue recursively into a nested switch--we only want
02019         // our own case and default statements!
02020         return false;
02021 
02022     CaseStmt *cs = dynamic_cast<CaseStmt *>(node);
02023     DefaultStmt *ds = dynamic_cast<DefaultStmt *>(node);
02024 
02025     SwitchVisitInfo *svi = (SwitchVisitInfo *)d;
02026     llvm::BasicBlock *bb = NULL;
02027     if (cs != NULL) {
02028         // Complain if we've seen a case statement with the same value
02029         // already
02030         for (int i = 0; i < (int)svi->caseBlocks.size(); ++i) {
02031             if (svi->caseBlocks[i].first == cs->value) {
02032                 Error(cs->pos, "Duplicate case value \"%d\".", cs->value); 
02033                 return true;
02034             }
02035         }
02036         
02037         // Otherwise create a new basic block for the code following this
02038         // 'case' statement and record the mappign between the case label
02039         // value and the basic block
02040         char buf[32];
02041         sprintf(buf, "case_%d", cs->value);
02042         bb = svi->ctx->CreateBasicBlock(buf);
02043         svi->caseBlocks.push_back(std::make_pair(cs->value, bb));
02044     }
02045     else if (ds != NULL) {
02046         // And complain if we've seen another 'default' label..
02047         if (svi->defaultBlock != NULL) {
02048             Error(ds->pos, "Multiple \"default\" lables in switch statement.");
02049             return true;
02050         }
02051         else {
02052             // Otherwise create a basic block for the code following the
02053             // "default".
02054             bb = svi->ctx->CreateBasicBlock("default");
02055             svi->defaultBlock = bb;
02056         }
02057     }
02058 
02059     // If we saw a "case" or "default" label, then update the map to record
02060     // that the block we just created follows the block created for the
02061     // previous label in the "switch".
02062     if (bb != NULL) {
02063         svi->nextBlock[svi->lastBlock] = bb;
02064         svi->lastBlock = bb;
02065     }
02066 
02067     return true;
02068 }

02071 void
02072 SwitchStmt::EmitCode(FunctionEmitContext *ctx) const {
02073     if (ctx->GetCurrentBasicBlock() == NULL)
02074         return;
02075 
02076     const Type *type;
02077     if (expr == NULL || ((type = expr->GetType()) == NULL)) {
02078         Assert(m->errorCount > 0);
02079         return;
02080     }
02081 
02082     // Basic block we'll end up after the switch statement
02083     llvm::BasicBlock *bbDone = ctx->CreateBasicBlock("switch_done");
02084 
02085     // Walk the AST of the statements after the 'switch' to collect a bunch
02086     // of information about the structure of the 'case' and 'default'
02087     // statements.
02088     SwitchVisitInfo svi(ctx);
02089     WalkAST(stmts, lSwitchASTPreVisit, NULL, &svi);
02090     // Record that the basic block following the last one created for a
02091     // case/default is the block after the end of the switch statement.
02092     svi.nextBlock[svi.lastBlock] = bbDone;
02093 
02094     llvm::Value *exprValue = expr->GetValue(ctx);
02095     if (exprValue == NULL) {
02096         Assert(m->errorCount > 0);
02097         return;
02098     }
02099 
02100     bool isUniformCF = (type->IsUniformType() &&
02101                         lHasVaryingBreakOrContinue(stmts) == false);
02102     ctx->StartSwitch(isUniformCF, bbDone);
02103     ctx->SwitchInst(exprValue, svi.defaultBlock ? svi.defaultBlock : bbDone,
02104                     svi.caseBlocks, svi.nextBlock);
02105 
02106     if (stmts != NULL)
02107         stmts->EmitCode(ctx);
02108 
02109     if (ctx->GetCurrentBasicBlock() != NULL)
02110         ctx->BranchInst(bbDone);
02111 
02112     ctx->SetCurrentBasicBlock(bbDone);
02113     ctx->EndSwitch();
02114 }

02385 /* Because the pointers to values that are passed to __do_print() are all
02386    void *s (and because ispc print() formatting strings statements don't
02387    encode types), we pass along a string to __do_print() where the i'th
02388    character encodes the type of the i'th value to be printed.  Needless to
02389    say, the encoding chosen here and the decoding code in __do_print() need
02390    to agree on the below!
02391  */
02392 static char
02393 lEncodeType(const Type *t) {
02394     if (Type::Equal(t, AtomicType::UniformBool))   return 'b';
02395     if (Type::Equal(t, AtomicType::VaryingBool))   return 'B';
02396     if (Type::Equal(t, AtomicType::UniformInt32))  return 'i';
02397     if (Type::Equal(t, AtomicType::VaryingInt32))  return 'I';
02398     if (Type::Equal(t, AtomicType::UniformUInt32)) return 'u';
02399     if (Type::Equal(t, AtomicType::VaryingUInt32)) return 'U';
02400     if (Type::Equal(t, AtomicType::UniformFloat))  return 'f';
02401     if (Type::Equal(t, AtomicType::VaryingFloat))  return 'F';
02402     if (Type::Equal(t, AtomicType::UniformInt64))  return 'l';
02403     if (Type::Equal(t, AtomicType::VaryingInt64))  return 'L';
02404     if (Type::Equal(t, AtomicType::UniformUInt64)) return 'v';
02405     if (Type::Equal(t, AtomicType::VaryingUInt64)) return 'V';
02406     if (Type::Equal(t, AtomicType::UniformDouble)) return 'd';
02407     if (Type::Equal(t, AtomicType::VaryingDouble)) return 'D';
02408     if (dynamic_cast<const PointerType *>(t) != NULL) {
02409         if (t->IsUniformType())
02410             return 'p';
02411         else
02412             return 'P';
02413     }
02414     else return '\0';
02415 }

02417 
02418 /** Given an Expr for a value to be printed, emit the code to evaluate the
02419     expression and store the result to alloca'd memory.  Update the
02420     argTypes string with the type encoding for this expression.
02421  */
02422 static llvm::Value *
02423 lProcessPrintArg(Expr *expr, FunctionEmitContext *ctx, std::string &argTypes) {
02424     const Type *type = expr->GetType();
02425     if (type == NULL)
02426         return NULL;
02427 
02428     if (dynamic_cast<const ReferenceType *>(type) != NULL) {
02429         expr = new DereferenceExpr(expr, expr->pos);
02430         type = expr->GetType();
02431         if (type == NULL)
02432             return NULL;
02433     }
02434 
02435     // Just int8 and int16 types to int32s...
02436     const Type *baseType = type->GetAsNonConstType()->GetAsUniformType();
02437     if (Type::Equal(baseType, AtomicType::UniformInt8) ||
02438         Type::Equal(baseType, AtomicType::UniformUInt8) ||
02439         Type::Equal(baseType, AtomicType::UniformInt16) ||
02440         Type::Equal(baseType, AtomicType::UniformUInt16)) {
02441         expr = new TypeCastExpr(type->IsUniformType() ? AtomicType::UniformInt32 :
02442                                                         AtomicType::VaryingInt32, 
02443                                 expr, expr->pos);
02444         type = expr->GetType();
02445     }
02446

02480     // in the code emitted below
02481     //
02482     // 1. the format string
02483     // 2. a string encoding the types of the values being printed, 
02484     //    one character per value
02485     // 3. the number of running program instances (i.e. the target's
02486     //    vector width)

02499         // stored in memory so that we set up the array of pointers to them

02502         int nArgs = elist ? elist->exprs.size() : 1;
02503 
02504         // Allocate space for the array of pointers to values to be printed 
02505         LLVM_TYPE_CONST llvm::Type *argPtrArrayType = 
02506             llvm::ArrayType::get(LLVMTypes::VoidPointerType, nArgs);
02507         llvm::Value *argPtrArray = ctx->AllocaInst(argPtrArrayType,
02508                                                    "print_arg_ptrs");

02510         // expects
02511         args[4] = ctx->BitCastInst(argPtrArray, 
02512                                    llvm::PointerType::get(LLVMTypes::VoidPointerType, 0));
02513 
02514         // Now, for each of the arguments, emit code to evaluate its value
02515         // and store the value into alloca'd storage.  Then store the
02516         // pointer to the alloca'd storage into argPtrArray.
02517         if (elist) {

02532             if (!ptr)

02535             ctx->StoreInst(ptr, arrayPtr);

02540     llvm::Function *printFunc = m->module->getFunction("__do_print");
02541     Assert(printFunc);

02548     args[3] = ctx->LaneMask(mask);

02727 /** This generates AST nodes for an __foreach_active statement.  This
02728     construct can be synthesized ouf of the existing ForStmt (and other AST
02729     nodes), so here we just build up the AST that we need rather than
02730     having a new Stmt implementation for __foreach_active.
02731 
02732     @param iterSym  Symbol for the iteration variable (e.g. "i" in
02733                     __foreach_active (i) { .. .}
02734     @param stmts    Statements to execute each time through the loop, for
02735                     each active program instance.
02736     @param pos      Position of the __foreach_active statement in the source 
02737                     file.
02738  */
02739 Stmt *
02740 CreateForeachActiveStmt(Symbol *iterSym, Stmt *stmts, SourcePos pos) {
02741     if (iterSym == NULL) {
02742         Assert(m->errorCount > 0);
02743         return NULL;
02744     }
02745 
02746     // loop initializer: set iter = 0
02747     std::vector<VariableDeclaration> var;
02748     ConstExpr *zeroExpr = new ConstExpr(AtomicType::UniformInt32, 0,
02749                                         iterSym->pos);
02750     var.push_back(VariableDeclaration(iterSym, zeroExpr));
02751     Stmt *initStmt = new DeclStmt(var, iterSym->pos);
02752 
02753     // loop test: (iter < programCount)
02754     ConstExpr *progCountExpr = 
02755         new ConstExpr(AtomicType::UniformInt32, g->target.vectorWidth,
02756                       pos);
02757     SymbolExpr *symExpr = new SymbolExpr(iterSym, iterSym->pos);
02758     Expr *testExpr = new BinaryExpr(BinaryExpr::Lt, symExpr, progCountExpr,
02759                                     pos);
02760     
02761     // loop step: ++iterSym
02762     UnaryExpr *incExpr = new UnaryExpr(UnaryExpr::PreInc, symExpr, pos);
02763     Stmt *stepStmt = new ExprStmt(incExpr, pos);
02764 
02765     // loop body
02766     // First, call __movmsk(__mask)) to get the mask as a set of bits.
02767     // This should be hoisted out of the loop
02768     Symbol *maskSym = m->symbolTable->LookupVariable("__mask");
02769     Assert(maskSym != NULL);
02770     Expr *maskVecExpr = new SymbolExpr(maskSym, pos);
02771     std::vector<Symbol *> mmFuns;
02772     m->symbolTable->LookupFunction("__movmsk", &mmFuns);
02773     Assert(mmFuns.size() == 2);
02774     FunctionSymbolExpr *movmskFunc = new FunctionSymbolExpr("__movmsk", mmFuns,
02775                                                             pos);
02776     ExprList *movmskArgs = new ExprList(maskVecExpr, pos);
02777     FunctionCallExpr *movmskExpr = new FunctionCallExpr(movmskFunc, movmskArgs,
02778                                                         pos);
02779 
02780     // Compute the per lane mask to test the mask bits against: (1 << iter)
02781     ConstExpr *oneExpr = new ConstExpr(AtomicType::UniformInt32, 1,
02782                                        iterSym->pos);
02783     Expr *shiftLaneExpr = new BinaryExpr(BinaryExpr::Shl, oneExpr, symExpr, 
02784                                          pos);
02785 
02786     // Compute the AND: movmsk & (1 << iter)
02787     Expr *maskAndLaneExpr = new BinaryExpr(BinaryExpr::BitAnd, movmskExpr,
02788                                            shiftLaneExpr, pos);
02789     // Test to see if it's non-zero: (mask & (1 << iter)) != 0
02790     Expr *ifTestExpr = new BinaryExpr(BinaryExpr::NotEqual, maskAndLaneExpr,
02791                                       zeroExpr, pos);
02792 
02793     // Now, enclose the provided statements in an if test such that they
02794     // only run if the mask is non-zero for the lane we're currently
02795     // handling in the loop.
02796     IfStmt *laneCheckIf = new IfStmt(ifTestExpr, stmts, NULL, false, pos);
02797 
02798     // And return a for loop that wires it all together.
02799     return new ForStmt(initStmt, testExpr, stepStmt, laneCheckIf, false, pos);
02800 }
