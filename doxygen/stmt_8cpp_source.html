00144 /** Utility routine that emits code to initialize a symbol given an
00145     initializer expression.
00146 
00147     @param lvalue    Memory location of storage for the symbol's data
00148     @param symName   Name of symbol (used in error messages)
00149     @param symType   Type of variable being initialized
00150     @param initExpr  Expression for the initializer
00151     @param ctx       FunctionEmitContext to use for generating instructions
00152     @param pos       Source file position of the variable being initialized
00153 */
00154 static void
00155 lInitSymbol(llvm::Value *lvalue, const char *symName, const Type *symType,
00156             Expr *initExpr, FunctionEmitContext *ctx, SourcePos pos) {
00157     if (initExpr == NULL)
00158         // leave it uninitialized
00159         return;
00160 
00161     // If the initializer is a straight up expression that isn't an
00162     // ExprList, then we'll see if we can type convert it to the type of
00163     // the variable.
00164     if (dynamic_cast<ExprList *>(initExpr) == NULL) {
00165         if (lPossiblyResolveFunctionOverloads(initExpr, symType) == false)
00166             return;
00167         initExpr = TypeConvertExpr(initExpr, symType, "initializer");
00168 
00169         if (initExpr != NULL) {
00170             llvm::Value *initializerValue = initExpr->GetValue(ctx);
00171             if (initializerValue != NULL)
00172                 // Bingo; store the value in the variable's storage
00173                 ctx->StoreInst(initializerValue, lvalue);
00174             return;
00175         }
00176     }
00177 
00178     // Atomic types and enums can't be initialized with { ... } initializer
00179     // expressions, so print an error and return if that's what we've got
00180     // here..
00181     if (dynamic_cast<const AtomicType *>(symType) != NULL ||
00182         dynamic_cast<const EnumType *>(symType) != NULL ||
00183         dynamic_cast<const PointerType *>(symType) != NULL) {
00184         ExprList *elist = dynamic_cast<ExprList *>(initExpr);
00185         if (elist != NULL) {
00186             if (elist->exprs.size() == 1)
00187                 lInitSymbol(lvalue, symName, symType, elist->exprs[0], ctx,
00188                             pos);
00189             else
00190                 Error(initExpr->pos, "Expression list initializers can't be used for "
00191                       "variable \"%s\' with type \"%s\".", symName,
00192                       symType->GetString().c_str());
00193         }
00194         return;
00195     }
00196 
00197     const ReferenceType *rt = dynamic_cast<const ReferenceType *>(symType);
00198     if (rt) {
00199         if (!Type::Equal(initExpr->GetType(), rt)) {
00200             Error(initExpr->pos, "Initializer for reference type \"%s\" must have same "
00201                   "reference type itself. \"%s\" is incompatible.", 
00202                   rt->GetString().c_str(), initExpr->GetType()->GetString().c_str());
00203             return;
00204         }
00205 
00206         llvm::Value *initializerValue = initExpr->GetValue(ctx);
00207         if (initializerValue)
00208             ctx->StoreInst(initializerValue, lvalue);
00209         return;
00210     }
00211 
00212     // There are two cases for initializing structs, arrays and vectors;
00213     // either a single initializer may be provided (float foo[3] = 0;), in
00214     // which case all of the elements are initialized to the given value,
00215     // or an initializer list may be provided (float foo[3] = { 1,2,3 }),
00216     // in which case the elements are initialized with the corresponding
00217     // values.
00218     const CollectionType *collectionType = 
00219         dynamic_cast<const CollectionType *>(symType);
00220     if (collectionType != NULL) {
00221         std::string name;
00222         if (dynamic_cast<const StructType *>(symType) != NULL)
00223             name = "struct";
00224         else if (dynamic_cast<const ArrayType *>(symType) != NULL) 
00225             name = "array";
00226         else if (dynamic_cast<const VectorType *>(symType) != NULL) 
00227             name = "vector";
00228         else 
00229             FATAL("Unexpected CollectionType in lInitSymbol()");
00230 
00231         ExprList *exprList = dynamic_cast<ExprList *>(initExpr);
00232         if (exprList != NULL) {
00233             // The { ... } case; make sure we have the same number of
00234             // expressions in the ExprList as we have struct members
00235             int nInits = exprList->exprs.size();
00236             if (nInits != collectionType->GetElementCount()) {
00237                 Error(initExpr->pos, "Initializer for %s \"%s\" requires "
00238                       "%d values; %d provided.", name.c_str(), symName, 
00239                       collectionType->GetElementCount(), nInits);
00240                 return;
00241             }
00242 
00243             // Initialize each element with the corresponding value from
00244             // the ExprList
00245             for (int i = 0; i < nInits; ++i) {
00246                 llvm::Value *ep;
00247                 if (dynamic_cast<const StructType *>(symType) != NULL)
00248                     ep = ctx->AddElementOffset(lvalue, i, NULL, "element");
00249                 else
00250                     ep = ctx->GetElementPtrInst(lvalue, LLVMInt32(0), LLVMInt32(i), 
00251                                                 PointerType::GetUniform(collectionType->GetElementType(i)), 
00252                                                 "gep");
00253 
00254                 lInitSymbol(ep, symName, collectionType->GetElementType(i), 
00255                             exprList->exprs[i], ctx, pos);
00256             }
00257         }
00258         else
00259             Error(initExpr->pos, "Can't assign type \"%s\" to \"%s\".",
00260                   initExpr->GetType()->GetString().c_str(),
00261                   collectionType->GetString().c_str());
00262         return;
00263     }
00264 
00265     FATAL("Unexpected Type in lInitSymbol()");
00266 }

00320                   sym->name.c_str());

00388 DeclStmt::Optimize() {
00389     for (unsigned int i = 0; i < vars.size(); ++i) {
00390         Expr *init = vars[i].init;
00391         if (init != NULL && dynamic_cast<ExprList *>(init) == NULL) {
00392             // If the variable is const-qualified, after we've optimized
00393             // the initializer expression, see if we have a ConstExpr.  If
00394             // so, save it in Symbol::constValue where it can be used in
00395             // optimizing later expressions that have this symbol in them.
00396             // Note that there are cases where the expression may be
00397             // constant but where we don't have a ConstExpr; an example is
00398             // const arrays--the ConstExpr implementation just can't
00399             // represent an array of values.
00400             //
00401             // All this is fine in terms of the code that's generated in
00402             // the end (LLVM's constant folding stuff is good), but it
00403             // means that the ispc compiler's ability to reason about what
00404             // is definitely a compile-time constant for things like
00405             // computing array sizes from non-trivial expressions is
00406             // consequently limited.
00407             Symbol *sym = vars[i].sym;
00408             if (sym->type && sym->type->IsConstType() && 
00409                 Type::Equal(init->GetType(), sym->type))
00410                 sym->constValue = dynamic_cast<ConstExpr *>(init);
00411         }
00412     }
00413     return this;
00414 }

00535         ctx->SetCurrentBasicBlock(bthen);

00641     if (dynamic_cast<AssertStmt *>(node) != NULL) {
00642         // While it's fine to run the assert for varying tests, it's not
00643         // desirable to check an assert on a uniform variable if all of the
00644         // lanes are off.
00645         *okPtr = false;
00646         return false;
00647     }
00648 
00649     if (g->target.allOffMaskIsSafe == true)
00650         // Don't worry about memory accesses if we have a target that can
00651         // safely run them with the mask all off
00652         return true;
00653 
00654     IndexExpr *ie;
00655     if ((ie = dynamic_cast<IndexExpr *>(node)) != NULL && ie->baseExpr != NULL) {
00656         const Type *type = ie->baseExpr->GetType();
00657         if (type == NULL)
00658             return true;
00659         if (dynamic_cast<const ReferenceType *>(type) != NULL)
00660             type = type->GetReferenceTarget();
00661 
00662         ConstExpr *ce = dynamic_cast<ConstExpr *>(ie->index);
00663         if (ce == NULL) {
00664             // indexing with a variable... -> not safe
00665             *okPtr = false;
00666             return false;
00667         }
00668 
00669         const PointerType *pointerType = 
00670             dynamic_cast<const PointerType *>(type);
00671         if (pointerType != NULL) {
00672             // pointer[index] -> can't be sure -> not safe
00673             *okPtr = false;
00674             return false;
00675         }
00676 
00677         const SequentialType *seqType = 
00678             dynamic_cast<const SequentialType *>(type);
00679         Assert(seqType != NULL);
00680         int nElements = seqType->GetElementCount();
00681         if (nElements == 0) {
00682             // Unsized array, so we can't be sure -> not safe
00683             *okPtr = false;
00684             return false;
00685         }
00686 
00687         int32_t indices[ISPC_MAX_NVEC];
00688         int count = ce->AsInt32(indices);
00689         for (int i = 0; i < count; ++i) {
00690             if (indices[i] < 0 || indices[i] >= nElements) {
00691                 // Index is out of bounds -> not safe
00692                 *okPtr = false;
00693                 return false;
00694             }
00695         }
00696 
00697         // All indices are in-bounds
00698     }
00699 
00700     return true;
00701 }

00704 /** Emit code for an if test that checks the mask and the test values and
00705     tries to be smart about jumping over code that doesn't need to be run.
00706  */
00707 void
00708 IfStmt::emitVaryingIf(FunctionEmitContext *ctx, llvm::Value *ltest) const {
00709     llvm::Value *oldMask = ctx->GetInternalMask();
00710     if (ctx->GetFullMask() == LLVMMaskAllOn && 
00711         !g->opt.disableCoherentControlFlow &&
00712         !g->opt.disableMaskAllOnOptimizations) {
00713         // We can tell that the mask is on statically at compile time; just
00714         // emit code for the 'if test with the mask all on' path
00715         llvm::BasicBlock *bDone = ctx->CreateBasicBlock("cif_done");
00716         emitMaskAllOn(ctx, ltest, bDone);
00717         ctx->SetCurrentBasicBlock(bDone);
00718     }
00719     else if (doAllCheck) {
00720         // We can't tell if the mask going into the if is all on at the
00721         // compile time.  Emit code to check for this and then either run
00722         // the code for the 'all on' or the 'mixed' case depending on the
00723         // mask's value at runtime.
00724         llvm::BasicBlock *bAllOn = ctx->CreateBasicBlock("cif_mask_all");
00725         llvm::BasicBlock *bMixedOn = ctx->CreateBasicBlock("cif_mask_mixed");
00726         llvm::BasicBlock *bDone = ctx->CreateBasicBlock("cif_done");
00727 
00728         // Jump to either bAllOn or bMixedOn, depending on the mask's value 
00729         llvm::Value *maskAllQ = ctx->All(ctx->GetFullMask());
00730         ctx->BranchInst(bAllOn, bMixedOn, maskAllQ);
00731 
00732         // Emit code for the 'mask all on' case
00733         ctx->SetCurrentBasicBlock(bAllOn);
00734         emitMaskAllOn(ctx, ltest, bDone);
00735         
00736         // And emit code for the mixed mask case
00737         ctx->SetCurrentBasicBlock(bMixedOn);
00738         emitMaskMixed(ctx, oldMask, ltest, bDone);
00739 
00740         // When done, set the current basic block to the block that the two
00741         // paths above jump to when they're done.
00742         ctx->SetCurrentBasicBlock(bDone);
00743     }
00744     else if (trueStmts != NULL || falseStmts != NULL) {
00745         // If there is nothing that is potentially unsafe to run with all
00746         // lanes off in the true and false statements and if the total
00747         // complexity of those two is relatively simple, then we'll go
00748         // ahead and emit straightline code that runs both sides, updating
00749         // the mask accordingly.  This is useful for efficiently compiling
00750         // things like:
00751         //
00752         // if (foo) x = 0;
00753         // else     ++x;
00754         //
00755         // Where the overhead of checking if any of the program instances wants
00756         // to run one side or the other is more than the actual computation.
00757         // The lSafeToRunWithAllLanesOff() checks to make sure that we don't do this
00758         // for potentially dangerous code like:
00759         //
00760         // if (index < count) array[index] = 0;
00761         //
00762         // where our use of blend for conditional assignments doesn't check
00763         // for the 'all lanes' off case.
00764         int trueFalseCost = (::EstimateCost(trueStmts) + 
00765                              ::EstimateCost(falseStmts));
00766         bool costIsAcceptable = (trueFalseCost <
00767                                  PREDICATE_SAFE_IF_STATEMENT_COST);
00768 
00769         bool safeToRunWithAllLanesOff = true;
00770         WalkAST(trueStmts, lCheckAllOffSafety, NULL, &safeToRunWithAllLanesOff);
00771         WalkAST(falseStmts, lCheckAllOffSafety, NULL, &safeToRunWithAllLanesOff);
00772 
00773         if (safeToRunWithAllLanesOff &&
00774             (costIsAcceptable || g->opt.disableCoherentControlFlow)) {
00775             ctx->StartVaryingIf(oldMask);
00776             emitMaskedTrueAndFalse(ctx, oldMask, ltest);
00777             Assert(ctx->GetCurrentBasicBlock());
00778             ctx->EndIf();
00779         }
00780         else {
00781             llvm::BasicBlock *bDone = ctx->CreateBasicBlock("if_done");
00782             emitMaskMixed(ctx, oldMask, ltest, bDone);
00783             ctx->SetCurrentBasicBlock(bDone);
00784         }
00785     }
00786 }

00789 /** Emits code for 'if' tests under the case where we know that the program
00790     mask is all on going into the 'if'.
00791  */
00792 void
00793 IfStmt::emitMaskAllOn(FunctionEmitContext *ctx, llvm::Value *ltest, 
00794                       llvm::BasicBlock *bDone) const {
00795     // We start by explicitly storing "all on" into the mask mask.  Note
00796     // that this doesn't change its actual value, but doing so lets the
00797     // compiler see what's going on so that subsequent optimizations for
00798     // code emitted here can operate with the knowledge that the mask is
00799     // definitely all on (until it modifies the mask itself).
00800     Assert(!g->opt.disableCoherentControlFlow);
00801     if (!g->opt.disableMaskAllOnOptimizations)
00802         ctx->SetInternalMask(LLVMMaskAllOn);
00803     llvm::Value *oldFunctionMask = ctx->GetFunctionMask();
00804     if (!g->opt.disableMaskAllOnOptimizations)
00805         ctx->SetFunctionMask(LLVMMaskAllOn);
00806 
00807     // First, check the value of the test.  If it's all on, then we jump to
00808     // a basic block that will only have code for the true case.
00809     llvm::BasicBlock *bTestAll = ctx->CreateBasicBlock("cif_test_all");
00810     llvm::BasicBlock *bTestNoneCheck = ctx->CreateBasicBlock("cif_test_none_check");
00811     llvm::Value *testAllQ = ctx->All(ltest);
00812     ctx->BranchInst(bTestAll, bTestNoneCheck, testAllQ);
00813 
00814     // Emit code for the 'test is all true' case
00815     ctx->SetCurrentBasicBlock(bTestAll);
00816     ctx->StartVaryingIf(LLVMMaskAllOn);
00817     lEmitIfStatements(ctx, trueStmts, "if: all on mask, expr all true");
00818     ctx->EndIf();
00819     if (ctx->GetCurrentBasicBlock() != NULL)
00820         // bblock may legitimately be NULL since if there's a return stmt
00821         // or break or continue we can actually jump and end emission since
00822         // we know all of the lanes are following this path...
00823         ctx->BranchInst(bDone);
00824 
00825     // The test isn't all true.  Now emit code to determine if it's all
00826     // false, or has mixed values.
00827     ctx->SetCurrentBasicBlock(bTestNoneCheck);
00828     llvm::BasicBlock *bTestNone = ctx->CreateBasicBlock("cif_test_none");
00829     llvm::BasicBlock *bTestMixed = ctx->CreateBasicBlock("cif_test_mixed");
00830     llvm::Value *testMixedQ = ctx->Any(ltest);
00831     ctx->BranchInst(bTestMixed, bTestNone, testMixedQ);
00832 
00833     // Emit code for the 'test is all false' case
00834     ctx->SetCurrentBasicBlock(bTestNone);
00835     ctx->StartVaryingIf(LLVMMaskAllOn);
00836     lEmitIfStatements(ctx, falseStmts, "if: all on mask, expr all false");
00837     ctx->EndIf();
00838     if (ctx->GetCurrentBasicBlock())
00839         // bblock may be NULL since if there's a return stmt or break or
00840         // continue we can actually jump or whatever and end emission...
00841         ctx->BranchInst(bDone);
00842 
00843     // Finally emit code for the 'mixed true/false' case.  We unavoidably
00844     // need to run both the true and the false statements.
00845     ctx->SetCurrentBasicBlock(bTestMixed);
00846     ctx->StartVaryingIf(LLVMMaskAllOn);
00847     emitMaskedTrueAndFalse(ctx, LLVMMaskAllOn, ltest);
00848     // In this case, return/break/continue isn't allowed to jump and end
00849     // emission.
00850     Assert(ctx->GetCurrentBasicBlock());
00851     ctx->EndIf();
00852     ctx->BranchInst(bDone);
00853 
00854     ctx->SetCurrentBasicBlock(bDone);
00855     ctx->SetFunctionMask(oldFunctionMask);
00856 }

00859 /** Emit code for an 'if' test where the lane mask is known to be mixed
00860     on/off going into it.
00861  */
00862 void
00863 IfStmt::emitMaskMixed(FunctionEmitContext *ctx, llvm::Value *oldMask, 
00864                       llvm::Value *ltest, llvm::BasicBlock *bDone) const {
00865     ctx->StartVaryingIf(oldMask);
00866     llvm::BasicBlock *bNext = ctx->CreateBasicBlock("safe_if_after_true");
00867     if (trueStmts != NULL) {
00868         llvm::BasicBlock *bRunTrue = ctx->CreateBasicBlock("safe_if_run_true");
00869         ctx->SetInternalMaskAnd(oldMask, ltest);
00870 
00871         // Do any of the program instances want to run the 'true'
00872         // block?  If not, jump ahead to bNext.
00873         llvm::Value *maskAnyQ = ctx->Any(ctx->GetFullMask());
00874         ctx->BranchInst(bRunTrue, bNext, maskAnyQ);
00875 
00876         // Emit statements for true
00877         ctx->SetCurrentBasicBlock(bRunTrue);
00878         lEmitIfStatements(ctx, trueStmts, "if: expr mixed, true statements");
00879         Assert(ctx->GetCurrentBasicBlock()); 
00880         ctx->BranchInst(bNext);
00881         ctx->SetCurrentBasicBlock(bNext);
00882     }
00883     if (falseStmts != NULL) {
00884         llvm::BasicBlock *bRunFalse = ctx->CreateBasicBlock("safe_if_run_false");
00885         bNext = ctx->CreateBasicBlock("safe_if_after_false");
00886         ctx->SetInternalMaskAndNot(oldMask, ltest);
00887 
00888         // Similarly, check to see if any of the instances want to
00889         // run the 'false' block...
00890         llvm::Value *maskAnyQ = ctx->Any(ctx->GetFullMask());
00891         ctx->BranchInst(bRunFalse, bNext, maskAnyQ);
00892 
00893         // Emit code for false
00894         ctx->SetCurrentBasicBlock(bRunFalse);
00895         lEmitIfStatements(ctx, falseStmts, "if: expr mixed, false statements");
00896         Assert(ctx->GetCurrentBasicBlock());
00897         ctx->BranchInst(bNext);
00898         ctx->SetCurrentBasicBlock(bNext);
00899     }
00900     ctx->BranchInst(bDone);
00901     ctx->SetCurrentBasicBlock(bDone);
00902     ctx->EndIf();
00903 }

00910     VaryingBCCheckInfo() {
00911         varyingControlFlowDepth = 0;
00912         foundVaryingBreakOrContinue = false;
00913     }
00914 
00915     int varyingControlFlowDepth;
00916     bool foundVaryingBreakOrContinue;

00920 /** Returns true if the given node is an 'if' statement where the test
00921     condition has varying type. */
00922 static bool
00923 lIsVaryingFor(ASTNode *node) {
00924     IfStmt *ifStmt;
00925     if ((ifStmt = dynamic_cast<IfStmt *>(node)) != NULL &&
00926         ifStmt->test != NULL) {
00927         const Type *type = ifStmt->test->GetType();
00928         return (type != NULL && type->IsVaryingType());
00929     }
00930     else
00931         return false;
00932 }

00934 
00935 /** Preorder callback function for checking for varying breaks or
00936     continues. */
00937 static bool
00938 lVaryingBCPreFunc(ASTNode *node, void *d) {
00939     VaryingBCCheckInfo *info = (VaryingBCCheckInfo *)d;
00940 
00941     // We found a break or continue statement; if we're under varying
00942     // control flow, then bingo.
00943     if ((dynamic_cast<BreakStmt *>(node) != NULL ||
00944          dynamic_cast<ContinueStmt *>(node) != NULL) &&
00945         info->varyingControlFlowDepth > 0) {
00946         info->foundVaryingBreakOrContinue = true;
00947         return false;
00948     }
00949 
00950     // Update the count of the nesting depth of varying control flow if
00951     // this is an if statement with a varying condition.
00952     if (lIsVaryingFor(node))
00953         ++info->varyingControlFlowDepth;
00954 
00955     if (dynamic_cast<ForStmt *>(node) != NULL ||
00956         dynamic_cast<DoStmt *>(node) != NULL ||
00957         dynamic_cast<ForeachStmt *>(node) != NULL)
00958         // Don't recurse into these guys, since we don't care about varying
00959         // breaks or continues within them...
00960         return false;
00961     else
00962         return true;
00963 }
00964 
00965 
00966 /** Postorder callback function for checking for varying breaks or
00967     continues; decrement the varying control flow depth after the node's
00968     children have been processed, if this is a varying if statement. */
00969 static ASTNode *
00970 lVaryingBCPostFunc(ASTNode *node, void *d) {
00971     VaryingBCCheckInfo *info = (VaryingBCCheckInfo *)d;
00972     if (lIsVaryingFor(node))
00973         --info->varyingControlFlowDepth;
00974     return node;
00975 }

00978 /** Given a statment, walk through it to see if there is a 'break' or
00979     'continue' statement inside if its children, under varying control
00980     flow.  We need to detect this case for loops since what might otherwise
00981     look like a 'uniform' loop needs to have code emitted to do all of the
00982     lane management stuff if this is the case.
00983  */ 
00984 static bool
00985 lHasVaryingBreakOrContinue(Stmt *stmt) {
00986     VaryingBCCheckInfo info;
00987     WalkAST(stmt, lVaryingBCPreFunc, lVaryingBCPostFunc, &info);
00988     return info.foundVaryingBreakOrContinue;
00989 }

00998 void DoStmt::EmitCode(FunctionEmitContext *ctx) const {
00999     // Check for things that could be NULL due to earlier errors during
01000     // compilation.
01001     if (!ctx->GetCurrentBasicBlock()) 
01002         return;
01003     if (!testExpr || !testExpr->GetType()) 
01004         return;
01005 
01006     bool uniformTest = testExpr->GetType()->IsUniformType();
01007     if (uniformTest && doCoherentCheck)
01008         Warning(pos, "Uniform condition supplied to \"cdo\" statement.");
01009 
01010     llvm::BasicBlock *bloop = ctx->CreateBasicBlock("do_loop");
01011     llvm::BasicBlock *bexit = ctx->CreateBasicBlock("do_exit");
01012     llvm::BasicBlock *btest = ctx->CreateBasicBlock("do_test");
01013 
01014     ctx->StartLoop(bexit, btest, uniformTest);
01015 
01016     // Start by jumping into the loop body
01017     ctx->BranchInst(bloop);
01018 
01019     // And now emit code for the loop body
01020     ctx->SetCurrentBasicBlock(bloop);
01021     ctx->SetLoopMask(ctx->GetInternalMask());
01022     ctx->SetDebugPos(pos);
01023     // FIXME: in the StmtList::EmitCode() method takes starts/stops a new
01024     // scope around the statements in the list.  So if the body is just a
01025     // single statement (and thus not a statement list), we need a new
01026     // scope, but we don't want two scopes in the StmtList case.
01027     if (!dynamic_cast<StmtList *>(bodyStmts))
01028         ctx->StartScope();
01029 
01030     ctx->AddInstrumentationPoint("do loop body");
01031     if (doCoherentCheck && !uniformTest) {
01032         // Check to see if the mask is all on
01033         llvm::BasicBlock *bAllOn = ctx->CreateBasicBlock("do_all_on");
01034         llvm::BasicBlock *bMixed = ctx->CreateBasicBlock("do_mixed");
01035         ctx->BranchIfMaskAll(bAllOn, bMixed);
01036 
01037         // If so, emit code for the 'mask all on' case.  In particular,
01038         // explicitly set the mask to 'all on' (see rationale in
01039         // IfStmt::emitCoherentTests()), and then emit the code for the
01040         // loop body.
01041         ctx->SetCurrentBasicBlock(bAllOn);
01042         if (!g->opt.disableMaskAllOnOptimizations)
01043             ctx->SetInternalMask(LLVMMaskAllOn);
01044         llvm::Value *oldFunctionMask = ctx->GetFunctionMask();
01045         if (!g->opt.disableMaskAllOnOptimizations)
01046             ctx->SetFunctionMask(LLVMMaskAllOn);
01047         if (bodyStmts)
01048             bodyStmts->EmitCode(ctx);
01049         Assert(ctx->GetCurrentBasicBlock());
01050         ctx->SetFunctionMask(oldFunctionMask);
01051         ctx->BranchInst(btest);
01052 
01053         // The mask is mixed.  Just emit the code for the loop body.
01054         ctx->SetCurrentBasicBlock(bMixed);
01055         if (bodyStmts)
01056             bodyStmts->EmitCode(ctx);
01057         Assert(ctx->GetCurrentBasicBlock());
01058         ctx->BranchInst(btest);
01059     }
01060     else {
01061         // Otherwise just emit the code for the loop body.  The current
01062         // mask is good.
01063         if (bodyStmts)
01064             bodyStmts->EmitCode(ctx);
01065         if (ctx->GetCurrentBasicBlock())
01066             ctx->BranchInst(btest);
01067     }
01068     // End the scope we started above, if needed.
01069     if (!dynamic_cast<StmtList *>(bodyStmts))
01070         ctx->EndScope();
01071 
01072     // Now emit code for the loop test.
01073     ctx->SetCurrentBasicBlock(btest);
01074     // First, emit code to restore the mask value for any lanes that
01075     // executed a 'continue' during the current loop before we go and emit
01076     // the code for the test.  This is only necessary for varying loops;
01077     // 'uniform' loops just jump when they hit a continue statement and
01078     // don't mess with the mask.
01079     if (!uniformTest)
01080         ctx->RestoreContinuedLanes();
01081     llvm::Value *testValue = testExpr->GetValue(ctx);
01082     if (!testValue)
01083         return;
01084 
01085     if (uniformTest)
01086         // For the uniform case, just jump to the top of the loop or the
01087         // exit basic block depending on the value of the test.
01088         ctx->BranchInst(bloop, bexit, testValue);
01089     else {
01090         // For the varying case, update the mask based on the value of the
01091         // test.  If any program instances still want to be running, jump
01092         // to the top of the loop.  Otherwise, jump out.
01093         llvm::Value *mask = ctx->GetInternalMask();
01094         ctx->SetInternalMaskAnd(mask, testValue);
01095         ctx->BranchIfMaskAny(bloop, bexit);
01096     }
01097 
01098     // ...and we're done.  Set things up for subsequent code to be emitted
01099     // in the right basic block.
01100     ctx->SetCurrentBasicBlock(bexit);
01101     ctx->EndLoop();
01102 }

01105 Stmt *
01106 DoStmt::TypeCheck() {
01107     const Type *testType;
01108     if (testExpr != NULL && (testType = testExpr->GetType()) != NULL) {
01109         if (!testType->IsNumericType() && !testType->IsBoolType()) {
01110             Error(testExpr->pos, "Type \"%s\" can't be converted to boolean for \"while\" "
01111                   "test in \"do\" loop.", testExpr->GetType()->GetString().c_str());
01112             return NULL;
01113         }
01114 
01115         // Should the test condition for the loop be uniform or varying?
01116         // It can be uniform only if three conditions are met:
01117         //
01118         // - First and foremost, the type of the test condition must be
01119         //   uniform.
01120         //
01121         // - Second, the user must not have set the dis-optimization option
01122         //   that disables uniform flow control.
01123         //
01124         // - Thirdly, and most subtlely, there must not be any break or
01125         //   continue statements inside the loop that are within the scope
01126         //   of a 'varying' if statement.  If there are, then we type cast
01127         //   the test to be 'varying', so that the code generated for the
01128         //   loop includes masking stuff, so that we can track which lanes
01129         //   actually want to be running, accounting for breaks/continues.
01130         //
01131         bool uniformTest = (testType->IsUniformType() &&
01132                             !g->opt.disableUniformControlFlow &&
01133                             !lHasVaryingBreakOrContinue(bodyStmts));
01134         testExpr = new TypeCastExpr(uniformTest ? AtomicType::UniformBool :
01135                                                   AtomicType::VaryingBool,
01136                                     testExpr, testExpr->pos);
01137     }
01138 
01139     return this;
01140 }

01177 void
01178 ForStmt::EmitCode(FunctionEmitContext *ctx) const {
01179     if (!ctx->GetCurrentBasicBlock()) 
01180         return;
01181 
01182     llvm::BasicBlock *btest = ctx->CreateBasicBlock("for_test");
01183     llvm::BasicBlock *bstep = ctx->CreateBasicBlock("for_step");
01184     llvm::BasicBlock *bloop = ctx->CreateBasicBlock("for_loop");
01185     llvm::BasicBlock *bexit = ctx->CreateBasicBlock("for_exit");
01186 
01187     bool uniformTest = test ? test->GetType()->IsUniformType() :
01188         (!g->opt.disableUniformControlFlow &&
01189          !lHasVaryingBreakOrContinue(stmts));
01190 
01191     ctx->StartLoop(bexit, bstep, uniformTest);
01192     ctx->SetDebugPos(pos);
01193 
01194     // If we have an initiailizer statement, start by emitting the code for
01195     // it and then jump into the loop test code.  (Also start a new scope
01196     // since the initiailizer may be a declaration statement).
01197     if (init) {
01198         Assert(dynamic_cast<StmtList *>(init) == NULL);
01199         ctx->StartScope();
01200         init->EmitCode(ctx);
01201     }
01202     ctx->BranchInst(btest);
01203 
01204     // Emit code to get the value of the loop test.  If no test expression
01205     // was provided, just go with a true value.
01206     ctx->SetCurrentBasicBlock(btest);
01207     llvm::Value *ltest = NULL;
01208     if (test) {
01209         ltest = test->GetValue(ctx);
01210         if (!ltest) {
01211             ctx->EndScope();
01212             ctx->EndLoop();
01213             return;
01214         }
01215     }
01216     else
01217         ltest = uniformTest ? LLVMTrue : LLVMBoolVector(true);
01218 
01219     // Now use the test's value.  For a uniform loop, we can either jump to
01220     // the loop body or the loop exit, based on whether it's true or false.
01221     // For a non-uniform loop, we update the mask and jump into the loop if
01222     // any of the mask values are true.
01223     if (uniformTest) {
01224         if (doCoherentCheck)
01225             Warning(pos, "Uniform condition supplied to cfor/cwhile statement.");
01226         Assert(ltest->getType() == LLVMTypes::BoolType);
01227         ctx->BranchInst(bloop, bexit, ltest);
01228     }
01229     else {
01230         llvm::Value *mask = ctx->GetInternalMask();
01231         ctx->SetInternalMaskAnd(mask, ltest);
01232         ctx->BranchIfMaskAny(bloop, bexit);
01233     }
01234 
01235     // On to emitting the code for the loop body.
01236     ctx->SetCurrentBasicBlock(bloop);
01237     ctx->SetLoopMask(ctx->GetInternalMask());
01238     ctx->AddInstrumentationPoint("for loop body");
01239     if (!dynamic_cast<StmtList *>(stmts))
01240         ctx->StartScope();
01241 
01242     if (doCoherentCheck && !uniformTest) {
01243         // For 'varying' loops with the coherence check, we start by
01244         // checking to see if the mask is all on, after it has been updated
01245         // based on the value of the test.
01246         llvm::BasicBlock *bAllOn = ctx->CreateBasicBlock("for_all_on");
01247         llvm::BasicBlock *bMixed = ctx->CreateBasicBlock("for_mixed");
01248         ctx->BranchIfMaskAll(bAllOn, bMixed);
01249 
01250         // Emit code for the mask being all on.  Explicitly set the mask to
01251         // be on so that the optimizer can see that it's on (i.e. now that
01252         // the runtime test has passed, make this fact clear for code
01253         // generation at compile time here.)
01254         ctx->SetCurrentBasicBlock(bAllOn);
01255         if (!g->opt.disableMaskAllOnOptimizations)
01256             ctx->SetInternalMask(LLVMMaskAllOn);
01257         llvm::Value *oldFunctionMask = ctx->GetFunctionMask();
01258         if (!g->opt.disableMaskAllOnOptimizations)
01259             ctx->SetFunctionMask(LLVMMaskAllOn);
01260         if (stmts)
01261             stmts->EmitCode(ctx);
01262         Assert(ctx->GetCurrentBasicBlock());
01263         ctx->SetFunctionMask(oldFunctionMask);
01264         ctx->BranchInst(bstep);
01265 
01266         // Emit code for the mask being mixed.  We should never run the
01267         // loop with the mask all off, based on the BranchIfMaskAny call
01268         // above.
01269         ctx->SetCurrentBasicBlock(bMixed);
01270         if (stmts)
01271             stmts->EmitCode(ctx);
01272         ctx->BranchInst(bstep);
01273     }
01274     else {
01275         // For both uniform loops and varying loops without the coherence
01276         // check, we know that at least one program instance wants to be
01277         // running the loop, so just emit code for the loop body and jump
01278         // to the loop step code.
01279         if (stmts)
01280             stmts->EmitCode(ctx);
01281         if (ctx->GetCurrentBasicBlock())
01282             ctx->BranchInst(bstep);
01283     }
01284     if (!dynamic_cast<StmtList *>(stmts))
01285         ctx->EndScope();
01286 
01287     // Emit code for the loop step.  First, restore the lane mask of any
01288     // program instances that executed a 'continue' during the previous
01289     // iteration.  Then emit code for the loop step and then jump to the
01290     // test code.
01291     ctx->SetCurrentBasicBlock(bstep);
01292     ctx->RestoreContinuedLanes();
01293     if (step)
01294         step->EmitCode(ctx);
01295     ctx->BranchInst(btest);
01296 
01297     // Set the current emission basic block to the loop exit basic block
01298     ctx->SetCurrentBasicBlock(bexit);
01299     if (init)
01300         ctx->EndScope();
01301     ctx->EndLoop();
01302 }

01455 /* Given a uniform counter value in the memory location pointed to by
01456    uniformCounterPtr, compute the corresponding set of varying counter
01457    values for use within the loop body.
01458  */
01459 static llvm::Value *
01460 lUpdateVaryingCounter(int dim, int nDims, FunctionEmitContext *ctx, 
01461                       llvm::Value *uniformCounterPtr,
01462                       llvm::Value *varyingCounterPtr,
01463                       const std::vector<int> &spans) {
01464     // Smear the uniform counter value out to be varying
01465     llvm::Value *counter = ctx->LoadInst(uniformCounterPtr);
01466     llvm::Value *smearCounter = 
01467         llvm::UndefValue::get(LLVMTypes::Int32VectorType);
01468     for (int i = 0; i < g->target.vectorWidth; ++i)
01469         smearCounter = 
01470             ctx->InsertInst(smearCounter, counter, i, "smear_counter");
01471 
01472     // Figure out the offsets; this is a little bit tricky.  As an example,
01473     // consider a 2D tiled foreach loop, where we're running 8-wide and
01474     // where the inner dimension has a stride of 4 and the outer dimension
01475     // has a stride of 2.  For the inner dimension, we want the offsets
01476     // (0,1,2,3,0,1,2,3), and for the outer dimension we want
01477     // (0,0,0,0,1,1,1,1).
01478     int32_t delta[ISPC_MAX_NVEC];
01479     for (int i = 0; i < g->target.vectorWidth; ++i) {
01480         int d = i;
01481         // First, account for the effect of any dimensions at deeper
01482         // nesting levels than the current one.
01483         int prevDimSpanCount = 1;
01484         for (int j = dim; j < nDims-1; ++j)
01485             prevDimSpanCount *= spans[j+1];
01486         d /= prevDimSpanCount;
01487 
01488         // And now with what's left, figure out our own offset
01489         delta[i] = d % spans[dim];
01490     }
01491 
01492     // Add the deltas to compute the varying counter values; store the
01493     // result to memory and then return it directly as well.
01494     llvm::Value *varyingCounter = 
01495         ctx->BinaryOperator(llvm::Instruction::Add, smearCounter,
01496                             LLVMInt32Vector(delta), "iter_val");
01497     ctx->StoreInst(varyingCounter, varyingCounterPtr);
01498     return varyingCounter;
01499 }

01502 /** Returns the integer log2 of the given integer. */
01503 static int
01504 lLog2(int i) {
01505     int ret = 0;
01506     while (i != 0) {
01507         ++ret;
01508         i >>= 1;
01509     }
01510     return ret-1;
01511 }

01514 /* Figure out how many elements to process in each dimension for each time
01515    through a foreach loop.  The untiled case is easy; all of the outer
01516    dimensions up until the innermost one have a span of 1, and the
01517    innermost one takes the entire vector width.  For the tiled case, we
01518    give wider spans to the innermost dimensions while also trying to
01519    generate relatively square domains.
01520 
01521    This code works recursively from outer dimensions to inner dimensions.
01522  */
01523 static void
01524 lGetSpans(int dimsLeft, int nDims, int itemsLeft, bool isTiled, int *a) {
01525     if (dimsLeft == 0) {
01526         // Nothing left to do but give all of the remaining work to the
01527         // innermost domain.
01528         *a = itemsLeft;
01529         return;
01530     }
01531 
01532     if (isTiled == false || (dimsLeft >= lLog2(itemsLeft)))
01533         // If we're not tiled, or if there are enough dimensions left that
01534         // giving this one any more than a span of one would mean that a
01535         // later dimension would have to have a span of one, give this one
01536         // a span of one to save the available items for later.
01537         *a = 1;
01538     else if (itemsLeft >= 16 && (dimsLeft == 1))
01539         // Special case to have 4x4 domains for the 2D case when running
01540         // 16-wide.
01541         *a = 4;
01542     else
01543         // Otherwise give this dimension a span of two. 
01544         *a = 2;
01545 
01546     lGetSpans(dimsLeft-1, nDims, itemsLeft / *a, isTiled, a+1);
01547 }

01550 /* Emit code for a foreach statement.  We effectively emit code to run the
01551    set of n-dimensional nested loops corresponding to the dimensionality of
01552    the foreach statement along with the extra logic to deal with mismatches
01553    between the vector width we're compiling to and the number of elements
01554    to process.
01555  */
01556 void
01557 ForeachStmt::EmitCode(FunctionEmitContext *ctx) const {
01558     if (ctx->GetCurrentBasicBlock() == NULL || stmts == NULL) 
01559         return;
01560 
01561     llvm::BasicBlock *bbCheckExtras = ctx->CreateBasicBlock("foreach_check_extras");
01562     llvm::BasicBlock *bbDoExtras = ctx->CreateBasicBlock("foreach_do_extras");
01563     llvm::BasicBlock *bbBody = ctx->CreateBasicBlock("foreach_body");
01564     llvm::BasicBlock *bbExit = ctx->CreateBasicBlock("foreach_exit");
01565 
01566     llvm::Value *oldMask = ctx->GetInternalMask();
01567 
01568     ctx->SetDebugPos(pos);
01569     ctx->StartScope();
01570 
01571     // This should be caught during typechecking
01572     Assert(startExprs.size() == dimVariables.size() && 
01573            endExprs.size() == dimVariables.size());
01574     int nDims = (int)dimVariables.size();
01575 
01576     ///////////////////////////////////////////////////////////////////////
01577     // Setup: compute the number of items we have to work on in each
01578     // dimension and a number of derived values.
01579     std::vector<llvm::BasicBlock *> bbReset, bbStep, bbTest;
01580     std::vector<llvm::Value *> startVals, endVals, uniformCounterPtrs;
01581     std::vector<llvm::Value *> nItems, nExtras, alignedEnd;
01582     std::vector<llvm::Value *> extrasMaskPtrs;
01583 
01584     std::vector<int> span(nDims, 0);
01585     lGetSpans(nDims-1, nDims, g->target.vectorWidth, isTiled, &span[0]);
01586 
01587     for (int i = 0; i < nDims; ++i) {
01588         // Basic blocks that we'll fill in later with the looping logic for
01589         // this dimension.
01590         bbReset.push_back(ctx->CreateBasicBlock("foreach_reset"));
01591         bbStep.push_back(ctx->CreateBasicBlock("foreach_step"));
01592         bbTest.push_back(ctx->CreateBasicBlock("foreach_test"));
01593 
01594         // Start and end value for this loop dimension
01595         llvm::Value *sv = startExprs[i]->GetValue(ctx);
01596         llvm::Value *ev = endExprs[i]->GetValue(ctx);
01597         if (sv == NULL || ev == NULL)
01598             return;
01599         startVals.push_back(sv);
01600         endVals.push_back(ev);
01601 
01602         // nItems = endVal - startVal
01603         nItems.push_back(ctx->BinaryOperator(llvm::Instruction::Sub, ev, sv,
01604                                              "nitems"));
01605 
01606         // nExtras = nItems % (span for this dimension)
01607         // This gives us the number of extra elements we need to deal with
01608         // at the end of the loop for this dimension that don't fit cleanly
01609         // into a vector width.
01610         nExtras.push_back(ctx->BinaryOperator(llvm::Instruction::SRem, nItems[i],
01611                                               LLVMInt32(span[i]), "nextras"));
01612 
01613         // alignedEnd = endVal - nExtras
01614         alignedEnd.push_back(ctx->BinaryOperator(llvm::Instruction::Sub, ev,
01615                                                  nExtras[i], "aligned_end"));
01616 
01617         ///////////////////////////////////////////////////////////////////////
01618         // Each dimension has a loop counter that is a uniform value that
01619         // goes from startVal to endVal, in steps of the span for this
01620         // dimension.  Its value is only used internally here for looping
01621         // logic and isn't directly available in the user's program code.
01622         uniformCounterPtrs.push_back(ctx->AllocaInst(LLVMTypes::Int32Type, 
01623                                                      "counter"));
01624         ctx->StoreInst(startVals[i], uniformCounterPtrs[i]);
01625 
01626         // There is also a varying variable that holds the set of index
01627         // values for each dimension in the current loop iteration; this is
01628         // the value that is program-visible.
01629         dimVariables[i]->storagePtr = ctx->AllocaInst(LLVMTypes::Int32VectorType, 
01630                                                   dimVariables[i]->name.c_str());
01631         dimVariables[i]->parentFunction = ctx->GetFunction();
01632         ctx->EmitVariableDebugInfo(dimVariables[i]);
01633 
01634         // Each dimension also maintains a mask that represents which of
01635         // the varying elements in the current iteration should be
01636         // processed.  (i.e. this is used to disable the lanes that have
01637         // out-of-bounds offsets.)
01638         extrasMaskPtrs.push_back(ctx->AllocaInst(LLVMTypes::MaskType, "extras mask"));
01639         ctx->StoreInst(LLVMMaskAllOn, extrasMaskPtrs[i]);
01640     }
01641 
01642     ctx->StartForeach(bbStep[nDims-1]);
01643 
01644     // On to the outermost loop's test
01645     ctx->BranchInst(bbTest[0]);
01646 
01647     ///////////////////////////////////////////////////////////////////////////
01648     // foreach_reset: this code runs when we need to reset the counter for
01649     // a given dimension in preparation for running through its loop again,
01650     // after the enclosing level advances its counter.
01651     for (int i = 0; i < nDims; ++i) {
01652         ctx->SetCurrentBasicBlock(bbReset[i]);
01653         if (i == 0)
01654             ctx->BranchInst(bbExit);
01655         else {
01656             ctx->StoreInst(LLVMMaskAllOn, extrasMaskPtrs[i]);
01657             ctx->StoreInst(startVals[i], uniformCounterPtrs[i]);
01658             ctx->BranchInst(bbStep[i-1]);
01659         }
01660     }
01661 
01662     ///////////////////////////////////////////////////////////////////////////
01663     // foreach_test
01664     std::vector<llvm::Value *> inExtras;
01665     for (int i = 0; i < nDims; ++i) {
01666         ctx->SetCurrentBasicBlock(bbTest[i]);
01667 
01668         llvm::Value *haveExtras = 
01669             ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SGT,
01670                          endVals[i], alignedEnd[i], "have_extras");
01671 
01672         llvm::Value *counter = ctx->LoadInst(uniformCounterPtrs[i], "counter");
01673         llvm::Value *atAlignedEnd = 
01674             ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_EQ,
01675                          counter, alignedEnd[i], "at_aligned_end");
01676         llvm::Value *inEx = 
01677             ctx->BinaryOperator(llvm::Instruction::And, haveExtras,
01678                                 atAlignedEnd, "in_extras");
01679 
01680         if (i == 0)
01681             inExtras.push_back(inEx);
01682         else
01683             inExtras.push_back(ctx->BinaryOperator(llvm::Instruction::Or, inEx,
01684                                                    inExtras[i-1], "in_extras_all"));
01685 
01686         llvm::Value *varyingCounter = 
01687             lUpdateVaryingCounter(i, nDims, ctx, uniformCounterPtrs[i], 
01688                                   dimVariables[i]->storagePtr, span);
01689 
01690         llvm::Value *smearEnd = llvm::UndefValue::get(LLVMTypes::Int32VectorType);
01691         for (int j = 0; j < g->target.vectorWidth; ++j)
01692             smearEnd = ctx->InsertInst(smearEnd, endVals[i], j, "smear_end");
01693         // Do a vector compare of its value to the end value to generate a
01694         // mask for this last bit of work.
01695         llvm::Value *emask = 
01696             ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SLT,
01697                          varyingCounter, smearEnd);
01698         emask = ctx->I1VecToBoolVec(emask);
01699 
01700         if (i == 0)
01701             ctx->StoreInst(emask, extrasMaskPtrs[i]);
01702         else {
01703             // FIXME: at least specialize the innermost loop to not do all
01704             // this mask stuff each time through the test...
01705             llvm::Value *oldMask = ctx->LoadInst(extrasMaskPtrs[i-1]);
01706             llvm::Value *newMask =
01707                 ctx->BinaryOperator(llvm::Instruction::And, oldMask, emask,
01708                                     "extras_mask");
01709             ctx->StoreInst(newMask, extrasMaskPtrs[i]);
01710         }
01711 
01712         llvm::Value *notAtEnd = 
01713             ctx->CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SLT,
01714                          counter, endVals[i]);
01715         if (i != nDims-1)
01716             ctx->BranchInst(bbTest[i+1], bbReset[i], notAtEnd);
01717         else
01718             ctx->BranchInst(bbCheckExtras, bbReset[i], notAtEnd);
01719     }
01720 
01721     ///////////////////////////////////////////////////////////////////////////
01722     // foreach_step: increment the uniform counter by the vector width.
01723     // Note that we don't increment the varying counter here as well but
01724     // just generate its value when we need it in the loop body.
01725     for (int i = 0; i < nDims; ++i) {
01726         ctx->SetCurrentBasicBlock(bbStep[i]);
01727         if (i == nDims-1)
01728             ctx->RestoreContinuedLanes();
01729         llvm::Value *counter = ctx->LoadInst(uniformCounterPtrs[i]);
01730         llvm::Value *newCounter =  
01731             ctx->BinaryOperator(llvm::Instruction::Add, counter,
01732                                 LLVMInt32(span[i]), "new_counter");
01733         ctx->StoreInst(newCounter, uniformCounterPtrs[i]);
01734         ctx->BranchInst(bbTest[i]);
01735     }
01736 
01737     ///////////////////////////////////////////////////////////////////////////
01738     // foreach_check_extras: see if we need to deal with any partial
01739     // vector's worth of work that's left.
01740     ctx->SetCurrentBasicBlock(bbCheckExtras);
01741     ctx->AddInstrumentationPoint("foreach loop check extras");
01742     ctx->BranchInst(bbDoExtras, bbBody, inExtras[nDims-1]);
01743 
01744     ///////////////////////////////////////////////////////////////////////////
01745     // foreach_body: do a full vector's worth of work.  We know that all
01746     // lanes will be running here, so we explicitly set the mask to be 'all
01747     // on'.  This ends up being relatively straightforward: just update the
01748     // value of the varying loop counter and have the statements in the
01749     // loop body emit their code.
01750     ctx->SetCurrentBasicBlock(bbBody);
01751     ctx->SetInternalMask(LLVMMaskAllOn);
01752     ctx->AddInstrumentationPoint("foreach loop body");
01753     stmts->EmitCode(ctx);
01754     Assert(ctx->GetCurrentBasicBlock() != NULL);
01755     ctx->BranchInst(bbStep[nDims-1]);
01756 
01757     ///////////////////////////////////////////////////////////////////////////
01758     // foreach_doextras: set the mask and have the statements emit their
01759     // code again.  Note that it's generally worthwhile having two copies
01760     // of the statements' code, since the code above is emitted with the
01761     // mask known to be all-on, which in turn leads to more efficient code
01762     // for that case.
01763     ctx->SetCurrentBasicBlock(bbDoExtras);
01764     llvm::Value *mask = ctx->LoadInst(extrasMaskPtrs[nDims-1]);
01765     ctx->SetInternalMask(mask);
01766     stmts->EmitCode(ctx);
01767     ctx->BranchInst(bbStep[nDims-1]);
01768 
01769     ///////////////////////////////////////////////////////////////////////////
01770     // foreach_exit: All done.  Restore the old mask and clean up
01771     ctx->SetCurrentBasicBlock(bbExit);
01772     ctx->SetInternalMask(oldMask);
01773 
01774     ctx->EndForeach();
01775     ctx->EndScope();
01776 }

01779 Stmt *
01780 ForeachStmt::TypeCheck() {
01781     bool anyErrors = false;
01782     for (unsigned int i = 0; i < startExprs.size(); ++i) {
01783         if (startExprs[i] != NULL)
01784             startExprs[i] = TypeConvertExpr(startExprs[i], 
01785                                             AtomicType::UniformInt32, 
01786                                             "foreach starting value");
01787         anyErrors |= (startExprs[i] == NULL);
01788     }
01789     for (unsigned int i = 0; i < endExprs.size(); ++i) {
01790         if (endExprs[i] != NULL)
01791             endExprs[i] = TypeConvertExpr(endExprs[i], AtomicType::UniformInt32,
01792                                           "foreach ending value");
01793         anyErrors |= (endExprs[i] == NULL);
01794     }
01795 
01796     if (startExprs.size() < dimVariables.size()) {
01797         Error(pos, "Not enough initial values provided for \"foreach\" loop; "
01798               "got %d, expected %d\n", (int)startExprs.size(), (int)dimVariables.size());
