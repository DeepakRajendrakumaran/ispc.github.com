00379 void
00380 FunctionEmitContext::EndIf() {
00381     // Make sure we match up with a Start{Uniform,Varying}If().
00382     assert(controlFlowInfo.size() > 0 && controlFlowInfo.back()->IsIf());
00383     CFInfo *ci = controlFlowInfo.back();
00384     controlFlowInfo.pop_back();
00385 
00386     // 'uniform' ifs don't change the mask so we only need to restore the
00387     // mask going into the if for 'varying' if statements
00388     if (!ci->IsUniform() && bblock != NULL) {
00389         // We can't just restore the mask as it was going into the 'if'
00390         // statement.  First we have to take into account any program
00391         // instances that have executed 'return' statements; the restored
00392         // mask must be off for those lanes.
00393         restoreMaskGivenReturns(ci->savedMask);
00394 
00395         // If the 'if' statement is inside a loop with a 'varying'
00396         // consdition, we also need to account for any break or continue
00397         // statements that executed inside the 'if' statmeent; we also must
00398         // leave the lane masks for the program instances that ran those
00399         // off after we restore the mask after the 'if'.  The code below
00400         // ends up being optimized out in the case that there were no break
00401         // or continue statements (and breakLanesPtr and continueLanesPtr
00402         // have their initial 'all off' values), so we don't need to check
00403         // for that here.
00404         if (continueLanesPtr != NULL) {
00405             // We want to compute:
00406             // newMask = (oldMask & ~(breakLanes | continueLanes))
00407             llvm::Value *oldMask = GetInternalMask();
00408             llvm::Value *continueLanes = LoadInst(continueLanesPtr,
00409                                                   "continue_lanes");
00410             llvm::Value *bcLanes = continueLanes;
00411 
00412             if (breakLanesPtr != NULL) {
00413                 // breakLanesPtr will be NULL if we're inside a 'foreach' loop
00414                 llvm::Value *breakLanes = LoadInst(breakLanesPtr, "break_lanes");
00415                 bcLanes = BinaryOperator(llvm::Instruction::Or, breakLanes, 
00416                                          continueLanes, "break|continue_lanes");
00417             }
00418 
00419             llvm::Value *notBreakOrContinue = 
00420                 NotOperator(bcLanes, "!(break|continue)_lanes");
00421             llvm::Value *newMask = 
00422                 BinaryOperator(llvm::Instruction::And, oldMask, 
00423                                notBreakOrContinue, "new_mask");
00424             SetInternalMask(newMask);
00425         }
00426     }
00427 }

00432                                bool uniformCF) {

00535 void
00536 FunctionEmitContext::Break(bool doCoherenceCheck) {
00537     if (breakTarget == NULL) {
00538         Error(currentPos, "\"break\" statement is illegal outside of "
00539               "for/while/do loops.");
00540         return;
00541     }
00542 
00543     // If all of the enclosing 'if' tests in the loop have uniform control
00544     // flow or if we can tell that the mask is all on, then we can just
00545     // jump to the break location.
00546     if (ifsInLoopAllUniform() || GetInternalMask() == LLVMMaskAllOn) {
00547         BranchInst(breakTarget);
00548         if (ifsInLoopAllUniform() && doCoherenceCheck)
00549             Warning(currentPos, "Coherent break statement not necessary in fully uniform "
00550                     "control flow.");
00551         // Set bblock to NULL since the jump has terminated the basic block
00552         bblock = NULL;
00553     }
00554     else {
00555         // Otherwise we need to update the mask of the lanes that have
00556         // executed a 'break' statement:
00557         // breakLanes = breakLanes | mask
00558         assert(breakLanesPtr != NULL);
00559         llvm::Value *mask = GetInternalMask();
00560         llvm::Value *breakMask = LoadInst(breakLanesPtr,
00561                                           "break_mask");
00562         llvm::Value *newMask = BinaryOperator(llvm::Instruction::Or,
00563                                               mask, breakMask, "mask|break_mask");
00564         StoreInst(newMask, breakLanesPtr);
00565 
00566         // Set the current mask to be all off, just in case there are any
00567         // statements in the same scope after the 'break'.  Most of time
00568         // this will be optimized away since we'll likely end the scope of
00569         // an 'if' statement and restore the mask then.
00570         SetInternalMask(LLVMMaskAllOff);
00571 
00572         if (doCoherenceCheck)
00573             // If the user has indicated that this is a 'coherent' break
00574             // statement, then check to see if the mask is all off.  If so,
00575             // we have to conservatively jump to the continueTarget, not
00576             // the breakTarget, since part of the reason the mask is all
00577             // off may be due to 'continue' statements that executed in the
00578             // current loop iteration.  
00579             // FIXME: if the loop only has break statements and no
00580             // continues, we can jump to breakTarget in that case.
00581             jumpIfAllLoopLanesAreDone(continueTarget);
00582     }
00583 }

00617         // statements in the same scope after the 'continue'
00618         SetInternalMask(LLVMMaskAllOff);

00620         if (doCoherenceCheck) 
00621             // If this is a 'coherent continue' statement, then emit the
00622             // code to see if all of the lanes are now off due to
00623             // breaks/continues and jump to the continue target if so.
00624             jumpIfAllLoopLanesAreDone(continueTarget);
00625     }
00626 }

00650 void
00651 FunctionEmitContext::jumpIfAllLoopLanesAreDone(llvm::BasicBlock *target) {
00652     llvm::Value *allDone = NULL;
00653     assert(continueLanesPtr != NULL);
00654     if (breakLanesPtr == NULL) {
00655         // In a foreach loop, break and return are illegal, and
00656         // breakLanesPtr is NULL.  In this case, the mask is guaranteed to
00657         // be all on at the start of each iteration, so we only need to
00658         // check if all lanes have continued..
00659         llvm::Value *continued = LoadInst(continueLanesPtr,
00660                                           "continue_lanes");
00661         allDone = All(continued);
00662     }
00663     else {
00664         // Check to see if (returned lanes | continued lanes | break lanes) is
00665         // equal to the value of mask at the start of the loop iteration.  If
00666         // so, everyone is done and we can jump to the given target
00667         llvm::Value *returned = LoadInst(returnedLanesPtr,
00668                                          "returned_lanes");
00669         llvm::Value *continued = LoadInst(continueLanesPtr,
00670                                           "continue_lanes");
00671         llvm::Value *breaked = LoadInst(breakLanesPtr, "break_lanes");
00672         llvm::Value *returnedOrContinued = BinaryOperator(llvm::Instruction::Or, 
00673                                                           returned, continued,
00674                                                           "returned|continued");
00675         llvm::Value *returnedOrContinuedOrBreaked = 
00676             BinaryOperator(llvm::Instruction::Or, returnedOrContinued,
00677                            breaked, "returned|continued");
00678 
00679         // Do we match the mask at loop entry?
00680         allDone = MasksAllEqual(returnedOrContinuedOrBreaked, loopMask);
00681     }
00682 
00683     llvm::BasicBlock *bAll = CreateBasicBlock("all_continued_or_breaked");
00684     llvm::BasicBlock *bNotAll = CreateBasicBlock("not_all_continued_or_breaked");
00685     BranchInst(bAll, bNotAll, allDone);
00686 
00687     // If so, have an extra basic block along the way to add
00688     // instrumentation, if the user asked for it.
00689     bblock = bAll;
00690     AddInstrumentationPoint("break/continue: all dynamically went");
00691     BranchInst(target);
00692 
00693     // And set the current basic block to a new one for future instructions
00694     // for the path where we weren't able to jump
00695     bblock = bNotAll;
00696     AddInstrumentationPoint("break/continue: not all went");
00697 }

00737 void
00738 FunctionEmitContext::CurrentLanesReturned(Expr *expr, bool doCoherenceCheck) {
00739     const Type *returnType = function->GetReturnType();
00740     if (returnType == AtomicType::Void) {
00741         if (expr != NULL)
00742             Error(expr->pos, "Can't return non-void type \"%s\" from void function.",
00743                   expr->GetType()->GetString().c_str());
00744     }
00745     else {
00746         if (expr == NULL) {
00747             Error(funcStartPos, "Must provide return value for return "
00748                   "statement for non-void function.");
00749             return;
00750         }
00751         
00752         expr = TypeConvertExpr(expr, returnType, "return statement");
00753         if (expr != NULL) {
00754             llvm::Value *retVal = expr->GetValue(this);
00755             if (retVal != NULL)
00756                 // Use a masked store to store the value of the expression
00757                 // in the return value memory; this preserves the return
00758                 // values from other lanes that may have executed return
00759                 // statements previously.
00760                 StoreInst(retVal, returnValuePtr, GetInternalMask(), 
00761                           PointerType::GetUniform(returnType));
00762         }
00763     }
00764 
00765     if (VaryingCFDepth() == 0) {
00766         // If there is only uniform control flow between us and the
00767         // function entry, then it's guaranteed that all lanes are running,
00768         // so we can just emit a true return instruction
00769         AddInstrumentationPoint("return: uniform control flow");
00770         ReturnInst();
00771     }
00772     else {
00773         // Otherwise we update the returnedLanes value by ANDing it with
00774         // the current lane mask.
00775         llvm::Value *oldReturnedLanes = 
00776             LoadInst(returnedLanesPtr, "old_returned_lanes");
00777         llvm::Value *newReturnedLanes = 
00778             BinaryOperator(llvm::Instruction::Or, oldReturnedLanes, 
00779                            GetInternalMask(), "old_mask|returned_lanes");
00780         
00781         // For 'coherent' return statements, emit code to check if all
00782         // lanes have returned
00783         if (doCoherenceCheck) {
00784             // if newReturnedLanes == functionMaskValue, get out of here!
00785             llvm::Value *cmp = MasksAllEqual(functionMaskValue, 
00786                                              newReturnedLanes);
00787             llvm::BasicBlock *bDoReturn = CreateBasicBlock("do_return");
00788             llvm::BasicBlock *bNoReturn = CreateBasicBlock("no_return");
00789             BranchInst(bDoReturn, bNoReturn, cmp);
00790 
00791             bblock = bDoReturn;
00792             AddInstrumentationPoint("return: all lanes have returned");
00793             ReturnInst();
00794 
00795             bblock = bNoReturn;
00796         }
00797         // Otherwise update returnedLanesPtr and turn off all of the lanes
00798         // in the current mask so that any subsequent statements in the
00799         // same scope after the return have no effect
00800         StoreInst(newReturnedLanes, returnedLanesPtr);
00801         AddInstrumentationPoint("return: some but not all lanes have returned");
00802         SetInternalMask(LLVMMaskAllOff);
00803     }
00804 }

01404 /** Utility routine used by the GetElementPtrInst() methods; given a
01405     pointer to some type (either uniform or varying) and an index (also
01406     either uniform or varying), this returns the new pointer (varying if
01407     appropriate) given by offsetting the base pointer by the index times
01408     the size of the object that the pointer points to.
01409  */
01410 llvm::Value *
01411 FunctionEmitContext::applyVaryingGEP(llvm::Value *basePtr, llvm::Value *index, 
01412                                      const Type *ptrType) {
01413     // Find the scale factor for the index (i.e. the size of the object
01414     // that the pointer(s) point(s) to.
01415     const Type *scaleType = ptrType->GetBaseType();
01416     llvm::Value *scale = g->target.SizeOf(scaleType->LLVMType(g->ctx));
01417 
01418     bool indexIsVarying = 
01419         llvm::isa<LLVM_TYPE_CONST llvm::VectorType>(index->getType());
01420     llvm::Value *offset = NULL;
01421     if (indexIsVarying == false) {
01422         // Truncate or sign extend the index as appropriate to a 32 or
01423         // 64-bit type.
01424         if ((g->target.is32Bit || g->opt.force32BitAddressing) && 
01425             index->getType() == LLVMTypes::Int64Type)
01426             index = TruncInst(index, LLVMTypes::Int32Type, "trunc_index");
01427         else if ((!g->target.is32Bit && !g->opt.force32BitAddressing) &&
01428                  index->getType() == LLVMTypes::Int32Type)
01429             index = SExtInst(index, LLVMTypes::Int64Type, "sext_index");
01430 
01431         // do a scalar multiply to get the offset as index * scale and then
01432         // smear the result out to be a vector; this is more efficient than
01433         // first promoting both the scale and the index to vectors and then
01434         // multiplying.
01435         offset = BinaryOperator(llvm::Instruction::Mul, scale, index);
01436         offset = SmearUniform(offset, "offset_smear");
01437     }
01438     else {
01439         // Similarly, truncate or sign extend the index to be a 32 or 64
01440         // bit vector type
01441         if ((g->target.is32Bit || g->opt.force32BitAddressing) && 
01442             index->getType() == LLVMTypes::Int64VectorType)
01443             index = TruncInst(index, LLVMTypes::Int32VectorType, "trunc_index");
01444         else if ((!g->target.is32Bit && !g->opt.force32BitAddressing) &&
01445                  index->getType() == LLVMTypes::Int32VectorType)
01446             index = SExtInst(index, LLVMTypes::Int64VectorType, "sext_index");
01447 
01448         scale = SmearUniform(scale, "scale_smear");
01449 
01450         // offset = index * scale
01451         offset = BinaryOperator(llvm::Instruction::Mul, scale, index, "offset");
01452     }
01453 
01454     // For 64-bit targets, if we've been doing our offset calculations in
01455     // 32 bits, we still have to convert to a 64-bit value before we
01456     // actually add the offset to the pointer.
01457     if (g->target.is32Bit == false && g->opt.force32BitAddressing == true)
01458         offset = SExtInst(offset, LLVMTypes::Int64VectorType, "offset_to_64");
01459 
01460     // Smear out the pointer to be varying; either the base pointer or the
01461     // index must be varying for this method to be called.
01462     bool baseIsUniform = 
01463         (llvm::isa<LLVM_TYPE_CONST llvm::PointerType>(basePtr->getType()));
01464     assert(baseIsUniform == false || indexIsVarying == true);
01465     llvm::Value *varyingPtr = baseIsUniform ? 
01466         SmearUniform(basePtr, "ptr_smear") : basePtr;
01467 
01468     // newPtr = ptr + offset
01469     return BinaryOperator(llvm::Instruction::Add, varyingPtr, offset, "new_ptr");
01470 }

01569 FunctionEmitContext::AddElementOffset(llvm::Value *basePtr, int elementNum,
01570                                       const Type *ptrType, const char *name) {
01571     if (ptrType == NULL || ptrType->IsUniformType() ||
01572         dynamic_cast<const ReferenceType *>(ptrType) != NULL) {
01573         // If the pointer is uniform or we have a reference (which is a
01574         // uniform pointer in the end), we can use the regular LLVM GEP.
01575         llvm::Value *offsets[2] = { LLVMInt32(0), LLVMInt32(elementNum) };
01576 #if defined(LLVM_3_0) || defined(LLVM_3_0svn) || defined(LLVM_3_1svn)
01577         llvm::ArrayRef<llvm::Value *> arrayRef(&offsets[0], &offsets[2]);
01578         return llvm::GetElementPtrInst::Create(basePtr, arrayRef,
01579                                                name ? name : "struct_offset", bblock);
01580 #else
01581         return llvm::GetElementPtrInst::Create(basePtr, &offsets[0], &offsets[2],
01582                                                name ? name : "struct_offset", bblock);
01583 #endif
01584 
01585     }
01586 
01587     if (dynamic_cast<const ReferenceType *>(ptrType) != NULL)
01588         ptrType = PointerType::GetUniform(ptrType->GetReferenceTarget());
01589     assert(dynamic_cast<const PointerType *>(ptrType) != NULL);
01590 
01591     // Otherwise do the math to find the offset and add it to the given
01592     // varying pointers
01593     const StructType *st = 
01594         dynamic_cast<const StructType *>(ptrType->GetBaseType());
01595     llvm::Value *offset = NULL;
01596     if (st != NULL)
01597         // If the pointer is to a structure, Target::StructOffset() gives
01598         // us the offset in bytes to the given element of the structure
01599         offset = g->target.StructOffset(st->LLVMType(g->ctx), elementNum);
01600     else {
01601         // Otherwise we should have a vector here and the offset is given
01602         // by the element number times the size of the element type of the
01603         // vector.
01604         const VectorType *vt = 
01605             dynamic_cast<const VectorType *>(ptrType->GetBaseType());
01606         assert(vt != NULL);
01607         llvm::Value *size = 
01608             g->target.SizeOf(vt->GetElementType()->LLVMType(g->ctx));
01609         llvm::Value *scale = (g->target.is32Bit || g->opt.force32BitAddressing) ?
01610             LLVMInt32(elementNum) : LLVMInt64(elementNum);
01611         offset = BinaryOperator(llvm::Instruction::Mul, size, scale);
01612     }
01613 
01614     offset = SmearUniform(offset, "offset_smear");
01615 
01616     if (g->target.is32Bit == false && g->opt.force32BitAddressing == true)
01617         // If we're doing 32 bit addressing with a 64 bit target, although
01618         // we did the math above in 32 bit, we need to go to 64 bit before
01619         // we add the offset to the varying pointers.
01620         offset = SExtInst(offset, LLVMTypes::Int64VectorType, "offset_to_64");
01621 
01622     return BinaryOperator(llvm::Instruction::Add, basePtr, offset, 
01623                           "struct_ptr_offset");
01624 }

01699 llvm::Value *
01700 FunctionEmitContext::gather(llvm::Value *ptr, const Type *ptrType, 
01701                             llvm::Value *mask, const char *name) {
01702     // We should have a varying lvalue if we get here...
01703     assert(ptrType->IsVaryingType() &&
01704            ptr->getType() == LLVMTypes::VoidPointerVectorType);
01705 
01706     const Type *returnType = ptrType->GetBaseType()->GetAsVaryingType();
01707     LLVM_TYPE_CONST llvm::Type *llvmReturnType = returnType->LLVMType(g->ctx);
01708 
01709     const CollectionType *collectionType = 
01710         dynamic_cast<const CollectionType *>(ptrType->GetBaseType());
01711     if (collectionType != NULL) {
01712         // For collections, recursively gather element wise to find the
01713         // result.
01714         llvm::Value *retValue = llvm::UndefValue::get(llvmReturnType);
01715         for (int i = 0; i < collectionType->GetElementCount(); ++i) {
01716             llvm::Value *eltPtr = AddElementOffset(ptr, i, ptrType);
01717             const Type *eltPtrType = 
01718                 PointerType::GetVarying(collectionType->GetElementType(i));
01719             eltPtr = addVaryingOffsetsIfNeeded(eltPtr, eltPtrType);
01720 
01721             // This in turn will be another gather
01722             llvm::Value *eltValues = LoadInst(eltPtr, mask, eltPtrType, name);
01723 
01724             retValue = InsertInst(retValue, eltValues, i, "set_value");
01725         }
01726         return retValue;
01727     }
01728     
01729     // Otherwise we should just have a basic scalar or pointer type and we
01730     // can go and do the actual gather
01731     AddInstrumentationPoint("gather");
01732 
01733     // Figure out which gather function to call based on the size of
01734     // the elements.
01735     const PointerType *pt = dynamic_cast<const PointerType *>(returnType);
01736     const char *funcName = NULL;
01737     if (pt != NULL)
01738         funcName = g->target.is32Bit ? "__pseudo_gather32_32" : 
01739             "__pseudo_gather64_64";
01740     else if (llvmReturnType == LLVMTypes::DoubleVectorType || 
01741              llvmReturnType == LLVMTypes::Int64VectorType)
01742         funcName = g->target.is32Bit ? "__pseudo_gather32_64" : 
01743             "__pseudo_gather64_64";
01744     else if (llvmReturnType == LLVMTypes::FloatVectorType || 
01745              llvmReturnType == LLVMTypes::Int32VectorType)
01746         funcName = g->target.is32Bit ? "__pseudo_gather32_32" : 
01747             "__pseudo_gather64_32";
01748     else if (llvmReturnType == LLVMTypes::Int16VectorType)
01749         funcName = g->target.is32Bit ? "__pseudo_gather32_16" : 
01750             "__pseudo_gather64_16";
01751     else {
01752         assert(llvmReturnType == LLVMTypes::Int8VectorType);
01753         funcName = g->target.is32Bit ? "__pseudo_gather32_8" : 
01754             "__pseudo_gather64_8";
01755     }
01756 
01757     llvm::Function *gatherFunc = m->module->getFunction(funcName);
01758     assert(gatherFunc != NULL);
01759 
01760     llvm::Value *call = CallInst(gatherFunc, NULL, ptr, mask, name);
01761 
01762     // Add metadata about the source file location so that the
01763     // optimization passes can print useful performance warnings if we
01764     // can't optimize out this gather
01765     addGSMetadata(call, currentPos);
01766 
01767     return BitCastInst(call, llvmReturnType, "gather_bitcast");
01768 }

01838 /** Code to store the given varying value to the given location, only
01839     storing the elements that correspond to active program instances as
01840     given by the provided storeMask value.  Note that the lvalue is only a
01841     single pointer, not a varying lvalue of one pointer per program
01842     instance (that case is handled by scatters).
01843  */
01844 void
01845 FunctionEmitContext::maskedStore(llvm::Value *value, llvm::Value *ptr,
01846                                  const Type *ptrType, llvm::Value *mask) {
01847     if (value == NULL || ptr == NULL) {
01848         assert(m->errorCount > 0);
01849         return;
01850     }
01851 
01852     assert(dynamic_cast<const PointerType *>(ptrType) != NULL);
01853     assert(ptrType->IsUniformType());
01854 
01855     const Type *valueType = ptrType->GetBaseType();
01856     const CollectionType *collectionType = 
01857         dynamic_cast<const CollectionType *>(valueType);
01858     if (collectionType != NULL) {
01859         // Assigning a structure / array / vector. Handle each element
01860         // individually with what turns into a recursive call to
01861         // makedStore()
01862         for (int i = 0; i < collectionType->GetElementCount(); ++i) {
01863             llvm::Value *eltValue = ExtractInst(value, i, "value_member");
01864             llvm::Value *eltPtr = 
01865                 AddElementOffset(ptr, i, ptrType, "struct_ptr_ptr");
01866             const Type *eltPtrType = 
01867                 PointerType::GetUniform(collectionType->GetElementType(i));
01868             StoreInst(eltValue, eltPtr, mask, eltPtrType);
01869         }
01870         return;
01871     }
01872 
01873     // We must have a regular atomic, enumerator, or pointer type at this
01874     // point.
01875     assert(dynamic_cast<const AtomicType *>(valueType) != NULL ||
01876            dynamic_cast<const EnumType *>(valueType) != NULL ||
01877            dynamic_cast<const PointerType *>(valueType) != NULL);
01878     valueType = valueType->GetAsNonConstType();
01879 
01880     llvm::Function *maskedStoreFunc = NULL;
01881     // Figure out if we need a 8, 16, 32 or 64-bit masked store.
01882     if (dynamic_cast<const PointerType *>(valueType) != NULL) {
01883         if (g->target.is32Bit)
01884             maskedStoreFunc = m->module->getFunction("__pseudo_masked_store_32");
01885         else
01886             maskedStoreFunc = m->module->getFunction("__pseudo_masked_store_64");
01887     }
01888     else if (valueType == AtomicType::VaryingDouble || 
01889              valueType == AtomicType::VaryingInt64 ||
01890              valueType == AtomicType::VaryingUInt64) {
01891         maskedStoreFunc = m->module->getFunction("__pseudo_masked_store_64");
01892         ptr = BitCastInst(ptr, LLVMTypes::Int64VectorPointerType, 
01893                              "ptr_to_int64vecptr");
01894         value = BitCastInst(value, LLVMTypes::Int64VectorType, 
01895                              "value_to_int64");
01896     }
01897     else if (valueType == AtomicType::VaryingFloat ||
01898              valueType == AtomicType::VaryingBool ||
01899              valueType == AtomicType::VaryingInt32 ||
01900              valueType == AtomicType::VaryingUInt32 ||
01901              dynamic_cast<const EnumType *>(valueType) != NULL) {
01902         maskedStoreFunc = m->module->getFunction("__pseudo_masked_store_32");
01903         ptr = BitCastInst(ptr, LLVMTypes::Int32VectorPointerType, 
01904                              "ptr_to_int32vecptr");
01905         if (valueType == AtomicType::VaryingFloat)
01906             value = BitCastInst(value, LLVMTypes::Int32VectorType, 
01907                                  "value_to_int32");
01908     }
01909     else if (valueType == AtomicType::VaryingInt16 ||
01910              valueType == AtomicType::VaryingUInt16) {
01911         maskedStoreFunc = m->module->getFunction("__pseudo_masked_store_16");
01912         ptr = BitCastInst(ptr, LLVMTypes::Int16VectorPointerType, 
01913                              "ptr_to_int16vecptr");
01914     }
01915     else if (valueType == AtomicType::VaryingInt8 ||
01916              valueType == AtomicType::VaryingUInt8) {
01917         maskedStoreFunc = m->module->getFunction("__pseudo_masked_store_8");
01918         ptr = BitCastInst(ptr, LLVMTypes::Int8VectorPointerType, 
01919                              "ptr_to_int8vecptr");
01920     }
01921     assert(maskedStoreFunc != NULL);
01922 
01923     std::vector<llvm::Value *> args;
01924     args.push_back(ptr);
01925     args.push_back(value);
01926     args.push_back(mask);
01927     CallInst(maskedStoreFunc, NULL, args);
01928 }

01932 /** Scatter the given varying value to the locations given by the varying
01933     lvalue (which should be an array of pointers with size equal to the
01934     target's vector width.  We want to store each rvalue element at the
01935     corresponding pointer's location, *if* the mask for the corresponding
01936     program instance are on.  If they're off, don't do anything.  
01937 */
01938 void
01939 FunctionEmitContext::scatter(llvm::Value *value, llvm::Value *ptr, 
01940                              const Type *ptrType, llvm::Value *mask) {
01941     assert(dynamic_cast<const PointerType *>(ptrType) != NULL);
01942     assert(ptrType->IsVaryingType());
01943 
01944     const Type *valueType = ptrType->GetBaseType();
01945 
01946     // I think this should be impossible
01947     assert(dynamic_cast<const ArrayType *>(valueType) == NULL);
01948 
01949     const CollectionType *collectionType = dynamic_cast<const CollectionType *>(valueType);
01950     if (collectionType != NULL) {
01951         // Scatter the collection elements individually
01952         for (int i = 0; i < collectionType->GetElementCount(); ++i) {
01953             llvm::Value *eltPtr = AddElementOffset(ptr, i, ptrType);
01954             llvm::Value *eltValue = ExtractInst(value, i);
01955             const Type *eltPtrType = 
01956                 PointerType::GetVarying(collectionType->GetElementType(i));
01957             eltPtr = addVaryingOffsetsIfNeeded(eltPtr, eltPtrType);
01958             scatter(eltValue, eltPtr, eltPtrType, mask);
01959         }
01960         return;
01961     }
01962 
01963     const PointerType *pt = dynamic_cast<const PointerType *>(valueType);
01964 
01965     // And everything should be a pointer or atomic from here on out...
01966     assert(pt != NULL || 
01967            dynamic_cast<const AtomicType *>(valueType) != NULL);
01968 
01969     LLVM_TYPE_CONST llvm::Type *type = value->getType();
01970     const char *funcName = NULL;
01971     if (pt != NULL)
01972         funcName = g->target.is32Bit ? "__pseudo_scatter32_32" :
01973             "__pseudo_scatter64_64";
01974     else if (type == LLVMTypes::DoubleVectorType || 
01975              type == LLVMTypes::Int64VectorType) {
01976         funcName = g->target.is32Bit ? "__pseudo_scatter32_64" :
01977             "__pseudo_scatter64_64";
01978         value = BitCastInst(value, LLVMTypes::Int64VectorType, "value2int");
01979     }
01980     else if (type == LLVMTypes::FloatVectorType || 
01981              type == LLVMTypes::Int32VectorType) {
01982         funcName = g->target.is32Bit ? "__pseudo_scatter32_32" :
01983             "__pseudo_scatter64_32";
01984         value = BitCastInst(value, LLVMTypes::Int32VectorType, "value2int");
01985     }
01986     else if (type == LLVMTypes::Int16VectorType)
01987         funcName = g->target.is32Bit ? "__pseudo_scatter32_16" :
01988             "__pseudo_scatter64_16";
01989     else if (type == LLVMTypes::Int8VectorType)
01990         funcName = g->target.is32Bit ? "__pseudo_scatter32_8" :
01991             "__pseudo_scatter64_8";
01992 
01993     llvm::Function *scatterFunc = m->module->getFunction(funcName);
01994     assert(scatterFunc != NULL);
01995     
01996     AddInstrumentationPoint("scatter");
01997 
01998     std::vector<llvm::Value *> args;
01999     args.push_back(ptr);
02000     args.push_back(value);
02001     args.push_back(mask);
02002     llvm::Value *inst = CallInst(scatterFunc, NULL, args);
02003     addGSMetadata(inst, currentPos);

02177 llvm::Value *
02178 FunctionEmitContext::CallInst(llvm::Value *func, const FunctionType *funcType,
02179                               const std::vector<llvm::Value *> &args,
02180                               const char *name) {
02181     if (func == NULL) {
02182         assert(m->errorCount > 0);
02183         return NULL;
02184     }
02185 
02186     std::vector<llvm::Value *> argVals = args;
02187     // Most of the time, the mask is passed as the last argument.  this
02188     // isn't the case for things like intrinsics, builtins, and extern "C"
02189     // functions from the application.  Add the mask if it's needed.
02190     unsigned int calleeArgCount = lCalleeArgCount(func, funcType);
02191     assert(argVals.size() + 1 == calleeArgCount ||
02192            argVals.size() == calleeArgCount);
02193     if (argVals.size() + 1 == calleeArgCount)
02194         argVals.push_back(GetFullMask());
02195 
02196     if (llvm::isa<LLVM_TYPE_CONST llvm::VectorType>(func->getType()) == false) {
02197         // Regular 'uniform' function call--just one function or function
02198         // pointer, so just emit the IR directly.
02199 #if defined(LLVM_3_0) || defined(LLVM_3_0svn) || defined(LLVM_3_1svn)
02200         llvm::Instruction *ci = 
02201             llvm::CallInst::Create(func, argVals, name ? name : "", bblock);
02202 #else
02203         llvm::Instruction *ci = 
02204             llvm::CallInst::Create(func, argVals.begin(), argVals.end(), 
02205                                    name ? name : "", bblock);
02206 #endif
02207         AddDebugPos(ci);
02208         return ci;
02209     }
02210     else {
02211         // Emit the code for a varying function call, where we have an
02212         // vector of function pointers, one for each program instance.  The
02213         // basic strategy is that we go through the function pointers, and
02214         // for the executing program instances, for each unique function
02215         // pointer that's in the vector, call that function with a mask
02216         // equal to the set of active program instances that also have that
02217         // function pointer.  When all unique function pointers have been
02218         // called, we're done.
02219 
02220         llvm::BasicBlock *bbTest = CreateBasicBlock("varying_funcall_test");
02221         llvm::BasicBlock *bbCall = CreateBasicBlock("varying_funcall_call");
02222         llvm::BasicBlock *bbDone = CreateBasicBlock("varying_funcall_done");
02223 
02224         // Get the current mask value so we can restore it later
02225         llvm::Value *origMask = GetInternalMask();
02226 
02227         // First allocate memory to accumulate the various program
02228         // instances' return values...
02229         const Type *returnType = funcType->GetReturnType();
02230         LLVM_TYPE_CONST llvm::Type *llvmReturnType = returnType->LLVMType(g->ctx);
02231         llvm::Value *resultPtr = NULL;
02232         if (llvmReturnType->isVoidTy() == false)
02233             resultPtr = AllocaInst(llvmReturnType);
02234 
02235         // The memory pointed to by maskPointer tracks the set of program
02236         // instances for which we still need to call the function they are
02237         // pointing to.  It starts out initialized with the mask of
02238         // currently running program instances.
02239         llvm::Value *maskPtr = AllocaInst(LLVMTypes::MaskType);
02240         StoreInst(GetFullMask(), maskPtr);
02241 
02242         // And now we branch to the test to see if there's more work to be
02243         // done.
02244         BranchInst(bbTest);
02245 
02246         // bbTest: are any lanes of the mask still on?  If so, jump to
02247         // bbCall
02248         SetCurrentBasicBlock(bbTest); {
02249             llvm::Value *maskLoad = LoadInst(maskPtr);
02250             llvm::Value *any = Any(maskLoad);
02251             BranchInst(bbCall, bbDone, any);
02252         }
02253 
02254         // bbCall: this is the body of the loop that calls out to one of
02255         // the active function pointer values.
02256         SetCurrentBasicBlock(bbCall); {
02257             // Figure out the first lane that still needs its function
02258             // pointer to be called.
02259             llvm::Value *currentMask = LoadInst(maskPtr);
02260             llvm::Function *cttz = 
02261                 m->module->getFunction("__count_trailing_zeros_i32");
02262             assert(cttz != NULL);
02263             llvm::Value *firstLane = CallInst(cttz, NULL, LaneMask(currentMask),
02264                                               "first_lane");
02265 
02266             // Get the pointer to the function we're going to call this
02267             // time through: ftpr = func[firstLane]
02268             llvm::Value *fptr = 
02269                 llvm::ExtractElementInst::Create(func, firstLane, 
02270                                                  "extract_fptr", bblock);
02271 
02272             // Smear it out into an array of function pointers
02273             llvm::Value *fptrSmear = SmearUniform(fptr, "func_ptr");
02274 
02275             // fpOverlap = (fpSmearAsVec == fpOrigAsVec).  This gives us a
02276             // mask for the set of program instances that have the same
02277             // value for their function pointer.
02278             llvm::Value *fpOverlap = 
02279                 CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_EQ,
02280                         fptrSmear, func);
02281             fpOverlap = I1VecToBoolVec(fpOverlap);
02282 
02283             // Figure out the mask to use when calling the function
02284             // pointer: we need to AND the current execution mask to handle
02285             // the case of any non-running program instances that happen to
02286             // have this function pointer value.
02287             // callMask = (currentMask & fpOverlap)
02288             llvm::Value *callMask = 
02289                 BinaryOperator(llvm::Instruction::And, currentMask, fpOverlap,
02290                                "call_mask");
02291 
02292             // Set the mask
02293             SetInternalMask(callMask);
02294 
02295             // bitcast the i32/64 function pointer to the actual function
02296             // pointer type (the variant that includes a mask).
02297             LLVM_TYPE_CONST llvm::Type *llvmFuncType =
02298                 funcType->LLVMFunctionType(g->ctx, true);
02299             LLVM_TYPE_CONST llvm::Type *llvmFPtrType = 
02300                 llvm::PointerType::get(llvmFuncType, 0);
02301             llvm::Value *fptrCast = IntToPtrInst(fptr, llvmFPtrType);
02302 
02303             // Call the function: callResult = call ftpr(args, args, call mask)
02304             llvm::Value *callResult = CallInst(fptrCast, funcType, args, name);
02305 
02306             // Now, do a masked store into the memory allocated to
02307             // accumulate the result using the call mask.
02308             if (callResult != NULL) {
02309                 assert(resultPtr != NULL);
02310                 StoreInst(callResult, resultPtr, callMask, 
02311                           PointerType::GetUniform(returnType));
02312             }
02313             else
02314                 assert(resultPtr == NULL);
02315 
02316             // Update the mask to turn off the program instances for which
02317             // we just called the function.
02318             // currentMask = currentMask & ~callmask
02319             llvm::Value *notCallMask = 
02320                 BinaryOperator(llvm::Instruction::Xor, callMask, LLVMMaskAllOn,
02321                                "~callMask");
02322             currentMask = BinaryOperator(llvm::Instruction::And, currentMask, 
02323                                          notCallMask, "currentMask&~callMask");
02324             StoreInst(currentMask, maskPtr);
02325 
02326             // And go back to the test to see if we need to do another
02327             // call.
02328             BranchInst(bbTest);
02329         }
02330 
02331         // bbDone: We're all done; clean up and return the result we've
02332         // accumulated in the result memory.
02333         SetCurrentBasicBlock(bbDone);
02334         SetInternalMask(origMask);
02335         return resultPtr ? LoadInst(resultPtr) : NULL;
02336     }
02337 }

02480 /** When we gathering from or scattering to a varying atomic type, we need
02481     to add an appropraite offset to the final address for each lane right
02482     before we use it.  Given a varying pointer we're about to use and its
02483     type, this function determines whether these offsets are needed and
02484     returns an updated pointer that incorporates these offsets if needed.
02485  */
02486 llvm::Value *
02487 FunctionEmitContext::addVaryingOffsetsIfNeeded(llvm::Value *ptr, 
02488                                                const Type *ptrType) {
02489     // This should only be called for varying pointers
02490     const PointerType *pt = dynamic_cast<const PointerType *>(ptrType);
02491     assert(pt && pt->IsVaryingType());
02492 
02493     const Type *baseType = ptrType->GetBaseType();
02494     assert(dynamic_cast<const AtomicType *>(baseType) != NULL ||
02495            dynamic_cast<const EnumType *>(baseType) != NULL ||
02496            dynamic_cast<const PointerType *>(baseType));
02497     if (baseType->IsUniformType())
02498         return ptr;
02499     
02500     // Find the size of a uniform element of the varying type
02501     LLVM_TYPE_CONST llvm::Type *llvmBaseUniformType = 
02502         baseType->GetAsUniformType()->LLVMType(g->ctx);
02503     llvm::Value *unifSize = g->target.SizeOf(llvmBaseUniformType);
02504     unifSize = SmearUniform(unifSize);
02505 
02506     // Compute offset = <0, 1, .. > * unifSize
02507     llvm::Value *varyingOffsets = llvm::UndefValue::get(unifSize->getType());
02508     for (int i = 0; i < g->target.vectorWidth; ++i) {
02509         llvm::Value *iValue = (g->target.is32Bit || g->opt.force32BitAddressing) ?
02510             LLVMInt32(i) : LLVMInt64(i);
02511         varyingOffsets = InsertInst(varyingOffsets, iValue, i, "varying_delta");
02512     }
02513     llvm::Value *offset = BinaryOperator(llvm::Instruction::Mul, unifSize, 
02514                                          varyingOffsets);
02515     
02516     if (g->opt.force32BitAddressing == true && g->target.is32Bit == false)
02517         // On 64-bit targets where we're doing 32-bit addressing
02518         // calculations, we need to convert to an i64 vector before adding
02519         // to the pointer
02520         offset = SExtInst(offset, LLVMTypes::Int64VectorType, "offset_to_64");
02521 
02522     return BinaryOperator(llvm::Instruction::Add, ptr, offset);
02523 }
