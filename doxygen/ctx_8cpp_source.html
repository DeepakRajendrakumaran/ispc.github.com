00478 void
00479 FunctionEmitContext::EndIf() {
00480     CFInfo *ci = popCFState();
00481     // Make sure we match up with a Start{Uniform,Varying}If().
00482     Assert(ci->IsIf());
00483 
00484     // 'uniform' ifs don't change the mask so we only need to restore the
00485     // mask going into the if for 'varying' if statements
00486     if (ci->IsUniform() || bblock == NULL)
00487         return;
00488 
00489     // We can't just restore the mask as it was going into the 'if'
00490     // statement.  First we have to take into account any program
00491     // instances that have executed 'return' statements; the restored
00492     // mask must be off for those lanes.
00493     restoreMaskGivenReturns(ci->savedMask);
00494 
00495     // If the 'if' statement is inside a loop with a 'varying'
00496     // condition, we also need to account for any break or continue
00497     // statements that executed inside the 'if' statmeent; we also must
00498     // leave the lane masks for the program instances that ran those
00499     // off after we restore the mask after the 'if'.  The code below
00500     // ends up being optimized out in the case that there were no break
00501     // or continue statements (and breakLanesPtr and continueLanesPtr
00502     // have their initial 'all off' values), so we don't need to check
00503     // for that here.
00504     // 
00505     // There are three general cases to deal with here:
00506     // - Loops: both break and continue are allowed, and thus the corresponding
00507     //   lane mask pointers are non-NULL
00508     // - Foreach: only continueLanesPtr may be non-NULL
00509     // - Switch: only breakLanesPtr may be non-NULL
00510     if (continueLanesPtr != NULL || breakLanesPtr != NULL) {
00511         // We want to compute:
00512         // newMask = (oldMask & ~(breakLanes | continueLanes)),
00513         // treading breakLanes or continueLanes as "all off" if the
00514         // corresponding pointer is NULL.
00515         llvm::Value *bcLanes = NULL;
00516 
00517         if (continueLanesPtr != NULL)
00518             bcLanes = LoadInst(continueLanesPtr, "continue_lanes");
00519         else
00520             bcLanes = LLVMMaskAllOff;
00521 
00522         if (breakLanesPtr != NULL) {
00523             llvm::Value *breakLanes = LoadInst(breakLanesPtr, "break_lanes");
00524             bcLanes = BinaryOperator(llvm::Instruction::Or, bcLanes, 
00525                                      breakLanes, "|break_lanes");
00526         }
00527 
00528         llvm::Value *notBreakOrContinue = 
00529             NotOperator(bcLanes, "!(break|continue)_lanes");
00530         llvm::Value *oldMask = GetInternalMask();
00531         llvm::Value *newMask = 
00532             BinaryOperator(llvm::Instruction::And, oldMask, 
00533                            notBreakOrContinue, "new_mask");
00534         SetInternalMask(newMask);
00535     }
00536 }

00541                                bool uniformCF) {
00542     // Store the current values of various loop-related state so that we
00543     // can restore it when we exit this loop.

00546                                               continueTarget, breakLanesPtr,
00547                                               continueLanesPtr, oldMask, loopMask));

00550         // which lanes 'break' or 'continue'; all of the running ones go

00559         StoreInst(LLVMMaskAllOff, breakLanesPtr);

00561

00563     continueTarget = ct;
00564     loopMask = NULL; // this better be set by the loop!

00643 void
00644 FunctionEmitContext::Break(bool doCoherenceCheck) {
00645     Assert(controlFlowInfo.size() > 0);
00646     if (breakTarget == NULL) {
00647         Error(currentPos, "\"break\" statement is illegal outside of "
00648               "for/while/do loops and \"switch\" statements.");
00649         return;
00650     }
00651 
00652     if (bblock == NULL)
00653         return;
00654 
00655     if (inSwitchStatement() == true &&
00656         switchConditionWasUniform == true && 
00657         ifsInCFAllUniform(CFInfo::Switch)) {
00658         // We know that all program instances are executing the break, so
00659         // just jump to the block immediately after the switch.
00660         Assert(breakTarget != NULL);
00661         BranchInst(breakTarget);
00662         bblock = NULL;
00663         return;
00664     }
00665 
00666     // If all of the enclosing 'if' tests in the loop have uniform control
00667     // flow or if we can tell that the mask is all on, then we can just
00668     // jump to the break location.
00669     if (inSwitchStatement() == false && 
00670         (ifsInCFAllUniform(CFInfo::Loop) || 
00671          GetInternalMask() == LLVMMaskAllOn)) {
00672         BranchInst(breakTarget);
00673         if (ifsInCFAllUniform(CFInfo::Loop) && doCoherenceCheck)
00674             Warning(currentPos, "Coherent break statement not necessary in "
00675                     "fully uniform control flow.");
00676         // Set bblock to NULL since the jump has terminated the basic block
00677         bblock = NULL;
00678     }
00679     else {
00680         // Varying switch, uniform switch where the 'break' is under
00681         // varying control flow, or a loop with varying 'if's above the
00682         // break.  In these cases, we need to update the mask of the lanes
00683         // that have executed a 'break' statement: 
00684         // breakLanes = breakLanes | mask
00685         Assert(breakLanesPtr != NULL);
00686         llvm::Value *mask = GetInternalMask();
00687         llvm::Value *breakMask = LoadInst(breakLanesPtr,
00688                                           "break_mask");
00689         llvm::Value *newMask = BinaryOperator(llvm::Instruction::Or,
00690                                               mask, breakMask, "mask|break_mask");
00691         StoreInst(newMask, breakLanesPtr);
00692 
00693         // Set the current mask to be all off, just in case there are any
00694         // statements in the same scope after the 'break'.  Most of time
00695         // this will be optimized away since we'll likely end the scope of
00696         // an 'if' statement and restore the mask then.
00697         SetInternalMask(LLVMMaskAllOff);
00698 
00699         if (doCoherenceCheck) {
00700             if (continueTarget != NULL)
00701                 // If the user has indicated that this is a 'coherent'
00702                 // break statement, then check to see if the mask is all
00703                 // off.  If so, we have to conservatively jump to the
00704                 // continueTarget, not the breakTarget, since part of the
00705                 // reason the mask is all off may be due to 'continue'
00706                 // statements that executed in the current loop iteration.
00707                 jumpIfAllLoopLanesAreDone(continueTarget);
00708             else if (breakTarget != NULL)
00709                 // Similarly handle these for switch statements, where we
00710                 // only have a break target.
00711                 jumpIfAllLoopLanesAreDone(breakTarget);
00712         }
00713     }
00714 }

00717 void
00718 FunctionEmitContext::Continue(bool doCoherenceCheck) {
00719     if (!continueTarget) {
00720         Error(currentPos, "\"continue\" statement illegal outside of "
00721               "for/while/do/foreach loops.");
00722         return;
00723     }
00724 
00725     if (ifsInCFAllUniform(CFInfo::Loop) || GetInternalMask() == LLVMMaskAllOn) {
00726         // Similarly to 'break' statements, we can immediately jump to the
00727         // continue target if we're only in 'uniform' control flow within
00728         // loop or if we can tell that the mask is all on.
00729         AddInstrumentationPoint("continue: uniform CF, jumped");
00730         if (ifsInCFAllUniform(CFInfo::Loop) && doCoherenceCheck)
00731             Warning(currentPos, "Coherent continue statement not necessary in "
00732                     "fully uniform control flow.");
00733         BranchInst(continueTarget);
00734         bblock = NULL;
00735     }
00736     else {
00737         // Otherwise update the stored value of which lanes have 'continue'd.
00738         // continueLanes = continueLanes | mask
00739         Assert(continueLanesPtr);
00740         llvm::Value *mask = GetInternalMask();
00741         llvm::Value *continueMask = 
00742             LoadInst(continueLanesPtr, "continue_mask");
00743         llvm::Value *newMask = 
00744             BinaryOperator(llvm::Instruction::Or, mask, continueMask,
00745                            "mask|continueMask");
00746         StoreInst(newMask, continueLanesPtr);
00747 
00748         // And set the current mask to be all off in case there are any
00749         // statements in the same scope after the 'continue'
00750         SetInternalMask(LLVMMaskAllOff);
00751 
00752         if (doCoherenceCheck) 
00753             // If this is a 'coherent continue' statement, then emit the
00754             // code to see if all of the lanes are now off due to
00755             // breaks/continues and jump to the continue target if so.
00756             jumpIfAllLoopLanesAreDone(continueTarget);
00757     }

00761 /** This function checks to see if all of the 'if' statements (if any)
00762     between the current scope and the first enclosing loop/switch of given
00763     control flow type have 'uniform' tests.
00764  */
00765 bool
00766 FunctionEmitContext::ifsInCFAllUniform(int type) const {
00767     Assert(controlFlowInfo.size() > 0);
00768     // Go backwards through controlFlowInfo, since we add new nested scopes
00769     // to the back.  Stop once we come to the first enclosing control flow
00770     // structure of the desired type.
00771     int i = controlFlowInfo.size() - 1;
00772     while (i >= 0 && controlFlowInfo[i]->type != type) {

00783 void
00784 FunctionEmitContext::jumpIfAllLoopLanesAreDone(llvm::BasicBlock *target) {
00785     llvm::Value *allDone = NULL;
00786     Assert(continueLanesPtr != NULL);
00787     if (breakLanesPtr == NULL) {
00788         // In a foreach loop, break and return are illegal, and
00789         // breakLanesPtr is NULL.  In this case, the mask is guaranteed to
00790         // be all on at the start of each iteration, so we only need to
00791         // check if all lanes have continued..
00792         llvm::Value *continued = LoadInst(continueLanesPtr,
00793                                           "continue_lanes");
00794         allDone = All(continued);
00795     }
00796     else {
00797         // Check to see if (returned lanes | continued lanes | break lanes) is
00798         // equal to the value of mask at the start of the loop iteration.  If
00799         // so, everyone is done and we can jump to the given target
00800         llvm::Value *returned = LoadInst(returnedLanesPtr,
00801                                          "returned_lanes");
00802         llvm::Value *continued = LoadInst(continueLanesPtr,
00803                                           "continue_lanes");
00804         llvm::Value *breaked = LoadInst(breakLanesPtr, "break_lanes");
00805         llvm::Value *returnedOrContinued = BinaryOperator(llvm::Instruction::Or, 
00806                                                           returned, continued,
00807                                                           "returned|continued");
00808         llvm::Value *returnedOrContinuedOrBreaked = 
00809             BinaryOperator(llvm::Instruction::Or, returnedOrContinued,
00810                            breaked, "returned|continued");
00811 
00812         // Do we match the mask at loop entry?
00813         allDone = MasksAllEqual(returnedOrContinuedOrBreaked, loopMask);
00814     }
00815 
00816     llvm::BasicBlock *bAll = CreateBasicBlock("all_continued_or_breaked");
00817     llvm::BasicBlock *bNotAll = CreateBasicBlock("not_all_continued_or_breaked");
00818     BranchInst(bAll, bNotAll, allDone);
00819 
00820     // If so, have an extra basic block along the way to add
00821     // instrumentation, if the user asked for it.
00822     bblock = bAll;
00823     AddInstrumentationPoint("break/continue: all dynamically went");
00824     BranchInst(target);
00825 
00826     // And set the current basic block to a new one for future instructions
00827     // for the path where we weren't able to jump
00828     bblock = bNotAll;
00829     AddInstrumentationPoint("break/continue: not all went");
00830 }

00921 void
00922 FunctionEmitContext::EmitDefaultLabel(bool checkMask, SourcePos pos) {
00923     if (inSwitchStatement() == false) {
00924         Error(pos, "\"default\" label illegal outside of \"switch\" "
00925               "statement.");
00926         return;
00927     }
00928 
00929     // If there's a default label in the switch, a basic block for it
00930     // should have been provided in the previous call to SwitchInst().
00931     Assert(defaultBlock != NULL);
00932 
00933     if (bblock != NULL)
00934         // The previous case in the switch fell through, or we're in a
00935         // varying switch; terminate the current block with a jump to the
00936         // block for the code for the default label.
00937         BranchInst(defaultBlock);
00938     SetCurrentBasicBlock(defaultBlock);
00939 
00940     if (switchConditionWasUniform)
00941         // Nothing more to do for this case; return back to the caller,
00942         // which will then emit the code for the default case.
00943         return;
00944 
00945     // For a varying switch, we need to update the execution mask.
00946     //
00947     // First, compute the mask that corresponds to which program instances
00948     // should execute the "default" code; this corresponds to the set of
00949     // program instances that don't match any of the case statements.
00950     // Therefore, we generate code that compares the value of the switch
00951     // expression to the value associated with each of the "case"
00952     // statements such that the surviving lanes didn't match any of them.
00953     llvm::Value *matchesDefault = getMaskAtSwitchEntry();
00954     for (int i = 0; i < (int)caseBlocks->size(); ++i) {
00955         int value = (*caseBlocks)[i].first;
00956         llvm::Value *valueVec = (switchExpr->getType() == LLVMTypes::Int32VectorType) ?
00957             LLVMInt32Vector(value) : LLVMInt64Vector(value);
00958         // TODO: for AVX2 at least, the following generates better code
00959         // than doing ICMP_NE and skipping the NotOperator() below; file a
00960         // LLVM bug?
00961         llvm::Value *matchesCaseValue = 
00962             CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_EQ, switchExpr,
00963                     valueVec, "cmp_case_value");
00964         matchesCaseValue = I1VecToBoolVec(matchesCaseValue);
00965 
00966         llvm::Value *notMatchesCaseValue = NotOperator(matchesCaseValue);
00967         matchesDefault = BinaryOperator(llvm::Instruction::And, matchesDefault, 
00968                                         notMatchesCaseValue, "default&~case_match");
00969     }
00970 
00971     // The mask may have some lanes on, which corresponds to the previous
00972     // label falling through; compute the updated mask by ANDing with the
00973     // current mask.
00974     llvm::Value *oldMask = GetInternalMask();
00975     llvm::Value *newMask = BinaryOperator(llvm::Instruction::Or, oldMask, 
00976                                           matchesDefault, "old_mask|matches_default");
00977     SetInternalMask(newMask);
00978 
00979     if (checkMask)
00980         addSwitchMaskCheck(newMask);
00981 }

00984 void
00985 FunctionEmitContext::EmitCaseLabel(int value, bool checkMask, SourcePos pos) {
00986     if (inSwitchStatement() == false) {
00987         Error(pos, "\"case\" label illegal outside of \"switch\" statement.");
00988         return;
00989     }
00990 
00991     // Find the basic block for this case statement.
00992     llvm::BasicBlock *bbCase = NULL;
00993     Assert(caseBlocks != NULL);
00994     for (int i = 0; i < (int)caseBlocks->size(); ++i)
00995         if ((*caseBlocks)[i].first == value) {
00996             bbCase = (*caseBlocks)[i].second;
00997             break;
00998         }
00999     Assert(bbCase != NULL);
01000 
01001     if (bblock != NULL)
01002         // fall through from the previous case
01003         BranchInst(bbCase);
01004     SetCurrentBasicBlock(bbCase);
01005 
01006     if (switchConditionWasUniform)
01007         return;
01008 
01009     // update the mask: first, get a mask that indicates which program
01010     // instances have a value for the switch expression that matches this
01011     // case statement.
01012     llvm::Value *valueVec = (switchExpr->getType() == LLVMTypes::Int32VectorType) ?
01013         LLVMInt32Vector(value) : LLVMInt64Vector(value);
01014     llvm::Value *matchesCaseValue = 
01015         CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_EQ, switchExpr,
01016                 valueVec, "cmp_case_value");
01017     matchesCaseValue = I1VecToBoolVec(matchesCaseValue);
01018 
01019     // If a lane was off going into the switch, we don't care if has a
01020     // value in the switch expression that happens to match this case.
01021     llvm::Value *entryMask = getMaskAtSwitchEntry();
01022     matchesCaseValue = BinaryOperator(llvm::Instruction::And, entryMask,
01023                                       matchesCaseValue, "entry_mask&case_match");
01024 
01025     // Take the surviving lanes and turn on the mask for them.
01026     llvm::Value *oldMask = GetInternalMask();
01027     llvm::Value *newMask = BinaryOperator(llvm::Instruction::Or, oldMask, 
01028                                           matchesCaseValue, "mask|case_match");
01029     SetInternalMask(newMask);
01030 
01031     if (checkMask)
01032         addSwitchMaskCheck(newMask);
01033 }

01036 void
01037 FunctionEmitContext::SwitchInst(llvm::Value *expr, llvm::BasicBlock *bbDefault,
01038                 const std::vector<std::pair<int, llvm::BasicBlock *> > &bbCases,
01039                 const std::map<llvm::BasicBlock *, llvm::BasicBlock *> &bbNext) {
01040     // The calling code should have called StartSwitch() before calling
01041     // SwitchInst().
01042     Assert(controlFlowInfo.size() &&
01043            controlFlowInfo.back()->IsSwitch());
01044 
01045     switchExpr = expr;
01046     defaultBlock = bbDefault;
01047     caseBlocks = new std::vector<std::pair<int, llvm::BasicBlock *> >(bbCases);
01048     nextBlocks = new std::map<llvm::BasicBlock *, llvm::BasicBlock *>(bbNext);
01049     switchConditionWasUniform = 
01050         (llvm::isa<LLVM_TYPE_CONST llvm::VectorType>(expr->getType()) == false);
01051 
01052     if (switchConditionWasUniform == true) {
01053         // For a uniform switch condition, just wire things up to the LLVM
01054         // switch instruction.
01055         llvm::SwitchInst *s = llvm::SwitchInst::Create(expr, bbDefault, 
01056                                                        bbCases.size(), bblock);
01057         for (int i = 0; i < (int)bbCases.size(); ++i) {
01058             if (expr->getType() == LLVMTypes::Int32Type)
01059                 s->addCase(LLVMInt32(bbCases[i].first), bbCases[i].second);
01060             else {
01061                 Assert(expr->getType() == LLVMTypes::Int64Type);
01062                 s->addCase(LLVMInt64(bbCases[i].first), bbCases[i].second);
01063             }
01064         }
01065 
01066         AddDebugPos(s);
01067         // switch is a terminator
01068         bblock = NULL;
01069     }
01070     else {
01071         // For a varying switch, we first turn off all lanes of the mask
01072         SetInternalMask(LLVMMaskAllOff);
01073 
01074         if (nextBlocks->size() > 0) {
01075             // If there are any labels inside the switch, jump to the first
01076             // one; any code before the first label won't be executed by
01077             // anyone.
01078             std::map<llvm::BasicBlock *, llvm::BasicBlock *>::const_iterator iter;
01079             iter = nextBlocks->find(NULL);
01080             Assert(iter != nextBlocks->end());
01081             llvm::BasicBlock *bbFirst = iter->second;
01082             BranchInst(bbFirst);
01083             bblock = NULL;
01084         }
01085     }
01086 }

01143 void
01144 FunctionEmitContext::CurrentLanesReturned(Expr *expr, bool doCoherenceCheck) {
01145     const Type *returnType = function->GetReturnType();
01146     if (returnType == AtomicType::Void) {
01147         if (expr != NULL)
01148             Error(expr->pos, "Can't return non-void type \"%s\" from void function.",
01149                   expr->GetType()->GetString().c_str());
01150     }
01151     else {
01152         if (expr == NULL) {
01153             Error(funcStartPos, "Must provide return value for return "
01154                   "statement for non-void function.");
01155             return;
01156         }
01157         
01158         expr = TypeConvertExpr(expr, returnType, "return statement");
01159         if (expr != NULL) {
01160             llvm::Value *retVal = expr->GetValue(this);
01161             if (retVal != NULL) {
01162                 if (returnType->IsUniformType() ||
01163                     dynamic_cast<const ReferenceType *>(returnType) != NULL)
01164                     StoreInst(retVal, returnValuePtr);
01165                 else {
01166                     // Use a masked store to store the value of the expression
01167                     // in the return value memory; this preserves the return
01168                     // values from other lanes that may have executed return
01169                     // statements previously.
01170                     StoreInst(retVal, returnValuePtr, GetInternalMask(), 
01171                               PointerType::GetUniform(returnType));
01172                 }
01173             }
01174         }
01175     }
01176 
01177     if (VaryingCFDepth() == 0) {
01178         // If there is only uniform control flow between us and the
01179         // function entry, then it's guaranteed that all lanes are running,
01180         // so we can just emit a true return instruction
01181         AddInstrumentationPoint("return: uniform control flow");
01182         ReturnInst();
01183     }
01184     else {
01185         // Otherwise we update the returnedLanes value by ANDing it with
01186         // the current lane mask.
01187         llvm::Value *oldReturnedLanes = 
01188             LoadInst(returnedLanesPtr, "old_returned_lanes");
01189         llvm::Value *newReturnedLanes = 
01190             BinaryOperator(llvm::Instruction::Or, oldReturnedLanes, 
01191                            GetInternalMask(), "old_mask|returned_lanes");
01192         
01193         // For 'coherent' return statements, emit code to check if all
01194         // lanes have returned
01195         if (doCoherenceCheck) {
01196             // if newReturnedLanes == functionMaskValue, get out of here!
01197             llvm::Value *cmp = MasksAllEqual(functionMaskValue, 
01198                                              newReturnedLanes);
01199             llvm::BasicBlock *bDoReturn = CreateBasicBlock("do_return");
01200             llvm::BasicBlock *bNoReturn = CreateBasicBlock("no_return");
01201             BranchInst(bDoReturn, bNoReturn, cmp);
01202 
01203             bblock = bDoReturn;
01204             AddInstrumentationPoint("return: all lanes have returned");
01205             ReturnInst();
01206 
01207             bblock = bNoReturn;
01208         }
01209         // Otherwise update returnedLanesPtr and turn off all of the lanes
01210         // in the current mask so that any subsequent statements in the
01211         // same scope after the return have no effect
01212         StoreInst(newReturnedLanes, returnedLanesPtr);
01213         AddInstrumentationPoint("return: some but not all lanes have returned");

01837 /** Utility routine used by the GetElementPtrInst() methods; given a
01838     pointer to some type (either uniform or varying) and an index (also
01839     either uniform or varying), this returns the new pointer (varying if
01840     appropriate) given by offsetting the base pointer by the index times
01841     the size of the object that the pointer points to.
01842  */
01843 llvm::Value *
01844 FunctionEmitContext::applyVaryingGEP(llvm::Value *basePtr, llvm::Value *index, 
01845                                      const Type *ptrType) {
01846     // Find the scale factor for the index (i.e. the size of the object
01847     // that the pointer(s) point(s) to.
01848     const Type *scaleType = ptrType->GetBaseType();
01849     llvm::Value *scale = g->target.SizeOf(scaleType->LLVMType(g->ctx), bblock);
01850 
01851     bool indexIsVarying = 
01852         llvm::isa<LLVM_TYPE_CONST llvm::VectorType>(index->getType());
01853     llvm::Value *offset = NULL;
01854     if (indexIsVarying == false) {
01855         // Truncate or sign extend the index as appropriate to a 32 or
01856         // 64-bit type.
01857         if ((g->target.is32Bit || g->opt.force32BitAddressing) && 
01858             index->getType() == LLVMTypes::Int64Type)
01859             index = TruncInst(index, LLVMTypes::Int32Type, "trunc_index");
01860         else if ((!g->target.is32Bit && !g->opt.force32BitAddressing) &&
01861                  index->getType() == LLVMTypes::Int32Type)
01862             index = SExtInst(index, LLVMTypes::Int64Type, "sext_index");
01863 
01864         // do a scalar multiply to get the offset as index * scale and then
01865         // smear the result out to be a vector; this is more efficient than
01866         // first promoting both the scale and the index to vectors and then
01867         // multiplying.
01868         offset = BinaryOperator(llvm::Instruction::Mul, scale, index);
01869         offset = SmearUniform(offset, "offset_smear");
01870     }
01871     else {
01872         // Similarly, truncate or sign extend the index to be a 32 or 64
01873         // bit vector type
01874         if ((g->target.is32Bit || g->opt.force32BitAddressing) && 
01875             index->getType() == LLVMTypes::Int64VectorType)
01876             index = TruncInst(index, LLVMTypes::Int32VectorType, "trunc_index");
01877         else if ((!g->target.is32Bit && !g->opt.force32BitAddressing) &&
01878                  index->getType() == LLVMTypes::Int32VectorType)
01879             index = SExtInst(index, LLVMTypes::Int64VectorType, "sext_index");
01880 
01881         scale = SmearUniform(scale, "scale_smear");
01882 
01883         // offset = index * scale
01884         offset = BinaryOperator(llvm::Instruction::Mul, scale, index, "offset");
01885     }
01886 
01887     // For 64-bit targets, if we've been doing our offset calculations in
01888     // 32 bits, we still have to convert to a 64-bit value before we
01889     // actually add the offset to the pointer.
01890     if (g->target.is32Bit == false && g->opt.force32BitAddressing == true)
01891         offset = SExtInst(offset, LLVMTypes::Int64VectorType, "offset_to_64");
01892 
01893     // Smear out the pointer to be varying; either the base pointer or the
01894     // index must be varying for this method to be called.
01895     bool baseIsUniform = 
01896         (llvm::isa<LLVM_TYPE_CONST llvm::PointerType>(basePtr->getType()));
01897     Assert(baseIsUniform == false || indexIsVarying == true);
01898     llvm::Value *varyingPtr = baseIsUniform ? 
01899         SmearUniform(basePtr, "ptr_smear") : basePtr;
01900 
01901     // newPtr = ptr + offset
01902     return BinaryOperator(llvm::Instruction::Add, varyingPtr, offset, "new_ptr");
01903 }

02044             LLVMInt32(elementNum) : LLVMInt64(elementNum);

02133 llvm::Value *
02134 FunctionEmitContext::gather(llvm::Value *ptr, const Type *ptrType, 
02135                             llvm::Value *mask, const char *name) {
02136     // We should have a varying lvalue if we get here...
02137     Assert(ptrType->IsVaryingType() &&
02138            ptr->getType() == LLVMTypes::VoidPointerVectorType);
02139 
02140     const Type *returnType = ptrType->GetBaseType()->GetAsVaryingType();
02141     LLVM_TYPE_CONST llvm::Type *llvmReturnType = returnType->LLVMType(g->ctx);
02142 
02143     const CollectionType *collectionType = 
02144         dynamic_cast<const CollectionType *>(ptrType->GetBaseType());
02145     if (collectionType != NULL) {
02146         // For collections, recursively gather element wise to find the
02147         // result.
02148         llvm::Value *retValue = llvm::UndefValue::get(llvmReturnType);
02149         for (int i = 0; i < collectionType->GetElementCount(); ++i) {
02150             llvm::Value *eltPtr = AddElementOffset(ptr, i, ptrType);
02151             const Type *eltPtrType = 
02152                 PointerType::GetVarying(collectionType->GetElementType(i));
02153             eltPtr = addVaryingOffsetsIfNeeded(eltPtr, eltPtrType);
02154 
02155             // This in turn will be another gather
02156             llvm::Value *eltValues = LoadInst(eltPtr, mask, eltPtrType, name);
02157 
02158             retValue = InsertInst(retValue, eltValues, i, "set_value");
02159         }
02160         return retValue;
02161     }
02162     
02163     // Otherwise we should just have a basic scalar or pointer type and we
02164     // can go and do the actual gather
02165     AddInstrumentationPoint("gather");
02166 
02167     // Figure out which gather function to call based on the size of
02168     // the elements.
02169     const PointerType *pt = dynamic_cast<const PointerType *>(returnType);
02170     const char *funcName = NULL;
02171     if (pt != NULL)
02172         funcName = g->target.is32Bit ? "__pseudo_gather32_32" : 
02173             "__pseudo_gather64_64";
02174     else if (llvmReturnType == LLVMTypes::DoubleVectorType || 
02175              llvmReturnType == LLVMTypes::Int64VectorType)
02176         funcName = g->target.is32Bit ? "__pseudo_gather32_64" : 
02177             "__pseudo_gather64_64";
02178     else if (llvmReturnType == LLVMTypes::FloatVectorType || 
02179              llvmReturnType == LLVMTypes::Int32VectorType)
02180         funcName = g->target.is32Bit ? "__pseudo_gather32_32" : 
02181             "__pseudo_gather64_32";
02182     else if (llvmReturnType == LLVMTypes::Int16VectorType)
02183         funcName = g->target.is32Bit ? "__pseudo_gather32_16" : 
02184             "__pseudo_gather64_16";
02185     else {
02186         Assert(llvmReturnType == LLVMTypes::Int8VectorType);
02187         funcName = g->target.is32Bit ? "__pseudo_gather32_8" : 
02188             "__pseudo_gather64_8";
02189     }
02190 
02191     llvm::Function *gatherFunc = m->module->getFunction(funcName);
02192     Assert(gatherFunc != NULL);
02193 
02194     llvm::Value *call = CallInst(gatherFunc, NULL, ptr, mask, name);
02195 
02196     // Add metadata about the source file location so that the
02197     // optimization passes can print useful performance warnings if we

02277 /** Code to store the given varying value to the given location, only
02278     storing the elements that correspond to active program instances as
02279     given by the provided storeMask value.  Note that the lvalue is only a
02280     single pointer, not a varying lvalue of one pointer per program
02281     instance (that case is handled by scatters).
02282  */
02283 void
02284 FunctionEmitContext::maskedStore(llvm::Value *value, llvm::Value *ptr,
02285                                  const Type *ptrType, llvm::Value *mask) {
02286     if (value == NULL || ptr == NULL) {
02287         Assert(m->errorCount > 0);
02288         return;
02289     }
02290 
02291     Assert(dynamic_cast<const PointerType *>(ptrType) != NULL);
02292     Assert(ptrType->IsUniformType());
02293 
02294     const Type *valueType = ptrType->GetBaseType();
02295     const CollectionType *collectionType = 
02296         dynamic_cast<const CollectionType *>(valueType);
02297     if (collectionType != NULL) {
02298         // Assigning a structure / array / vector. Handle each element
02299         // individually with what turns into a recursive call to
02300         // makedStore()
02301         for (int i = 0; i < collectionType->GetElementCount(); ++i) {
02302             llvm::Value *eltValue = ExtractInst(value, i, "value_member");
02303             llvm::Value *eltPtr = 
02304                 AddElementOffset(ptr, i, ptrType, "struct_ptr_ptr");
02305             const Type *eltPtrType = 
02306                 PointerType::GetUniform(collectionType->GetElementType(i));
02307             StoreInst(eltValue, eltPtr, mask, eltPtrType);
02308         }
02309         return;
02310     }
02311 
02312     // We must have a regular atomic, enumerator, or pointer type at this
02313     // point.
02314     Assert(dynamic_cast<const AtomicType *>(valueType) != NULL ||
02315            dynamic_cast<const EnumType *>(valueType) != NULL ||
02316            dynamic_cast<const PointerType *>(valueType) != NULL);
02317     valueType = valueType->GetAsNonConstType();
02318 
02319     llvm::Function *maskedStoreFunc = NULL;
02320     // Figure out if we need a 8, 16, 32 or 64-bit masked store.
02321     if (dynamic_cast<const PointerType *>(valueType) != NULL) {
02322         if (g->target.is32Bit)
02323             maskedStoreFunc = m->module->getFunction("__pseudo_masked_store_32");
02324         else
02325             maskedStoreFunc = m->module->getFunction("__pseudo_masked_store_64");
02326     }
02327     else if (valueType == AtomicType::VaryingBool &&
02328              g->target.maskBitCount == 1) {
02329         llvm::Value *notMask = BinaryOperator(llvm::Instruction::Xor, mask,
02330                                               LLVMMaskAllOn, "~mask");
02331         llvm::Value *old = LoadInst(ptr);
02332         llvm::Value *maskedOld = BinaryOperator(llvm::Instruction::And, old,
02333                                                 notMask, "old&~mask");
02334         llvm::Value *maskedNew = BinaryOperator(llvm::Instruction::And, value,
02335                                                 mask, "new&mask");
02336         llvm::Value *final = BinaryOperator(llvm::Instruction::Or, maskedOld,
02337                                             maskedNew, "old_new_result");
02338         StoreInst(final, ptr);
02339         return;
02340     }
02341     else if (valueType == AtomicType::VaryingDouble || 
02342              valueType == AtomicType::VaryingInt64 ||
02343              valueType == AtomicType::VaryingUInt64) {
02344         maskedStoreFunc = m->module->getFunction("__pseudo_masked_store_64");
02345         ptr = BitCastInst(ptr, LLVMTypes::Int64VectorPointerType, 
02346                              "ptr_to_int64vecptr");
02347         value = BitCastInst(value, LLVMTypes::Int64VectorType, 
02348                              "value_to_int64");
02349     }
02350     else if (valueType == AtomicType::VaryingFloat ||
02351              valueType == AtomicType::VaryingBool ||
02352              valueType == AtomicType::VaryingInt32 ||
02353              valueType == AtomicType::VaryingUInt32 ||
02354              dynamic_cast<const EnumType *>(valueType) != NULL) {
02355         maskedStoreFunc = m->module->getFunction("__pseudo_masked_store_32");
02356         ptr = BitCastInst(ptr, LLVMTypes::Int32VectorPointerType, 
02357                              "ptr_to_int32vecptr");
02358         if (valueType == AtomicType::VaryingFloat)
02359             value = BitCastInst(value, LLVMTypes::Int32VectorType, 
02360                                  "value_to_int32");
02361     }
02362     else if (valueType == AtomicType::VaryingInt16 ||
02363              valueType == AtomicType::VaryingUInt16) {
02364         maskedStoreFunc = m->module->getFunction("__pseudo_masked_store_16");
02365         ptr = BitCastInst(ptr, LLVMTypes::Int16VectorPointerType, 
02366                              "ptr_to_int16vecptr");
02367     }
02368     else if (valueType == AtomicType::VaryingInt8 ||
02369              valueType == AtomicType::VaryingUInt8) {
02370         maskedStoreFunc = m->module->getFunction("__pseudo_masked_store_8");
02371         ptr = BitCastInst(ptr, LLVMTypes::Int8VectorPointerType, 
02372                              "ptr_to_int8vecptr");
02373     }
02374     Assert(maskedStoreFunc != NULL);
02375 
02376     std::vector<llvm::Value *> args;
02377     args.push_back(ptr);
02378     args.push_back(value);
02379     args.push_back(mask);
02380     CallInst(maskedStoreFunc, NULL, args);

02385 /** Scatter the given varying value to the locations given by the varying
02386     lvalue (which should be an array of pointers with size equal to the
02387     target's vector width.  We want to store each rvalue element at the
02388     corresponding pointer's location, *if* the mask for the corresponding
02389     program instance are on.  If they're off, don't do anything.  
02390 */
02391 void
02392 FunctionEmitContext::scatter(llvm::Value *value, llvm::Value *ptr, 
02393                              const Type *ptrType, llvm::Value *mask) {
02394     Assert(dynamic_cast<const PointerType *>(ptrType) != NULL);
02395     Assert(ptrType->IsVaryingType());
02396 
02397     const Type *valueType = ptrType->GetBaseType();
02398 
02399     // I think this should be impossible
02400     Assert(dynamic_cast<const ArrayType *>(valueType) == NULL);
02401 
02402     const CollectionType *collectionType = dynamic_cast<const CollectionType *>(valueType);
02403     if (collectionType != NULL) {
02404         // Scatter the collection elements individually
02405         for (int i = 0; i < collectionType->GetElementCount(); ++i) {
02406             llvm::Value *eltPtr = AddElementOffset(ptr, i, ptrType);
02407             llvm::Value *eltValue = ExtractInst(value, i);
02408             const Type *eltPtrType = 
02409                 PointerType::GetVarying(collectionType->GetElementType(i));
02410             eltPtr = addVaryingOffsetsIfNeeded(eltPtr, eltPtrType);
02411             scatter(eltValue, eltPtr, eltPtrType, mask);
02412         }
02413         return;
02414     }
02415 
02416     const PointerType *pt = dynamic_cast<const PointerType *>(valueType);
02417 
02418     // And everything should be a pointer or atomic from here on out...
02419     Assert(pt != NULL || 
02420            dynamic_cast<const AtomicType *>(valueType) != NULL);
02421 
02422     LLVM_TYPE_CONST llvm::Type *type = value->getType();
02423     const char *funcName = NULL;
02424     if (pt != NULL)
02425         funcName = g->target.is32Bit ? "__pseudo_scatter32_32" :
02426             "__pseudo_scatter64_64";
02427     else if (type == LLVMTypes::DoubleVectorType || 
02428              type == LLVMTypes::Int64VectorType) {
02429         funcName = g->target.is32Bit ? "__pseudo_scatter32_64" :
02430             "__pseudo_scatter64_64";
02431         value = BitCastInst(value, LLVMTypes::Int64VectorType, "value2int");
02432     }
02433     else if (type == LLVMTypes::FloatVectorType || 
02434              type == LLVMTypes::Int32VectorType) {
02435         funcName = g->target.is32Bit ? "__pseudo_scatter32_32" :
02436             "__pseudo_scatter64_32";
02437         value = BitCastInst(value, LLVMTypes::Int32VectorType, "value2int");
02438     }
02439     else if (type == LLVMTypes::Int16VectorType)
02440         funcName = g->target.is32Bit ? "__pseudo_scatter32_16" :
02441             "__pseudo_scatter64_16";
02442     else if (type == LLVMTypes::Int8VectorType)
02443         funcName = g->target.is32Bit ? "__pseudo_scatter32_8" :
02444             "__pseudo_scatter64_8";
02445 
02446     llvm::Function *scatterFunc = m->module->getFunction(funcName);
02447     Assert(scatterFunc != NULL);
02448     
02449     AddInstrumentationPoint("scatter");
02450 
02451     std::vector<llvm::Value *> args;
02452     args.push_back(ptr);
02453     args.push_back(value);
02454     args.push_back(mask);
02455     llvm::Value *inst = CallInst(scatterFunc, NULL, args);
02456     addGSMetadata(inst, currentPos);
02457 }

02607     pointer to a function to be called, figure out how many arguments the
02608     function has. */

02610 lCalleeArgCount(llvm::Value *callee, const FunctionType *funcType) {
02611     LLVM_TYPE_CONST llvm::FunctionType *ft = 
02612         llvm::dyn_cast<LLVM_TYPE_CONST llvm::FunctionType>(callee->getType());

02616             llvm::dyn_cast<LLVM_TYPE_CONST llvm::PointerType>(callee->getType());
02617         if (pt == NULL) {
02618             // varying--in this case, it must be the version of the

02620             return funcType->GetNumParameters() + 1;

02623     }

02630 llvm::Value *
02631 FunctionEmitContext::CallInst(llvm::Value *func, const FunctionType *funcType,
02632                               const std::vector<llvm::Value *> &args,
02633                               const char *name) {
02634     if (func == NULL) {
02635         Assert(m->errorCount > 0);
02636         return NULL;
02637     }
02638 
02639     std::vector<llvm::Value *> argVals = args;
02640     // Most of the time, the mask is passed as the last argument.  this
02641     // isn't the case for things like intrinsics, builtins, and extern "C"
02642     // functions from the application.  Add the mask if it's needed.
02643     unsigned int calleeArgCount = lCalleeArgCount(func, funcType);
02644     Assert(argVals.size() + 1 == calleeArgCount ||
02645            argVals.size() == calleeArgCount);
02646     if (argVals.size() + 1 == calleeArgCount)
02647         argVals.push_back(GetFullMask());
02648 
02649     if (llvm::isa<LLVM_TYPE_CONST llvm::VectorType>(func->getType()) == false) {
02650         // Regular 'uniform' function call--just one function or function
02651         // pointer, so just emit the IR directly.
02652 #if defined(LLVM_3_0) || defined(LLVM_3_0svn) || defined(LLVM_3_1svn)
02653         llvm::Instruction *ci = 
02654             llvm::CallInst::Create(func, argVals, name ? name : "", bblock);
02655 #else
02656         llvm::Instruction *ci = 
02657             llvm::CallInst::Create(func, argVals.begin(), argVals.end(), 
02658                                    name ? name : "", bblock);
02659 #endif
02660         AddDebugPos(ci);
02661         return ci;
02662     }
02663     else {
02664         // Emit the code for a varying function call, where we have an
02665         // vector of function pointers, one for each program instance.  The
02666         // basic strategy is that we go through the function pointers, and
02667         // for the executing program instances, for each unique function
02668         // pointer that's in the vector, call that function with a mask
02669         // equal to the set of active program instances that also have that
02670         // function pointer.  When all unique function pointers have been
02671         // called, we're done.
02672 
02673         llvm::BasicBlock *bbTest = CreateBasicBlock("varying_funcall_test");
02674         llvm::BasicBlock *bbCall = CreateBasicBlock("varying_funcall_call");
02675         llvm::BasicBlock *bbDone = CreateBasicBlock("varying_funcall_done");
02676 
02677         // Get the current mask value so we can restore it later
02678         llvm::Value *origMask = GetInternalMask();
02679 
02680         // First allocate memory to accumulate the various program
02681         // instances' return values...
02682         const Type *returnType = funcType->GetReturnType();
02683         LLVM_TYPE_CONST llvm::Type *llvmReturnType = returnType->LLVMType(g->ctx);
02684         llvm::Value *resultPtr = NULL;
02685         if (llvmReturnType->isVoidTy() == false)
02686             resultPtr = AllocaInst(llvmReturnType);
02687 
02688         // The memory pointed to by maskPointer tracks the set of program
02689         // instances for which we still need to call the function they are
02690         // pointing to.  It starts out initialized with the mask of
02691         // currently running program instances.
02692         llvm::Value *maskPtr = AllocaInst(LLVMTypes::MaskType);
02693         StoreInst(GetFullMask(), maskPtr);
02694 
02695         // And now we branch to the test to see if there's more work to be
02696         // done.
02697         BranchInst(bbTest);
02698 
02699         // bbTest: are any lanes of the mask still on?  If so, jump to
02700         // bbCall
02701         SetCurrentBasicBlock(bbTest); {
02702             llvm::Value *maskLoad = LoadInst(maskPtr);
02703             llvm::Value *any = Any(maskLoad);
02704             BranchInst(bbCall, bbDone, any);
02705         }
02706 
02707         // bbCall: this is the body of the loop that calls out to one of
02708         // the active function pointer values.
02709         SetCurrentBasicBlock(bbCall); {
02710             // Figure out the first lane that still needs its function
02711             // pointer to be called.
02712             llvm::Value *currentMask = LoadInst(maskPtr);
02713             llvm::Function *cttz = 
02714                 m->module->getFunction("__count_trailing_zeros_i32");
02715             Assert(cttz != NULL);
02716             llvm::Value *firstLane = CallInst(cttz, NULL, LaneMask(currentMask),
02717                                               "first_lane");
02718 
02719             // Get the pointer to the function we're going to call this
02720             // time through: ftpr = func[firstLane]
02721             llvm::Value *fptr = 
02722                 llvm::ExtractElementInst::Create(func, firstLane, 
02723                                                  "extract_fptr", bblock);
02724 
02725             // Smear it out into an array of function pointers
02726             llvm::Value *fptrSmear = SmearUniform(fptr, "func_ptr");
02727 
02728             // fpOverlap = (fpSmearAsVec == fpOrigAsVec).  This gives us a
02729             // mask for the set of program instances that have the same
02730             // value for their function pointer.
02731             llvm::Value *fpOverlap = 
02732                 CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_EQ,
02733                         fptrSmear, func);
02734             fpOverlap = I1VecToBoolVec(fpOverlap);
02735 
02736             // Figure out the mask to use when calling the function
02737             // pointer: we need to AND the current execution mask to handle
02738             // the case of any non-running program instances that happen to
02739             // have this function pointer value.
02740             // callMask = (currentMask & fpOverlap)
02741             llvm::Value *callMask = 
02742                 BinaryOperator(llvm::Instruction::And, currentMask, fpOverlap,
02743                                "call_mask");
02744 
02745             // Set the mask
02746             SetInternalMask(callMask);
02747 
02748             // bitcast the i32/64 function pointer to the actual function
02749             // pointer type (the variant that includes a mask).
02750             LLVM_TYPE_CONST llvm::Type *llvmFuncType =
02751                 funcType->LLVMFunctionType(g->ctx, true);
02752             LLVM_TYPE_CONST llvm::Type *llvmFPtrType = 
02753                 llvm::PointerType::get(llvmFuncType, 0);
02754             llvm::Value *fptrCast = IntToPtrInst(fptr, llvmFPtrType);
02755 
02756             // Call the function: callResult = call ftpr(args, args, call mask)
02757             llvm::Value *callResult = CallInst(fptrCast, funcType, args, name);
02758 
02759             // Now, do a masked store into the memory allocated to
02760             // accumulate the result using the call mask.
02761             if (callResult != NULL) {
02762                 Assert(resultPtr != NULL);
02763                 StoreInst(callResult, resultPtr, callMask, 
02764                           PointerType::GetUniform(returnType));
02765             }
02766             else
02767                 Assert(resultPtr == NULL);
02768 
02769             // Update the mask to turn off the program instances for which
02770             // we just called the function.
02771             // currentMask = currentMask & ~callmask
02772             llvm::Value *notCallMask = 
02773                 BinaryOperator(llvm::Instruction::Xor, callMask, LLVMMaskAllOn,
02774                                "~callMask");
02775             currentMask = BinaryOperator(llvm::Instruction::And, currentMask, 
02776                                          notCallMask, "currentMask&~callMask");
02777             StoreInst(currentMask, maskPtr);
02778 
02779             // And go back to the test to see if we need to do another
02780             // call.
02781             BranchInst(bbTest);
02782         }
02783 
02784         // bbDone: We're all done; clean up and return the result we've
02785         // accumulated in the result memory.
02786         SetCurrentBasicBlock(bbDone);
02787         SetInternalMask(origMask);
02788         return resultPtr ? LoadInst(resultPtr) : NULL;
02789     }
02790 }

02837 llvm::Value *
02838 FunctionEmitContext::LaunchInst(llvm::Value *callee, 
02839                                 std::vector<llvm::Value *> &argVals,
02840                                 llvm::Value *launchCount) {
02841     if (callee == NULL) {
02842         Assert(m->errorCount > 0);
02843         return NULL;
02844     }
02845 
02846     launchedTasks = true;
02847 
02848     Assert(llvm::isa<llvm::Function>(callee));
02849     LLVM_TYPE_CONST llvm::Type *argType = 
02850         (llvm::dyn_cast<llvm::Function>(callee))->arg_begin()->getType();
02851     Assert(llvm::PointerType::classof(argType));
02852     LLVM_TYPE_CONST llvm::PointerType *pt = 
02853         llvm::dyn_cast<LLVM_TYPE_CONST llvm::PointerType>(argType);
02854     Assert(llvm::StructType::classof(pt->getElementType()));
02855     LLVM_TYPE_CONST llvm::StructType *argStructType = 
02856         static_cast<LLVM_TYPE_CONST llvm::StructType *>(pt->getElementType());
02857     Assert(argStructType->getNumElements() == argVals.size() + 1);
02858 
02859     llvm::Function *falloc = m->module->getFunction("ISPCAlloc");
02860     Assert(falloc != NULL);
02861     llvm::Value *structSize = g->target.SizeOf(argStructType, bblock);
02862     if (structSize->getType() != LLVMTypes::Int64Type)
02863         // ISPCAlloc expects the size as an uint64_t, but on 32-bit
02864         // targets, SizeOf returns a 32-bit value
02865         structSize = ZExtInst(structSize, LLVMTypes::Int64Type,
02866                               "struct_size_to_64");
02867     int align = 4 * RoundUpPow2(g->target.nativeVectorWidth);
02868 
02869     std::vector<llvm::Value *> allocArgs;
02870     allocArgs.push_back(launchGroupHandlePtr);
02871     allocArgs.push_back(structSize);
02872     allocArgs.push_back(LLVMInt32(align));
02873     llvm::Value *voidmem = CallInst(falloc, NULL, allocArgs, "args_ptr");
02874     llvm::Value *argmem = BitCastInst(voidmem, pt);
02875 
02876     // Copy the values of the parameters into the appropriate place in
02877     // the argument block
02878     for (unsigned int i = 0; i < argVals.size(); ++i) {
02879         llvm::Value *ptr = AddElementOffset(argmem, i, NULL, "funarg");
02880         // don't need to do masked store here, I think
02881         StoreInst(argVals[i], ptr);
02882     }
02883 
02884     // copy in the mask
02885     llvm::Value *mask = GetFullMask();
02886     llvm::Value *ptr = AddElementOffset(argmem, argVals.size(), NULL,
02887                                         "funarg_mask");
02888     StoreInst(mask, ptr);
02889 
02890     // And emit the call to the user-supplied task launch function, passing
02891     // a pointer to the task function being called and a pointer to the
02892     // argument block we just filled in
02893     llvm::Value *fptr = BitCastInst(callee, LLVMTypes::VoidPointerType);
02894     llvm::Function *flaunch = m->module->getFunction("ISPCLaunch");
02895     Assert(flaunch != NULL);
02896     std::vector<llvm::Value *> args;
02897     args.push_back(launchGroupHandlePtr);
02898     args.push_back(fptr);
02899     args.push_back(voidmem);
02900     args.push_back(launchCount);
02901     return CallInst(flaunch, NULL, args, "");
02902 }

02905 void
02906 FunctionEmitContext::SyncInst() {
02907     llvm::Value *launchGroupHandle = LoadInst(launchGroupHandlePtr);
02908     llvm::Value *nullPtrValue = 
02909         llvm::Constant::getNullValue(LLVMTypes::VoidPointerType);
02910     llvm::Value *nonNull = CmpInst(llvm::Instruction::ICmp,
02911                                    llvm::CmpInst::ICMP_NE,
02912                                    launchGroupHandle, nullPtrValue);
02913     llvm::BasicBlock *bSync = CreateBasicBlock("call_sync");
02914     llvm::BasicBlock *bPostSync = CreateBasicBlock("post_sync");
02915     BranchInst(bSync, bPostSync, nonNull);
02916 
02917     SetCurrentBasicBlock(bSync);
02918     llvm::Function *fsync = m->module->getFunction("ISPCSync");
02919     if (fsync == NULL)
02920         FATAL("Couldn't find ISPCSync declaration?!");
02921     CallInst(fsync, NULL, launchGroupHandle, "");
02922 
02923     // zero out the handle so that if ISPCLaunch is called again in this
02924     // function, it knows it's starting out from scratch
02925     StoreInst(nullPtrValue, launchGroupHandlePtr);
02926 
02927     BranchInst(bPostSync);
02928 
02929     SetCurrentBasicBlock(bPostSync);
02930 }

02940 FunctionEmitContext::addVaryingOffsetsIfNeeded(llvm::Value *ptr,

02948         dynamic_cast<const EnumType *>(baseType) == NULL &&
