00480 void
00481 FunctionEmitContext::EndIf() {
00482     CFInfo *ci = popCFState();
00483     // Make sure we match up with a Start{Uniform,Varying}If().
00484     Assert(ci->IsIf());
00485 
00486     // 'uniform' ifs don't change the mask so we only need to restore the
00487     // mask going into the if for 'varying' if statements
00488     if (ci->IsUniform() || bblock == NULL)
00489         return;
00490 
00491     // We can't just restore the mask as it was going into the 'if'
00492     // statement.  First we have to take into account any program
00493     // instances that have executed 'return' statements; the restored
00494     // mask must be off for those lanes.
00495     restoreMaskGivenReturns(ci->savedMask);
00496 
00497     // If the 'if' statement is inside a loop with a 'varying'
00498     // condition, we also need to account for any break or continue
00499     // statements that executed inside the 'if' statmeent; we also must
00500     // leave the lane masks for the program instances that ran those
00501     // off after we restore the mask after the 'if'.  The code below
00502     // ends up being optimized out in the case that there were no break
00503     // or continue statements (and breakLanesPtr and continueLanesPtr
00504     // have their initial 'all off' values), so we don't need to check
00505     // for that here.
00506     // 
00507     // There are three general cases to deal with here:
00508     // - Loops: both break and continue are allowed, and thus the corresponding
00509     //   lane mask pointers are non-NULL
00510     // - Foreach: only continueLanesPtr may be non-NULL
00511     // - Switch: only breakLanesPtr may be non-NULL
00512     if (continueLanesPtr != NULL || breakLanesPtr != NULL) {
00513         // We want to compute:
00514         // newMask = (oldMask & ~(breakLanes | continueLanes)),
00515         // treading breakLanes or continueLanes as "all off" if the
00516         // corresponding pointer is NULL.
00517         llvm::Value *bcLanes = NULL;
00518 
00519         if (continueLanesPtr != NULL)
00520             bcLanes = LoadInst(continueLanesPtr, "continue_lanes");
00521         else
00522             bcLanes = LLVMMaskAllOff;
00523 
00524         if (breakLanesPtr != NULL) {
00525             llvm::Value *breakLanes = LoadInst(breakLanesPtr, "break_lanes");
00526             bcLanes = BinaryOperator(llvm::Instruction::Or, bcLanes, 
00527                                      breakLanes, "|break_lanes");
00528         }
00529 
00530         llvm::Value *notBreakOrContinue = 
00531             NotOperator(bcLanes, "!(break|continue)_lanes");
00532         llvm::Value *oldMask = GetInternalMask();
00533         llvm::Value *newMask = 
00534             BinaryOperator(llvm::Instruction::And, oldMask, 
00535                            notBreakOrContinue, "new_mask");
00536         SetInternalMask(newMask);
00537     }
00538 }

00645 void
00646 FunctionEmitContext::Break(bool doCoherenceCheck) {
00647     if (breakTarget == NULL) {
00648         Error(currentPos, "\"break\" statement is illegal outside of "
00649               "for/while/do loops and \"switch\" statements.");
00650         return;
00651     }
00652     Assert(controlFlowInfo.size() > 0);
00653 
00654     if (bblock == NULL)
00655         return;
00656 
00657     if (inSwitchStatement() == true &&
00658         switchConditionWasUniform == true && 
00659         ifsInCFAllUniform(CFInfo::Switch)) {
00660         // We know that all program instances are executing the break, so
00661         // just jump to the block immediately after the switch.
00662         Assert(breakTarget != NULL);
00663         BranchInst(breakTarget);
00664         bblock = NULL;
00665         return;
00666     }
00667 
00668     // If all of the enclosing 'if' tests in the loop have uniform control
00669     // flow or if we can tell that the mask is all on, then we can just
00670     // jump to the break location.
00671     if (inSwitchStatement() == false && 
00672         (ifsInCFAllUniform(CFInfo::Loop) || 
00673          GetInternalMask() == LLVMMaskAllOn)) {
00674         BranchInst(breakTarget);
00675         if (ifsInCFAllUniform(CFInfo::Loop) && doCoherenceCheck)
00676             Warning(currentPos, "Coherent break statement not necessary in "
00677                     "fully uniform control flow.");
00678         // Set bblock to NULL since the jump has terminated the basic block
00679         bblock = NULL;
00680     }
00681     else {
00682         // Varying switch, uniform switch where the 'break' is under
00683         // varying control flow, or a loop with varying 'if's above the
00684         // break.  In these cases, we need to update the mask of the lanes
00685         // that have executed a 'break' statement: 
00686         // breakLanes = breakLanes | mask
00687         Assert(breakLanesPtr != NULL);
00688         llvm::Value *mask = GetInternalMask();
00689         llvm::Value *breakMask = LoadInst(breakLanesPtr,
00690                                           "break_mask");
00691         llvm::Value *newMask = BinaryOperator(llvm::Instruction::Or,
00692                                               mask, breakMask, "mask|break_mask");
00693         StoreInst(newMask, breakLanesPtr);
00694 
00695         // Set the current mask to be all off, just in case there are any
00696         // statements in the same scope after the 'break'.  Most of time
00697         // this will be optimized away since we'll likely end the scope of
00698         // an 'if' statement and restore the mask then.
00699         SetInternalMask(LLVMMaskAllOff);
00700 
00701         if (doCoherenceCheck) {
00702             if (continueTarget != NULL)
00703                 // If the user has indicated that this is a 'coherent'
00704                 // break statement, then check to see if the mask is all
00705                 // off.  If so, we have to conservatively jump to the
00706                 // continueTarget, not the breakTarget, since part of the
00707                 // reason the mask is all off may be due to 'continue'
00708                 // statements that executed in the current loop iteration.
00709                 jumpIfAllLoopLanesAreDone(continueTarget);
00710             else if (breakTarget != NULL)
00711                 // Similarly handle these for switch statements, where we
00712                 // only have a break target.
00713                 jumpIfAllLoopLanesAreDone(breakTarget);
00714         }
00715     }
00716 }

00719 void
00720 FunctionEmitContext::Continue(bool doCoherenceCheck) {
00721     if (!continueTarget) {
00722         Error(currentPos, "\"continue\" statement illegal outside of "
00723               "for/while/do/foreach loops.");
00724         return;
00725     }
00726     Assert(controlFlowInfo.size() > 0);
00727 
00728     if (ifsInCFAllUniform(CFInfo::Loop) || GetInternalMask() == LLVMMaskAllOn) {
00729         // Similarly to 'break' statements, we can immediately jump to the
00730         // continue target if we're only in 'uniform' control flow within
00731         // loop or if we can tell that the mask is all on.
00732         AddInstrumentationPoint("continue: uniform CF, jumped");
00733         if (ifsInCFAllUniform(CFInfo::Loop) && doCoherenceCheck)
00734             Warning(currentPos, "Coherent continue statement not necessary in "
00735                     "fully uniform control flow.");
00736         BranchInst(continueTarget);
00737         bblock = NULL;
00738     }
00739     else {
00740         // Otherwise update the stored value of which lanes have 'continue'd.
00741         // continueLanes = continueLanes | mask
00742         Assert(continueLanesPtr);
00743         llvm::Value *mask = GetInternalMask();
00744         llvm::Value *continueMask = 
00745             LoadInst(continueLanesPtr, "continue_mask");
00746         llvm::Value *newMask = 
00747             BinaryOperator(llvm::Instruction::Or, mask, continueMask,
00748                            "mask|continueMask");
00749         StoreInst(newMask, continueLanesPtr);
00750 
00751         // And set the current mask to be all off in case there are any
00752         // statements in the same scope after the 'continue'
00753         SetInternalMask(LLVMMaskAllOff);
00754 
00755         if (doCoherenceCheck) 
00756             // If this is a 'coherent continue' statement, then emit the
00757             // code to see if all of the lanes are now off due to
00758             // breaks/continues and jump to the continue target if so.
00759             jumpIfAllLoopLanesAreDone(continueTarget);
00760     }
00761 }

00764 /** This function checks to see if all of the 'if' statements (if any)
00765     between the current scope and the first enclosing loop/switch of given
00766     control flow type have 'uniform' tests.
00767  */
00768 bool
00769 FunctionEmitContext::ifsInCFAllUniform(int type) const {
00770     Assert(controlFlowInfo.size() > 0);
00771     // Go backwards through controlFlowInfo, since we add new nested scopes
00772     // to the back.  Stop once we come to the first enclosing control flow
00773     // structure of the desired type.
00774     int i = controlFlowInfo.size() - 1;
00775     while (i >= 0 && controlFlowInfo[i]->type != type) {
00776         if (controlFlowInfo[i]->isUniform == false)
00777             // Found a scope due to an 'if' statement with a varying test

00786 void
00787 FunctionEmitContext::jumpIfAllLoopLanesAreDone(llvm::BasicBlock *target) {
00788     llvm::Value *allDone = NULL;
00789     Assert(continueLanesPtr != NULL);
00790     if (breakLanesPtr == NULL) {
00791         // In a foreach loop, break and return are illegal, and
00792         // breakLanesPtr is NULL.  In this case, the mask is guaranteed to
00793         // be all on at the start of each iteration, so we only need to
00794         // check if all lanes have continued..
00795         llvm::Value *continued = LoadInst(continueLanesPtr,
00796                                           "continue_lanes");
00797         allDone = All(continued);
00798     }
00799     else {
00800         // Check to see if (returned lanes | continued lanes | break lanes) is
00801         // equal to the value of mask at the start of the loop iteration.  If
00802         // so, everyone is done and we can jump to the given target
00803         llvm::Value *returned = LoadInst(returnedLanesPtr,
00804                                          "returned_lanes");
00805         llvm::Value *continued = LoadInst(continueLanesPtr,
00806                                           "continue_lanes");
00807         llvm::Value *breaked = LoadInst(breakLanesPtr, "break_lanes");
00808         llvm::Value *returnedOrContinued = BinaryOperator(llvm::Instruction::Or, 
00809                                                           returned, continued,
00810                                                           "returned|continued");
00811         llvm::Value *returnedOrContinuedOrBreaked = 
00812             BinaryOperator(llvm::Instruction::Or, returnedOrContinued,
00813                            breaked, "returned|continued");
00814 
00815         // Do we match the mask at loop entry?
00816         allDone = MasksAllEqual(returnedOrContinuedOrBreaked, loopMask);
00817     }
00818 
00819     llvm::BasicBlock *bAll = CreateBasicBlock("all_continued_or_breaked");
00820     llvm::BasicBlock *bNotAll = CreateBasicBlock("not_all_continued_or_breaked");
00821     BranchInst(bAll, bNotAll, allDone);
00822 
00823     // If so, have an extra basic block along the way to add
00824     // instrumentation, if the user asked for it.
00825     bblock = bAll;
00826     AddInstrumentationPoint("break/continue: all dynamically went");
00827     BranchInst(target);
00828 
00829     // And set the current basic block to a new one for future instructions
00830     // for the path where we weren't able to jump
00831     bblock = bNotAll;
00832     AddInstrumentationPoint("break/continue: not all went");
00833 }

00924 void
00925 FunctionEmitContext::EmitDefaultLabel(bool checkMask, SourcePos pos) {
00926     if (inSwitchStatement() == false) {
00927         Error(pos, "\"default\" label illegal outside of \"switch\" "
00928               "statement.");
00929         return;
00930     }
00931 
00932     // If there's a default label in the switch, a basic block for it
00933     // should have been provided in the previous call to SwitchInst().
00934     Assert(defaultBlock != NULL);
00935 
00936     if (bblock != NULL)
00937         // The previous case in the switch fell through, or we're in a
00938         // varying switch; terminate the current block with a jump to the
00939         // block for the code for the default label.
00940         BranchInst(defaultBlock);
00941     SetCurrentBasicBlock(defaultBlock);
00942 
00943     if (switchConditionWasUniform)
00944         // Nothing more to do for this case; return back to the caller,
00945         // which will then emit the code for the default case.
00946         return;
00947 
00948     // For a varying switch, we need to update the execution mask.
00949     //
00950     // First, compute the mask that corresponds to which program instances
00951     // should execute the "default" code; this corresponds to the set of
00952     // program instances that don't match any of the case statements.
00953     // Therefore, we generate code that compares the value of the switch
00954     // expression to the value associated with each of the "case"
00955     // statements such that the surviving lanes didn't match any of them.
00956     llvm::Value *matchesDefault = getMaskAtSwitchEntry();
00957     for (int i = 0; i < (int)caseBlocks->size(); ++i) {
00958         int value = (*caseBlocks)[i].first;
00959         llvm::Value *valueVec = (switchExpr->getType() == LLVMTypes::Int32VectorType) ?
00960             LLVMInt32Vector(value) : LLVMInt64Vector(value);
00961         // TODO: for AVX2 at least, the following generates better code
00962         // than doing ICMP_NE and skipping the NotOperator() below; file a
00963         // LLVM bug?
00964         llvm::Value *matchesCaseValue = 
00965             CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_EQ, switchExpr,
00966                     valueVec, "cmp_case_value");
00967         matchesCaseValue = I1VecToBoolVec(matchesCaseValue);
00968 
00969         llvm::Value *notMatchesCaseValue = NotOperator(matchesCaseValue);
00970         matchesDefault = BinaryOperator(llvm::Instruction::And, matchesDefault, 
00971                                         notMatchesCaseValue, "default&~case_match");
00972     }
00973 
00974     // The mask may have some lanes on, which corresponds to the previous
00975     // label falling through; compute the updated mask by ANDing with the
00976     // current mask.
00977     llvm::Value *oldMask = GetInternalMask();
00978     llvm::Value *newMask = BinaryOperator(llvm::Instruction::Or, oldMask, 
00979                                           matchesDefault, "old_mask|matches_default");
00980     SetInternalMask(newMask);
00981 
00982     if (checkMask)
00983         addSwitchMaskCheck(newMask);
00984 }

00987 void
00988 FunctionEmitContext::EmitCaseLabel(int value, bool checkMask, SourcePos pos) {
00989     if (inSwitchStatement() == false) {
00990         Error(pos, "\"case\" label illegal outside of \"switch\" statement.");
00991         return;
00992     }
00993 
00994     // Find the basic block for this case statement.
00995     llvm::BasicBlock *bbCase = NULL;
00996     Assert(caseBlocks != NULL);
00997     for (int i = 0; i < (int)caseBlocks->size(); ++i)
00998         if ((*caseBlocks)[i].first == value) {
00999             bbCase = (*caseBlocks)[i].second;
01000             break;
01001         }
01002     Assert(bbCase != NULL);
01003 
01004     if (bblock != NULL)
01005         // fall through from the previous case
01006         BranchInst(bbCase);
01007     SetCurrentBasicBlock(bbCase);
01008 
01009     if (switchConditionWasUniform)
01010         return;
01011 
01012     // update the mask: first, get a mask that indicates which program
01013     // instances have a value for the switch expression that matches this
01014     // case statement.
01015     llvm::Value *valueVec = (switchExpr->getType() == LLVMTypes::Int32VectorType) ?
01016         LLVMInt32Vector(value) : LLVMInt64Vector(value);
01017     llvm::Value *matchesCaseValue = 
01018         CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_EQ, switchExpr,
01019                 valueVec, "cmp_case_value");
01020     matchesCaseValue = I1VecToBoolVec(matchesCaseValue);
01021 
01022     // If a lane was off going into the switch, we don't care if has a
01023     // value in the switch expression that happens to match this case.
01024     llvm::Value *entryMask = getMaskAtSwitchEntry();
01025     matchesCaseValue = BinaryOperator(llvm::Instruction::And, entryMask,
01026                                       matchesCaseValue, "entry_mask&case_match");
01027 
01028     // Take the surviving lanes and turn on the mask for them.
01029     llvm::Value *oldMask = GetInternalMask();
01030     llvm::Value *newMask = BinaryOperator(llvm::Instruction::Or, oldMask, 
01031                                           matchesCaseValue, "mask|case_match");
01032     SetInternalMask(newMask);
01033 
01034     if (checkMask)
01035         addSwitchMaskCheck(newMask);
01036 }

01039 void
01040 FunctionEmitContext::SwitchInst(llvm::Value *expr, llvm::BasicBlock *bbDefault,
01041                 const std::vector<std::pair<int, llvm::BasicBlock *> > &bbCases,
01042                 const std::map<llvm::BasicBlock *, llvm::BasicBlock *> &bbNext) {
01043     // The calling code should have called StartSwitch() before calling
01044     // SwitchInst().
01045     Assert(controlFlowInfo.size() &&
01046            controlFlowInfo.back()->IsSwitch());
01047 
01048     switchExpr = expr;
01049     defaultBlock = bbDefault;
01050     caseBlocks = new std::vector<std::pair<int, llvm::BasicBlock *> >(bbCases);
01051     nextBlocks = new std::map<llvm::BasicBlock *, llvm::BasicBlock *>(bbNext);
01052     switchConditionWasUniform = 
01053         (llvm::isa<llvm::VectorType>(expr->getType()) == false);
01054 
01055     if (switchConditionWasUniform == true) {
01056         // For a uniform switch condition, just wire things up to the LLVM
01057         // switch instruction.
01058         llvm::SwitchInst *s = llvm::SwitchInst::Create(expr, bbDefault, 
01059                                                        bbCases.size(), bblock);
01060         for (int i = 0; i < (int)bbCases.size(); ++i) {
01061             if (expr->getType() == LLVMTypes::Int32Type)
01062                 s->addCase(LLVMInt32(bbCases[i].first), bbCases[i].second);
01063             else {
01064                 Assert(expr->getType() == LLVMTypes::Int64Type);
01065                 s->addCase(LLVMInt64(bbCases[i].first), bbCases[i].second);
01066             }
01067         }
01068 
01069         AddDebugPos(s);
01070         // switch is a terminator
01071         bblock = NULL;
01072     }
01073     else {
01074         // For a varying switch, we first turn off all lanes of the mask
01075         SetInternalMask(LLVMMaskAllOff);
01076 
01077         if (nextBlocks->size() > 0) {
01078             // If there are any labels inside the switch, jump to the first
01079             // one; any code before the first label won't be executed by
01080             // anyone.
01081             std::map<llvm::BasicBlock *, llvm::BasicBlock *>::const_iterator iter;
01082             iter = nextBlocks->find(NULL);
01083             Assert(iter != nextBlocks->end());
01084             llvm::BasicBlock *bbFirst = iter->second;
01085             BranchInst(bbFirst);
01086             bblock = NULL;
01087         }
01088     }
01089 }

01159 void
01160 FunctionEmitContext::CurrentLanesReturned(Expr *expr, bool doCoherenceCheck) {
01161     const Type *returnType = function->GetReturnType();
01162     if (Type::Equal(returnType, AtomicType::Void)) {
01163         if (expr != NULL)
01164             Error(expr->pos, "Can't return non-void type \"%s\" from void function.",
01165                   expr->GetType()->GetString().c_str());
01166     }
01167     else {
01168         if (expr == NULL) {
01169             Error(funcStartPos, "Must provide return value for return "
01170                   "statement for non-void function.");
01171             return;
01172         }
01173         
01174         expr = TypeConvertExpr(expr, returnType, "return statement");
01175         if (expr != NULL) {
01176             llvm::Value *retVal = expr->GetValue(this);
01177             if (retVal != NULL) {
01178                 if (returnType->IsUniformType() ||
01179                     dynamic_cast<const ReferenceType *>(returnType) != NULL)
01180                     StoreInst(retVal, returnValuePtr);
01181                 else {
01182                     // Use a masked store to store the value of the expression
01183                     // in the return value memory; this preserves the return
01184                     // values from other lanes that may have executed return
01185                     // statements previously.
01186                     StoreInst(retVal, returnValuePtr, GetInternalMask(), 
01187                               returnType, PointerType::GetUniform(returnType));
01188                 }
01189             }
01190         }
01191     }
01192 
01193     if (VaryingCFDepth() == 0) {
01194         // If there is only uniform control flow between us and the
01195         // function entry, then it's guaranteed that all lanes are running,
01196         // so we can just emit a true return instruction
01197         AddInstrumentationPoint("return: uniform control flow");
01198         ReturnInst();
01199     }
01200     else {
01201         // Otherwise we update the returnedLanes value by ANDing it with
01202         // the current lane mask.
01203         llvm::Value *oldReturnedLanes = 
01204             LoadInst(returnedLanesPtr, "old_returned_lanes");
01205         llvm::Value *newReturnedLanes = 
01206             BinaryOperator(llvm::Instruction::Or, oldReturnedLanes, 
01207                            GetInternalMask(), "old_mask|returned_lanes");
01208         
01209         // For 'coherent' return statements, emit code to check if all
01210         // lanes have returned
01211         if (doCoherenceCheck) {
01212             // if newReturnedLanes == functionMaskValue, get out of here!
01213             llvm::Value *cmp = MasksAllEqual(functionMaskValue, 
01214                                              newReturnedLanes);
01215             llvm::BasicBlock *bDoReturn = CreateBasicBlock("do_return");
01216             llvm::BasicBlock *bNoReturn = CreateBasicBlock("no_return");
01217             BranchInst(bDoReturn, bNoReturn, cmp);
01218 
01219             bblock = bDoReturn;

01865 /** Utility routine used by the GetElementPtrInst() methods; given a
01866     pointer to some type (either uniform or varying) and an index (also
01867     either uniform or varying), this returns the new pointer (varying if
01868     appropriate) given by offsetting the base pointer by the index times
01869     the size of the object that the pointer points to.
01870  */
01871 llvm::Value *
01872 FunctionEmitContext::applyVaryingGEP(llvm::Value *basePtr, llvm::Value *index, 
01873                                      const Type *ptrType) {
01874     // Find the scale factor for the index (i.e. the size of the object
01875     // that the pointer(s) point(s) to.
01876     const Type *scaleType = ptrType->GetBaseType();
01877     llvm::Value *scale = g->target.SizeOf(scaleType->LLVMType(g->ctx), bblock);
01878 
01879     bool indexIsVarying = 
01880         llvm::isa<llvm::VectorType>(index->getType());
01881     llvm::Value *offset = NULL;
01882     if (indexIsVarying == false) {
01883         // Truncate or sign extend the index as appropriate to a 32 or
01884         // 64-bit type.
01885         if ((g->target.is32Bit || g->opt.force32BitAddressing) && 
01886             index->getType() == LLVMTypes::Int64Type)
01887             index = TruncInst(index, LLVMTypes::Int32Type);
01888         else if ((!g->target.is32Bit && !g->opt.force32BitAddressing) &&
01889                  index->getType() == LLVMTypes::Int32Type)
01890             index = SExtInst(index, LLVMTypes::Int64Type);
01891 
01892         // do a scalar multiply to get the offset as index * scale and then
01893         // smear the result out to be a vector; this is more efficient than
01894         // first promoting both the scale and the index to vectors and then
01895         // multiplying.
01896         offset = BinaryOperator(llvm::Instruction::Mul, scale, index);
01897         offset = SmearUniform(offset);
01898     }
01899     else {
01900         // Similarly, truncate or sign extend the index to be a 32 or 64
01901         // bit vector type
01902         if ((g->target.is32Bit || g->opt.force32BitAddressing) && 
01903             index->getType() == LLVMTypes::Int64VectorType)
01904             index = TruncInst(index, LLVMTypes::Int32VectorType); 
01905         else if ((!g->target.is32Bit && !g->opt.force32BitAddressing) &&
01906                  index->getType() == LLVMTypes::Int32VectorType)
01907             index = SExtInst(index, LLVMTypes::Int64VectorType);
01908 
01909         scale = SmearUniform(scale);
01910 
01911         // offset = index * scale
01912         offset = BinaryOperator(llvm::Instruction::Mul, scale, index, 
01913                                 LLVMGetName("mul", scale, index));
01914     }
01915 
01916     // For 64-bit targets, if we've been doing our offset calculations in
01917     // 32 bits, we still have to convert to a 64-bit value before we
01918     // actually add the offset to the pointer.
01919     if (g->target.is32Bit == false && g->opt.force32BitAddressing == true)
01920         offset = SExtInst(offset, LLVMTypes::Int64VectorType, 
01921                           LLVMGetName(offset, "_to_64"));
01922 
01923     // Smear out the pointer to be varying; either the base pointer or the
01924     // index must be varying for this method to be called.
01925     bool baseIsUniform = 
01926         (llvm::isa<llvm::PointerType>(basePtr->getType()));
01927     Assert(baseIsUniform == false || indexIsVarying == true);
01928     llvm::Value *varyingPtr = baseIsUniform ? SmearUniform(basePtr) : basePtr;
01929 
01930     // newPtr = ptr + offset
01931     return BinaryOperator(llvm::Instruction::Add, varyingPtr, offset, 
01932                           LLVMGetName(basePtr, "_offset"));
01933 }

01936 void
01937 FunctionEmitContext::MatchIntegerTypes(llvm::Value **v0, llvm::Value **v1) {
01938     llvm::Type *type0 = (*v0)->getType();
01939     llvm::Type *type1 = (*v1)->getType();
01940 
01941     // First, promote to a vector type if one of the two values is a vector
01942     // type
01943     if (llvm::isa<llvm::VectorType>(type0) &&
01944         !llvm::isa<llvm::VectorType>(type1)) {
01945         *v1 = SmearUniform(*v1, "smear_v1");
01946         type1 = (*v1)->getType();
01947     }
01948     if (!llvm::isa<llvm::VectorType>(type0) &&
01949         llvm::isa<llvm::VectorType>(type1)) {
01950         *v0 = SmearUniform(*v0, "smear_v0");
01951         type0 = (*v0)->getType();
01952     }
01953 
01954     // And then update to match bit widths
01955     if (type0 == LLVMTypes::Int32VectorType &&
01956         type1 == LLVMTypes::Int64VectorType)
01957         *v0 = SExtInst(*v0, LLVMTypes::Int64VectorType);
01958     else if (type1 == LLVMTypes::Int32VectorType &&
01959              type0 == LLVMTypes::Int64VectorType)
01960         *v1 = SExtInst(*v1, LLVMTypes::Int64VectorType);
01961 }

01964 /** Given an integer index in indexValue that's indexing into an array of
01965     soa<> structures with given soaWidth, compute the two sub-indices we
01966     need to do the actual indexing calculation:
01967 
01968     subIndices[0] = (indexValue >> log(soaWidth))
01969     subIndices[1] = (indexValue & (soaWidth-1))
01970  */
01971 static llvm::Value *
01972 lComputeSliceIndex(FunctionEmitContext *ctx, int soaWidth,
01973                    llvm::Value *indexValue, llvm::Value *ptrSliceOffset, 
01974                    llvm::Value **newSliceOffset) {
01975     // Compute the log2 of the soaWidth.
01976     Assert(soaWidth > 0);
01977     int logWidth = 0, sw = soaWidth;
01978     while (sw > 1) {
01979         ++logWidth;
01980         sw >>= 1;
01981     }
01982     Assert((1 << logWidth) == soaWidth);
01983 
01984     ctx->MatchIntegerTypes(&indexValue, &ptrSliceOffset);
01985 
01986     llvm::Type *indexType = indexValue->getType();
01987     llvm::Value *shift = LLVMIntAsType(logWidth, indexType);
01988     llvm::Value *mask = LLVMIntAsType(soaWidth-1, indexType);
01989 
01990     llvm::Value *indexSum = 
01991         ctx->BinaryOperator(llvm::Instruction::Add, indexValue, ptrSliceOffset,
01992                             "index_sum");
01993 
01994     // minor index = (index & (soaWidth - 1))
01995     *newSliceOffset = ctx->BinaryOperator(llvm::Instruction::And, indexSum,
01996                                           mask, "slice_index_minor");
01997     // slice offsets are always 32 bits...
01998     if ((*newSliceOffset)->getType() == LLVMTypes::Int64Type)
01999         *newSliceOffset = ctx->TruncInst(*newSliceOffset, LLVMTypes::Int32Type);
02000     else if ((*newSliceOffset)->getType() == LLVMTypes::Int64VectorType)
02001         *newSliceOffset = ctx->TruncInst(*newSliceOffset, LLVMTypes::Int32VectorType);
02002 
02003     // major index = (index >> logWidth)
02004     return ctx->BinaryOperator(llvm::Instruction::AShr, indexSum,
02005                                shift, "slice_index_major");
02006 }

02027 llvm::Value *
02028 FunctionEmitContext::GetElementPtrInst(llvm::Value *basePtr, llvm::Value *index, 
02029                                        const Type *ptrRefType, const char *name) {
02030     if (basePtr == NULL || index == NULL) {
02031         Assert(m->errorCount > 0);
02032         return NULL;
02033     }
02034 
02035     // Regularize to a standard pointer type for basePtr's type
02036     const PointerType *ptrType;
02037     if (dynamic_cast<const ReferenceType *>(ptrRefType) != NULL)
02038         ptrType = PointerType::GetUniform(ptrRefType->GetReferenceTarget());
02039     else {
02040         ptrType = dynamic_cast<const PointerType *>(ptrRefType);
02041         Assert(ptrType != NULL);
02042     }
02043 
02044     if (ptrType->IsSlice()) {
02045         Assert(llvm::isa<llvm::StructType>(basePtr->getType()));
02046 
02047         llvm::Value *ptrSliceOffset = ExtractInst(basePtr, 1);
02048         if (ptrType->IsFrozenSlice() == false) {
02049             // For slice pointers that aren't frozen, we compute a new
02050             // index based on the given index plus the offset in the slice
02051             // pointer.  This gives us an updated integer slice index for
02052             // the resulting slice pointer and then an index to index into
02053             // the soa<> structs with.
02054             llvm::Value *newSliceOffset;
02055             int soaWidth = ptrType->GetBaseType()->GetSOAWidth();
02056             index = lComputeSliceIndex(this, soaWidth, index, 
02057                                        ptrSliceOffset, &newSliceOffset);
02058             ptrSliceOffset = newSliceOffset;
02059         }
02060 
02061         // Handle the indexing into the soa<> structs with the major
02062         // component of the index through a recursive call
02063         llvm::Value *p = GetElementPtrInst(ExtractInst(basePtr, 0), index,
02064                                            ptrType->GetAsNonSlice(), name);
02065 
02066         // And mash the results together for the return value
02067         return MakeSlicePointer(p, ptrSliceOffset);
02068     }
02069 
02070     // Double-check consistency between the given pointer type and its LLVM
02071     // type.
02072     if (ptrType->IsUniformType())
02073         Assert(llvm::isa<llvm::PointerType>(basePtr->getType()));
02074     else if (ptrType->IsVaryingType())
02075         Assert(llvm::isa<llvm::VectorType>(basePtr->getType()));
02076 
02077     bool indexIsVaryingType = 
02078         llvm::isa<llvm::VectorType>(index->getType());
02079 
02080     if (indexIsVaryingType == false && ptrType->IsUniformType() == true) {
02081         // The easy case: both the base pointer and the indices are
02082         // uniform, so just emit the regular LLVM GEP instruction
02083         llvm::Value *ind[1] = { index };
02084         llvm::ArrayRef<llvm::Value *> arrayRef(&ind[0], &ind[1]);
02085         llvm::Instruction *inst = 
02086             llvm::GetElementPtrInst::Create(basePtr, arrayRef,
02087                                             name ? name : "gep", bblock);
02088         AddDebugPos(inst);
02089         return inst;
02090     }
02091     else
02092         return applyVaryingGEP(basePtr, index, ptrType);
02093 }

02096 llvm::Value *
02097 FunctionEmitContext::GetElementPtrInst(llvm::Value *basePtr, llvm::Value *index0, 
02098                                        llvm::Value *index1, const Type *ptrRefType,
02099                                        const char *name) {
02100     if (basePtr == NULL || index0 == NULL || index1 == NULL) {
02101         Assert(m->errorCount > 0);
02102         return NULL;
02103     }
02104 
02105     // Regaularize the pointer type for basePtr
02106     const PointerType *ptrType = NULL;
02107     if (dynamic_cast<const ReferenceType *>(ptrRefType) != NULL)
02108         ptrType = PointerType::GetUniform(ptrRefType->GetReferenceTarget());
02109     else {
02110         ptrType = dynamic_cast<const PointerType *>(ptrRefType);
02111         Assert(ptrType != NULL);
02112     }
02113 
02114     if (ptrType->IsSlice()) {
02115         // Similar to the 1D GEP implementation above, for non-frozen slice
02116         // pointers we do the two-step indexing calculation and then pass
02117         // the new major index on to a recursive GEP call.
02118         Assert(llvm::isa<llvm::StructType>(basePtr->getType()));
02119         llvm::Value *ptrSliceOffset = ExtractInst(basePtr, 1);
02120         if (ptrType->IsFrozenSlice() == false) {
02121             llvm::Value *newSliceOffset;
02122             int soaWidth = ptrType->GetBaseType()->GetSOAWidth();
02123             index1 = lComputeSliceIndex(this, soaWidth, index1,
02124                                         ptrSliceOffset, &newSliceOffset);
02125             ptrSliceOffset = newSliceOffset;
02126         }
02127 
02128         llvm::Value *p = GetElementPtrInst(ExtractInst(basePtr, 0), index0,
02129                                            index1, ptrType->GetAsNonSlice(), 
02130                                            name);
02131         return MakeSlicePointer(p, ptrSliceOffset);
02132     }
02133 
02134     bool index0IsVaryingType = 
02135         llvm::isa<llvm::VectorType>(index0->getType());
02136     bool index1IsVaryingType = 
02137         llvm::isa<llvm::VectorType>(index1->getType());
02138 
02139     if (index0IsVaryingType == false && index1IsVaryingType == false && 
02140         ptrType->IsUniformType() == true) {
02141         // The easy case: both the base pointer and the indices are
02142         // uniform, so just emit the regular LLVM GEP instruction
02143         llvm::Value *indices[2] = { index0, index1 };
02144         llvm::ArrayRef<llvm::Value *> arrayRef(&indices[0], &indices[2]);
02145         llvm::Instruction *inst = 
02146             llvm::GetElementPtrInst::Create(basePtr, arrayRef,
02147                                             name ? name : "gep", bblock);
02148         AddDebugPos(inst);

02150     }
02151     else {
02152         // Handle the first dimension with index0
02153         llvm::Value *ptr0 = GetElementPtrInst(basePtr, index0, ptrType);
02154 
02155         // Now index into the second dimension with index1.  First figure
02156         // out the type of ptr0.
02157         const Type *baseType = ptrType->GetBaseType();
02158         const SequentialType *st = dynamic_cast<const SequentialType *>(baseType);
02159         Assert(st != NULL);
02160 
02161         bool ptr0IsUniform = 
02162             llvm::isa<llvm::PointerType>(ptr0->getType());
02163         const Type *ptr0BaseType = st->GetElementType();
02164         const Type *ptr0Type = ptr0IsUniform ?
02165             PointerType::GetUniform(ptr0BaseType) : 
02166             PointerType::GetVarying(ptr0BaseType);
02167 
02168         return applyVaryingGEP(ptr0, index1, ptr0Type);
02169     }

02173 llvm::Value *
02174 FunctionEmitContext::AddElementOffset(llvm::Value *fullBasePtr, int elementNum,
02175                                       const Type *ptrRefType, const char *name,
02176                                       const PointerType **resultPtrType) {
02177     if (resultPtrType != NULL)
02178         Assert(ptrRefType != NULL);
02179 
02180     // (Unfortunately) it's not required to pass a non-NULL ptrRefType, but
02181     // if we have one, regularize into a pointer type.
02182     const PointerType *ptrType = NULL;
02183     if (ptrRefType != NULL) {
02184         // Normalize references to uniform pointers
02185         if (dynamic_cast<const ReferenceType *>(ptrRefType) != NULL)
02186             ptrType = PointerType::GetUniform(ptrRefType->GetReferenceTarget());
02187         else
02188             ptrType = dynamic_cast<const PointerType *>(ptrRefType);
02189         Assert(ptrType != NULL);
02190     }
02191 
02192     // Similarly, we have to see if the pointer type is a struct to see if
02193     // we have a slice pointer instead of looking at ptrType; this is also
02194     // unfortunate...
02195     llvm::Value *basePtr = fullBasePtr;
02196     bool baseIsSlicePtr = 
02197         llvm::isa<llvm::StructType>(fullBasePtr->getType());
02198     const PointerType *rpt;
02199     if (baseIsSlicePtr) {
02200         Assert(ptrType != NULL);
02201         // Update basePtr to just be the part that actually points to the
02202         // start of an soa<> struct for now; the element offset computation
02203         // doesn't change the slice offset, so we'll incorporate that into
02204         // the final value right before this method returns.
02205         basePtr = ExtractInst(fullBasePtr, 0);
02206         if (resultPtrType == NULL)
02207             resultPtrType = &rpt;
02208     }
02209 
02210     // Return the pointer type of the result of this call, for callers that
02211     // want it.
02212     if (resultPtrType != NULL) {
02213         Assert(ptrType != NULL);
02214         const CollectionType *ct = 
02215             dynamic_cast<const CollectionType *>(ptrType->GetBaseType());
02216         Assert(ct != NULL);
02217         *resultPtrType = new PointerType(ct->GetElementType(elementNum),
02218                                          ptrType->GetVariability(),
02219                                          ptrType->IsConstType(),
02220                                          ptrType->IsSlice());
02221     }
02222 
02223     llvm::Value *resultPtr = NULL;
02224     if (ptrType == NULL || ptrType->IsUniformType()) {
02225         // If the pointer is uniform, we can use the regular LLVM GEP.
02226         llvm::Value *offsets[2] = { LLVMInt32(0), LLVMInt32(elementNum) };
02227         llvm::ArrayRef<llvm::Value *> arrayRef(&offsets[0], &offsets[2]);
02228         resultPtr = 
02229             llvm::GetElementPtrInst::Create(basePtr, arrayRef,
02230                                             name ? name : "struct_offset", bblock);
02231     }
02232     else {
02233         // Otherwise do the math to find the offset and add it to the given
02234         // varying pointers
02235         const StructType *st = 
02236             dynamic_cast<const StructType *>(ptrType->GetBaseType());
02237         llvm::Value *offset = NULL;
02238         if (st != NULL)
02239             // If the pointer is to a structure, Target::StructOffset() gives
02240             // us the offset in bytes to the given element of the structure
02241             offset = g->target.StructOffset(st->LLVMType(g->ctx), elementNum,
02242                                             bblock);
02243         else {
02244             // Otherwise we should have a vector or array here and the offset
02245             // is given by the element number times the size of the element
02246             // type of the vector.
02247             const SequentialType *st = 
02248                 dynamic_cast<const SequentialType *>(ptrType->GetBaseType());
02249             Assert(st != NULL);
02250             llvm::Value *size = 
02251                 g->target.SizeOf(st->GetElementType()->LLVMType(g->ctx), bblock);
02252             llvm::Value *scale = (g->target.is32Bit || g->opt.force32BitAddressing) ?
02253                 LLVMInt32(elementNum) : LLVMInt64(elementNum);
02254             offset = BinaryOperator(llvm::Instruction::Mul, size, scale);
02255         }
02256 
02257         offset = SmearUniform(offset, "offset_smear");
02258 
02259         if (g->target.is32Bit == false && g->opt.force32BitAddressing == true)
02260             // If we're doing 32 bit addressing with a 64 bit target, although
02261             // we did the math above in 32 bit, we need to go to 64 bit before
02262             // we add the offset to the varying pointers.
02263             offset = SExtInst(offset, LLVMTypes::Int64VectorType, "offset_to_64");
02264 
02265         resultPtr = BinaryOperator(llvm::Instruction::Add, basePtr, offset, 
02266                                    "struct_ptr_offset");
02267     }
02268 
02269     // Finally, if had a slice pointer going in, mash back together with
02270     // the original (unchanged) slice offset.
02271     if (baseIsSlicePtr)
02272         return MakeSlicePointer(resultPtr, ExtractInst(fullBasePtr, 1));
02273     else
02274         return resultPtr;
02275 }

02410             // can just assume the natural alignment (0 here), but for varying
02411             // atomic types, we need to make sure that the compiler emits
02412             // unaligned vector loads, so we specify a reduced alignment here.
02413             int align = 0;
02414             const AtomicType *atomicType = 
02415                 dynamic_cast<const AtomicType *>(ptrType->GetBaseType());
02416             if (atomicType != NULL && atomicType->IsVaryingType())
02417                 // We actually just want to align to the vector element
02418                 // alignment, but can't easily get that here, so just tell LLVM
02419                 // it's totally unaligned.  (This shouldn't make any difference
02420                 // vs the proper alignment in practice.)
02421                 align = 1;
02422             llvm::Instruction *inst = new llvm::LoadInst(ptr, name,
02423                                                          false /* not volatile */,
02424                                                          align, bblock);
02425             AddDebugPos(inst);
02426             return inst;
02427         }
02428     }
02429     else {
02430         // Otherwise we should have a varying ptr and it's time for a
02431         // gather.
02432         return gather(ptr, ptrType, GetFullMask(), name);
02433     }
02434 }

02437 llvm::Value *
02438 FunctionEmitContext::gather(llvm::Value *ptr, const PointerType *ptrType, 
02439                             llvm::Value *mask, const char *name) {
02440     // We should have a varying pointer if we get here...
02441     Assert(ptrType->IsVaryingType());
02442 
02443     const Type *returnType = ptrType->GetBaseType()->GetAsVaryingType();
02444     llvm::Type *llvmReturnType = returnType->LLVMType(g->ctx);
02445 
02446     const CollectionType *collectionType = 
02447         dynamic_cast<const CollectionType *>(ptrType->GetBaseType());
02448     if (collectionType != NULL) {
02449         // For collections, recursively gather element wise to find the
02450         // result.
02451         llvm::Value *retValue = llvm::UndefValue::get(llvmReturnType);
02452 
02453         for (int i = 0; i < collectionType->GetElementCount(); ++i) {
02454             const PointerType *eltPtrType;
02455             llvm::Value *eltPtr = 
02456                 AddElementOffset(ptr, i, ptrType, "gather_elt_ptr", &eltPtrType);
02457 
02458             eltPtr = addVaryingOffsetsIfNeeded(eltPtr, eltPtrType);
02459 
02460             // This in turn will be another gather
02461             llvm::Value *eltValues = LoadInst(eltPtr, mask, eltPtrType, name);
02462 
02463             retValue = InsertInst(retValue, eltValues, i, "set_value");
02464         }
02465         return retValue;
02466     }
02467     else if (ptrType->IsSlice()) {
02468         // If we have a slice pointer, we need to add the final slice
02469         // offset here right before issuing the actual gather
02470         //
02471         // FIXME: would it be better to do the corresponding same thing for
02472         // all of the varying offsets stuff here (and in scatter)?
02473         ptr = lFinalSliceOffset(this, ptr, &ptrType);
02474     }
02475 
02476     // Otherwise we should just have a basic scalar or pointer type and we
02477     // can go and do the actual gather
02478     AddInstrumentationPoint("gather");
02479 
02480     // Figure out which gather function to call based on the size of
02481     // the elements.
02482     const PointerType *pt = dynamic_cast<const PointerType *>(returnType);
02483     const char *funcName = NULL;
02484     if (pt != NULL)
02485         funcName = g->target.is32Bit ? "__pseudo_gather32_32" : 
02486             "__pseudo_gather64_64";
02487     else if (llvmReturnType == LLVMTypes::DoubleVectorType || 
02488              llvmReturnType == LLVMTypes::Int64VectorType)
02489         funcName = g->target.is32Bit ? "__pseudo_gather32_64" : 
02490             "__pseudo_gather64_64";
02491     else if (llvmReturnType == LLVMTypes::FloatVectorType || 
02492              llvmReturnType == LLVMTypes::Int32VectorType)
02493         funcName = g->target.is32Bit ? "__pseudo_gather32_32" : 
02494             "__pseudo_gather64_32";
02495     else if (llvmReturnType == LLVMTypes::Int16VectorType)
02496         funcName = g->target.is32Bit ? "__pseudo_gather32_16" : 
02497             "__pseudo_gather64_16";
02498     else {
02499         Assert(llvmReturnType == LLVMTypes::Int8VectorType);
02500         funcName = g->target.is32Bit ? "__pseudo_gather32_8" : 
02501             "__pseudo_gather64_8";
02502     }
02503 
02504     llvm::Function *gatherFunc = m->module->getFunction(funcName);
02505     Assert(gatherFunc != NULL);
02506 
02507     llvm::Value *call = CallInst(gatherFunc, NULL, ptr, mask, name);
02508 
02509     // Add metadata about the source file location so that the
02510     // optimization passes can print useful performance warnings if we
02511     // can't optimize out this gather
02512     if (disableGSWarningCount == 0)
02513         addGSMetadata(call, currentPos);
02514 
02515     return BitCastInst(call, llvmReturnType, LLVMGetName(call, "_gather_bitcast"));
02516 }

02591 /** Code to store the given varying value to the given location, only
02592     storing the elements that correspond to active program instances as
02593     given by the provided storeMask value.  Note that the lvalue is only a
02594     single pointer, not a varying lvalue of one pointer per program
02595     instance (that case is handled by scatters).
02596  */
02597 void
02598 FunctionEmitContext::maskedStore(llvm::Value *value, llvm::Value *ptr,
02599                                  const Type *ptrType, llvm::Value *mask) {
02600     if (value == NULL || ptr == NULL) {
02601         Assert(m->errorCount > 0);
02602         return;
02603     }
02604 
02605     Assert(dynamic_cast<const PointerType *>(ptrType) != NULL);
02606     Assert(ptrType->IsUniformType());
02607 
02608     const Type *valueType = ptrType->GetBaseType();
02609     const CollectionType *collectionType = 
02610         dynamic_cast<const CollectionType *>(valueType);
02611     if (collectionType != NULL) {
02612         // Assigning a structure / array / vector. Handle each element
02613         // individually with what turns into a recursive call to
02614         // makedStore()
02615         for (int i = 0; i < collectionType->GetElementCount(); ++i) {
02616             llvm::Value *eltValue = ExtractInst(value, i, "value_member");
02617             llvm::Value *eltPtr = 
02618                 AddElementOffset(ptr, i, ptrType, "struct_ptr_ptr");
02619             const Type *eltType = collectionType->GetElementType(i);
02620             const Type *eltPtrType = PointerType::GetUniform(eltType);
02621             StoreInst(eltValue, eltPtr, mask, eltType, eltPtrType);
02622         }
02623         return;
02624     }
02625 
02626     // We must have a regular atomic, enumerator, or pointer type at this
02627     // point.
02628     Assert(Type::IsBasicType(valueType));
02629     valueType = valueType->GetAsNonConstType();
02630 
02631     // Figure out if we need a 8, 16, 32 or 64-bit masked store.
02632     llvm::Function *maskedStoreFunc = NULL;
02633 
02634     const PointerType *pt = dynamic_cast<const PointerType *>(valueType);
02635     if (pt != NULL) {
02636         if (pt->IsSlice()) {
02637             // Masked store of (varying) slice pointer.
02638             Assert(pt->IsVaryingType());
02639                     
02640             // First, extract the pointer from the slice struct and masked
02641             // store that.
02642             llvm::Value *v0 = ExtractInst(value, 0);
02643             llvm::Value *p0 = AddElementOffset(ptr, 0, ptrType);
02644             maskedStore(v0, p0, PointerType::GetUniform(pt->GetAsNonSlice()),
02645                         mask);
02646 
02647             // And then do same for the integer offset
02648             llvm::Value *v1 = ExtractInst(value, 1);
02649             llvm::Value *p1 = AddElementOffset(ptr, 1, ptrType);
02650             const Type *offsetType = AtomicType::VaryingInt32;
02651             maskedStore(v1, p1, PointerType::GetUniform(offsetType), mask);
02652 
02653             return;
02654         }
02655 
02656         if (g->target.is32Bit)
02657             maskedStoreFunc = m->module->getFunction("__pseudo_masked_store_32");
02658         else
02659             maskedStoreFunc = m->module->getFunction("__pseudo_masked_store_64");
02660     }
02661     else if (Type::Equal(valueType, AtomicType::VaryingBool) &&
02662              g->target.maskBitCount == 1) {
02663         llvm::Value *notMask = BinaryOperator(llvm::Instruction::Xor, mask,
02664                                               LLVMMaskAllOn, "~mask");
02665         llvm::Value *old = LoadInst(ptr);
02666         llvm::Value *maskedOld = BinaryOperator(llvm::Instruction::And, old,
02667                                                 notMask, "old&~mask");
02668         llvm::Value *maskedNew = BinaryOperator(llvm::Instruction::And, value,
02669                                                 mask, "new&mask");
02670         llvm::Value *final = BinaryOperator(llvm::Instruction::Or, maskedOld,
02671                                             maskedNew, "old_new_result");
02672         StoreInst(final, ptr);
02673         return;
02674     }
02675     else if (Type::Equal(valueType, AtomicType::VaryingDouble) || 
02676              Type::Equal(valueType, AtomicType::VaryingInt64) ||
02677              Type::Equal(valueType, AtomicType::VaryingUInt64)) {
02678         maskedStoreFunc = m->module->getFunction("__pseudo_masked_store_64");
02679         ptr = BitCastInst(ptr, LLVMTypes::Int64VectorPointerType, 
02680                           LLVMGetName(ptr, "_to_int64vecptr"));
02681         value = BitCastInst(value, LLVMTypes::Int64VectorType, 
02682                             LLVMGetName(value, "_to_int64"));
02683     }
02684     else if (Type::Equal(valueType, AtomicType::VaryingFloat) ||
02685              Type::Equal(valueType, AtomicType::VaryingBool) ||
02686              Type::Equal(valueType, AtomicType::VaryingInt32) ||
02687              Type::Equal(valueType, AtomicType::VaryingUInt32) ||
02688              dynamic_cast<const EnumType *>(valueType) != NULL) {
02689         maskedStoreFunc = m->module->getFunction("__pseudo_masked_store_32");
02690         ptr = BitCastInst(ptr, LLVMTypes::Int32VectorPointerType, 
02691                           LLVMGetName(ptr, "_to_int32vecptr"));
02692         if (Type::Equal(valueType, AtomicType::VaryingFloat))
02693             value = BitCastInst(value, LLVMTypes::Int32VectorType, 
02694                                 LLVMGetName(value, "_to_int32"));
02695     }
02696     else if (Type::Equal(valueType, AtomicType::VaryingInt16) ||
02697              Type::Equal(valueType, AtomicType::VaryingUInt16)) {
02698         maskedStoreFunc = m->module->getFunction("__pseudo_masked_store_16");
02699         ptr = BitCastInst(ptr, LLVMTypes::Int16VectorPointerType, 
02700                           LLVMGetName(ptr, "_to_int16vecptr"));
02701     }
02702     else if (Type::Equal(valueType, AtomicType::VaryingInt8) ||
02703              Type::Equal(valueType, AtomicType::VaryingUInt8)) {
02704         maskedStoreFunc = m->module->getFunction("__pseudo_masked_store_8");
02705         ptr = BitCastInst(ptr, LLVMTypes::Int8VectorPointerType, 
02706                           LLVMGetName(ptr, "_to_int8vecptr"));
02707     }
02708     Assert(maskedStoreFunc != NULL);
02709 
02710     std::vector<llvm::Value *> args;
02711     args.push_back(ptr);
02712     args.push_back(value);
02713     args.push_back(mask);
02714     CallInst(maskedStoreFunc, NULL, args);
02715 }

02719 /** Scatter the given varying value to the locations given by the varying
02720     lvalue (which should be an array of pointers with size equal to the
02721     target's vector width.  We want to store each rvalue element at the
02722     corresponding pointer's location, *if* the mask for the corresponding
02723     program instance are on.  If they're off, don't do anything.  
02724 */
02725 void
02726 FunctionEmitContext::scatter(llvm::Value *value, llvm::Value *ptr, 
02727                              const Type *valueType, const Type *origPt,
02728                              llvm::Value *mask) {
02729     const PointerType *ptrType = dynamic_cast<const PointerType *>(origPt);
02730     Assert(ptrType != NULL);
02731     Assert(ptrType->IsVaryingType());
02732 
02733     const CollectionType *srcCollectionType = 
02734         dynamic_cast<const CollectionType *>(valueType);
02735     if (srcCollectionType != NULL) {
02736         // We're scattering a collection type--we need to keep track of the
02737         // source type (the type of the data values to be stored) and the
02738         // destination type (the type of objects in memory that will be
02739         // stored into) separately.  This is necessary so that we can get
02740         // all of the addressing calculations right if we're scattering
02741         // from a varying struct to an array of uniform instances of the
02742         // same struct type, versus scattering into an array of varying
02743         // instances of the struct type, etc.
02744         const CollectionType *dstCollectionType =
02745             dynamic_cast<const CollectionType *>(ptrType->GetBaseType());
02746         Assert(dstCollectionType != NULL);
02747             
02748         // Scatter the collection elements individually
02749         for (int i = 0; i < srcCollectionType->GetElementCount(); ++i) {
02750             // First, get the values for the current element out of the
02751             // source.
02752             llvm::Value *eltValue = ExtractInst(value, i);
02753             const Type *srcEltType = srcCollectionType->GetElementType(i);
02754 
02755             // We may be scattering a uniform atomic element; in this case
02756             // we'll smear it out to be varying before making the recursive
02757             // scatter() call below.
02758             if (srcEltType->IsUniformType() && Type::IsBasicType(srcEltType)) {
02759                 eltValue = SmearUniform(eltValue, "to_varying");
02760                 srcEltType = srcEltType->GetAsVaryingType();
02761             }
02762 
02763             // Get the (varying) pointer to the i'th element of the target
02764             // collection
02765             llvm::Value *eltPtr = AddElementOffset(ptr, i, ptrType);
02766 
02767             // The destination element type may be uniform (e.g. if we're
02768             // scattering to an array of uniform structs).  Thus, we need
02769             // to be careful about passing the correct type to
02770             // addVaryingOffsetsIfNeeded() here.
02771             const Type *dstEltType = dstCollectionType->GetElementType(i);
02772             const PointerType *dstEltPtrType = PointerType::GetVarying(dstEltType);
02773             if (ptrType->IsSlice())
02774                 dstEltPtrType = dstEltPtrType->GetAsSlice();
02775 
02776             eltPtr = addVaryingOffsetsIfNeeded(eltPtr, dstEltPtrType);
02777 
02778             // And recursively scatter() until we hit a basic type, at
02779             // which point the actual memory operations can be performed...
02780             scatter(eltValue, eltPtr, srcEltType, dstEltPtrType, mask);
02781         }
02782         return;
02783     }
02784     else if (ptrType->IsSlice()) {
02785         // As with gather, we need to add the final slice offset finally
02786         // once we get to a terminal SOA array of basic types..
02787         ptr = lFinalSliceOffset(this, ptr, &ptrType);
02788     }
02789 
02790     const PointerType *pt = dynamic_cast<const PointerType *>(valueType);
02791 
02792     // And everything should be a pointer or atomic from here on out...
02793     Assert(pt != NULL || 
02794            dynamic_cast<const AtomicType *>(valueType) != NULL);
02795 
02796     llvm::Type *type = value->getType();
02797     const char *funcName = NULL;
02798     if (pt != NULL)
02799         funcName = g->target.is32Bit ? "__pseudo_scatter32_32" :
02800             "__pseudo_scatter64_64";
02801     else if (type == LLVMTypes::DoubleVectorType || 
02802              type == LLVMTypes::Int64VectorType) {
02803         funcName = g->target.is32Bit ? "__pseudo_scatter32_64" :
02804             "__pseudo_scatter64_64";
02805         value = BitCastInst(value, LLVMTypes::Int64VectorType, "value2int");
02806     }
02807     else if (type == LLVMTypes::FloatVectorType || 
02808              type == LLVMTypes::Int32VectorType) {
02809         funcName = g->target.is32Bit ? "__pseudo_scatter32_32" :
02810             "__pseudo_scatter64_32";
02811         value = BitCastInst(value, LLVMTypes::Int32VectorType, "value2int");
02812     }
02813     else if (type == LLVMTypes::Int16VectorType)
02814         funcName = g->target.is32Bit ? "__pseudo_scatter32_16" :
02815             "__pseudo_scatter64_16";
02816     else if (type == LLVMTypes::Int8VectorType)
02817         funcName = g->target.is32Bit ? "__pseudo_scatter32_8" :
02818             "__pseudo_scatter64_8";
02819 
02820     llvm::Function *scatterFunc = m->module->getFunction(funcName);
02821     Assert(scatterFunc != NULL);
02822     
02823     AddInstrumentationPoint("scatter");
02824 
02825     std::vector<llvm::Value *> args;
02826     args.push_back(ptr);
02827     args.push_back(value);
02828     args.push_back(mask);
02829     llvm::Value *inst = CallInst(scatterFunc, NULL, args);
02830 
02831     if (disableGSWarningCount == 0)
02832         addGSMetadata(inst, currentPos);
02833 }

02859 void
02860 FunctionEmitContext::StoreInst(llvm::Value *value, llvm::Value *ptr,
02861                                llvm::Value *mask, const Type *valueType,
02862                                const Type *ptrRefType) {
02863     if (value == NULL || ptr == NULL) {
02864         // may happen due to error elsewhere
02865         Assert(m->errorCount > 0);
02866         return;
02867     }
02868 
02869     const PointerType *ptrType;
02870     if (dynamic_cast<const ReferenceType *>(ptrRefType) != NULL)
02871         ptrType = PointerType::GetUniform(ptrRefType->GetReferenceTarget());
02872     else {
02873         ptrType = dynamic_cast<const PointerType *>(ptrRefType);
02874         Assert(ptrType != NULL);
02875     }
02876 
02877     // Figure out what kind of store we're doing here
02878     if (ptrType->IsUniformType()) {
02879         if (ptrType->IsSlice())
02880             // storing a uniform value to a single slice of a SOA type
02881             storeUniformToSOA(value, ptr, mask, valueType, ptrType);
02882         else if (ptrType->GetBaseType()->IsUniformType())
02883             // the easy case
02884             StoreInst(value, ptr);
02885         else if (mask == LLVMMaskAllOn && !g->opt.disableMaskAllOnOptimizations)
02886             // Otherwise it is a masked store unless we can determine that the
02887             // mask is all on...  (Unclear if this check is actually useful.)
02888             StoreInst(value, ptr);
02889         else
02890             maskedStore(value, ptr, ptrType, mask);
02891     }
02892     else {
02893         Assert(ptrType->IsVaryingType());
02894         // We have a varying ptr (an array of pointers), so it's time to
02895         // scatter
02896         scatter(value, ptr, valueType, ptrType, GetFullMask());
02897     }
02898 }

02901 /** Store a uniform type to SOA-laid-out memory.
02902  */
02903 void
02904 FunctionEmitContext::storeUniformToSOA(llvm::Value *value, llvm::Value *ptr,
02905                                        llvm::Value *mask, const Type *valueType,
02906                                        const PointerType *ptrType) {
02907     Assert(Type::EqualIgnoringConst(ptrType->GetBaseType()->GetAsUniformType(), 
02908                                     valueType));
02909 
02910     const CollectionType *ct = dynamic_cast<const CollectionType *>(valueType);
02911     if (ct != NULL) {
02912         // Handle collections element wise...
02913         for (int i = 0; i < ct->GetElementCount(); ++i) {
02914             llvm::Value *eltValue = ExtractInst(value, i);
02915             const Type *eltType = ct->GetElementType(i);
02916             const PointerType *dstEltPtrType;
02917             llvm::Value *dstEltPtr = 
02918                 AddElementOffset(ptr, i, ptrType, "slice_offset",
02919                                  &dstEltPtrType);
02920             StoreInst(eltValue, dstEltPtr, mask, eltType, dstEltPtrType);
02921         }
02922     }
02923     else {
02924         // We're finally at a leaf SOA array; apply the slice offset and
02925         // then we can do a final regular store
02926         Assert(Type::IsBasicType(valueType));
02927         ptr = lFinalSliceOffset(this, ptr, &ptrType);
02928         StoreInst(value, ptr);
02929     }
02930 }

02937     src = BitCastInst(src, LLVMTypes::VoidPointerType);
02938     if (count->getType() != LLVMTypes::Int64Type) {
02939         Assert(count->getType() == LLVMTypes::Int32Type);
02940         count = ZExtInst(count, LLVMTypes::Int64Type, "count_to_64");
02941     }

02943         align = LLVMInt32(1);
02944 
02945     llvm::Constant *mcFunc = 
02946         m->module->getOrInsertFunction("llvm.memcpy.p0i8.p0i8.i64", 
02947                                        LLVMTypes::VoidType, LLVMTypes::VoidPointerType,
02948                                        LLVMTypes::VoidPointerType, LLVMTypes::Int64Type,
02949                                        LLVMTypes::Int32Type, LLVMTypes::BoolType, NULL);
02950     Assert(mcFunc != NULL);
02951     Assert(llvm::isa<llvm::Function>(mcFunc));

03085 llvm::Value *
03086 FunctionEmitContext::CallInst(llvm::Value *func, const FunctionType *funcType,
03087                               const std::vector<llvm::Value *> &args,
03088                               const char *name) {
03089     if (func == NULL) {
03090         Assert(m->errorCount > 0);
03091         return NULL;
03092     }
03093 
03094     std::vector<llvm::Value *> argVals = args;
03095     // Most of the time, the mask is passed as the last argument.  this
03096     // isn't the case for things like intrinsics, builtins, and extern "C"
03097     // functions from the application.  Add the mask if it's needed.
03098     unsigned int calleeArgCount = lCalleeArgCount(func, funcType);
03099     Assert(argVals.size() + 1 == calleeArgCount ||
03100            argVals.size() == calleeArgCount);
03101     if (argVals.size() + 1 == calleeArgCount)
03102         argVals.push_back(GetFullMask());
03103 
03104     if (llvm::isa<llvm::VectorType>(func->getType()) == false) {
03105         // Regular 'uniform' function call--just one function or function
03106         // pointer, so just emit the IR directly.
03107         llvm::Instruction *ci = 
03108             llvm::CallInst::Create(func, argVals, name ? name : "", bblock);
03109         AddDebugPos(ci);
03110         return ci;
03111     }
03112     else {
03113         // Emit the code for a varying function call, where we have an
03114         // vector of function pointers, one for each program instance.  The
03115         // basic strategy is that we go through the function pointers, and
03116         // for the executing program instances, for each unique function
03117         // pointer that's in the vector, call that function with a mask
03118         // equal to the set of active program instances that also have that
03119         // function pointer.  When all unique function pointers have been
03120         // called, we're done.
03121 
03122         llvm::BasicBlock *bbTest = CreateBasicBlock("varying_funcall_test");
03123         llvm::BasicBlock *bbCall = CreateBasicBlock("varying_funcall_call");
03124         llvm::BasicBlock *bbDone = CreateBasicBlock("varying_funcall_done");
03125 
03126         // Get the current mask value so we can restore it later
03127         llvm::Value *origMask = GetInternalMask();
03128 
03129         // First allocate memory to accumulate the various program
03130         // instances' return values...
03131         const Type *returnType = funcType->GetReturnType();
03132         llvm::Type *llvmReturnType = returnType->LLVMType(g->ctx);
03133         llvm::Value *resultPtr = NULL;
03134         if (llvmReturnType->isVoidTy() == false)
03135             resultPtr = AllocaInst(llvmReturnType);
03136 
03137         // The memory pointed to by maskPointer tracks the set of program
03138         // instances for which we still need to call the function they are
03139         // pointing to.  It starts out initialized with the mask of
03140         // currently running program instances.
03141         llvm::Value *maskPtr = AllocaInst(LLVMTypes::MaskType);
03142         StoreInst(GetFullMask(), maskPtr);
03143 
03144         // And now we branch to the test to see if there's more work to be
03145         // done.
03146         BranchInst(bbTest);
03147 
03148         // bbTest: are any lanes of the mask still on?  If so, jump to
03149         // bbCall
03150         SetCurrentBasicBlock(bbTest); {
03151             llvm::Value *maskLoad = LoadInst(maskPtr);
03152             llvm::Value *any = Any(maskLoad);
03153             BranchInst(bbCall, bbDone, any);
03154         }
03155 
03156         // bbCall: this is the body of the loop that calls out to one of
03157         // the active function pointer values.
03158         SetCurrentBasicBlock(bbCall); {
03159             // Figure out the first lane that still needs its function
03160             // pointer to be called.
03161             llvm::Value *currentMask = LoadInst(maskPtr);
03162             llvm::Function *cttz = 
03163                 m->module->getFunction("__count_trailing_zeros_i32");
03164             Assert(cttz != NULL);
03165             llvm::Value *firstLane = CallInst(cttz, NULL, LaneMask(currentMask),
03166                                               "first_lane");
03167 
03168             // Get the pointer to the function we're going to call this
03169             // time through: ftpr = func[firstLane]
03170             llvm::Value *fptr = 
03171                 llvm::ExtractElementInst::Create(func, firstLane, 
03172                                                  "extract_fptr", bblock);
03173 
03174             // Smear it out into an array of function pointers
03175             llvm::Value *fptrSmear = SmearUniform(fptr, "func_ptr");
03176 
03177             // fpOverlap = (fpSmearAsVec == fpOrigAsVec).  This gives us a
03178             // mask for the set of program instances that have the same
03179             // value for their function pointer.
03180             llvm::Value *fpOverlap = 
03181                 CmpInst(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_EQ,
03182                         fptrSmear, func);
03183             fpOverlap = I1VecToBoolVec(fpOverlap);
03184 
03185             // Figure out the mask to use when calling the function
03186             // pointer: we need to AND the current execution mask to handle
03187             // the case of any non-running program instances that happen to
03188             // have this function pointer value.
03189             // callMask = (currentMask & fpOverlap)
03190             llvm::Value *callMask = 
03191                 BinaryOperator(llvm::Instruction::And, currentMask, fpOverlap,
03192                                "call_mask");
03193 
03194             // Set the mask
03195             SetInternalMask(callMask);
03196 
03197             // bitcast the i32/64 function pointer to the actual function
03198             // pointer type (the variant that includes a mask).
03199             llvm::Type *llvmFuncType =
03200                 funcType->LLVMFunctionType(g->ctx, true);
03201             llvm::Type *llvmFPtrType = 
03202                 llvm::PointerType::get(llvmFuncType, 0);
03203             llvm::Value *fptrCast = IntToPtrInst(fptr, llvmFPtrType);
03204 
03205             // Call the function: callResult = call ftpr(args, args, call mask)
03206             llvm::Value *callResult = CallInst(fptrCast, funcType, args, name);
03207 
03208             // Now, do a masked store into the memory allocated to
03209             // accumulate the result using the call mask.
03210             if (callResult != NULL) {
03211                 Assert(resultPtr != NULL);
03212                 StoreInst(callResult, resultPtr, callMask, returnType,
03213                           PointerType::GetUniform(returnType));
03214             }
03215             else
03216                 Assert(resultPtr == NULL);
03217 
03218             // Update the mask to turn off the program instances for which
03219             // we just called the function.
03220             // currentMask = currentMask & ~callmask
03221             llvm::Value *notCallMask = 
03222                 BinaryOperator(llvm::Instruction::Xor, callMask, LLVMMaskAllOn,
03223                                "~callMask");
03224             currentMask = BinaryOperator(llvm::Instruction::And, currentMask, 
03225                                          notCallMask, "currentMask&~callMask");
03226             StoreInst(currentMask, maskPtr);
03227 
03228             // And go back to the test to see if we need to do another
03229             // call.
03230             BranchInst(bbTest);
03231         }
03232 
03233         // bbDone: We're all done; clean up and return the result we've
03234         // accumulated in the result memory.
03235         SetCurrentBasicBlock(bbDone);
03236         SetInternalMask(origMask);
03237         return resultPtr ? LoadInst(resultPtr) : NULL;
03238     }
03239 }
