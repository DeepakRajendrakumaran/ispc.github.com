00069 /** Utility routine to create a llvm DIArray type of the given number of
00070     the given element type. */
00071 static llvm::DIType 
00072 lCreateDIArray(llvm::DIType eltType, int count) {
00073     int lowerBound = 0, upperBound = count-1;
00074 
00075     if (count == 0) {
00076         // unsized array -> indicate with low > high
00077         lowerBound = 1;
00078         upperBound = 0;
00079     }
00080 
00081     llvm::Value *sub = m->diBuilder->getOrCreateSubrange(lowerBound, upperBound);
00082     std::vector<llvm::Value *> subs;
00083     subs.push_back(sub);
00084 #ifdef LLVM_2_9
00085     llvm::DIArray subArray = m->diBuilder->getOrCreateArray(&subs[0], subs.size());
00086 #else
00087     llvm::DIArray subArray = m->diBuilder->getOrCreateArray(subs);
00088 #endif
00089 
00090     uint64_t size = eltType.getSizeInBits() * count;
00091     uint64_t align = eltType.getAlignInBits();
00092 
00093     return m->diBuilder->createArrayType(size, align, eltType, subArray);

01106 LLVM_TYPE_CONST llvm::Type *
01107 PointerType::LLVMType(llvm::LLVMContext *ctx) const {
01108     if (baseType == NULL) {
01109         Assert(m->errorCount > 0);
01110         return NULL;
01111     }
01112 
01113     if (isSlice)
01114         // Slice pointers are represented as a structure with a pointer and
01115         // an integer offset; the corresponding ispc type is returned by
01116         // GetSliceStructType().
01117         return GetSliceStructType()->LLVMType(ctx);
01118 
01119     switch (variability.type) {
01120     case Variability::Uniform: {
01121         LLVM_TYPE_CONST llvm::Type *ptype = NULL;
01122         const FunctionType *ftype = dynamic_cast<const FunctionType *>(baseType);

01124             // Get the type of the function variant that takes the mask as the

01127             ptype = llvm::PointerType::get(ftype->LLVMFunctionType(ctx, true), 0);
01128         else {
01129             if (baseType == AtomicType::Void)
01130                 ptype = LLVMTypes::VoidPointerType;

01136     case Variability::Varying:
01137         // always the same, since we currently use int vectors for varying
01138         // pointers
01139         return LLVMTypes::VoidPointerVectorType;

01141         ArrayType at(GetAsUniformType(), variability.soaWidth);

01460 const Type *
01461 ArrayType::SizeUnsizedArrays(const Type *type, Expr *initExpr) {
01462     const ArrayType *at = dynamic_cast<const ArrayType *>(type);
01463     if (at == NULL)
01464         return type;
01465 
01466     ExprList *exprList = dynamic_cast<ExprList *>(initExpr);
01467     if (exprList == NULL || exprList->exprs.size() == 0)
01468         return type;
01469 
01470     // If the current dimension is unsized, then size it according to the
01471     // length of the expression list
01472     if (at->GetElementCount() == 0) {
01473         type = at->GetSizedArray(exprList->exprs.size());
01474         at = dynamic_cast<const ArrayType *>(type);
01475     }
01476 
01477     // Is there another nested level of expression lists?  If not, bail out
01478     // now.  Otherwise we'll use the first one to size the next dimension
01479     // (after checking below that it has the same length as all of the
01480     // other ones.
01481     ExprList *nextList = dynamic_cast<ExprList *>(exprList->exprs[0]);
01482     if (nextList == NULL)
01483         return type;
01484 
01485     const Type *nextType = at->GetElementType();
01486     const ArrayType *nextArrayType = dynamic_cast<const ArrayType *>(nextType);
01487     if (nextArrayType != NULL && nextArrayType->GetElementCount() == 0) {
01488         // If the recursive call to SizeUnsizedArrays at the bottom of the
01489         // function is going to size an unsized dimension, make sure that
01490         // all of the sub-expression lists are the same length--i.e. issue
01491         // an error if we have something like
01492         // int x[][] = { { 1 }, { 1, 2, 3, 4 } };
01493         unsigned int nextSize = nextList->exprs.size();
01494         for (unsigned int i = 1; i < exprList->exprs.size(); ++i) {
01495             if (exprList->exprs[i] == NULL) {
01496                 // We should have seen an error earlier in this case.
01497                 Assert(m->errorCount > 0);
01498                 continue;
01499             }
01500 
01501             ExprList *el = dynamic_cast<ExprList *>(exprList->exprs[i]);
01502             if (el == NULL || el->exprs.size() != nextSize) {
01503                 Error(Union(exprList->exprs[0]->pos, exprList->exprs[i]->pos), 
01504                       "Inconsistent initializer expression list lengths "
01505                       "make it impossible to size unsized array dimensions.");
01506                 return NULL;
01507             }
01508         }
01509     }
01510 
01511     // Recursively call SizeUnsizedArrays() to get the child type for the
01512     // array that we were able to size here.
01513     return new ArrayType(SizeUnsizedArrays(at->GetElementType(), nextList),
01514                          at->GetElementCount());
01515 }

01654         Assert(m->errorCount > 0);

01684 VectorType::GetDIType(llvm::DIDescriptor scope) const {
01685     llvm::DIType eltType = base->GetDIType(scope);
01686     llvm::Value *sub = m->diBuilder->getOrCreateSubrange(0, numElements-1);

01698     // explicitly aligned to the machines natural vector alignment.
01699     uint64_t align = eltType.getAlignInBits();
01700     if (IsUniformType())
01701         align = 4 * g->target.nativeVectorWidth;
01702

01716 int
01717 VectorType::getVectorMemoryCount() const {
01718     if (base->IsVaryingType())
01719         return numElements;
01720     else if (base->IsUniformType()) {
01721         int nativeWidth = g->target.nativeVectorWidth;
01722         if (Type::Equal(base->GetAsUniformType(), AtomicType::UniformInt64) ||
01723             Type::Equal(base->GetAsUniformType(), AtomicType::UniformUInt64) ||
01724             Type::Equal(base->GetAsUniformType(), AtomicType::UniformDouble))
01725             // target.nativeVectorWidth should be in terms of 32-bit
01726             // values, so for the 64-bit guys, it takes half as many of
01727             // them to fill the native width
01728             nativeWidth /= 2;
01729         // and now round up the element count to be a multiple of
01730         // nativeWidth
01731         return (numElements + (nativeWidth - 1)) & ~(nativeWidth-1);
01732     }
01733     else if (base->IsSOAType()) {
01734         FATAL("VectorType SOA getVectorMemoryCount");
01735         return -1;
01736     }
01737     else {
01738         FATAL("Unexpected variability in VectorType::getVectorMemoryCount()");
01739         return -1;
01740     }

01954 llvm::DIType
01955 StructType::GetDIType(llvm::DIDescriptor scope) const {
01956     uint64_t currentSize = 0, align = 0;
01957 
01958     std::vector<llvm::Value *> elementLLVMTypes;
01959     // Walk through the elements of the struct; for each one figure out its
01960     // alignment and size, using that to figure out its offset w.r.t. the
01961     // start of the structure.
01962     for (unsigned int i = 0; i < elementTypes.size(); ++i) {
01963         llvm::DIType eltType = GetElementType(i)->GetDIType(scope);
01964         uint64_t eltAlign = eltType.getAlignInBits();
01965         uint64_t eltSize = eltType.getSizeInBits();
01966 
01967         // The alignment for the entire structure is the maximum of the
01968         // required alignments of its elements
01969         align = std::max(align, eltAlign);
01970 
01971         // Move the current size forward if needed so that the current
01972         // element starts at an offset that's the correct alignment.
01973         if (currentSize > 0 && (currentSize % eltAlign))
01974             currentSize += eltAlign - (currentSize % eltAlign);
01975         Assert((currentSize == 0) || (currentSize % eltAlign) == 0);
01976 
01977         llvm::DIFile diFile = elementPositions[i].GetDIFile();
01978         int line = elementPositions[i].first_line;
01979 #ifdef LLVM_2_9
01980         llvm::DIType fieldType = 
01981             m->diBuilder->createMemberType(elementNames[i], diFile, line,
01982                                            eltSize, eltAlign, currentSize, 0,
01983                                            eltType);
01984 #else
01985         llvm::DIType fieldType = 
01986             m->diBuilder->createMemberType(scope, elementNames[i], diFile, 
01987                                            line, eltSize, eltAlign, 
01988                                            currentSize, 0, eltType);
01989 #endif // LLVM_2_9
01990         elementLLVMTypes.push_back(fieldType);
01991 
01992         currentSize += eltSize;
01993     }
01994 
01995     // Round up the struct's entire size so that it's a multiple of the
01996     // required alignment that we figured out along the way...
01997     if (currentSize > 0 && (currentSize % align))
01998         currentSize += align - (currentSize % align);
01999 
02000 #ifdef LLVM_2_9
02001     llvm::DIArray elements = m->diBuilder->getOrCreateArray(&elementLLVMTypes[0], 
02002                                                             elementLLVMTypes.size());
02003 #else
02004     llvm::DIArray elements = m->diBuilder->getOrCreateArray(elementLLVMTypes);
02005 #endif
02006     llvm::DIFile diFile = pos.GetDIFile();
02007     return m->diBuilder->createStructType(scope, name, diFile, pos.first_line, currentSize, 
02008                                           align, 0, elements);
02009 }

02602         callTypes.push_back(LLVMTypes::Int32Type); // threadIndex

02609         callTypes = llvmArgTypes;
02610

02612         Assert(m->errorCount > 0);
02613         return NULL;
02614     }
02615

02665 /** Given an atomic or vector type, return a vector type of the given
02666     vecSize.  Issue an error if given a vector type that isn't already that

02671     const VectorType *vt = dynamic_cast<const VectorType *>(type);
02672     if (vt) {
02673         if (vt->GetElementCount() != vecSize) {
02674             Error(pos, "Implicit conversion between from vector type "

02692 
02693 const Type *
02694 Type::MoreGeneralType(const Type *t0, const Type *t1, SourcePos pos, const char *reason, 
02695                       bool forceVarying, int vecSize) {
02696     Assert(reason != NULL);
02697 
02698     // First, if one or both types are function types, convert them to
02699     // pointer to function types and then try again.
02700     if (dynamic_cast<const FunctionType *>(t0) ||
02701         dynamic_cast<const FunctionType *>(t1)) {
02702         if (dynamic_cast<const FunctionType *>(t0))
02703             t0 = PointerType::GetUniform(t0);
02704         if (dynamic_cast<const FunctionType *>(t1))
02705             t1 = PointerType::GetUniform(t1);
02706         return MoreGeneralType(t0, t1, pos, reason, forceVarying, vecSize);
02707     }
02708 
02709     // First, if we need to go varying, promote both of the types to be
02710     // varying.
02711     if (t0->IsVaryingType() || t1->IsVaryingType() || forceVarying) {
02712         t0 = t0->GetAsVaryingType();
02713         t1 = t1->GetAsVaryingType();
02714     }
02715 
02716     // And similarly, promote them both to vectors if the caller requested
02717     // a particular vector size
02718     if (vecSize > 0) {
02719         t0 = lVectorConvert(t0, pos, reason, vecSize);
02720         t1 = lVectorConvert(t1, pos, reason, vecSize);
02721         if (!t0 || !t1)
02722             return NULL;
02723     }
02724 
02725     // Are they both the same type?  If so, we're done, QED.
02726     if (Type::Equal(t0, t1)) 
02727         return t0;
02728     
02729     // If they're function types, it's hopeless if they didn't match in the
02730     // Type::Equal() call above.  Fail here so that we don't get into
02731     // trouble calling GetAsConstType()...
02732     if (dynamic_cast<const FunctionType *>(t0) ||
02733         dynamic_cast<const FunctionType *>(t1)) {
02734         Error(pos, "Incompatible function types \"%s\" and \"%s\" in %s.",
02735               t0->GetString().c_str(), t1->GetString().c_str(), reason);
02736         return NULL;
02737     }
02738 
02739     // Not the same types, but only a const/non-const difference?  Return
02740     // the non-const type as the more general one.
02741     if (Type::EqualIgnoringConst(t0, t1))
02742         return t0->GetAsNonConstType();
02743 
02744     const PointerType *pt0 = dynamic_cast<const PointerType *>(t0);
02745     const PointerType *pt1 = dynamic_cast<const PointerType *>(t1);
02746     if (pt0 != NULL && pt1 != NULL) {
02747         if (PointerType::IsVoidPointer(pt0))
02748             return pt1;
02749         else if (PointerType::IsVoidPointer(pt1))
02750             return pt0;
02751         else {
02752             Error(pos, "Conversion between incompatible pointer types \"%s\" "
02753                   "and \"%s\" isn't possible.", t0->GetString().c_str(),
02754                   t1->GetString().c_str());
02755             return NULL;
02756         }
02757     }
02758 
02759     const VectorType *vt0 = dynamic_cast<const VectorType *>(t0);
02760     const VectorType *vt1 = dynamic_cast<const VectorType *>(t1);
02761     if (vt0 && vt1) {
02762         // both are vectors; convert their base types and make a new vector
02763         // type, as long as their lengths match
02764         if (vt0->GetElementCount() != vt1->GetElementCount()) {
02765             Error(pos, "Implicit conversion between differently sized vector types "
02766                   "(%s, %s) for %s is not possible.", t0->GetString().c_str(),
02767                   t1->GetString().c_str(), reason);
02768             return NULL;
02769         }
02770         const Type *t = MoreGeneralType(vt0->GetElementType(), vt1->GetElementType(),
02771                                         pos, reason, forceVarying);
02772         if (!t) 
02773             return NULL;
02774 
02775         // The 'more general' version of the two vector element types must
02776         // be an AtomicType (that's all that vectors can hold...)
02777         const AtomicType *at = dynamic_cast<const AtomicType *>(t);
02778         Assert(at != NULL);
02779 
02780         return new VectorType(at, vt0->GetElementCount());
02781     }
02782     else if (vt0) {
02783         // If one type is a vector type but the other isn't, see if we can
02784         // promote the other one to a vector type.  This will fail and
02785         // return NULL if t1 is e.g. an array type and it's illegal to have
02786         // a vector of it..
02787         const Type *t = MoreGeneralType(vt0->GetElementType(), t1, pos, 
02788                                         reason, forceVarying);
02789         if (!t) 
02790             return NULL;
02791 
02792         const AtomicType *at = dynamic_cast<const AtomicType *>(t);
02793         Assert(at != NULL);
02794         return new VectorType(at, vt0->GetElementCount());
02795     }
02796     else if (vt1) {
02797         // As in the above case, see if we can promote t0 to make a vector
02798         // that matches vt1.
02799         const Type *t = MoreGeneralType(t0, vt1->GetElementType(), pos, 
02800                                         reason, forceVarying);
02801         if (!t) 
02802             return NULL;
02803 
02804         const AtomicType *at = dynamic_cast<const AtomicType *>(t);
02805         Assert(at != NULL);
02806         return new VectorType(at, vt1->GetElementCount());
02807     }
02808 
02809     // TODO: what do we need to do about references here, if anything??
02810 
02811     const AtomicType *at0 = dynamic_cast<const AtomicType *>(t0->GetReferenceTarget());
02812     const AtomicType *at1 = dynamic_cast<const AtomicType *>(t1->GetReferenceTarget());
02813 
02814     const EnumType *et0 = dynamic_cast<const EnumType *>(t0->GetReferenceTarget());
02815     const EnumType *et1 = dynamic_cast<const EnumType *>(t1->GetReferenceTarget());
02816     if (et0 != NULL && et1 != NULL) {
02817         // Two different enum types -> make them uint32s...
02818         Assert(et0->IsVaryingType() == et1->IsVaryingType());
02819         return et0->IsVaryingType() ? AtomicType::VaryingUInt32 :
02820                 AtomicType::UniformUInt32;
02821     }
02822     else if (et0 != NULL) {
02823         if (at1 != NULL)
02824             // Enum type and atomic type -> convert the enum to the atomic type
02825             // TODO: should we return uint32 here, unless the atomic type is
02826             // a 64-bit atomic type, in which case we return that?
02827             return at1;
02828         else {
02829             Error(pos, "Implicit conversion from enum type \"%s\" to "
02830                   "non-atomic type \"%s\" for %s not possible.",
02831                   t0->GetString().c_str(), t1->GetString().c_str(), reason);
02832             return NULL;
02833         }
02834     }
02835     else if (et1 != NULL) {
02836         if (at0 != NULL)
02837             // Enum type and atomic type; see TODO above here as well...
02838             return at0;
02839         else {
02840             Error(pos, "Implicit conversion from enum type \"%s\" to "
02841                   "non-atomic type \"%s\" for %s not possible.",
02842                   t1->GetString().c_str(), t0->GetString().c_str(), reason);
02843             return NULL;
02844         }
02845     }
02846 
02847     // Now all we can do is promote atomic types...
02848     if (at0 == NULL || at1 == NULL) {
02849         Assert(reason != NULL);
02850         Error(pos, "Implicit conversion from type \"%s\" to \"%s\" for %s not possible.",
02851               t0->GetString().c_str(), t1->GetString().c_str(), reason);
02852         return NULL;
02853     }
02854 
02855     // Finally, to determine which of the two atomic types is more general,
02856     // use the ordering of entries in the AtomicType::BasicType enumerator.
02857     return (int(at0->basicType) >= int(at1->basicType)) ? at0 : at1;
02858 }
