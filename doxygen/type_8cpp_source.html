00062         return false;

01145 const Type *
01146 ArrayType::SizeUnsizedArrays(const Type *type, Expr *initExpr) {
01147     const ArrayType *at = dynamic_cast<const ArrayType *>(type);
01148     if (at == NULL)
01149         return type;
01150 
01151     ExprList *exprList = dynamic_cast<ExprList *>(initExpr);
01152     if (exprList == NULL || exprList->exprs.size() == 0)
01153         return type;
01154 
01155     // If the current dimension is unsized, then size it according to the
01156     // length of the expression list
01157     if (at->GetElementCount() == 0)
01158         type = at->GetSizedArray(exprList->exprs.size());
01159 
01160     // Is there another nested level of expression lists?  If not, bail out
01161     // now.  Otherwise we'll use the first one to size the next dimension
01162     // (after checking below that it has the same length as all of the
01163     // other ones.
01164     ExprList *nextList = dynamic_cast<ExprList *>(exprList->exprs[0]);
01165     if (nextList == NULL)
01166         return type;
01167 
01168     const Type *nextType = at->GetElementType();
01169     const ArrayType *nextArrayType = 
01170         dynamic_cast<const ArrayType *>(nextType);
01171     if (nextArrayType != NULL && nextArrayType->GetElementCount() == 0) {
01172         // If the recursive call to SizeUnsizedArrays at the bottom of the
01173         // function is going to size an unsized dimension, make sure that
01174         // all of the sub-expression lists are the same length--i.e. issue
01175         // an error if we have something like
01176         // int x[][] = { { 1 }, { 1, 2, 3, 4 } };
01177         unsigned int nextSize = nextList->exprs.size();
01178         for (unsigned int i = 1; i < exprList->exprs.size(); ++i) {
01179             if (exprList->exprs[i] == NULL) {
01180                 // We should have seen an error earlier in this case.
01181                 Assert(m->errorCount > 0);
01182                 continue;
01183             }
01184 
01185             ExprList *el = dynamic_cast<ExprList *>(exprList->exprs[i]);
01186             if (el == NULL || el->exprs.size() != nextSize) {
01187                 Error(Union(exprList->exprs[0]->pos, exprList->exprs[i]->pos), 
01188                       "Inconsistent initializer expression list lengths "
01189                       "make it impossible to size unsized array dimensions.");
01190                 return NULL;
01191             }
01192         }
01193     }
01194 
01195     // Recursively call SizeUnsizedArrays() to get the child type for the
01196     // array that we were able to size here.
01197     return new ArrayType(SizeUnsizedArrays(at->GetElementType(), nextList),
01198                          exprList->exprs.size());
01199 }

01457 LLVM_TYPE_CONST llvm::Type *
01458 VectorType::LLVMType(llvm::LLVMContext *ctx) const {
01459     LLVM_TYPE_CONST llvm::Type *bt = base->LLVMType(ctx);
01460     if (!bt)
01461         return NULL;
01462 
01463     if (base->IsUniformType())
01464         // vectors of uniform types are laid out across LLVM vectors, with
01465         // the llvm vector size set to be a multiple of the machine's
01466         // natural vector size (e.g. 4 on SSE).  This is a roundabout way
01467         // of ensuring that LLVM lays them out into machine vector
01468         // registers so that e.g. if we want to add two uniform 4 float
01469         // vectors, that is turned into a single addps on SSE.
01470         return llvm::VectorType::get(bt, getVectorMemoryCount());
01471     else

01479 VectorType::GetDIType(llvm::DIDescriptor scope) const {

01482 #ifdef LLVM_2_9

01502 int
01503 VectorType::getVectorMemoryCount() const {
01504     if (base->IsVaryingType())
01505         return numElements;
01506     else {
01507         int nativeWidth = g->target.nativeVectorWidth;
01508         if (base->GetAsUniformType() == AtomicType::UniformInt64 ||
01509             base->GetAsUniformType() == AtomicType::UniformUInt64 ||
01510             base->GetAsUniformType() == AtomicType::UniformDouble)
01511             // target.nativeVectorWidth should be in terms of 32-bit
01512             // values, so for the 64-bit guys, it takes half as many of
01513             // them to fill the native width
01514             nativeWidth /= 2;
01515         // and now round up the element count to be a multiple of
01516         // nativeWidth
01517         return (numElements + (nativeWidth - 1)) & ~(nativeWidth-1);
01518     }

01700 
01701 llvm::DIType
01702 StructType::GetDIType(llvm::DIDescriptor scope) const {
01703     uint64_t currentSize = 0, align = 0;
01704 
01705     std::vector<llvm::Value *> elementLLVMTypes;
01706     // Walk through the elements of the struct; for each one figure out its
01707     // alignment and size, using that to figure out its offset w.r.t. the
01708     // start of the structure.
01709     for (unsigned int i = 0; i < elementTypes.size(); ++i) {
01710         llvm::DIType eltType = GetElementType(i)->GetDIType(scope);
01711         uint64_t eltAlign = eltType.getAlignInBits();
01712         uint64_t eltSize = eltType.getSizeInBits();
01713 
01714         // The alignment for the entire structure is the maximum of the
01715         // required alignments of its elements
01716         align = std::max(align, eltAlign);
01717 
01718         // Move the current size forward if needed so that the current
01719         // element starts at an offset that's the correct alignment.
01720         if (currentSize > 0 && (currentSize % eltAlign))
01721             currentSize += eltAlign - (currentSize % eltAlign);
01722         Assert((currentSize == 0) || (currentSize % eltAlign) == 0);
01723 
01724         llvm::DIFile diFile = elementPositions[i].GetDIFile();
01725         int line = elementPositions[i].first_line;
01726 #ifdef LLVM_2_9
01727         llvm::DIType fieldType = 
01728             m->diBuilder->createMemberType(elementNames[i], diFile, line,
01729                                            eltSize, eltAlign, currentSize, 0,
01730                                            eltType);
01731 #else
01732         llvm::DIType fieldType = 
01733             m->diBuilder->createMemberType(scope, elementNames[i], diFile, 
01734                                            line, eltSize, eltAlign, 
01735                                            currentSize, 0, eltType);
01736 #endif // LLVM_2_9
01737         elementLLVMTypes.push_back(fieldType);
01738 
01739         currentSize += eltSize;
01740     }
01741 
01742     // Round up the struct's entire size so that it's a multiple of the
01743     // required alignment that we figured out along the way...
01744     if (currentSize > 0 && (currentSize % align))
01745         currentSize += align - (currentSize % align);
01746 
01747 #ifdef LLVM_2_9
01748     llvm::DIArray elements = m->diBuilder->getOrCreateArray(&elementLLVMTypes[0], 
01749                                                             elementLLVMTypes.size());
01750 #else
01751     llvm::DIArray elements = m->diBuilder->getOrCreateArray(elementLLVMTypes);
01752 #endif
01753     llvm::DIFile diFile = pos.GetDIFile();
01754     return m->diBuilder->createStructType(scope, name, diFile, pos.first_line, currentSize, 
01755                                           align, 0, elements);
01756 }

02252         t0 = t0->GetAsVaryingType();

02257     // a particular vector size

02265     // Are they both the same type?  If so, we're done, QED.

02280     // the non-const type as the more general one.

02285     const PointerType *pt1 = dynamic_cast<const PointerType *>(t1);

02300     const VectorType *vt1 = dynamic_cast<const VectorType *>(t1);

02303         // type, as long as their lengths match

02316         // be an AtomicType (that's all that vectors can hold...)

02324         // promote the other one to a vector type.  This will fail and
02325         // return NULL if t1 is e.g. an array type and it's illegal to have
02326         // a vector of it..

02328                                         reason, forceVarying);
02329         if (!t)

02333         Assert(at != NULL);

02338         // that matches vt1.

02340                                         reason, forceVarying);
