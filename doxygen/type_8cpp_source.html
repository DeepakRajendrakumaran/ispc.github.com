01052 PointerType::LLVMType(llvm::LLVMContext *ctx) const {
01053     if (baseType == NULL) {
01054         Assert(m->errorCount > 0);
01055         return NULL;
01056     }
01057 
01058     if (isSlice) {
01059         llvm::Type *types[2];
01060         types[0] = GetAsNonSlice()->LLVMType(ctx);
01061         
01062         switch (variability.type) {
01063         case Variability::Uniform:
01064             types[1] = LLVMTypes::Int32Type;
01065             break;
01066         case Variability::Varying:
01067             types[1] = LLVMTypes::Int32VectorType;
01068             break;
01069         case Variability::SOA:
01070             types[1] = llvm::ArrayType::get(LLVMTypes::Int32Type,
01071                                             variability.soaWidth);
01072             break;
01073         default:
01074             FATAL("unexpected variability for slice pointer in "
01075                   "PointerType::LLVMType");

01079             llvm::ArrayRef<llvm::Type *>(types, 2);
01080         return llvm::StructType::get(*g->ctx, typesArrayRef);
01081     }
01082 
01083     switch (variability.type) {
01084     case Variability::Uniform: {
01085         llvm::Type *ptype = NULL;
01086         const FunctionType *ftype = dynamic_cast<const FunctionType *>(baseType);
01087         if (ftype != NULL) 
01088             // Get the type of the function variant that takes the mask as the

01090             // exported functions.
01091             ptype = llvm::PointerType::get(ftype->LLVMFunctionType(ctx, true), 0);
01092         else {
01093             if (baseType == AtomicType::Void)
01094                 ptype = LLVMTypes::VoidPointerType;
01095             else
01096                 ptype = llvm::PointerType::get(baseType->LLVMType(ctx), 0);
01097         }

01100     case Variability::Varying:
01101         // always the same, since we currently use int vectors for varying
01102         // pointers

01104     case Variability::SOA: {
01105         ArrayType at(GetAsUniformType(), variability.soaWidth);

01426 const Type *
01427 ArrayType::SizeUnsizedArrays(const Type *type, Expr *initExpr) {
01428     const ArrayType *at = dynamic_cast<const ArrayType *>(type);
01429     if (at == NULL)
01430         return type;
01431 
01432     ExprList *exprList = dynamic_cast<ExprList *>(initExpr);
01433     if (exprList == NULL || exprList->exprs.size() == 0)
01434         return type;
01435 
01436     // If the current dimension is unsized, then size it according to the
01437     // length of the expression list
01438     if (at->GetElementCount() == 0) {
01439         type = at->GetSizedArray(exprList->exprs.size());
01440         at = dynamic_cast<const ArrayType *>(type);
01441     }
01442 
01443     // Is there another nested level of expression lists?  If not, bail out
01444     // now.  Otherwise we'll use the first one to size the next dimension
01445     // (after checking below that it has the same length as all of the
01446     // other ones.
01447     ExprList *nextList = dynamic_cast<ExprList *>(exprList->exprs[0]);
01448     if (nextList == NULL)
01449         return type;
01450 
01451     const Type *nextType = at->GetElementType();
01452     const ArrayType *nextArrayType = dynamic_cast<const ArrayType *>(nextType);
01453     if (nextArrayType != NULL && nextArrayType->GetElementCount() == 0) {
01454         // If the recursive call to SizeUnsizedArrays at the bottom of the
01455         // function is going to size an unsized dimension, make sure that
01456         // all of the sub-expression lists are the same length--i.e. issue
01457         // an error if we have something like
01458         // int x[][] = { { 1 }, { 1, 2, 3, 4 } };
01459         unsigned int nextSize = nextList->exprs.size();
01460         for (unsigned int i = 1; i < exprList->exprs.size(); ++i) {
01461             if (exprList->exprs[i] == NULL) {
01462                 // We should have seen an error earlier in this case.
01463                 Assert(m->errorCount > 0);
01464                 continue;
01465             }
01466 
01467             ExprList *el = dynamic_cast<ExprList *>(exprList->exprs[i]);
01468             if (el == NULL || el->exprs.size() != nextSize) {
01469                 Error(Union(exprList->exprs[0]->pos, exprList->exprs[i]->pos), 
01470                       "Inconsistent initializer expression list lengths "
01471                       "make it impossible to size unsized array dimensions.");
01472                 return NULL;
01473             }
01474         }
01475     }
01476 
01477     // Recursively call SizeUnsizedArrays() to get the child type for the
01478     // array that we were able to size here.
01479     return new ArrayType(SizeUnsizedArrays(at->GetElementType(), nextList),
01480                          at->GetElementCount());
01481 }

01618 VectorType::LLVMType(llvm::LLVMContext *ctx) const {
01619     if (base == NULL) {
01620         Assert(m->errorCount > 0);
01621         return NULL;
01622     }

01677 int
01678 VectorType::getVectorMemoryCount() const {
01679     if (base->IsVaryingType())
01680         return numElements;
01681     else if (base->IsUniformType()) {
01682         int nativeWidth = g->target.nativeVectorWidth;
01683         if (Type::Equal(base->GetAsUniformType(), AtomicType::UniformInt64) ||
01684             Type::Equal(base->GetAsUniformType(), AtomicType::UniformUInt64) ||
01685             Type::Equal(base->GetAsUniformType(), AtomicType::UniformDouble))
01686             // target.nativeVectorWidth should be in terms of 32-bit
01687             // values, so for the 64-bit guys, it takes half as many of
01688             // them to fill the native width
01689             nativeWidth /= 2;
01690         // and now round up the element count to be a multiple of
01691         // nativeWidth
01692         return (numElements + (nativeWidth - 1)) & ~(nativeWidth-1);
01693     }
01694     else if (base->IsSOAType()) {
01695         FATAL("VectorType SOA getVectorMemoryCount");
01696         return -1;
01697     }
01698     else {
01699         FATAL("Unexpected variability in VectorType::getVectorMemoryCount()");
01700         return -1;
01701     }

01707 
01708 // We maintain a map from struct names to LLVM struct types so that we can
01709 // uniquely get the llvm::StructType * for a given ispc struct type.  Note
01710 // that we need to mangle the name a bit so that we can e.g. differentiate
01711 // between the uniform and varying variants of a given struct type.  This
01712 // is handled by lMangleStructName() below.
01713 static std::map<std::string, llvm::StructType *> lStructTypeMap;
01714 
01715 /** Using a struct's name, its variability, and the vector width for the
01716     current compilation target, this function generates a string that
01717     encodes that full structure type, for use in the lStructTypeMap.  Note
01718     that the vector width is needed in order to differentiate between
01719     'varying' structs with different compilation targets, which have
01720     different memory layouts...
01721  */
01722 static std::string
01723 lMangleStructName(const std::string &name, Variability variability) {
01724     char buf[32];
01725     std::string n;
01726 
01727     // Encode vector width
01728     sprintf(buf, "v%d", g->target.vectorWidth);
01729     n += buf;
01730 
01731     // Variability
01732     switch (variability.type) {
01733     case Variability::Uniform:
01734         n += "_uniform_";
01735         break;
01736     case Variability::Varying:
01737         n += "_varying_";
01738         break;
01739     case Variability::SOA:
01740         sprintf(buf, "_soa%d_", variability.soaWidth);
01741         n += buf;
01742         break;
01743     default:
01744         FATAL("Unexpected varaibility in lMangleStructName()");
01745     }
01746     
01747     // And stuff the name at the end....
01748     n += name;
01749     return n;
01750 }

01752         
01753 StructType::StructType(const std::string &n, const std::vector<const Type *> &elts, 
01754                        const std::vector<std::string> &en,
01755                        const std::vector<SourcePos> &ep,
01756                        bool ic, Variability v, SourcePos p) 
01757     : name(n), elementTypes(elts), elementNames(en), elementPositions(ep),
01758       variability(v), isConst(ic), pos(p) {
01759     if (variability != Variability::Unbound) {
01760         // For structs with non-unbound variability, we'll create the
01761         // correspoing LLVM struct type now, if one hasn't been made
01762         // already.
01763 
01764         // Create a unique anonymous struct name if we have an anonymous
01765         // struct (name == ""), or if we are creating a derived type from
01766         // an anonymous struct (e.g. the varying variant--name == '$').
01767         if (name == "" || name[0] == '$') {
01768             char buf[16];
01769             static int count = 0;
01770             sprintf(buf, "$anon%d", count);
01771             name = buf;
01772             ++count;
01773         }
01774 
01775         // If a non-opaque LLVM struct for this type has already been
01776         // created, we're done.  For an opaque struct type, we'll override
01777         // the old definition now that we have a full definition.
01778         std::string mname = lMangleStructName(name, variability);
01779         if (lStructTypeMap.find(mname) != lStructTypeMap.end() &&
01780             lStructTypeMap[mname]->isOpaque() == false)
01781             return;
01782 
01783         // Actually make the LLVM struct
01784         std::vector<llvm::Type *> elementTypes;
01785         for (int i = 0; i < GetElementCount(); ++i) {
01786             const Type *type = GetElementType(i);
01787             if (type == NULL) {
01788                 Assert(m->errorCount > 0);
01789                 return;
01790             }
01791             elementTypes.push_back(type->LLVMType(g->ctx));
01792         }
01793 
01794         if (lStructTypeMap.find(mname) == lStructTypeMap.end()) {
01795             // New struct definition
01796             llvm::StructType *st =
01797                 llvm::StructType::create(*g->ctx, elementTypes, mname);
01798             lStructTypeMap[mname] = st;
01799         }
01800         else {
01801             // Definition for what was before just a declaration
01802             lStructTypeMap[mname]->setBody(elementTypes);
01803         }
01804     }
01805 }

02012 llvm::DIType
02013 StructType::GetDIType(llvm::DIDescriptor scope) const {
02014     uint64_t currentSize = 0, align = 0;
02015 
02016     std::vector<llvm::Value *> elementLLVMTypes;
02017     // Walk through the elements of the struct; for each one figure out its
02018     // alignment and size, using that to figure out its offset w.r.t. the
02019     // start of the structure.
02020     for (unsigned int i = 0; i < elementTypes.size(); ++i) {
02021         llvm::DIType eltType = GetElementType(i)->GetDIType(scope);
02022         uint64_t eltAlign = eltType.getAlignInBits();
02023         uint64_t eltSize = eltType.getSizeInBits();
02024         Assert(eltAlign != 0);
02025 
02026         // The alignment for the entire structure is the maximum of the
02027         // required alignments of its elements
02028         align = std::max(align, eltAlign);
02029 
02030         // Move the current size forward if needed so that the current
02031         // element starts at an offset that's the correct alignment.
02032         if (currentSize > 0 && (currentSize % eltAlign))
02033             currentSize += eltAlign - (currentSize % eltAlign);
02034         Assert((currentSize == 0) || (currentSize % eltAlign) == 0);
02035 
02036         llvm::DIFile diFile = elementPositions[i].GetDIFile();
02037         int line = elementPositions[i].first_line;
02038         llvm::DIType fieldType = 
02039             m->diBuilder->createMemberType(scope, elementNames[i], diFile, 
02040                                            line, eltSize, eltAlign, 
02041                                            currentSize, 0, eltType);
02042         elementLLVMTypes.push_back(fieldType);
02043 
02044         currentSize += eltSize;
02045     }
02046 
02047     // Round up the struct's entire size so that it's a multiple of the
02048     // required alignment that we figured out along the way...
02049     if (currentSize > 0 && (currentSize % align))
02050         currentSize += align - (currentSize % align);
02051 
02052     llvm::DIArray elements = m->diBuilder->getOrCreateArray(elementLLVMTypes);
02053     llvm::DIFile diFile = pos.GetDIFile();
02054     return m->diBuilder->createStructType(scope, name, diFile, pos.first_line, currentSize, 
02055                                           align, 0, elements);
02056 }

02791 llvm::FunctionType *
02792 FunctionType::LLVMFunctionType(llvm::LLVMContext *ctx, bool includeMask) const {
02793     if (isTask == true) 
02794         Assert(includeMask == true);
02795 
02796     // Get the LLVM Type *s for the function arguments
02797     std::vector<llvm::Type *> llvmArgTypes;
02798     for (unsigned int i = 0; i < paramTypes.size(); ++i) {
02799         if (paramTypes[i] == NULL) {
02800             Assert(m->errorCount > 0);
02801             return NULL;
02802         }
02803         Assert(Type::Equal(paramTypes[i], AtomicType::Void) == false);
02804 
02805         llvm::Type *t = paramTypes[i]->LLVMType(ctx);
02806         if (t == NULL) {
02807             Assert(m->errorCount > 0);
02808             return NULL;
02809         }
02810         llvmArgTypes.push_back(t);
02811     }
02812 
02813     // And add the function mask, if asked for
02814     if (includeMask)
02815         llvmArgTypes.push_back(LLVMTypes::MaskType);
02816 
02817     std::vector<llvm::Type *> callTypes;
02818     if (isTask) {
02819         // Tasks take three arguments: a pointer to a struct that holds the
02820         // actual task arguments, the thread index, and the total number of
02821         // threads the tasks system has running.  (Task arguments are
02822         // marshalled in a struct so that it's easy to allocate space to
02823         // hold them until the task actually runs.)
02824         llvm::Type *st = llvm::StructType::get(*ctx, llvmArgTypes);
02825         callTypes.push_back(llvm::PointerType::getUnqual(st));
02826         callTypes.push_back(LLVMTypes::Int32Type); // threadIndex
02827         callTypes.push_back(LLVMTypes::Int32Type); // threadCount
02828         callTypes.push_back(LLVMTypes::Int32Type); // taskIndex
02829         callTypes.push_back(LLVMTypes::Int32Type); // taskCount
02830     }
02831     else
02832         // Otherwise we already have the types of the arguments 
02833         callTypes = llvmArgTypes;
02834 
02835     if (returnType == NULL) {
02836         Assert(m->errorCount > 0);
02837         return NULL;
02838     }
02839 
02840     return llvm::FunctionType::get(returnType->LLVMType(g->ctx), callTypes, false);
02841 }

02917 const Type *
02918 Type::MoreGeneralType(const Type *t0, const Type *t1, SourcePos pos, const char *reason, 
02919                       bool forceVarying, int vecSize) {
02920     Assert(reason != NULL);
02921 
02922     // First, if one or both types are function types, convert them to
02923     // pointer to function types and then try again.
02924     if (dynamic_cast<const FunctionType *>(t0) ||
02925         dynamic_cast<const FunctionType *>(t1)) {
02926         if (dynamic_cast<const FunctionType *>(t0))
02927             t0 = PointerType::GetUniform(t0);
02928         if (dynamic_cast<const FunctionType *>(t1))
02929             t1 = PointerType::GetUniform(t1);
02930         return MoreGeneralType(t0, t1, pos, reason, forceVarying, vecSize);
02931     }
02932 
02933     // First, if we need to go varying, promote both of the types to be
02934     // varying.

03058     }

03061             // Enum type and atomic type; see TODO above here as well...
03062             return at0;
03063         else {

03065                   "non-atomic type \"%s\" for %s not possible.",

03101         if (dynamic_cast<const FunctionType *>(b) == NULL)
03102             b = b->GetAsNonConstType();

03104     else if (a->IsConstType() != b->IsConstType())

03108     const AtomicType *atb = dynamic_cast<const AtomicType *>(b);
03109     if (ata != NULL && atb != NULL) {

03111                 (ata->GetVariability() == atb->GetVariability()));
03112     }
03113 
03114     // For all of the other types, we need to see if we have the same two
03115     // general types.  If so, then we dig into the details of the type and
03116     // see if all of the relevant bits are equal...

03118     const EnumType *etb = dynamic_cast<const EnumType *>(b);
03119     if (eta != NULL && etb != NULL)
03120         // Kind of goofy, but this sufficies to check

03122                 eta->GetVariability() == etb->GetVariability());

03135                 lCheckTypeEquality(vta->GetElementType(), vtb->GetElementType(),
03136                                    ignoreConst));

03138     const StructType *sta = dynamic_cast<const StructType *>(a);

03142     const UndefinedStructType *ustb =
03143         dynamic_cast<const UndefinedStructType *>(b);
03144     if ((sta != NULL || usta != NULL) && (stb != NULL || ustb != NULL)) {
03145         // Report both defuned and undefined structs as equal if their

03147         if (a->GetVariability() != b->GetVariability())
03148             return false;
03149

03151         std::string nameb = stb ? stb->GetStructName() : ustb->GetStructName();

03156     const PointerType *ptb = dynamic_cast<const PointerType *>(b);

03165     const ReferenceType *rtb = dynamic_cast<const ReferenceType *>(b);

03167         return (lCheckTypeEquality(rta->GetReferenceTarget(),

03169 
03170     const FunctionType *fta = dynamic_cast<const FunctionType *>(a);
03171     const FunctionType *ftb = dynamic_cast<const FunctionType *>(b);
03172     if (fta != NULL && ftb != NULL) {
03173         // Both the return types and all of the argument types must match
03174         // for function types to match
03175         if (!lCheckTypeEquality(fta->GetReturnType(), ftb->GetReturnType(), 
03176                                 ignoreConst))

03178 
03179         if (fta->isTask != ftb->isTask ||
03180             fta->isExported != ftb->isExported ||
