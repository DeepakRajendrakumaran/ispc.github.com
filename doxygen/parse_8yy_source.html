01798 lYYTNameErr (char *yyres, const char *yystr)
01799 {
01800   extern std::map<std::string, std::string> tokenNameRemap;
01801   Assert(tokenNameRemap.size() > 0);
01802   if (tokenNameRemap.find(yystr) != tokenNameRemap.end()) {
01803       std::string n = tokenNameRemap[yystr];
01804       if (yyres == NULL)
01805           return n.size();
01806       else
01807           return yystpcpy(yyres, n.c_str()) - yyres;
01808   }
01809   
01810   if (*yystr == '"')
01811     {
01812       YYSIZE_T yyn = 0;
01813       char const *yyp = yystr;
01814 
01815       for (;;)
01816     switch (*++yyp)
01817       {
01818       case '\'':
01819       case ',':
01820         goto do_not_strip_quotes;
01821 
01822       case '\\':
01823         if (*++yyp != '\\')
01824           goto do_not_strip_quotes;
01825         /* Fall through.  */
01826       default:
01827         if (yyres)
01828           yyres[yyn] = *yyp;
01829         yyn++;
01830         break;
01831 
01832       case '"':
01833         if (yyres)
01834           yyres[yyn] = '\0';
01835         return yyn;
01836       }
01837     do_not_strip_quotes: ;
01838     }
01839 
01840   if (! yyres)
01841     return yystrlen (yystr);
01842 
01843   return yystpcpy (yyres, yystr) - yyres;
