01702     in the source file will already have ConstExpr * set; we just need
01703     to set the values for the others here.

01707                            const EnumType *enumType) {

01712        zero and goes up with each successive enumerant.  If any of them
01713        has a value specified, then nextVal is ignored for that one and is
01714        set to one plus that one's value for the default value for the next
01715        one. */

01717 
01718     for (unsigned int i = 0; i < enums.size(); ++i) {
01719         enums[i]->type = enumType;

01730                the actual enum type here and optimize it, which will have
01731                us end up with a ConstExpr with the desired EnumType... */

01733                                               false, enums[i]->pos);

01736             assert(enums[i]->constValue != NULL);
