00472 /** Conservative test to see if two llvm::Values are equal.  There are
00473     (potentially many) cases where the two values actually are equal but
00474     this will return false.  However, if it does return true, the two
00475     vectors definitely are equal.
00476 
00477     @todo This seems to catch all of the cases we currently need it for in
00478     practice, but it's be nice to make it a little more robust/general.  In
00479     general, though, a little something called the halting problem means we
00480     won't get all of them.
00481 */
00482 static bool
00483 lValuesAreEqual(llvm::Value *v0, llvm::Value *v1, 
00484                 std::vector<llvm::PHINode *> &seenPhi0,
00485                 std::vector<llvm::PHINode *> &seenPhi1) {
00486     // Thanks to the fact that LLVM hashes and returns the same pointer for
00487     // constants (of all sorts, even constant expressions), this first test
00488     // actually catches a lot of cases.  LLVM's SSA form also helps a lot
00489     // with this..
00490     if (v0 == v1)
00491         return true;
00492 
00493     Assert(seenPhi0.size() == seenPhi1.size());
00494     for (unsigned int i = 0; i < seenPhi0.size(); ++i)
00495         if (v0 == seenPhi0[i] && v1 == seenPhi1[i])
00496             return true;
00497 
00498     llvm::BinaryOperator *bo0 = llvm::dyn_cast<llvm::BinaryOperator>(v0);
00499     llvm::BinaryOperator *bo1 = llvm::dyn_cast<llvm::BinaryOperator>(v1);
00500     if (bo0 != NULL && bo1 != NULL) {
00501         if (bo0->getOpcode() != bo1->getOpcode())
00502             return false;
00503         return (lValuesAreEqual(bo0->getOperand(0), bo1->getOperand(0),
00504                                 seenPhi0, seenPhi1) &&
00505                 lValuesAreEqual(bo0->getOperand(1), bo1->getOperand(1),
00506                                 seenPhi0, seenPhi1));
00507     }
00508 
00509     llvm::PHINode *phi0 = llvm::dyn_cast<llvm::PHINode>(v0);
00510     llvm::PHINode *phi1 = llvm::dyn_cast<llvm::PHINode>(v1);
00511     if (phi0 != NULL && phi1 != NULL) {
00512         if (phi0->getNumIncomingValues() != phi1->getNumIncomingValues())
00513             return false;
00514 
00515         seenPhi0.push_back(phi0);
00516         seenPhi1.push_back(phi1);
00517 
00518         unsigned int numIncoming = phi0->getNumIncomingValues();
00519         // Check all of the incoming values: if all of them are all equal,
00520         // then we're good.
00521         bool anyFailure = false;
00522         for (unsigned int i = 0; i < numIncoming; ++i) {
00523             Assert(phi0->getIncomingBlock(i) == phi1->getIncomingBlock(i));
00524             if (!lValuesAreEqual(phi0->getIncomingValue(i), 
00525                                  phi1->getIncomingValue(i), seenPhi0, seenPhi1)) {
00526                 anyFailure = true;
00527                 break;
00528             }
00529         }
00530 
00531         seenPhi0.pop_back();
00532         seenPhi1.pop_back();
00533 
00534         return !anyFailure;
00535     }
00536 
00537     return false;
00538 }
