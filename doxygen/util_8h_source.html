00049 /** Rounds up the given value to the next power of two, if it isn't a power
00050     of two already. */
00051 inline uint32_t RoundUpPow2(uint32_t v) {
00052     v--;
00053     v |= v >> 1;    
00054     v |= v >> 2;
00055     v |= v >> 4;    
00056     v |= v >> 8;
00057     v |= v >> 16;
00058     return v+1;
00059 }
