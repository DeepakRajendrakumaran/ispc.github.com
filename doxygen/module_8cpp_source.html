00728 /** Small structure used in representing dependency graphs of structures
00729     (i.e. given a StructType, which other structure types does it have as
00730     elements).
00731  */ 
00732 struct StructDAGNode {
00733     StructDAGNode()
00734         : visited(false) { }
00735 
00736     bool visited;
00737     std::vector<const StructType *> dependents;
00738 };
00739 
00740 
00741 /** Visit a node for the topological sort.
00742  */
00743 static void
00744 lVisitNode(const StructType *structType, 
00745            std::map<const StructType *, StructDAGNode *> &structToNode,
00746            std::vector<const StructType *> &sortedTypes) {
00747     Assert(structToNode.find(structType) != structToNode.end());
00748     // Get the node that encodes the structs that this one is immediately
00749     // dependent on.
00750     StructDAGNode *node = structToNode[structType];
00751     if (node->visited)
00752         return;
00753 
00754     node->visited = true;
00755     // Depth-first traversal: visit all of the dependent nodes...
00756     for (unsigned int i = 0; i < node->dependents.size(); ++i)
00757         lVisitNode(node->dependents[i], structToNode, sortedTypes);
00758     // ...and then add this one to the sorted list
00759     sortedTypes.push_back(structType);
00760 }
00761            
00762 
00763 /** Given a set of structures that we want to print C declarations of in a
00764     header file, order them so that any struct that is used as a member
00765     variable in another struct is printed before the struct that uses it
00766     and then print them to the given file.
00767  */
00768 static void
00769 lEmitStructDecls(std::vector<const StructType *> &structTypes, FILE *file) {
00770     // First, build a DAG among the struct types where there is an edge
00771     // from node A to node B if struct type A depends on struct type B
00772 
00773     // Records the struct types that have incoming edges in the
00774     // DAG--i.e. the ones that one or more other struct types depend on
00775     std::set<const StructType *> hasIncomingEdges;
00776     // Records the mapping between struct type pointers and the
00777     // StructDagNode structures
00778     std::map<const StructType *, StructDAGNode *> structToNode;
00779     for (unsigned int i = 0; i < structTypes.size(); ++i) {
00780         // For each struct type, create its DAG node and record the
00781         // relationship between it and its node
00782         const StructType *st = structTypes[i];
00783         StructDAGNode *node = new StructDAGNode;
00784         structToNode[st] = node;
00785 
00786         for (int j = 0; j < st->GetElementCount(); ++j) {
00787             const StructType *elementStructType = 
00788                 dynamic_cast<const StructType *>(st->GetElementType(j));
00789             // If this element is a struct type and we haven't already
00790             // processed it for the current struct type, then upate th
00791             // dependencies and record that this element type has other
00792             // struct types that depend on it.
00793             if (elementStructType != NULL &&
00794                 (std::find(node->dependents.begin(), node->dependents.end(), 
00795                            elementStructType) == node->dependents.end())) {
00796                 node->dependents.push_back(elementStructType);
00797                 hasIncomingEdges.insert(elementStructType);
00798             }
00799         }
00800     }
00801 
00802     // Perform a topological sort of the struct types.  Kick it off by
00803     // visiting nodes with no incoming edges; i.e. the struct types that no
00804     // other struct types depend on.
00805     std::vector<const StructType *> sortedTypes;
00806     for (unsigned int i = 0; i < structTypes.size(); ++i) {
00807         const StructType *structType = structTypes[i];
00808         if (hasIncomingEdges.find(structType) == hasIncomingEdges.end())
00809             lVisitNode(structType, structToNode, sortedTypes);
00810     }
00811     Assert(sortedTypes.size() == structTypes.size());
00812 
00813     // And finally we can emit the struct declarations by going through the
00814     // sorted ones in order.
00815     for (unsigned int i = 0; i < sortedTypes.size(); ++i) {
00816         const StructType *st = sortedTypes[i];
00817         fprintf(file, "struct %s {\n", st->GetStructName().c_str());
00818         for (int j = 0; j < st->GetElementCount(); ++j) {
00819             const Type *type = st->GetElementType(j)->GetAsNonConstType();
00820             std::string d = type->GetCDeclaration(st->GetElementName(j));
00821             fprintf(file, "    %s;\n", d.c_str());
00822         }
00823         fprintf(file, "};\n\n");
00824     }
00825 }

00902 /** Add the given type to the vector, if that type isn't already in there.
00903  */
00904 template <typename T> static void
00905 lAddTypeIfNew(const Type *type, std::vector<const T *> *exportedTypes) {
00906     type = type->GetAsNonConstType();
00907 
00908     // Linear search, so this ends up being n^2.  It's unlikely this will
00909     // matter in practice, though.
00910     for (unsigned int i = 0; i < exportedTypes->size(); ++i)
00911         if (Type::Equal((*exportedTypes)[i], type))
00912             return;
00913 
00914     const T *castType = dynamic_cast<const T *>(type);
00915     Assert(castType != NULL);
00916     exportedTypes->push_back(castType);
00917 }

00920 /** Given an arbitrary type that appears in the app/ispc interface, add it
00921     to an appropriate vector if it is a struct, enum, or short vector type.
00922     Then, if it's a struct, recursively process its members to do the same.
00923  */
00924 static void
00925 lGetExportedTypes(const Type *type, 
00926                   std::vector<const StructType *> *exportedStructTypes,
00927                   std::vector<const EnumType *> *exportedEnumTypes,
00928                   std::vector<const VectorType *> *exportedVectorTypes) {
00929     const ArrayType *arrayType = dynamic_cast<const ArrayType *>(type);
00930     const StructType *structType = dynamic_cast<const StructType *>(type);
00931 
00932     if (dynamic_cast<const ReferenceType *>(type) != NULL)
00933         lGetExportedTypes(type->GetReferenceTarget(), exportedStructTypes, 
00934                           exportedEnumTypes, exportedVectorTypes);
00935     else if (dynamic_cast<const PointerType *>(type) != NULL)
00936         lGetExportedTypes(type->GetBaseType(), exportedStructTypes,
00937                           exportedEnumTypes, exportedVectorTypes);
00938     else if (arrayType != NULL)
00939         lGetExportedTypes(arrayType->GetElementType(), exportedStructTypes, 
00940                           exportedEnumTypes, exportedVectorTypes);
00941     else if (structType != NULL) {
00942         lAddTypeIfNew(type, exportedStructTypes);
00943         for (int i = 0; i < structType->GetElementCount(); ++i)
00944             lGetExportedTypes(structType->GetElementType(i), exportedStructTypes,
00945                               exportedEnumTypes, exportedVectorTypes);
00946     }
00947     else if (dynamic_cast<const EnumType *>(type) != NULL)
00948         lAddTypeIfNew(type, exportedEnumTypes);
00949     else if (dynamic_cast<const VectorType *>(type) != NULL)
00950         lAddTypeIfNew(type, exportedVectorTypes);
00951     else
00952         Assert(dynamic_cast<const AtomicType *>(type) != NULL);
00953 }

01042         if (isdigit(*p))

01051

01405     This function checks to see which vector ISAs the system the 
01406     code is running on supports and calls out to the best available
01407     variant that was generated at compile time.

01409     @param module      Module in which to create the dispatch function.

01414                              variable, which holds a value of the Target::ISA
01415                              enumerant giving the most capable ISA that the
01416                              system supports.

01418                        dispatch function

01421 static void
01422 lCreateDispatchFunction(llvm::Module *module, llvm::Function *setISAFunc,
01423                         llvm::Value *systemBestISAPtr, const std::string &name, 
01424                         FunctionTargetVariants &funcs) {
01425     // The llvm::Function pointers in funcs are pointers to functions in
01426     // different llvm::Modules, so we can't call them directly.  Therefore,
01427     // we'll start by generating an 'extern' declaration of each one that
01428     // we have in the current module so that we can then call out to that.
01429     llvm::Function *targetFuncs[Target::NUM_ISAS];
01430     LLVM_TYPE_CONST llvm::FunctionType *ftype = NULL;
01431 
01432     for (int i = 0; i < Target::NUM_ISAS; ++i) {
01433         if (funcs.func[i] == NULL) {
01434             targetFuncs[i] = NULL;
01435             continue;
01436         }
01437 
01438         // Grab the type of the function as well.
01439         if (ftype != NULL)
01440             Assert(ftype == funcs.func[i]->getFunctionType());
01441         else
01442             ftype = funcs.func[i]->getFunctionType();
01443 
01444         targetFuncs[i] = 
01445             llvm::Function::Create(ftype, llvm::GlobalValue::ExternalLinkage, 
01446                                    funcs.func[i]->getName(), module);
01447     }
01448 
01449     bool voidReturn = ftype->getReturnType()->isVoidTy();
01450 
01451     // Now we can emit the definition of the dispatch function..
01452     llvm::Function *dispatchFunc =

01454                                name.c_str(), module);

01456         llvm::BasicBlock::Create(*g->ctx, "entry", dispatchFunc);
01457 
01458     // Start by calling out to the function that determines the system's

01464         new llvm::LoadInst(systemBestISAPtr, "system_isa", bblock);
01465 
01466     // Now emit code that works backwards though the available variants of
01467     // the function.  We'll call out to the first one we find that will run
01468     // successfully on the system the code is running on.  In working
01469     // through the candidate ISAs here backward, we're taking advantage of
01470     // the expectation that they are ordered in the Target::ISA enumerant
01471     // from least to most capable.
01472     for (int i = Target::NUM_ISAS-1; i >= 0; --i) {
01473         if (targetFuncs[i] == NULL)

01480             llvm::CmpInst::Create(llvm::Instruction::ICmp, llvm::CmpInst::ICMP_SGE,

01485             llvm::BasicBlock::Create(*g->ctx, "next_try", dispatchFunc);

01488         // Emit the code to make the call call in callBBlock.
01489         // Just pass through all of the args from the dispatch function to
01490         // the target-specific function.
01491         std::vector<llvm::Value *> args;
01492         llvm::Function::arg_iterator argIter = dispatchFunc->arg_begin(); 
01493         for (; argIter != dispatchFunc->arg_end(); ++argIter)
01494             args.push_back(argIter);
01495         if (voidReturn) {
01496 #if defined(LLVM_3_0) || defined(LLVM_3_0svn) || defined(LLVM_3_1svn)
01497             llvm::CallInst::Create(targetFuncs[i], args, "", callBBlock);
01498 #else

01500                                    "", callBBlock);

01502             llvm::ReturnInst::Create(*g->ctx, callBBlock);
01503         }
01504         else {
01505 #if defined(LLVM_3_0) || defined(LLVM_3_0svn) || defined(LLVM_3_1svn)

01507                 llvm::CallInst::Create(targetFuncs[i], args, "ret_value", 
01508                                        callBBlock);

01511                 llvm::CallInst::Create(targetFuncs[i], args.begin(), args.end(),
01512                                        "ret_value", callBBlock);
01513 #endif
01514             llvm::ReturnInst::Create(*g->ctx, retValue, callBBlock);
01515         }

01517         // Otherwise we'll go on to the next candidate and see about that
01518         // one...
01519         bblock = nextBBlock;
01520     }
01521 
01522     // We couldn't find a match that the current system was capable of
01523     // running.  We'll call abort(); this is a bit of a blunt hammer--it
01524     // might be preferable to call a user-supplied callback--ISPCError(...)
01525     // or some such, but we don't want to start imposing too much of a
01526     // runtime library requirement either...
01527     llvm::Function *abortFunc = module->getFunction("abort");

01533     // terminated...
