00097 bool
00098 SymbolTable::AddVariable(Symbol *symbol) {
00099     Assert(symbol != NULL);
00100 
00101     // Check to see if a symbol of the same name has already been declared.
00102     for (int i = (int)variables.size() - 1; i >= 0; --i) {
00103         SymbolMapType &sm = *(variables[i]);
00104         if (sm.find(symbol->name) != sm.end()) {
00105             if (i == (int)variables.size()-1) {
00106                 // If a symbol of the same name was declared in the
00107                 // same scope, it's an error.
00108                 Error(symbol->pos, "Ignoring redeclaration of symbol \"%s\".", 
00109                       symbol->name.c_str());
00110                 return false;
00111             }
00112             else {
00113                 // Otherwise it's just shadowing something else, which
00114                 // is legal but dangerous..
00115                 Warning(symbol->pos, 
00116                         "Symbol \"%s\" shadows symbol declared in outer scope.",
00117                         symbol->name.c_str());
00118                 (*variables.back())[symbol->name] = symbol;
00119                 return true;
00120             }
00121         }
00122     }
00123 
00124     // No matches, so go ahead and add it...
00125     (*variables.back())[symbol->name] = symbol;
00126     return true;
00127 }

00130 Symbol *
00131 SymbolTable::LookupVariable(const char *name) {
00132     // Note that we iterate through the variables vectors backwards, since
00133     // we want to search from the innermost scope to the outermost, so that
00134     // we get the right symbol if we have multiple variables in different
00135     // scopes that shadow each other.
00136     for (int i = (int)variables.size() - 1; i >= 0; --i) {
00137         SymbolMapType &sm = *(variables[i]);
00138         SymbolMapType::iterator iter = sm.find(name);
00139         if (iter != sm.end())
00140             return iter->second;
00141     }
00142     return NULL;

00218 std::vector<std::string>
00219 SymbolTable::ClosestVariableOrFunctionMatch(const char *str) const {
00220     // This is a little wasteful, but we'll look through all of the
00221     // variable and function symbols and compute the edit distance from the
00222     // given string to them.  If the edit distance is under maxDelta, then
00223     // it goes in the entry of the matches[] array corresponding to its
00224     // edit distance.
00225     const int maxDelta = 2;
00226     std::vector<std::string> matches[maxDelta+1];
00227 
00228     for (int i = 0; i < (int)variables.size(); ++i) {
00229         const SymbolMapType &sv = *(variables[i]);
00230         SymbolMapType::const_iterator iter;
00231         for (iter = sv.begin(); iter != sv.end(); ++iter) {
00232             const Symbol *sym = iter->second;
00233             int dist = StringEditDistance(str, sym->name, maxDelta+1);
00234             if (dist <= maxDelta)
00235                 matches[dist].push_back(sym->name);
00236         }
00237     }
00238 
00239     FunctionMapType::const_iterator iter;
00240     for (iter = functions.begin(); iter != functions.end(); ++iter) {
00241         int dist = StringEditDistance(str, iter->first, maxDelta+1);
00242         if (dist <= maxDelta)
00243             matches[dist].push_back(iter->first);
00244     }
00245 
00246     // Now, return the first entry of matches[] that is non-empty, if any.
00247     for (int i = 0; i <= maxDelta; ++i) {
00248         if (matches[i].size())
00249             return matches[i];
00250     }
00251 
00252     // Otherwise, no joy.
00253     return std::vector<std::string>();
00254 }

00269 std::vector<std::string>
00270 SymbolTable::closestTypeMatch(const char *str, bool structsVsEnums) const {
00271     // This follows the same approach as ClosestVariableOrFunctionMatch()
00272     // above; compute all edit distances, keep the ones shorter than
00273     // maxDelta, return the first non-empty vector of one or more sets of
00274     // alternatives with minimal edit distance.
00275     const int maxDelta = 2;
00276     std::vector<std::string> matches[maxDelta+1];
00277 
00278     TypeMapType::const_iterator iter;
00279     for (iter = types.begin(); iter != types.end(); ++iter) {
00280         // Skip over either StructTypes or EnumTypes, depending on the
00281         // value of the structsVsEnums parameter
00282         bool isEnum = (CastType<EnumType>(iter->second) != NULL);
00283         if (isEnum && structsVsEnums)
00284             continue;
00285         else if (!isEnum && !structsVsEnums)
00286             continue;
00287 
00288         int dist = StringEditDistance(str, iter->first, maxDelta+1);
00289         if (dist <= maxDelta)
00290             matches[dist].push_back(iter->first);
00291     }
00292 
00293     for (int i = 0; i <= maxDelta; ++i) {
00294         if (matches[i].size())
00295             return matches[i];
00296     }
00297     return std::vector<std::string>();
00298 }
