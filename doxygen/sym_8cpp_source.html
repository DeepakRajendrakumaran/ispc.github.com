00100 SymbolTable::AddVariable(Symbol *symbol) {
00101     Assert(symbol != NULL);
00102 
00103     // Check to see if a symbol of the same name has already been declared.
00104     for (int i = (int)variables.size() - 1; i >= 0; --i) {
00105         SymbolMapType &sm = *(variables[i]);
00106         if (sm.find(symbol->name) != sm.end()) {
00107             if (i == (int)variables.size()-1) {
00108                 // If a symbol of the same name was declared in the
00109                 // same scope, it's an error.
00110                 Error(symbol->pos, "Ignoring redeclaration of symbol \"%s\".", 
00111                       symbol->name.c_str());
00112                 return false;
00113             }
00114             else {
00115                 // Otherwise it's just shadowing something else, which
00116                 // is legal but dangerous..
00117                 Warning(symbol->pos, 
00118                         "Symbol \"%s\" shadows symbol declared in outer scope.",
00119                         symbol->name.c_str());
00120                 (*variables.back())[symbol->name] = symbol;
00121                 return true;
00122             }
00123         }
00124     }
00125 
00126     // No matches, so go ahead and add it...
00127     (*variables.back())[symbol->name] = symbol;
00128     return true;

00133 SymbolTable::LookupVariable(const char *name) {
00134     // Note that we iterate through the variables vectors backwards, since
00135     // we want to search from the innermost scope to the outermost, so that
00136     // we get the right symbol if we have multiple variables in different
00137     // scopes that shadow each other.

00139         SymbolMapType &sm = *(variables[i]);

00194 SymbolTable::AddType(const char *name, const Type *type, SourcePos pos) {
00195     // Like AddVariable(), we go backwards through the type maps, working
00196     // from innermost scope to outermost.
00197     for (int i = types.size()-1; i >= 0; --i) {
00198         TypeMapType &sm = *(types[i]);
00199         if (sm.find(name) != sm.end()) {
00200             if (i == (int)types.size() - 1) {
00201                 Error(pos, "Ignoring redefinition of type \"%s\".", name);
00202                 return false;
00203             }
00204             else {
00205                 Warning(pos, "Type \"%s\" shadows type declared in outer scope.", name);
00206                 TypeMapType &sm = *(types.back());

00208                 return true;
00209             }
00210         }
00211     }

00214     sm[name] = type;

00231 std::vector<std::string>
00232 SymbolTable::ClosestVariableOrFunctionMatch(const char *str) const {
00233     // This is a little wasteful, but we'll look through all of the
00234     // variable and function symbols and compute the edit distance from the
00235     // given string to them.  If the edit distance is under maxDelta, then
00236     // it goes in the entry of the matches[] array corresponding to its
00237     // edit distance.
00238     const int maxDelta = 2;
00239     std::vector<std::string> matches[maxDelta+1];
00240 
00241     for (int i = 0; i < (int)variables.size(); ++i) {
00242         const SymbolMapType &sv = *(variables[i]);
00243         SymbolMapType::const_iterator iter;
00244         for (iter = sv.begin(); iter != sv.end(); ++iter) {
00245             const Symbol *sym = iter->second;
00246             int dist = StringEditDistance(str, sym->name, maxDelta+1);
00247             if (dist <= maxDelta)
00248                 matches[dist].push_back(sym->name);
00249         }
00250     }
00251 
00252     FunctionMapType::const_iterator iter;
00253     for (iter = functions.begin(); iter != functions.end(); ++iter) {
00254         int dist = StringEditDistance(str, iter->first, maxDelta+1);
00255         if (dist <= maxDelta)
00256             matches[dist].push_back(iter->first);
00257     }
00258 
00259     // Now, return the first entry of matches[] that is non-empty, if any.
00260     for (int i = 0; i <= maxDelta; ++i) {
00261         if (matches[i].size())
00262             return matches[i];
00263     }
00264 
00265     // Otherwise, no joy.
00266     return std::vector<std::string>();
00267 }

00284     // This follows the same approach as ClosestVariableOrFunctionMatch()
00285     // above; compute all edit distances, keep the ones shorter than
00286     // maxDelta, return the first non-empty vector of one or more sets of
00287     // alternatives with minimal edit distance.
00288     const int maxDelta = 2;
00289     std::vector<std::string> matches[maxDelta+1];
00290 
00291     for (unsigned int i = 0; i < types.size(); ++i) {
00292         TypeMapType::const_iterator iter;
00293         for (iter = types[i]->begin(); iter != types[i]->end(); ++iter) {
00294             // Skip over either StructTypes or EnumTypes, depending on the
00295             // value of the structsVsEnums parameter
00296             bool isEnum = (dynamic_cast<const EnumType *>(iter->second) != NULL);
00297             if (isEnum && structsVsEnums)
00298                 continue;
00299             else if (!isEnum && !structsVsEnums)
00300                 continue;
00301 
00302             int dist = StringEditDistance(str, iter->first, maxDelta+1);
00303             if (dist <= maxDelta)
00304                 matches[dist].push_back(iter->first);
00305         }
00306     }
00307 
00308     for (int i = 0; i <= maxDelta; ++i) {
00309         if (matches[i].size())
00310             return matches[i];
00311     }
00312     return std::vector<std::string>();
