00063 /** Given a Type and a set of type qualifiers, apply the type qualifiers to
00064     the type, returning the type that is the result. 
00065 */
00066 static const Type *
00067 lApplyTypeQualifiers(int typeQualifiers, const Type *type, SourcePos pos) {
00068     if (type == NULL)
00069         return NULL;
00070 
00071     if ((typeQualifiers & TYPEQUAL_CONST) != 0)
00072         type = type->GetAsConstType();
00073 
00074     if ((typeQualifiers & TYPEQUAL_UNIFORM) != 0) {
00075         if (Type::Equal(type, AtomicType::Void))
00076             Error(pos, "\"uniform\" qualifier is illegal with \"void\" type.");
00077         else
00078             type = type->GetAsUniformType();
00079     }
00080     else if ((typeQualifiers & TYPEQUAL_VARYING) != 0) {
00081         if (Type::Equal(type, AtomicType::Void))
00082             Error(pos, "\"varying\" qualifier is illegal with \"void\" type.");
00083         else
00084             type = type->GetAsVaryingType();
00085     }
00086     else
00087         if (Type::Equal(type, AtomicType::Void) == false)
00088             type = type->GetAsUnboundVariabilityType();
00089 
00090     if ((typeQualifiers & TYPEQUAL_UNSIGNED) != 0) {
00091         if ((typeQualifiers & TYPEQUAL_SIGNED) != 0)
00092             Error(pos, "Illegal to apply both \"signed\" and \"unsigned\" "
00093                   "qualifiers.");
00094 
00095         const Type *unsignedType = type->GetAsUnsignedType();
00096         if (unsignedType != NULL)
00097             type = unsignedType;
00098         else {
00099             const Type *resolvedType = 
00100                 type->ResolveUnboundVariability(Variability::Varying);
00101             Error(pos, "\"unsigned\" qualifier is illegal with \"%s\" type.",
00102                   resolvedType->GetString().c_str());
00103         }
00104     }
00105 
00106     if ((typeQualifiers & TYPEQUAL_SIGNED) != 0 && type->IsIntType() == false) {
00107         const Type *resolvedType = 
00108             type->ResolveUnboundVariability(Variability::Varying);
00109         Error(pos, "\"signed\" qualifier is illegal with non-integer type "
00110               "\"%s\".", resolvedType->GetString().c_str());
00111     }
00112 
00113     return type;
00114 }

00301         dynamic_cast<const FunctionType *>(GetType(ds));
00302     if (type == NULL)
00303         return NULL;
00304 
00305     Symbol *declSym = GetSymbol();

00309     // already have been added to the symbol table by AddGlobal() by the
00310     // time we get here.)
00311     Symbol *funSym = m->symbolTable->LookupFunction(declSym->name.c_str(), type);
00312     if (funSym == NULL)
00313         // May be NULL due to error earlier in compilation
00314         Assert(m->errorCount > 0);
00315     else
00316         funSym->pos = pos;
00317 
00318     // Walk down to the declarator for the function.  (We have to get past
00319     // the stuff that specifies the function's return type before we get to
00320     // the function's declarator.)
00321     Declarator *d = this;
00322     while (d != NULL && d->kind != DK_FUNCTION)
00323         d = d->child;

00325 
00326     for (unsigned int i = 0; i < d->functionParams.size(); ++i) {
00327         Symbol *sym = d->GetSymbolForFunctionParameter(i);
00328         if (sym->type == NULL) {
00329             Assert(m->errorCount > 0);
00330             continue;
00331         }
00332         else
00333             sym->type = sym->type->ResolveUnboundVariability(Variability::Varying);
00334 
00335         funArgs->push_back(sym);
00336     }
00337 
00338     if (funSym != NULL)
00339         funSym->type = funSym->type->ResolveUnboundVariability(Variability::Varying);
00340 
00341     return funSym;

00346 Declarator::GetType(const Type *base, DeclSpecs *ds) const {

00348     bool hasVaryingQual = ((typeQualifiers & TYPEQUAL_VARYING) != 0);
00349     bool isTask =         ((typeQualifiers & TYPEQUAL_TASK) != 0);
00350     bool isExported =     ((typeQualifiers & TYPEQUAL_EXPORT) != 0);
00351     bool isConst =        ((typeQualifiers & TYPEQUAL_CONST) != 0);
00352 
00353     if (hasUniformQual && hasVaryingQual) {
00354         Error(pos, "Can't provide both \"uniform\" and \"varying\" qualifiers.");
00355         return NULL;
00356     }
00357     if (kind != DK_FUNCTION && isTask)
00358         Error(pos, "\"task\" qualifier illegal in variable declaration.");
00359     if (kind != DK_FUNCTION && isExported)
00360         Error(pos, "\"export\" qualifier illegal in variable declaration.");
00361 
00362     Variability variability(Variability::Unbound);
00363     if (hasUniformQual)
00364         variability = Variability::Uniform;
00365     else if (hasVaryingQual)
00366         variability = Variability::Varying;
00367 
00368     const Type *type = base;
00369     switch (kind) {
00370     case DK_BASE:
00371         // All of the type qualifiers should be in the DeclSpecs for the
00372         // base declarator
00373         Assert(typeQualifiers == 0);
00374         Assert(child == NULL);
00375         return type;
00376 
00377     case DK_POINTER:
00378         /* For now, any pointer to an SOA type gets the slice property; if
00379            we add the capability to declare pointers as slices or not,
00380            we'll want to set this based on a type qualifier here. */
00381         type = new PointerType(type, variability, isConst, type->IsSOAType());
00382         if (child != NULL)
00383             return child->GetType(type, ds);
00384         else
00385             return type;
00386         break;
00387 
00388     case DK_REFERENCE:
00389         if (hasUniformQual)
00390             Error(pos, "\"uniform\" qualifier is illegal to apply to references.");
00391         if (hasVaryingQual)
00392             Error(pos, "\"varying\" qualifier is illegal to apply to references.");
00393         if (isConst)
00394             Error(pos, "\"const\" qualifier is to illegal apply to references.");
00395 
00396         // The parser should disallow this already, but double check.
00397         if (dynamic_cast<const ReferenceType *>(type) != NULL) {
00398             Error(pos, "References to references are illegal.");
00399             return NULL;
00400         }
00401 
00402         type = new ReferenceType(type);
00403         if (child != NULL)
00404             return child->GetType(type, ds);
00405         else
00406             return type;
00407         break;
00408 
00409     case DK_ARRAY:
00410         if (Type::Equal(type, AtomicType::Void)) {
00411             Error(pos, "Arrays of \"void\" type are illegal.");
00412             return NULL;
00413         }
00414         if (dynamic_cast<const ReferenceType *>(type)) {
00415             Error(pos, "Arrays of references (type \"%s\") are illegal.",
00416                   type->GetString().c_str());
00417             return NULL;
00418         }
00419 
00420         type = new ArrayType(type, arraySize);
00421         if (child)
00422             return child->GetType(type, ds);
00423         else
00424             return type;
00425         break;
00426 
00427     case DK_FUNCTION: {

00429         std::vector<std::string> argNames;
