00049 /** Given a Type and a set of type qualifiers, apply the type qualifiers to
00050     the type, returning the type that is the result. 
00051 */
00052 static const Type *
00053 lApplyTypeQualifiers(int typeQualifiers, const Type *type, SourcePos pos) {
00054     if (type == NULL)
00055         return NULL;
00056 
00057     if ((typeQualifiers & TYPEQUAL_UNSIGNED) != 0) {
00058         if ((typeQualifiers & TYPEQUAL_SIGNED) != 0)
00059             Error(pos, "Illegal to apply both \"signed\" and \"unsigned\" "
00060                   "qualifiers.");
00061 
00062         const Type *unsignedType = type->GetAsUnsignedType();
00063         if (unsignedType != NULL)
00064             type = unsignedType;
00065         else
00066             Error(pos, "\"unsigned\" qualifier is illegal with \"%s\" type.",
00067               type->GetString().c_str());
00068 
00069     }
00070 
00071     if ((typeQualifiers & TYPEQUAL_SIGNED) != 0 && type->IsIntType() == false)
00072         Error(pos, "\"signed\" qualifier is illegal with non-integer type "
00073               "\"%s\".", type->GetString().c_str());
00074 
00075     if ((typeQualifiers & TYPEQUAL_CONST) != 0)
00076         type = type->GetAsConstType();
00077 
00078     if ((typeQualifiers & TYPEQUAL_UNIFORM) != 0)
00079         type = type->GetAsUniformType();
00080     else if ((typeQualifiers & TYPEQUAL_VARYING) != 0)
00081         type = type->GetAsVaryingType();
00082     else {
00083         // otherwise, structs are uniform by default and everything
00084         // else is varying by default
00085         if (dynamic_cast<const StructType *>(type->GetBaseType()) != NULL)
00086             type = type->GetAsUniformType();
00087         else
00088             type = type->GetAsVaryingType();
00089     }
00090 
00091     return type;
00092 }
