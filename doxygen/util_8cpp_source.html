00066 /** Returns the width of the terminal where the compiler is running.
00067     Finding this out may fail in a variety of reasonable situations (piping
00068     compiler output to 'less', redirecting output to a file, running the
00069     compiler under a debuffer; in this case, just return a reasonable
00070     default.
00071  */
00072 static int
00073 lTerminalWidth() {
00074     if (g->disableLineWrap)
00075         return 1<<30;
00076 
00077 #if defined(ISPC_IS_WINDOWS)
00078     HANDLE h = GetStdHandle(STD_OUTPUT_HANDLE);
00079     if (h == INVALID_HANDLE_VALUE || h == NULL)
00080         return 80;
00081     CONSOLE_SCREEN_BUFFER_INFO bufferInfo = { {0} };
00082     GetConsoleScreenBufferInfo(h, &bufferInfo);
00083     return bufferInfo.dwSize.X;
00084 #else
00085     struct winsize w;
00086     if (ioctl(STDOUT_FILENO, TIOCGWINSZ, &w) < 0)
00087         return 80;

00089 #endif // ISPC_IS_WINDOWS
00090 }

00146 /** When printing error messages, we sometimes want to include the source
00147     file line for context.  This function print the line(s) of the file
00148     corresponding to the provided SourcePos and underlines the range of the
00149     SourcePos with '^' symbols.
00150 */
00151 static void
00152 lPrintFileLineContext(SourcePos p) {
00153     if (p.first_line == 0)
00154         return;
00155 
00156     FILE *f = fopen(p.name, "r");
00157     if (!f)
00158         return;
00159 
00160     int c, curLine = 1;
00161     while ((c = fgetc(f)) != EOF) {
00162         // Don't print more than three lines of context.  (More than that,
00163         // and we're probably doing the wrong thing...)
00164         if (curLine >= std::max(p.first_line, p.last_line-2) && 
00165             curLine <= p.last_line)
00166             fputc(c, stderr);
00167         if (c == '\n')
00168             ++curLine;
00169         if (curLine > p.last_line)
00170             break;
00171     }
00172 
00173     int i = 1;
00174     for (; i < p.first_column; ++i)
00175         fputc(' ', stderr);
00176     fputc('^', stderr);
00177     ++i;
00178     for (; i < p.last_column; ++i)
00179         fputc('^', stderr);
00180     fputc('\n', stderr);
00181     fputc('\n', stderr);
00182 
00183     fclose(f);
00184 }

00187 /** Counts the number of characters into the buf at which the numColons
00188     colon character is found.  Skips over ANSI escape sequences and doesn't
00189     include their characters in the final count.
00190  */
00191 static int
00192 lFindIndent(int numColons, const char *buf) {
00193     int indent = 0;
00194     while (*buf != '\0') {
00195         if (*buf == '\033') {
00196             while (*buf != '\0' && *buf != 'm')
00197                 ++buf;
00198             if (*buf == 'm')
00199                 ++buf;
00200         }
00201         else {
00202             if (*buf == ':') {
00203                 if (--numColons == 0)
00204                     break;
00205             }
00206             ++indent;
00207             ++buf;
00208         }
00209     }
00210     return indent + 2;
00211 }

00214 /** Print the given string to the given FILE, assuming the given output
00215     column width.  Break words as needed to avoid words spilling past the
00216     last column.  */
00217 static void
00218 lPrintWithWordBreaks(const char *buf, int indent, int columnWidth, FILE *out) {
00219 #ifdef ISPC_IS_WINDOWS
00220     fputs(buf, out);
00221 #else
00222     int column = 0;
00223     int width = std::max(40, columnWidth - 2);
00224 
00225     // Collect everything into a string and print it all at once at the end
00226     // -> try to avoid troubles with mangled error messages with
00227     // multi-threaded builds.
00228     std::string outStr;
00229 
00230     const char *msgPos = buf;
00231     while (true) {
00232         if (*msgPos == '\033') {
00233             // handle ANSI color escape: copy it to the output buffer
00234             // without charging for the characters it uses
00235             do {
00236                 outStr.push_back(*msgPos++);
00237             } while (*msgPos != '\0' && *msgPos != 'm');
00238             continue;
00239         }
00240 
00241         while (*msgPos != '\0' && isspace(*msgPos))
00242             ++msgPos;
00243         if (*msgPos == '\0')
00244             break;
00245 
00246         const char *wordEnd = lFindWordEnd(msgPos);
00247         if (column > indent && column + wordEnd - msgPos > width) {
00248             // This word would overflow, so start a new line
00249             column = indent;
00250             outStr.push_back('\n');
00251             // Indent to the same column as the ":" at the start of the
00252             // message.
00253             for (int i = 0; i < indent; ++i)
00254                 outStr.push_back(' ');
00255         }
00256 
00257         // Finally go and copy the word
00258         while (msgPos != wordEnd) {
00259             outStr.push_back(*msgPos++);
00260             ++column;
00261         }
00262         outStr.push_back(' ');
00263         ++column;
00264     }
00265     outStr.push_back('\n');
00266     fputs(outStr.c_str(), out);
00267 #endif
00268 }

00297 /** Helper function for Error(), Warning(), etc.
00298 
00299     @param type   The type of message being printed (e.g. "Warning")
00300     @param p      Position in source file that is connected to the message
00301                   being printed
00302     @param fmt    printf()-style format string
00303     @param args   Arguments with values for format string % entries
00304 */
00305 static void
00306 lPrint(const char *type, bool isError, SourcePos p, const char *fmt, 
00307        va_list args) {
00308     char *errorBuf, *formattedBuf;
00309     if (vasprintf(&errorBuf, fmt, args) == -1) {
00310         fprintf(stderr, "vasprintf() unable to allocate memory!\n");
00311         abort();
00312     }
00313 
00314     int indent = 0;
00315     if (p.first_line == 0) {
00316         // We don't have a valid SourcePos, so create a message without it
00317         if (asprintf(&formattedBuf, "%s%s%s%s%s: %s%s\n", lStartBold(),
00318                      isError ? lStartRed() : lStartBlue(), type,
00319                      lResetColor(), lStartBold(), errorBuf, 
00320                      lResetColor()) == -1) {
00321             fprintf(stderr, "asprintf() unable to allocate memory!\n");
00322             exit(1);
00323         }
00324         indent = lFindIndent(1, formattedBuf);
00325     }
00326     else {
00327         // Create an error message that includes the file and line number
00328         if (asprintf(&formattedBuf, "%s%s:%d:%d: %s%s%s%s: %s%s\n", 
00329                      lStartBold(), p.name, p.first_line, p.first_column, 
00330                      isError ? lStartRed() : lStartBlue(), type, 
00331                      lResetColor(), lStartBold(), errorBuf, 
00332                      lResetColor()) == -1) {
00333             fprintf(stderr, "asprintf() unable to allocate memory!\n");
00334             exit(1);
00335         }
00336         indent = lFindIndent(3, formattedBuf);
00337     }
00338 
00339     // Now that we've done all that work, see if we've already printed the
00340     // exact same error message.  If so, return, so we don't redundantly
00341     // print it and annoy the user.
00342     static std::set<std::string> printed;
00343     if (printed.find(formattedBuf) != printed.end())
00344         return;
00345     printed.insert(formattedBuf);
00346 
00347     lPrintWithWordBreaks(formattedBuf, indent, lTerminalWidth(), stderr);
00348     lPrintFileLineContext(p);
00349 
00350     free(errorBuf);
00351     free(formattedBuf);
00352 }

00423 // http://en.wikipedia.org/wiki/Levenshtein_distance
00424 int
00425 StringEditDistance(const std::string &str1, const std::string &str2, int maxDist) {
00426     // Small hack: don't return 0 if the strings are the same; if we've
00427     // gotten here, there's been a parsing error, and suggesting the same
00428     // string isn't going to actually help things.
00429     if (str1 == str2)
00430         return maxDist;
00431 
00432     int n1 = (int)str1.size(), n2 = (int)str2.size();
00433     int nmax = std::max(n1, n2);
00434 
00435     int *current =  (int *)alloca((nmax+1) * sizeof(int));
00436     int *previous = (int *)alloca((nmax+1) * sizeof(int));
00437 
00438     for (int i = 0; i <= n2; ++i)
00439         previous[i] = i;
00440 
00441     for (int y = 1; y <= n1; ++y) {
00442         current[0] = y;
00443         int rowBest = y;
00444 
00445         for (int x = 1; x <= n2; ++x) {
00446             current[x] = std::min(previous[x-1] + (str1[y-1] == str2[x-1] ? 0 : 1),
00447                                   std::min(current[x-1], previous[x])+1);
00448             rowBest = std::min(rowBest, current[x]);
00449         }
00450 
00451         if (maxDist != 0 && rowBest > maxDist)
00452             return maxDist + 1;
00453 
00454         std::swap(current, previous);
00455     }
00456 
00457     return previous[n2];
00458 }

00461 std::vector<std::string> 
00462 MatchStrings(const std::string &str, const std::vector<std::string> &options) {
00463     if (str.size() == 0 || (str.size() == 1 && !isalpha(str[0])))
00464         // don't even try...
00465         return std::vector<std::string>();
00466 
00467     const int maxDelta = 2;
00468     std::vector<std::string> matches[maxDelta+1];
00469 
00470     // For all of the options that are up to maxDelta edit distance, store
00471     // them in the element of matches[] that corresponds to their edit
00472     // distance.
00473     for (int i = 0; i < (int)options.size(); ++i) {
00474         int dist = StringEditDistance(str, options[i], maxDelta+1);
00475         if (dist <= maxDelta)
00476             matches[dist].push_back(options[i]);
00477     }
00478 
00479     // And return the first one of them, if any, that has at least one
00480     // match.
00481     for (int i = 0; i <= maxDelta; ++i) {
00482         if (matches[i].size())
00483             return matches[i];
00484     }
00485     return std::vector<std::string>();
00486 }
