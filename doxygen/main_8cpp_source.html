00143 /** We take arguments from both the command line as well as from the
00144     ISPC_ARGS environment variable.  This function returns a new set of
00145     arguments representing the ones from those two sources merged together.
00146  */ 
00147 static void lGetAllArgs(int Argc, char *Argv[], int &argc, char *argv[128]) {
00148     // Copy over the command line arguments (passed in)
00149     for (int i = 0; i < Argc; ++i)
00150         argv[i] = Argv[i];
00151     argc = Argc;
00152 
00153     // See if we have any set via the environment variable
00154     const char *env = getenv("ISPC_ARGS");
00155     if (!env)
00156         return;
00157     while (true) {
00158         // Look for the next space in the string, which delimits the end of
00159         // the current argument
00160         const char *end = strchr(env, ' ');
00161         if (end == NULL)
00162             end = env + strlen(env);
00163         int len = end - env;
00164 
00165         // Copy the argument into a newly allocated memory (so we can
00166         // NUL-terminate it).
00167         char *ptr = new char[len+1];
00168         strncpy(ptr, env, len);
00169         ptr[len] = '\0';
00170 
00171         // Add it to the args array and get out of here 
00172         argv[argc++] = ptr;
00173         if (*end == '\0')
00174             break;
00175 
00176         // Advance the starting pointer of the string to the next non-space
00177         // character
00178         env = end+1;
00179         while (*env == ' ')
00180             ++env;
00181 
00182         // Hit the end of the string; get out of here
00183         if (*env == '\0')
00184             break;
00185     }
00186 }
