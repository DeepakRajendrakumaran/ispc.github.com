00299     // have it do this remapping, which would presumably be a bit less
00300     // error prone?
00301     if (arch == "x86")
00302         triple.setArchName("i386");
00303     else if (arch == "x86-64")
00304         triple.setArchName("x86_64");

00369 static bool
00370 lGenericTypeLayoutIndeterminate(LLVM_TYPE_CONST llvm::Type *type) {
00371     if (type->isPrimitiveType() || type->isIntegerTy())
00372         return false;
00373 
00374     if (type == LLVMTypes::BoolVectorType ||
00375         type == LLVMTypes::MaskType ||
00376         type == LLVMTypes::Int1VectorType)
00377         return true;
00378 
00379     LLVM_TYPE_CONST llvm::ArrayType *at = 
00380         llvm::dyn_cast<LLVM_TYPE_CONST llvm::ArrayType>(type);
00381     if (at != NULL)
00382         return lGenericTypeLayoutIndeterminate(at->getElementType());
00383 
00384     LLVM_TYPE_CONST llvm::PointerType *pt = 
00385         llvm::dyn_cast<LLVM_TYPE_CONST llvm::PointerType>(type);
00386     if (pt != NULL)
00387         return false;
00388 
00389     LLVM_TYPE_CONST llvm::StructType *st =
00390         llvm::dyn_cast<LLVM_TYPE_CONST llvm::StructType>(type);
00391     if (st != NULL) {
00392         for (int i = 0; i < (int)st->getNumElements(); ++i)
00393             if (lGenericTypeLayoutIndeterminate(st->getElementType(i)))
00394                 return true;
00395         return false;
00396     }
00397 
00398     Assert(llvm::isa<LLVM_TYPE_CONST llvm::VectorType>(type));
00399     return true;
00400 }
