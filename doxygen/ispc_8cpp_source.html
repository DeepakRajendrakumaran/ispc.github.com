00082   __asm__ __volatile__ ("xchg{l}\t{%%}ebx, %1\n\t"
00083                         "cpuid\n\t"

00085                         : "=a" (info[0]), "=r" (info[1]), "=c" (info[2]), "=d" (info[3])

00091 static const char *
00092 lGetSystemISA() {
00093     int info[4];
00094     __cpuid(info, 1);
00095 
00096     if ((info[2] & (1 << 28)) != 0) {
00097         // AVX1 for sure. Do we have AVX2?
00098         // Call cpuid with eax=7, ecx=0
00099         __cpuidex(info, 7, 0);
00100         if ((info[1] & (1 << 5)) != 0)
00101             return "avx2";
00102         else
00103             return "avx";
00104     }
00105     else if ((info[2] & (1 << 19)) != 0)
00106         return "sse4";
00107     else if ((info[3] & (1 << 26)) != 0)
00108         return "sse2";
00109     else {
00110         fprintf(stderr, "Unable to detect supported SSE/AVX ISA.  Exiting.\n");
00111         exit(1);
00112     }
00113 }

00368 std::string
00369 Target::GetTripleString() const {
00370     llvm::Triple triple;
00371     // Start with the host triple as the default
00372 #if defined(LLVM_3_1) || defined(LLVM_3_1svn)
00373     triple.setTriple(llvm::sys::getDefaultTargetTriple());
00374 #else
00375     triple.setTriple(llvm::sys::getHostTriple());
00376 #endif
00377 
00378     // And override the arch in the host triple based on what the user
00379     // specified.  Here we need to deal with the fact that LLVM uses one
00380     // naming convention for targets TargetRegistry, but wants some
00381     // slightly different ones for the triple.  TODO: is there a way to
00382     // have it do this remapping, which would presumably be a bit less
00383     // error prone?
00384     if (arch == "x86")
00385         triple.setArchName("i386");
00386     else if (arch == "x86-64")
00387         triple.setArchName("x86_64");
00388     else
00389         triple.setArchName(arch);
00390 
00391     return triple.str();
00392 }

00459 static bool
00460 lGenericTypeLayoutIndeterminate(llvm::Type *type) {
00461     if (type->isPrimitiveType() || type->isIntegerTy())
00462         return false;
00463 
00464     if (type == LLVMTypes::BoolVectorType ||
00465         type == LLVMTypes::MaskType ||
00466         type == LLVMTypes::Int1VectorType)
00467         return true;
00468 
00469     llvm::ArrayType *at = 
00470         llvm::dyn_cast<llvm::ArrayType>(type);
00471     if (at != NULL)
00472         return lGenericTypeLayoutIndeterminate(at->getElementType());
00473 
00474     llvm::PointerType *pt = 
00475         llvm::dyn_cast<llvm::PointerType>(type);
00476     if (pt != NULL)
00477         return false;
00478 
00479     llvm::StructType *st =
00480         llvm::dyn_cast<llvm::StructType>(type);
00481     if (st != NULL) {
00482         for (int i = 0; i < (int)st->getNumElements(); ++i)
00483             if (lGenericTypeLayoutIndeterminate(st->getElementType(i)))
00484                 return true;
00485         return false;
00486     }
00487 
00488     Assert(llvm::isa<llvm::VectorType>(type));
00489     return true;
00490 }
