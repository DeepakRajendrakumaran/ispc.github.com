00129     always generate debugging information but we do always generate this

00134     FunctionEmitContext::addGSMetadata(), which encodes the source position during

00455 /** Given an llvm::Value represinting a vector mask, see if the value is a
00456     constant.  If so, return the integer mask found by taking the high bits
00457     of the mask values in turn and concatenating them into a single integer.
00458     In other words, given the 4-wide mask: < 0xffffffff, 0, 0, 0xffffffff >, 
00459     we have 0b1001 = 9.
00460 
00461     @todo This will break if we ever do 32-wide compilation, in which case
00462     it don't be possible to distinguish between -1 for "don't know" and
00463     "known and all bits on".
00464  */

00466 lGetMask(llvm::Value *factor) {

00468     if (cv) {
00469         int mask = 0;
00470         llvm::SmallVector<llvm::Constant *, ISPC_MAX_NVEC> elements;
00471         cv->getVectorElements(elements);
00472 
00473         for (unsigned int i = 0; i < elements.size(); ++i) {
00474             llvm::APInt intMaskValue;
00475             // SSE has the "interesting" approach of encoding blending
00476             // masks as <n x float>.
00477             llvm::ConstantFP *cf = llvm::dyn_cast<llvm::ConstantFP>(elements[i]);
00478             if (cf) {
00479                 llvm::APFloat apf = cf->getValueAPF();
00480                 intMaskValue = apf.bitcastToAPInt();
00481             }
00482             else {
00483                 // Otherwise get it as an int
00484                 llvm::ConstantInt *ci = llvm::dyn_cast<llvm::ConstantInt>(elements[i]);
00485                 assert(ci != NULL);  // vs return -1 if NULL?
00486                 intMaskValue = ci->getValue();
00487             }
00488             // Is the high-bit set?  If so, OR in the appropriate bit in
00489             // the result mask
00490             if (intMaskValue.countLeadingOnes() > 0)
00491                 mask |= (1 << i);
00492         }
00493         return mask;
00494     }
00495     else if (llvm::isa<llvm::ConstantAggregateZero>(factor))
00496         return 0;
00497     else {
00498 #if 0
00499         llvm::ConstantExpr *ce = llvm::dyn_cast<llvm::ConstantExpr>(factor);
00500         if (ce != NULL) {

00556         if (!callInst)
00557             continue;
00558 
00559         BlendInstruction *blend = matchingBlendInstruction(callInst->getCalledFunction());
00560         if (blend != NULL) {
00561             llvm::Value *v[2] = { callInst->getArgOperand(blend->op0), 
00562                                   callInst->getArgOperand(blend->op1) };
00563             llvm::Value *factor = callInst->getArgOperand(blend->opFactor);
00564 
00565             // If the values are the same, then no need to blend..
00566             if (v[0] == v[1]) {
00567                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(), 
00568                                            iter, v[0]);
00569                 modifiedAny = true;
00570                 goto restart;
00571             }
00572 
00573             // If one of the two is undefined, we're allowed to replace
00574             // with the value of the other.  (In other words, the only
00575             // valid case is that the blend factor ends up having a value
00576             // that only selects from the defined one of the two operands,
00577             // otherwise the result is undefined and any value is fine,
00578             // ergo the defined one is an acceptable result.)
00579             if (lIsUndef(v[0])) {
00580                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(), 
00581                                            iter, v[1]);
00582                 modifiedAny = true;
00583                 goto restart;
00584             }
00585             if (lIsUndef(v[1])) {
00586                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(), 
00587                                            iter, v[0]);
00588                 modifiedAny = true;
00589                 goto restart;
00590             }
00591 
00592             int mask = lGetMask(factor);
00593             llvm::Value *value = NULL;
00594             if (mask == 0)
00595                 // Mask all off -> replace with the first blend value
00596                 value = v[0];
00597             else if (mask == blend->allOnMask)
00598                 // Mask all on -> replace with the second blend value
00599                 value = v[1];
00600 
00601             if (value != NULL) {
00602                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(), 
00603                                            iter, value);
00604                 modifiedAny = true;
00605                 goto restart;
00606             }
00607         }
00608         else if (matchesMaskInstruction(callInst->getCalledFunction())) {
00609             llvm::Value *factor = callInst->getArgOperand(0);
00610             int mask = lGetMask(factor);
00611             if (mask != -1) {
00612                 // If the vector-valued mask has a known value, replace it
00613                 // with the corresponding integer mask from its elements
00614                 // high bits.
00615                 llvm::Value *value = LLVMInt32(mask);
00616                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(),
00617                                            iter, value);
00618                 modifiedAny = true;
00619                 goto restart;
00620             }
00621         }
00622 #if defined(LLVM_3_0) || defined(LLVM_3_0svn) || defined(LLVM_3_1svn)
00623         else if (callInst->getCalledFunction() == avxMaskedLoad32 ||
00624                  callInst->getCalledFunction() == avxMaskedLoad64) {
00625             llvm::Value *factor = callInst->getArgOperand(1);
00626             int mask = lGetMask(factor);
00627             if (mask == 0) {
00628                 // nothing being loaded, replace with undef value
00629                 llvm::Type *returnType = callInst->getType();
00630                 assert(llvm::isa<llvm::VectorType>(returnType));
00631                 llvm::Value *undefValue = llvm::UndefValue::get(returnType);
00632                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(),
00633                                            iter, undefValue);
00634                 modifiedAny = true;
00635                 goto restart;
00636             }
00637             else if (mask == 0xff) {
00638                 // all lanes active; replace with a regular load
00639                 llvm::Type *returnType = callInst->getType();
00640                 assert(llvm::isa<llvm::VectorType>(returnType));
00641                 // cast the i8 * to the appropriate type
00642                 llvm::Value *castPtr = 
00643                     new llvm::BitCastInst(callInst->getArgOperand(0),
00644                                           llvm::PointerType::get(returnType, 0), 
00645                                           "ptr2vec", callInst);
00646                 lCopyMetadata(castPtr, callInst);
00647                 int align = callInst->getCalledFunction() == avxMaskedLoad32 ? 4 : 8;
00648                 llvm::Instruction *loadInst = 
00649                     new llvm::LoadInst(castPtr, "load", false /* not volatile */,
00650                                        align, (llvm::Instruction *)NULL);
00651                 lCopyMetadata(loadInst, callInst);
00652                 llvm::ReplaceInstWithInst(callInst, loadInst);
00653                 modifiedAny = true;
00654                 goto restart;
00655             }
00656         }
00657         else if (callInst->getCalledFunction() == avxMaskedStore32 ||
00658                  callInst->getCalledFunction() == avxMaskedStore64) {
00659             // NOTE: mask is the 2nd parameter, not the 3rd one!!
00660             llvm::Value *factor = callInst->getArgOperand(1);
00661             int mask = lGetMask(factor);
00662             if (mask == 0) {
00663                 // nothing actually being stored, just remove the inst
00664                 callInst->eraseFromParent();
00665                 modifiedAny = true;
00666                 goto restart;
00667             }
00668             else if (mask == 0xff) {
00669                 // all lanes storing, so replace with a regular store
00670                 llvm::Value *rvalue = callInst->getArgOperand(2);
00671                 llvm::Type *storeType = rvalue->getType();
00672                 llvm::Value *castPtr = 
00673                     new llvm::BitCastInst(callInst->getArgOperand(0),
00674                                           llvm::PointerType::get(storeType, 0), 
00675                                           "ptr2vec", callInst);
00676                 lCopyMetadata(castPtr, callInst);
00677 
00678                 llvm::StoreInst *storeInst = 
00679                     new llvm::StoreInst(rvalue, castPtr, (llvm::Instruction *)NULL);
00680                 int align = callInst->getCalledFunction() == avxMaskedStore32 ? 4 : 8;
00681                 storeInst->setAlignment(align);
00682                 lCopyMetadata(storeInst, callInst);
00683                 llvm::ReplaceInstWithInst(callInst, storeInst);
00684 
00685                 modifiedAny = true;
00686                 goto restart;
00687             }
00688         }
00689 #endif
00690     }
00691     return modifiedAny;
00692 }

00748 /** Given an llvm::Value known to be an integer, return its value as
00749     an int64_t.
00750 */
00751 static int64_t
00752 lGetIntValue(llvm::Value *offset) {
00753     llvm::ConstantInt *intOffset = llvm::dyn_cast<llvm::ConstantInt>(offset);
00754     assert(intOffset && (intOffset->getBitWidth() == 32 ||
00755                          intOffset->getBitWidth() == 64));
00756     return intOffset->getSExtValue();
00757 }

00759 /** This function takes chains of InsertElement instructions along the
00760     lines of:
00761 
00762     %v0 = insertelement undef, value_0, i32 index_0
00763     %v1 = insertelement %v1,   value_1, i32 index_1
00764     ...
00765     %vn = insertelement %vn-1, value_n-1, i32 index_n-1
00766 
00767     and initializes the provided elements array such that the i'th
00768     llvm::Value * in the array is the element that was inserted into the
00769     i'th element of the vector.
00770 */
00771 static void
00772 lFlattenInsertChain(llvm::InsertElementInst *ie, int vectorWidth,
00773                     llvm::Value **elements) {
00774     for (int i = 0; i < vectorWidth; ++i)
00775         elements[i] = NULL;
00776 
00777     while (ie != NULL) {
00778         int64_t iOffset = lGetIntValue(ie->getOperand(2));
00779         assert(iOffset >= 0 && iOffset < vectorWidth);
00780         assert(elements[iOffset] == NULL);
00781 
00782         elements[iOffset] = ie->getOperand(1);
00783 
00784         llvm::Value *insertBase = ie->getOperand(0);
00785         ie = llvm::dyn_cast<llvm::InsertElementInst>(insertBase);
00786         if (ie == NULL)
00787             assert(llvm::isa<llvm::UndefValue>(insertBase));
00788     }
00789 }

00792 /** Check to make sure that this value is actually a pointer in the end.
00793     We need to make sure that given an expression like vec(offset) +
00794     ptr2int(ptr), lGetBasePointer() doesn't return vec(offset) for the base
00795     pointer such that we then treat ptr2int(ptr) as an offset.  This ends
00796     up being important so that we don't generate LLVM GEP instructions like
00797     "gep inttoptr 8, i64 %ptr", which in turn can lead to incorrect code
00798     since LLVM's pointer aliasing analysis assumes that operands after the
00799     first one to a GEP aren't pointers.
00800  */
00801 static llvm::Value *
00802 lCheckForActualPointer(llvm::Value *v) {
00803     if (v == NULL)
00804         return NULL;
00805     else if (llvm::isa<LLVM_TYPE_CONST llvm::PointerType>(v->getType()))
00806         return v;
00807     else if (llvm::isa<llvm::PtrToIntInst>(v))
00808         return v;
00809     else {
00810         llvm::ConstantExpr *uce = 
00811             llvm::dyn_cast<llvm::ConstantExpr>(v);
00812         if (uce != NULL &&
00813             uce->getOpcode() == llvm::Instruction::PtrToInt)
00814             return v;
00815         return NULL;
00816     }
00817 }

00820 /** Given a llvm::Value representing a varying pointer, this function
00821     checks to see if all of the elements of the vector have the same value
00822     (i.e. there's a common base pointer).  If so, it returns the common
00823     pointer value; otherwise it returns NULL.
00824  */
00825 static llvm::Value *
00826 lGetBasePointer(llvm::Value *v) {
00827     llvm::InsertElementInst *ie = llvm::dyn_cast<llvm::InsertElementInst>(v);
00828     if (ie != NULL) {
00829         llvm::Value *elements[ISPC_MAX_NVEC];
00830         lFlattenInsertChain(ie, g->target.vectorWidth, elements);
00831 
00832         // Make sure none of the elements is undefined.
00833         // TODO: it's probably ok to allow undefined elements and return
00834         // the base pointer if all of the other elements have the same
00835         // value.
00836         for (int i = 0; i < g->target.vectorWidth; ++i)
00837             if (elements[i] == NULL)
00838                 return NULL;
00839 
00840         // Do all of the elements have the same value?
00841         for (int i = 0; i < g->target.vectorWidth-1; ++i)
00842             if (elements[i] != elements[i+1])
00843                 return NULL;
00844 
00845         return lCheckForActualPointer(elements[0]);
00846     }
00847 
00848     // This case comes up with global/static arrays
00849     llvm::ConstantVector *cv = llvm::dyn_cast<llvm::ConstantVector>(v);
00850     if (cv != NULL)
00851         return lCheckForActualPointer(cv->getSplatValue());
00852 
00853     return NULL;
00854 }

00857 /** Given the two operands to a constant add expression, see if we have the
00858     form "base pointer + offset", whee op0 is the base pointer and op1 is
00859     the offset; if so return the base and the offset. */
00860 static llvm::Constant *
00861 lGetConstantAddExprBaseOffset(llvm::Constant *op0, llvm::Constant *op1, 
00862                               llvm::Constant **delta) {
00863     llvm::ConstantExpr *op = llvm::dyn_cast<llvm::ConstantExpr>(op0);
00864     if (op == NULL || op->getOpcode() != llvm::Instruction::PtrToInt)
00865         // the first operand isn't a pointer
00866         return NULL;
00867 
00868     llvm::ConstantInt *opDelta = llvm::dyn_cast<llvm::ConstantInt>(op1);
00869     if (opDelta == NULL)
00870         // the second operand isn't an integer operand
00871         return NULL;
00872 
00873     *delta = opDelta;
00874     return op0;
00875 }

00878 /** Given a varying pointer in ptrs, this function checks to see if it can
00879     be determined to be indexing from a common uniform base pointer.  If
00880     so, the function returns the base pointer llvm::Value and initializes
00881     *offsets with an int vector of the per-lane offsets
00882  */
00883 static llvm::Value *
00884 lGetBasePtrAndOffsets(llvm::Value *ptrs, llvm::Value **offsets) {
00885     llvm::Value *base = lGetBasePointer(ptrs);
00886     if (base != NULL) {
00887         // We have a straight up varying pointer with no indexing that's
00888         // actually all the same value.
00889         if (g->target.is32Bit)
00890             *offsets = LLVMInt32Vector(0);
00891         else
00892             *offsets = LLVMInt64Vector((int64_t)0);
00893         return base;
00894     }
00895 
00896     llvm::BinaryOperator *bop = llvm::dyn_cast<llvm::BinaryOperator>(ptrs);
00897     if (bop != NULL && bop->getOpcode() == llvm::Instruction::Add) {
00898         // If we have a common pointer plus something, then we're also
00899         // good.
00900         if ((base = lGetBasePointer(bop->getOperand(0))) != NULL) {
00901             *offsets = bop->getOperand(1);
00902             return base;
00903         }
00904         else if ((base = lGetBasePointer(bop->getOperand(1))) != NULL) {
00905             *offsets = bop->getOperand(0);
00906             return base;
00907         }
00908     }
00909 
00910     llvm::ConstantVector *cv = llvm::dyn_cast<llvm::ConstantVector>(ptrs);
00911     if (cv != NULL) {
00912         // Indexing into global arrays can lead to this form, with
00913         // ConstantVectors..
00914         llvm::SmallVector<llvm::Constant *, ISPC_MAX_NVEC> elements;
00915         cv->getVectorElements(elements);
00916 
00917         llvm::Constant *delta[ISPC_MAX_NVEC];
00918         for (unsigned int i = 0; i < elements.size(); ++i) {
00919             // For each element, try to decompose it into either a straight
00920             // up base pointer, or a base pointer plus an integer value.
00921             llvm::ConstantExpr *ce = llvm::dyn_cast<llvm::ConstantExpr>(elements[i]);
00922             if (ce == NULL)
00923                 return NULL;
00924 
00925             delta[i] = NULL;
00926             llvm::Value *elementBase = NULL; // base pointer for this element
00927             if (ce->getOpcode() == llvm::Instruction::PtrToInt) {
00928                 // If the element is just a ptr to int instruction, treat
00929                 // it as having an offset of zero
00930                 elementBase = ce;
00931                 delta[i] = g->target.is32Bit ? LLVMInt32(0) : LLVMInt64(0);
00932             }
00933             else if (ce->getOpcode() == llvm::Instruction::Add) {
00934                 // Try both orderings of the operands to see if we can get
00935                 // a pointer+offset out of them.
00936                 elementBase =
00937                     lGetConstantAddExprBaseOffset(ce->getOperand(0), 
00938                                                   ce->getOperand(1),
00939                                                   &delta[i]);
00940                 if (elementBase == NULL)
00941                     elementBase = 
00942                         lGetConstantAddExprBaseOffset(ce->getOperand(1), 
00943                                                       ce->getOperand(0),
00944                                                       &delta[i]);
00945             }
00946 
00947             // We weren't able to find a base pointer in the above.  (We
00948             // don't expect this to happen; if it does, it may be necessary
00949             // to handle more cases in the decomposition above.)
00950             if (elementBase == NULL)
00951                 return NULL;
00952 
00953             assert(delta[i] != NULL);
00954             if (base == NULL)
00955                 // The first time we've found a base pointer
00956                 base = elementBase;
00957             else if (base != elementBase)
00958                 // Different program instances have different base
00959                 // pointers, so no luck.
00960                 return NULL;
00961         }
00962 
00963         assert(base != NULL);
00964         llvm::ArrayRef<llvm::Constant *> deltas(&delta[0], 
00965                                                 &delta[elements.size()]);
00966         *offsets = llvm::ConstantVector::get(deltas);
00967         return base;
00968     }
00969 
00970     return NULL;
00971 }

00993         GSInfo("__pseudo_gather32_16", "__pseudo_gather_base_offsets32_16",

00999         GSInfo("__pseudo_scatter32_8",  "__pseudo_scatter_base_offsets32_8",

01001         GSInfo("__pseudo_scatter32_16", "__pseudo_scatter_base_offsets32_16", 
01002                "__pseudo_scatter_base_offsets32_16", false),
01003         GSInfo("__pseudo_scatter32_32", "__pseudo_scatter_base_offsets32_32", 
01004                "__pseudo_scatter_base_offsets32_32", false),
01005         GSInfo("__pseudo_scatter32_64", "__pseudo_scatter_base_offsets32_64", 
01006                "__pseudo_scatter_base_offsets32_64", false),
01007         GSInfo("__pseudo_gather64_8",  "__pseudo_gather_base_offsets64_8", 
01008                "__pseudo_gather_base_offsets32_8", true),
01009         GSInfo("__pseudo_gather64_16", "__pseudo_gather_base_offsets64_16", 
01010                "__pseudo_gather_base_offsets32_16", true),
01011         GSInfo("__pseudo_gather64_32", "__pseudo_gather_base_offsets64_32", 
01012                "__pseudo_gather_base_offsets32_32", true),
01013         GSInfo("__pseudo_gather64_64", "__pseudo_gather_base_offsets64_64", 
01014                "__pseudo_gather_base_offsets32_64", true),
01015         GSInfo("__pseudo_scatter64_8",  "__pseudo_scatter_base_offsets64_8", 
01016                "__pseudo_scatter_base_offsets32_8", false),
01017         GSInfo("__pseudo_scatter64_16", "__pseudo_scatter_base_offsets64_16", 
01018                "__pseudo_scatter_base_offsets32_16", false),
01019         GSInfo("__pseudo_scatter64_32", "__pseudo_scatter_base_offsets64_32", 
01020                "__pseudo_scatter_base_offsets32_32", false),
01021         GSInfo("__pseudo_scatter64_64", "__pseudo_scatter_base_offsets64_64", 
01022                "__pseudo_scatter_base_offsets32_64", false),
01023     };
01024     int numGSFuncs = sizeof(gsFuncs) / sizeof(gsFuncs[0]);
01025     for (int i = 0; i < numGSFuncs; ++i)
01026         assert(gsFuncs[i].func != NULL && gsFuncs[i].baseOffsetsFunc != NULL &&
01027                gsFuncs[i].baseOffsets32Func != NULL);
01028 
01029     bool modifiedAny = false;
01030  restart:
01031     // Iterate through all of the instructions in the basic block.
01032     for (llvm::BasicBlock::iterator iter = bb.begin(), e = bb.end(); iter != e; ++iter) {
01033         llvm::CallInst *callInst = llvm::dyn_cast<llvm::CallInst>(&*iter);
01034         // If we don't have a call to one of the
01035         // __pseudo_{gather,scatter}_* functions, then just go on to the
01036         // next instruction.
01037         if (callInst == NULL)
01038             continue;
01039         GSInfo *info = NULL;
01040         for (int i = 0; i < numGSFuncs; ++i)
01041             if (gsFuncs[i].func != NULL &&
01042                 callInst->getCalledFunction() == gsFuncs[i].func) {
01043                 info = &gsFuncs[i];
01044                 break;
01045             }
01046         if (info == NULL)
01047             continue;
01048 
01049         // Try to transform the array of pointers to a single base pointer
01050         // and an array of int32 offsets.  (All the hard work is done by
01051         // lGetBasePtrAndOffsets).
01052         llvm::Value *ptrs = callInst->getArgOperand(0);
01053         llvm::Value *offsetVector = NULL;
01054         llvm::Value *basePtr = lGetBasePtrAndOffsets(ptrs, &offsetVector);
01055 
01056         if (basePtr == NULL || offsetVector == NULL)
01057             // It's actually a fully general gather/scatter with a varying
01058             // set of base pointers, so leave it as is and continune onward
01059             // to the next instruction...
01060             continue;
01061 
01062         // Cast the base pointer to a void *, since that's what the
01063         // __pseudo_*_base_offsets_* functions want.
01064         basePtr = new llvm::IntToPtrInst(basePtr, LLVMTypes::VoidPointerType,
01065                                          "base2void", callInst);
01066         lCopyMetadata(basePtr, callInst);
01067 
01068         llvm::Function *gatherScatterFunc = info->baseOffsetsFunc;
01069 
01070         if (g->opt.force32BitAddressing) {
01071             // If we're doing 32-bit addressing on a 64-bit target, here we
01072             // will see if we can call one of the 32-bit variants of the
01073             // pseudo gather/scatter functions.  Specifically, if the
01074             // offset vector turns out to be an i32 value that was sext'ed
01075             // to be i64 immediately before the scatter/gather, then we
01076             // walk past the sext to get the i32 offset values and then
01077             // call out to the corresponding 32-bit gather/scatter
01078             // function.
01079             llvm::SExtInst *sext = llvm::dyn_cast<llvm::SExtInst>(offsetVector);
01080             if (sext != NULL && 
01081                 sext->getOperand(0)->getType() == LLVMTypes::Int32VectorType) {
01082                 offsetVector = sext->getOperand(0);
01083                 gatherScatterFunc = info->baseOffsets32Func;
01084             }
01085         }
01086 
01087         if (info->isGather) {
01088             llvm::Value *mask = callInst->getArgOperand(1);
01089 
01090             // Generate a new function call to the next pseudo gather
01091             // base+offsets instruction.  Note that we're passing a NULL
01092             // llvm::Instruction to llvm::CallInst::Create; this means that
01093             // the instruction isn't inserted into a basic block and that
01094             // way we can then call ReplaceInstWithInst().
01095             llvm::Value *newArgs[3] = { basePtr, offsetVector, mask };
01096 #if defined(LLVM_3_0) || defined(LLVM_3_0svn) || defined(LLVM_3_1svn)

01429 /** Conservative test to see if two llvm::Values are equal.  There are
01430     (potentially many) cases where the two values actually are equal but
01431     this will return false.  However, if it does return true, the two
01432     vectors definitely are equal.
01433 
01434     @todo This seems to catch all of the cases we currently need it for in
01435     practice, but it's be nice to make it a little more robust/general.  In
01436     general, though, a little something called the halting problem means we
01437     won't get all of them.
01438 */
01439 static bool
01440 lValuesAreEqual(llvm::Value *v0, llvm::Value *v1, 
01441                 std::vector<llvm::PHINode *> &seenPhi0,
01442                 std::vector<llvm::PHINode *> &seenPhi1) {
01443     // Thanks to the fact that LLVM hashes and returns the same pointer for
01444     // constants (of all sorts, even constant expressions), this first test
01445     // actually catches a lot of cases.  LLVM's SSA form also helps a lot
01446     // with this..
01447     if (v0 == v1)
01448         return true;
01449 
01450     assert(seenPhi0.size() == seenPhi1.size());
01451     for (unsigned int i = 0; i < seenPhi0.size(); ++i)
01452         if (v0 == seenPhi0[i] && v1 == seenPhi1[i])
01453             return true;
01454 
01455     llvm::BinaryOperator *bo0 = llvm::dyn_cast<llvm::BinaryOperator>(v0);
01456     llvm::BinaryOperator *bo1 = llvm::dyn_cast<llvm::BinaryOperator>(v1);
01457     if (bo0 != NULL && bo1 != NULL) {
01458         if (bo0->getOpcode() != bo1->getOpcode())
01459             return false;
01460         return (lValuesAreEqual(bo0->getOperand(0), bo1->getOperand(0),
01461                                 seenPhi0, seenPhi1) &&
01462                 lValuesAreEqual(bo0->getOperand(1), bo1->getOperand(1),
01463                                 seenPhi0, seenPhi1));
01464     }
01465 
01466     llvm::PHINode *phi0 = llvm::dyn_cast<llvm::PHINode>(v0);
01467     llvm::PHINode *phi1 = llvm::dyn_cast<llvm::PHINode>(v1);
01468     if (phi0 != NULL && phi1 != NULL) {
01469         if (phi0->getNumIncomingValues() != phi1->getNumIncomingValues())
01470             return false;
01471 
01472         seenPhi0.push_back(phi0);
01473         seenPhi1.push_back(phi1);
01474 
01475         unsigned int numIncoming = phi0->getNumIncomingValues();
01476         // Check all of the incoming values: if all of them are all equal,
01477         // then we're good.
01478         bool anyFailure = false;
01479         for (unsigned int i = 0; i < numIncoming; ++i) {
01480             assert(phi0->getIncomingBlock(i) == phi1->getIncomingBlock(i));
01481             if (!lValuesAreEqual(phi0->getIncomingValue(i), 
01482                                  phi1->getIncomingValue(i), seenPhi0, seenPhi1)) {
01483                 anyFailure = true;
01484                 break;
01485             }
01486         }
01487 
01488         seenPhi0.pop_back();
01489         seenPhi1.pop_back();
01490 
01491         return !anyFailure;
01492     }
01493 
01494     return false;
01495 }

01605     elements.

01616     if (ci == NULL)

01632             return false;

01687     if (bop != NULL) {
01688         // FIXME: is it right to pass the seenPhis to the all equal check as well??
01689         llvm::Value *op0 = bop->getOperand(0), *op1 = bop->getOperand(1);
01690 
01691         if (bop->getOpcode() == llvm::Instruction::Add)
01692             // There are two cases to check if we have an add:
01693             //
01694             // programIndex + unif -> ascending linear seqeuence
01695             // unif + programIndex -> ascending linear sequence
01696             return ((lVectorIsLinear(op0, vectorLength, stride, seenPhis) &&
01697                      lVectorValuesAllEqual(op1, vectorLength, seenPhis)) ||
01698                     (lVectorIsLinear(op1, vectorLength, stride, seenPhis) &&
01699                      lVectorValuesAllEqual(op0, vectorLength, seenPhis)));
01700         else if (bop->getOpcode() == llvm::Instruction::Sub)
01701             // For subtraction, we only match:
01702             //
01703             // programIndex - unif -> ascending linear seqeuence
01704             //
01705             // In the future, we could also look for:
01706             // unif - programIndex -> *descending* linear seqeuence
01707             // And generate code for that as a vector load + shuffle.
01708             return (lVectorIsLinear(bop->getOperand(0), vectorLength,
01709                                     stride, seenPhis) &&
01710                     lVectorValuesAllEqual(bop->getOperand(1), vectorLength,
01711                                           seenPhis));
01712         else if (bop->getOpcode() == llvm::Instruction::Mul)
01713             // Multiplies are a bit trickier, so are handled in a separate
01714             // function.
01715             return (lCheckMulForLinear(op0, op1, vectorLength, stride, seenPhis) ||
01716                     lCheckMulForLinear(op1, op0, vectorLength, stride, seenPhis));
01717         else
01718             return false;
01719     }
01720 
01721     llvm::CastInst *ci = llvm::dyn_cast<llvm::CastInst>(v);
01722     if (ci != NULL)
01723         return lVectorIsLinear(ci->getOperand(0), vectorLength,
01724                                stride, seenPhis);
01725 
01726     if (llvm::isa<llvm::CallInst>(v) || llvm::isa<llvm::LoadInst>(v))
01727         return false;
01728 
01729     llvm::PHINode *phi = llvm::dyn_cast<llvm::PHINode>(v);
01730     if (phi != NULL) {
01731         for (unsigned int i = 0; i < seenPhis.size(); ++i)
01732             if (seenPhis[i] == phi)
01733                 return true;
01734 
01735         seenPhis.push_back(phi);
01736 
01737         unsigned int numIncoming = phi->getNumIncomingValues();
01738         // Check all of the incoming values: if all of them are all equal,
01739         // then we're good.
01740         for (unsigned int i = 0; i < numIncoming; ++i) {
01741             if (!lVectorIsLinear(phi->getIncomingValue(i), vectorLength, stride,
01742                                  seenPhis)) {
01743                 seenPhis.pop_back();
01744                 return false;
01745             }
01746         }
01747 
01748         seenPhis.pop_back();
01749         return true;
01750     }
01751 
01752     // TODO: is any reason to worry about these?
01753     if (llvm::isa<llvm::InsertElementInst>(v))
01754         return false;
01755 
01756     // TODO: we could also handle shuffles, but we haven't yet seen any
01757     // cases where doing so would detect cases where actually have a linear
01758     // vector.
01759     llvm::ShuffleVectorInst *shuffle = llvm::dyn_cast<llvm::ShuffleVectorInst>(v);
01760     if (shuffle != NULL)
01761         return false;
01762 
01763 #if 0
01764     fprintf(stderr, "linear check: ");
01765     v->dump();
01766     fprintf(stderr, "\n");
01767     llvm::Instruction *inst = llvm::dyn_cast<llvm::Instruction>(v);
01768     if (inst) {
01769         inst->getParent()->dump();
01770         fprintf(stderr, "\n");
01771         fprintf(stderr, "\n");
01772     }
01773 #endif
01774 
01775     return false;
01776 }

01872             if (sInfo[i].pseudoFunc != NULL &&
01873                 calledFunc == sInfo[i].pseudoFunc) {
01874                 scatterInfo = &sInfo[i];
01875                 break;
01876             }
01877         }

01879             continue;
01880 
01881         SourcePos pos;

01883         assert(ok);

01886         llvm::Value *offsets = callInst->getArgOperand(1);

01888 
01889         {

01955         }
01956         }
01957 
01958         int step = gatherInfo ? gatherInfo->align : scatterInfo->align;
01959         std::vector<llvm::PHINode *> seenPhis;
01960         if (lVectorIsLinear(offsets, g->target.vectorWidth, step, seenPhis)) {
01961             // We have a linear sequence of memory locations being accessed
01962             // starting with the location given by the offset from
01963             // offsetElements[0], with stride of 4 or 8 bytes (for 32 bit
01964             // and 64 bit gather/scatters, respectively.)
01965 
01966             // Get the base pointer using the first guy's offset.
01967             llvm::Value *firstOffset = 
01968                 llvm::ExtractElementInst::Create(offsets, LLVMInt32(0), "first_offset",
01969                                                  callInst);
01970             llvm::Value *indices[1] = { firstOffset };
01971 #if defined(LLVM_3_0) || defined(LLVM_3_0svn) || defined(LLVM_3_1svn)
01972             llvm::ArrayRef<llvm::Value *> arrayRef(&indices[0], &indices[1]);
01973             llvm::Value *ptr = 
01974                 llvm::GetElementPtrInst::Create(base, arrayRef, "ptr", callInst);
01975 #else
01976             llvm::Value *ptr = 
01977                 llvm::GetElementPtrInst::Create(base, &indices[0], &indices[1],
01978                                                 "ptr", callInst);
01979 #endif
01980             lCopyMetadata(ptr, callInst);
01981 
01982             if (gatherInfo != NULL) {
01983                 Debug(pos, "Transformed gather to unaligned vector load!");
01984                 llvm::Value *args[2] = { ptr, mask };
01985 #if defined(LLVM_3_0) || defined(LLVM_3_0svn) || defined(LLVM_3_1svn)
01986                 llvm::ArrayRef<llvm::Value *> argArray(&args[0], &args[2]);
01987                 llvm::Instruction *newCall = 
01988                     llvm::CallInst::Create(gatherInfo->loadMaskedFunc, argArray, 
01989                                            "load_masked", (llvm::Instruction *)NULL);
01990 #else
01991                 llvm::Instruction *newCall = 
01992                     llvm::CallInst::Create(gatherInfo->loadMaskedFunc, &args[0],
01993                                            &args[2], "load_masked");
01994 #endif
01995                 lCopyMetadata(newCall, callInst);
01996                 llvm::ReplaceInstWithInst(callInst, newCall);
01997             }
01998             else {
01999                 Debug(pos, "Transformed scatter to unaligned vector store!");
02000                 llvm::Value *rvalue = callInst->getArgOperand(2);
02001                 ptr = new llvm::BitCastInst(ptr, scatterInfo->vecPtrType, "ptrcast", 
02002                                             callInst);
02003 
02004                 llvm::Value *args[3] = { ptr, rvalue, mask };
02005 #if defined(LLVM_3_0) || defined(LLVM_3_0svn) || defined(LLVM_3_1svn)
02006                 llvm::ArrayRef<llvm::Value *> argArray(&args[0], &args[3]);
02007                 llvm::Instruction *newCall = 
02008                     llvm::CallInst::Create(scatterInfo->maskedStoreFunc, argArray,
02009                                            "", (llvm::Instruction *)NULL);
02010 #else
02011                 llvm::Instruction *newCall = 
02012                     llvm::CallInst::Create(scatterInfo->maskedStoreFunc,
02013                                            &args[0], &args[3], "");
02014 #endif
02015                 lCopyMetadata(newCall, callInst);
02016                 llvm::ReplaceInstWithInst(callInst, newCall);
02017             }
02018 
02019             modifiedAny = true;
02020             goto restart;
02021         }
02022     }
02023 
02024     return modifiedAny;
02025 }
