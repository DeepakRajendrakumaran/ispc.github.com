00153 /** We have a protocol with the front-end LLVM IR code generation process

00181     if (!filename || !first_line || !first_column || !last_line || !last_column)

00613 /** Given a vector of constant values (int, float, or bool) representing an
00614     execution mask, convert it to a bitvector where the 0th bit corresponds
00615     to the first vector value and so forth.
00616 */
00617 static int
00618 lConstElementsToMask(const llvm::SmallVector<llvm::Constant *, 
00619                                              ISPC_MAX_NVEC> &elements) {
00620     Assert(elements.size() <= 32);
00621 
00622     int mask = 0;
00623     for (unsigned int i = 0; i < elements.size(); ++i) {
00624         llvm::APInt intMaskValue;
00625         // SSE has the "interesting" approach of encoding blending
00626         // masks as <n x float>.
00627         llvm::ConstantFP *cf = llvm::dyn_cast<llvm::ConstantFP>(elements[i]);
00628         if (cf != NULL) {
00629             llvm::APFloat apf = cf->getValueAPF();
00630             intMaskValue = apf.bitcastToAPInt();
00631         }
00632         else {
00633             // Otherwise get it as an int
00634             llvm::ConstantInt *ci = llvm::dyn_cast<llvm::ConstantInt>(elements[i]);
00635             Assert(ci != NULL);  // vs return -1 if NULL?
00636             intMaskValue = ci->getValue();
00637         }
00638         // Is the high-bit set?  If so, OR in the appropriate bit in
00639         // the result mask
00640         if (intMaskValue.countLeadingOnes() > 0)
00641             mask |= (1 << i);
00642     }
00643     return mask;
00644 }

00647 /** Given an llvm::Value represinting a vector mask, see if the value is a
00648     constant.  If so, return the integer mask found by taking the high bits
00649     of the mask values in turn and concatenating them into a single integer.
00650     In other words, given the 4-wide mask: < 0xffffffff, 0, 0, 0xffffffff >, 
00651     we have 0b1001 = 9.
00652  */

00654 lGetMask(llvm::Value *factor) {

00656        it don't be possible to distinguish between -1 for "don't know" and
00657        "known and all bits on". */

00659 
00660 #ifdef LLVM_3_1svn
00661     llvm::ConstantDataVector *cdv = llvm::dyn_cast<llvm::ConstantDataVector>(factor);
00662     if (cdv != NULL) {
00663         llvm::SmallVector<llvm::Constant *, ISPC_MAX_NVEC> elements;
00664         for (int i = 0; i < (int)cdv->getNumElements(); ++i)
00665             elements.push_back(cdv->getElementAsConstant(i));

00673  #ifdef LLVM_3_1svn

00675             llvm::Constant *c = 
00676                 llvm::dyn_cast<llvm::Constant>(cv->getOperand(i));
00677             if (c == NULL)
00678                 return NULL;
00679             elements.push_back(c);
00680         }
00681 #else
00682         cv->getVectorElements(elements);
00683 #endif
00684         return lConstElementsToMask(elements);
00685     }
00686     else if (llvm::isa<llvm::ConstantAggregateZero>(factor))
00687         return 0;
00688     else {
00689 #if 0

00728 bool
00729 IntrinsicsOpt::runOnBasicBlock(llvm::BasicBlock &bb) {
00730     DEBUG_START_PASS("IntrinsicsOpt");
00731 
00732     llvm::Function *avxMaskedLoad32 = 
00733         llvm::Intrinsic::getDeclaration(m->module, llvm::Intrinsic::x86_avx_maskload_ps_256);
00734     llvm::Function *avxMaskedLoad64 = 
00735         llvm::Intrinsic::getDeclaration(m->module, llvm::Intrinsic::x86_avx_maskload_pd_256);
00736     llvm::Function *avxMaskedStore32 = 
00737         llvm::Intrinsic::getDeclaration(m->module, llvm::Intrinsic::x86_avx_maskstore_ps_256);
00738     llvm::Function *avxMaskedStore64 = 
00739         llvm::Intrinsic::getDeclaration(m->module, llvm::Intrinsic::x86_avx_maskstore_pd_256);
00740     Assert(avxMaskedLoad32 != NULL && avxMaskedStore32 != NULL);
00741     Assert(avxMaskedLoad64 != NULL && avxMaskedStore64 != NULL);
00742 
00743     bool modifiedAny = false;
00744  restart:
00745     for (llvm::BasicBlock::iterator iter = bb.begin(), e = bb.end(); iter != e; ++iter) {
00746         llvm::CallInst *callInst = llvm::dyn_cast<llvm::CallInst>(&*iter);
00747         if (callInst == NULL || callInst->getCalledFunction() == NULL)
00748             continue;
00749 
00750         BlendInstruction *blend = matchingBlendInstruction(callInst->getCalledFunction());
00751         if (blend != NULL) {
00752             llvm::Value *v[2] = { callInst->getArgOperand(blend->op0), 
00753                                   callInst->getArgOperand(blend->op1) };
00754             llvm::Value *factor = callInst->getArgOperand(blend->opFactor);
00755 
00756             // If the values are the same, then no need to blend..
00757             if (v[0] == v[1]) {
00758                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(), 
00759                                            iter, v[0]);
00760                 modifiedAny = true;
00761                 goto restart;
00762             }
00763 
00764             // If one of the two is undefined, we're allowed to replace
00765             // with the value of the other.  (In other words, the only
00766             // valid case is that the blend factor ends up having a value
00767             // that only selects from the defined one of the two operands,
00768             // otherwise the result is undefined and any value is fine,
00769             // ergo the defined one is an acceptable result.)
00770             if (lIsUndef(v[0])) {
00771                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(), 
00772                                            iter, v[1]);
00773                 modifiedAny = true;
00774                 goto restart;
00775             }
00776             if (lIsUndef(v[1])) {
00777                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(), 
00778                                            iter, v[0]);
00779                 modifiedAny = true;
00780                 goto restart;
00781             }
00782 
00783             int mask = lGetMask(factor);
00784             llvm::Value *value = NULL;
00785             if (mask == 0)
00786                 // Mask all off -> replace with the first blend value
00787                 value = v[0];
00788             else if (mask == blend->allOnMask)
00789                 // Mask all on -> replace with the second blend value
00790                 value = v[1];
00791 
00792             if (value != NULL) {
00793                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(), 
00794                                            iter, value);
00795                 modifiedAny = true;
00796                 goto restart;
00797             }
00798         }
00799         else if (matchesMaskInstruction(callInst->getCalledFunction())) {
00800             llvm::Value *factor = callInst->getArgOperand(0);
00801             int mask = lGetMask(factor);
00802             if (mask != -1) {
00803                 // If the vector-valued mask has a known value, replace it
00804                 // with the corresponding integer mask from its elements
00805                 // high bits.
00806                 llvm::Value *value = LLVMInt32(mask);
00807                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(),
00808                                            iter, value);
00809                 modifiedAny = true;
00810                 goto restart;
00811             }
00812         }
00813         else if (callInst->getCalledFunction() == avxMaskedLoad32 ||
00814                  callInst->getCalledFunction() == avxMaskedLoad64) {
00815             llvm::Value *factor = callInst->getArgOperand(1);
00816             int mask = lGetMask(factor);
00817             if (mask == 0) {
00818                 // nothing being loaded, replace with undef value
00819                 llvm::Type *returnType = callInst->getType();
00820                 Assert(llvm::isa<llvm::VectorType>(returnType));
00821                 llvm::Value *undefValue = llvm::UndefValue::get(returnType);
00822                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(),
00823                                            iter, undefValue);
00824                 modifiedAny = true;
00825                 goto restart;
00826             }
00827             else if (mask == 0xff) {
00828                 // all lanes active; replace with a regular load
00829                 llvm::Type *returnType = callInst->getType();
00830                 Assert(llvm::isa<llvm::VectorType>(returnType));
00831                 // cast the i8 * to the appropriate type
00832                 const char *name = LLVMGetName(callInst->getArgOperand(0), "_cast");
00833                 llvm::Value *castPtr = 
00834                     new llvm::BitCastInst(callInst->getArgOperand(0),
00835                                           llvm::PointerType::get(returnType, 0), 
00836                                           name, callInst);
00837                 lCopyMetadata(castPtr, callInst);
00838                 int align = callInst->getCalledFunction() == avxMaskedLoad32 ? 4 : 8;
00839                 name = LLVMGetName(callInst->getArgOperand(0), "_load");
00840                 llvm::Instruction *loadInst = 
00841                     new llvm::LoadInst(castPtr, name, false /* not volatile */,
00842                                        align, (llvm::Instruction *)NULL);
00843                 lCopyMetadata(loadInst, callInst);
00844                 llvm::ReplaceInstWithInst(callInst, loadInst);
00845                 modifiedAny = true;
00846                 goto restart;
00847             }
00848         }
00849         else if (callInst->getCalledFunction() == avxMaskedStore32 ||
00850                  callInst->getCalledFunction() == avxMaskedStore64) {
00851             // NOTE: mask is the 2nd parameter, not the 3rd one!!
00852             llvm::Value *factor = callInst->getArgOperand(1);
00853             int mask = lGetMask(factor);
00854             if (mask == 0) {
00855                 // nothing actually being stored, just remove the inst
00856                 callInst->eraseFromParent();
00857                 modifiedAny = true;
00858                 goto restart;
00859             }
00860             else if (mask == 0xff) {
00861                 // all lanes storing, so replace with a regular store
00862                 llvm::Value *rvalue = callInst->getArgOperand(2);
00863                 llvm::Type *storeType = rvalue->getType();
00864                 const char *name = LLVMGetName(callInst->getArgOperand(0),
00865                                                "_ptrcast");
00866                 llvm::Value *castPtr = 
00867                     new llvm::BitCastInst(callInst->getArgOperand(0),
00868                                           llvm::PointerType::get(storeType, 0), 
00869                                           name, callInst);
00870                 lCopyMetadata(castPtr, callInst);
00871 
00872                 llvm::StoreInst *storeInst = 
00873                     new llvm::StoreInst(rvalue, castPtr, (llvm::Instruction *)NULL);
00874                 int align = callInst->getCalledFunction() == avxMaskedStore32 ? 4 : 8;
00875                 storeInst->setAlignment(align);
00876                 lCopyMetadata(storeInst, callInst);
00877                 llvm::ReplaceInstWithInst(callInst, storeInst);
00878 
00879                 modifiedAny = true;
00880                 goto restart;
00881             }
00882         }
00883     }
00884 
00885     DEBUG_END_PASS("IntrinsicsOpt");
00886 
00887     return modifiedAny;
00888 }

01053             uce->getOpcode() == llvm::Instruction::PtrToInt)
01054             return v;

01075         // value.

01077             if (elements[i] == NULL)

01261 /** Given a vector expression in vec, separate it into a compile-time
01262     constant component and a variable component, returning the two parts in
01263     *constOffset and *variableOffset.  (It should be the case that the sum
01264     of these two is exactly equal to the original vector.)

01267     will fail and return components that are actually compile-time

01271     components, the corresponding return value may be set to NULL.
01272  */

01274 lExtractConstantOffset(llvm::Value *vec, llvm::Value **constOffset,
01275                        llvm::Value **variableOffset, 
01276                        llvm::Instruction *insertBefore) {

01278 #ifdef LLVM_3_1svn
01279         llvm::isa<llvm::ConstantDataVector>(vec) ||
01280 #endif

01283         *variableOffset = NULL;

01288     if (sext != NULL) {

01290         llvm::Value *co, *vo;

01368

02134 /** This routine attempts to determine if the given pointer in lvalue is
02135     pointing to stack-allocated memory.  It's conservative in that it
02136     should never return true for non-stack allocated memory, but may return
02137     false for memory that actually is stack allocated.  The basic strategy
02138     is to traverse through the operands and see if the pointer originally
02139     comes from an AllocaInst.
02140 */
02141 static bool
02142 lIsSafeToBlend(llvm::Value *lvalue) {
02143     llvm::BitCastInst *bc = llvm::dyn_cast<llvm::BitCastInst>(lvalue);
02144     if (bc != NULL)
02145         return lIsSafeToBlend(bc->getOperand(0));
02146     else {
02147         llvm::AllocaInst *ai = llvm::dyn_cast<llvm::AllocaInst>(lvalue);
02148         if (ai) {
02149             llvm::Type *type = ai->getType();
02150             llvm::PointerType *pt = 
02151                 llvm::dyn_cast<llvm::PointerType>(type);
02152             assert(pt != NULL);
02153             type = pt->getElementType();
02154             llvm::ArrayType *at;
02155             while ((at = llvm::dyn_cast<llvm::ArrayType>(type))) {
02156                 type = at->getElementType();
02157             }
02158             llvm::VectorType *vt = 
02159                 llvm::dyn_cast<llvm::VectorType>(type);
02160             return (vt != NULL && 
02161                     (int)vt->getNumElements() == g->target.vectorWidth);
02162         }
02163         else {
02164             llvm::GetElementPtrInst *gep = 
02165                 llvm::dyn_cast<llvm::GetElementPtrInst>(lvalue);
02166             if (gep != NULL)
02167                 return lIsSafeToBlend(gep->getOperand(0));
02168             else
02169                 return false;
02170         }
02171     }

02397                 break;

02399         }

02409         llvm::Value *constOffsets = callInst->getArgOperand(3);

02421             if (varyingOffsets->getType() == LLVMTypes::Int64VectorType)
02422                 scales.push_back(LLVMInt64(scaleValue));
02423             else
02424                 scales.push_back(LLVMInt32((int32_t)scaleValue));

02427 
02428         llvm::Value *scaledVarying =

02430                                          varyingOffsets, "scaled_varying", callInst);

02441             // pointer they all represent based on the first one of them

02472                 // we arbitrarily write one of the values, but we need to

02474                 // the value from an executing program instance in that

02476                 // do redundant writes, since this isn't important to make
02477                 // fast anyway...
02478             }
02479         }
02480         else {

02494                         lCallInst(gatherInfo->loadMaskedFunc, ptr, mask, 
02495                                   LLVMGetName(ptr, "_masked_load"));

02497                     llvm::ReplaceInstWithInst(callInst, newCall);
02498                 }
02499                 else {
02500                     Debug(pos, "Transformed scatter to unaligned vector store!");

02502                                                 callInst);
02503                     llvm::Instruction *newCall =
02504                         lCallInst(scatterInfo->maskedStoreFunc, ptr, storeValue, 
02505                                   mask, "");
02506                     lCopyMetadata(newCall, callInst);
02507                     llvm::ReplaceInstWithInst(callInst, newCall);
02508                 }
02509 
02510                 modifiedAny = true;
02511                 goto restart;
02512             }
02513         }
02514     }

02516     DEBUG_END_PASS("GSToLoadStorePass");
02517

02528 ///////////////////////////////////////////////////////////////////////////
02529 // GatherCoalescePass
02530 
02531 // This pass implements two optimizations to improve the performance of
02532 // gathers; currently only gathers of 32-bit values where it can be
02533 // determined at compile time that the mask is all on are supported, though
02534 // both of those limitations may be generalized in the future.
02535 //
02536 //  First, for any single gather, see if it's worthwhile to break it into
02537 //  any of scalar, 2-wide (i.e. 64-bit), 4-wide, or 8-wide loads.  Further,
02538 //  we generate code that shuffles these loads around.  Doing fewer, larger
02539 //  loads in this manner, when possible, can be more efficient.
02540 //    
02541 //  Second, this pass can coalesce memory accesses across multiple
02542 //  gathers. If we have a series of gathers without any memory writes in
02543 //  the middle, then we try to analyze their reads collectively and choose
02544 //  an efficient set of loads for them.  Not only does this help if
02545 //  different gathers reuse values from the same location in memory, but
02546 //  it's specifically helpful when data with AOS layout is being accessed;
02547 //  in this case, we're often able to generate wide vector loads and
02548 //  appropriate shuffles automatically.
02549

02551 public:
02552     static char ID;
02553     GatherCoalescePass() : BasicBlockPass(ID) { }
02554 
02555     const char *getPassName() const { return "Gather Coalescing"; }
02556     bool runOnBasicBlock(llvm::BasicBlock &BB);
02557 };

02601 /** This function determines whether it makes sense (and is safe) to
02602     generate a vector load of width vectorWidth, starting at *iter.  It
02603     returns true if so, setting *newIter to point to the next element in
02604     the set that isn't taken care of by the generated load.  If a vector
02605     load of the given width doesn't make sense, then false is returned.
02606  */
02607 static bool
02608 lVectorLoadIsEfficient(std::set<int64_t>::iterator iter, 
02609                        std::set<int64_t>::iterator end,
02610                        std::set<int64_t>::iterator *newIter, int vectorWidth) {
02611     // We're considering a vector load of width vectorWidth, starting at
02612     // the offset "start".
02613     int64_t start = *iter;
02614 
02615     // The basic idea is that we'll look at the subsequent elements in the
02616     // load set after the initial one at start.  As long as subsequent
02617     // elements:
02618     //
02619     // 1. Aren't so far separated that they no longer fit into the range
02620     //    [start, start+vectorWidth)
02621     //
02622     // 2. And don't have too large a gap in between them (e.g., it's not
02623     //    worth generating an 8-wide load for two elements with offsets 0
02624     //    and 7, but no loads requested in between).  
02625     //
02626     // Then we continue moving forward through the elements until we either
02627     // fill up the vector or run out of elements.
02628 
02629     // lastAccepted holds the last offset we've processed and accepted as
02630     // valid for the vector load underconsideration
02631     int64_t lastAccepted = start;
02632 
02633     while (iter != end) {
02634         // What is the separation in offset values from the last element we
02635         // added to the set for this load?
02636         int64_t delta = *iter - lastAccepted;
02637         if (delta > 3)
02638             // If there's too big a gap, then we won't issue the load
02639             return false;
02640 
02641         int64_t span = *iter - start + 1;
02642 
02643         if (span == vectorWidth) {
02644             // We've extended far enough that we have exactly filled up the
02645             // entire vector width; we can't go any further, so return with
02646             // success.  (Update *newIter to point at the next element
02647             // after the last one accepted here.)
02648             *newIter = ++iter;
02649             return true;
02650         }
02651         else if (span > vectorWidth) {
02652             // The current offset won't fit into a vectorWidth-wide load
02653             // starting from start.  It's still generally worthwhile
02654             // issuing the load we've been considering, though, since it
02655             // will provide values for a number of previous offsets.  This
02656             // load will have one or more elements at the end of its range
02657             // that is not needed by any of the offsets under
02658             // consideration.  As such, there are three cases where issuing
02659             // this load is a bad idea:
02660             //
02661             // 1. 2-wide loads: we know that we haven't completely filled
02662             //    the 2-wide vector, since otherwise the if() test above
02663             //    would have succeeded previously.  Therefore, we must have
02664             //    a situation with offsets like (4,6,...); it would be a
02665             //    silly idea to issue a 2-wide load to get the value for
02666             //    the 4 offset, versus failing here and issuing a scalar
02667             //    load instead.
02668             //
02669             // 2. If there are too many unnecessary values at the end of
02670             //    the load extent (defined as more than half of them)--in
02671             //    this case, it'd be better to issue a vector load of
02672             //    smaller width anyway.
02673             //
02674             // 3. If the gap between the last accepted offset and the
02675             //    current one under consideration is more than the page
02676             //    size.  In this case we can't be sure whether or not some
02677             //    of the unused elements at the end of the load will
02678             //    straddle a page boundary and thus lead to an undesirable
02679             //    fault.  (It's hard to imagine this happening in practice,
02680             //    except under contrived circumstances, but better safe
02681             //    than sorry.)
02682             const int pageSize = 4096;
02683             if (vectorWidth != 2 && 
02684                 (lastAccepted - start) > (vectorWidth / 2) &&
02685                 (*iter - lastAccepted) < pageSize) {
02686                 *newIter = iter;
02687                 return true;
02688             }
02689             else
02690                 return false;
02691         }
02692 
02693         // Continue moving forward 
02694         lastAccepted = *iter;
02695         ++iter;
02696     }
02697 
02698     return false;
02699 }

02702 /** Given a set of offsets from a common base pointer that we need to get
02703     loaded into memory, determine a reasonable set of load operations that
02704     gets all of the corresponding values in memory (ideally, including as
02705     many as possible wider vector loads rather than scalar loads).  Return
02706     a CoalescedLoadOp for each one in the *loads array.
02707  */
02708 static void
02709 lSelectLoads(const std::vector<int64_t> &loadOffsets,
02710              std::vector<CoalescedLoadOp> *loads) {
02711     // First, get a sorted set of unique offsets to load from.
02712     std::set<int64_t> allOffsets;
02713     for (unsigned int i = 0; i < loadOffsets.size(); ++i)
02714         allOffsets.insert(loadOffsets[i]);
02715 
02716     std::set<int64_t>::iterator iter = allOffsets.begin();
02717     while (iter != allOffsets.end()) {
02718         Debug(SourcePos(), "Load needed at %" PRId64 ".", *iter);
02719         ++iter;
02720     }
02721 
02722     // Now, iterate over the offsets from low to high.  Starting at the
02723     // current offset, we see if a vector load starting from that offset
02724     // will cover loads at subsequent offsets as well.
02725     iter = allOffsets.begin();
02726     while (iter != allOffsets.end()) {
02727         // Consider vector loads of width of each of the elements of
02728         // spanSizes[], in order.
02729         int vectorWidths[] = { 8, 4, 2 };
02730         int nVectorWidths = sizeof(vectorWidths) / sizeof(vectorWidths[0]);
02731         bool gotOne = false;
02732         for (int i = 0; i < nVectorWidths; ++i) {
02733             // See if a load of vector with width vectorWidths[i] would be
02734             // effective (i.e. would cover a reasonable number of the
02735             // offsets that need to be loaded from).
02736             std::set<int64_t>::iterator newIter;
02737             if (lVectorLoadIsEfficient(iter, allOffsets.end(), &newIter, 
02738                                        vectorWidths[i])) {
02739                 // Yes: create the corresponding coalesced load and update
02740                 // the iterator to the returned iterator; doing so skips
02741                 // over the additional offsets that are taken care of by
02742                 // this load.
02743                 loads->push_back(CoalescedLoadOp(*iter, vectorWidths[i]));
02744                 iter = newIter;
02745                 gotOne = true;
02746                 break;
02747             }
02748         }
02749         
02750         if (gotOne == false) {
02751             // We couldn't find a vector load starting from this offset
02752             // that made sense, so emit a scalar load and continue onward.
02753             loads->push_back(CoalescedLoadOp(*iter, 1));
02754             ++iter;
02755         }
02756     }
02757 }

02845 lEmitLoads(llvm::Value *basePtr, std::vector<CoalescedLoadOp> &loadOps, 
02846            int elementSize, llvm::Instruction *insertBefore) {
02847     Debug(SourcePos(), "Coalesce doing %d loads.", (int)loadOps.size());
02848     for (int i = 0; i < (int)loadOps.size(); ++i) {
02849         Debug(SourcePos(), "Load #%d @ %" PRId64 ", %d items", i, loadOps[i].start, 
02850               loadOps[i].count);
02851 
02852         // basePtr is an i8 *, so the offset from it should be in terms of
02853         // bytes, not underlying i32 elements.
02854         int64_t start = loadOps[i].start * elementSize;
02855 
02856         int align = 4;
02857         switch (loadOps[i].count) {
02858         case 1:
02859             // Single 32-bit scalar load
02860             loadOps[i].load = lGEPAndLoad(basePtr, start, align, insertBefore,
02861                                           LLVMTypes::Int32Type);
02862             break;
02863         case 2: {
02864             // Emit 2 x i32 loads as i64 loads and then break the result
02865             // into two 32-bit parts.
02866             loadOps[i].load = lGEPAndLoad(basePtr, start, align, insertBefore,
02867                                           LLVMTypes::Int64Type);
02868             // element0 = (int32)value;
02869             loadOps[i].element0 = 
02870                 new llvm::TruncInst(loadOps[i].load, LLVMTypes::Int32Type,
02871                                     "load64_elt0", insertBefore);
02872             // element1 = (int32)(value >> 32)
02873             llvm::Value *shift =
02874                 llvm::BinaryOperator::Create(llvm::Instruction::LShr,
02875                                              loadOps[i].load, LLVMInt64(32),
02876                                              "load64_shift", insertBefore);
02877             loadOps[i].element1 = 
02878                 new llvm::TruncInst(shift, LLVMTypes::Int32Type,
02879                                     "load64_elt1", insertBefore);
02880             break;
02881         }
02882         case 4: {
02883             // 4-wide vector load
02884             llvm::VectorType *vt =
02885                 llvm::VectorType::get(LLVMTypes::Int32Type, 4);
02886             loadOps[i].load = lGEPAndLoad(basePtr, start, align,
02887                                           insertBefore, vt);
02888             break;
02889         }
02890         case 8: {
02891             // 8-wide vector load
02892             llvm::VectorType *vt =
02893                 llvm::VectorType::get(LLVMTypes::Int32Type, 8);
02894             loadOps[i].load = lGEPAndLoad(basePtr, start, align, 
02895                                           insertBefore, vt);
02896             break;
02897         }
02898         default:
02899             FATAL("Unexpected load count in lEmitLoads()");
02900         }
02901     }
02902 }

02943             llvm::Instruction *insertBefore) {

02946             offsets[elt] < load.start + load.count) {

02948                   "(value %" PRId64 ")", load.start, elt, offsets[elt]);
02949             // If this load gives one of the values that we need, then we
02950             // can just insert it in directly
02951             Assert(set[elt] == false);
02952             result = 
02953                 llvm::InsertElementInst::Create(result, load.load, LLVMInt32(elt),
02954                                                 "insert_load", insertBefore);
02955             set[elt] = true;
02956         }
02957     }

02963 /** Similarly, incorporate the values from a 2-wide load into any vector
02964     elements that they apply to. */
02965 static llvm::Value *
02966 lApplyLoad2(llvm::Value *result, const CoalescedLoadOp &load,
02967             const int64_t offsets[4], bool set[4], 
02968             llvm::Instruction *insertBefore) {
02969     for (int elt = 0; elt < 4; ++elt) {
02970         // First, try to do a 64-bit-wide insert into the result vector.
02971         // We can do this when we're currently at an even element, when the
02972         // current and next element have consecutive values, and where the
02973         // original 64-bit load is at the offset needed by the current
02974         // element.
02975         if ((elt & 1) == 0 &&
02976             offsets[elt] + 1 == offsets[elt+1] &&
02977             offsets[elt] == load.start) {
02978             Debug(SourcePos(), "Load 2 @ %" PRId64 " matches for elements #%d,%d "
02979                   "(values %" PRId64 ",%" PRId64 ")", load.start, elt, elt+1,
02980                   offsets[elt], offsets[elt+1]);
02981             Assert(set[elt] == false && set[elt+1] == false);
02982 
02983             // In this case, we bitcast from a 4xi32 to a 2xi64 vector
02984             llvm::Type *vec2x64Type = 
02985                 llvm::VectorType::get(LLVMTypes::Int64Type, 2);
02986             result = new llvm::BitCastInst(result, vec2x64Type, "to2x64",
02987                                            insertBefore);
02988 
02989             // And now we can insert the 64-bit wide value into the
02990             // appropriate elment
02991             result = llvm::InsertElementInst::Create(result, load.load,
02992                                                      LLVMInt32(elt/2), 
02993                                                      "insert64", insertBefore);
02994             
02995             // And back to 4xi32.
02996             llvm::Type *vec4x32Type = 
02997                 llvm::VectorType::get(LLVMTypes::Int32Type, 4);
02998             result = new llvm::BitCastInst(result, vec4x32Type, "to4x32",
02999                                            insertBefore);
03000 
03001             set[elt] = set[elt+1] = true;
03002             // Advance elt one extra time, since we just took care of two
03003             // elements
03004             ++elt;
03005         }
03006         else if (offsets[elt] >= load.start && 
03007                  offsets[elt] < load.start + load.count) {
03008             Debug(SourcePos(), "Load 2 @ %" PRId64 " matches for element #%d "
03009                   "(value %" PRId64 ")", load.start, elt, offsets[elt]);
03010             // Otherwise, insert one of the 32-bit pieces into an element
03011             // of the final vector
03012             Assert(set[elt] == false);
03013             llvm::Value *toInsert = (offsets[elt] == load.start) ?
03014                 load.element0 : load.element1;
03015             result =                         
03016                 llvm::InsertElementInst::Create(result, toInsert, LLVMInt32(elt),
03017                                                 "insert_load", insertBefore);
03018             set[elt] = true;
03019         }
03020     }
03021 
03022     return result;
03023 }

03325     int width = g->target.vectorWidth;

03341 /** Actually do the coalescing.  We have a set of gathers all accessing
03342     addresses of the form:
03343 
03344     (ptr + {1,2,4,8} * varyingOffset) + constOffset, a.k.a.
03345     basePtr + constOffset
03346 
03347     where varyingOffset actually has the same value across all of the SIMD
03348     lanes and where the part in parenthesis has the same value for all of
03349     the gathers in the group.
03350  */
03351 static bool
03352 lCoalesceGathers(const std::vector<llvm::CallInst *> &coalesceGroup) {
03353     llvm::Instruction *insertBefore = coalesceGroup[0];
03354 
03355     // First, compute the shared base pointer for all of the gathers
03356     llvm::Value *basePtr = lComputeBasePtr(coalesceGroup[0], insertBefore);
03357 
03358     int elementSize;
03359     if (coalesceGroup[0]->getType() == LLVMTypes::Int32VectorType)
03360         elementSize = 4;
03361     else if (coalesceGroup[0]->getType() == LLVMTypes::Int64VectorType)
03362         elementSize = 8;
03363     else
03364         FATAL("Unexpected gather type in lCoalesceGathers");
03365 
03366     // Extract the constant offsets from the gathers into the constOffsets
03367     // vector: the first vectorWidth elements will be those for the first
03368     // gather, the next vectorWidth those for the next gather, and so
03369     // forth.
03370     std::vector<int64_t> constOffsets;
03371     lExtractConstOffsets(coalesceGroup, elementSize, &constOffsets);
03372 
03373     // Determine a set of loads to perform to get all of the values we need
03374     // loaded.
03375     std::vector<CoalescedLoadOp> loadOps;
03376     lSelectLoads(constOffsets, &loadOps);
03377 
03378     lCoalescePerfInfo(coalesceGroup, loadOps);
03379 
03380     // Actually emit load instructions for them
03381     lEmitLoads(basePtr, loadOps, elementSize, insertBefore);
03382 
03383     // Now, for any loads that give us <8 x i32> vectors, split their
03384     // values into two <4 x i32> vectors; it turns out that LLVM gives us
