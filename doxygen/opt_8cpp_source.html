00101 /** This utility routine copies the metadata (if any) attached to the
00102     'from' instruction in the IR to the 'to' instruction.  
00103 
00104     For flexibility, this function takes an llvm::Value rather than an
00105     llvm::Instruction for the 'to' parameter; at some places in the code
00106     below, we sometimes use a llvm::Value to start out storing a value and
00107     then later store instructions.  If a llvm::Value is passed to this, the

00112 lCopyMetadata(llvm::Value *vto, const llvm::Instruction *from) {
00113     llvm::Instruction *to = llvm::dyn_cast<llvm::Instruction>(vto);
00114     if (!to)

00116                                                               
00117     llvm::SmallVector<std::pair<unsigned int, llvm::MDNode *>, 8> metadata;
00118     from->getAllMetadata(metadata);
00119     for (unsigned int i = 0; i < metadata.size(); ++i)
00120         to->setMetadata(metadata[i].first, metadata[i].second);

00124 /** We have a protocol with the front-end LLVM IR code generation process
00125     that allows us to encode the source file position that corresponds with
00126     instructions.  (For example, this allows us to issue performance
00127     warnings related to things like scatter and gather after optimization
00128     has been performed, so that we aren't warning about scatters and
00129     gathers that have been improved to stores and loads by optimization
00130     passes.)  Note that this is slightly redundant with the source file
00131     position encoding generated for debugging symbols, though we don't
00132     always generate debugging information but we do always generate this
00133     position data.
00134 
00135     This function finds the SourcePos that the metadata in the instruction
00136     (if present) corresponds to.  See the implementation of
00137     FunctionEmitContext::addGSMetadata(), which encodes the source position during
00138     code generation.
00139 
00140     @param inst   Instruction to try to find the source position of
00141     @param pos    Output variable in which to store the position
00142     @returns      True if source file position metadata was present and *pos
00143                   has been set.  False otherwise.
00144 */
00145 static bool
00146 lGetSourcePosFromMetadata(const llvm::Instruction *inst, SourcePos *pos) {
00147     llvm::MDNode *filename = inst->getMetadata("filename");
00148     llvm::MDNode *first_line = inst->getMetadata("first_line");
00149     llvm::MDNode *first_column = inst->getMetadata("first_column");
00150     llvm::MDNode *last_line = inst->getMetadata("last_line");
00151     llvm::MDNode *last_column = inst->getMetadata("last_column");
00152     if (!filename || !first_line || !first_column || !last_line || !last_column)
00153         return false;
00154 
00155     // All of these asserts are things that FunctionEmitContext::addGSMetadata() is
00156     // expected to have done in its operation
00157     llvm::MDString *str = llvm::dyn_cast<llvm::MDString>(filename->getOperand(0));
00158     Assert(str);
00159     llvm::ConstantInt *first_lnum = 
00160         llvm::dyn_cast<llvm::ConstantInt>(first_line->getOperand(0));
00161     Assert(first_lnum);
00162     llvm::ConstantInt *first_colnum = 
00163         llvm::dyn_cast<llvm::ConstantInt>(first_column->getOperand(0));
00164     Assert(first_column);
00165     llvm::ConstantInt *last_lnum = 
00166         llvm::dyn_cast<llvm::ConstantInt>(last_line->getOperand(0));
00167     Assert(last_lnum);
00168     llvm::ConstantInt *last_colnum = 
00169         llvm::dyn_cast<llvm::ConstantInt>(last_column->getOperand(0));
00170     Assert(last_column);
00171 
00172     *pos = SourcePos(str->getString().data(), (int)first_lnum->getZExtValue(),
00173                      (int)first_colnum->getZExtValue(), (int)last_lnum->getZExtValue(),
00174                      (int)last_colnum->getZExtValue());
00175     return true;
00176 }

00620 /** Given an llvm::Value represinting a vector mask, see if the value is a
00621     constant.  If so, return the integer mask found by taking the high bits
00622     of the mask values in turn and concatenating them into a single integer.
00623     In other words, given the 4-wide mask: < 0xffffffff, 0, 0, 0xffffffff >, 
00624     we have 0b1001 = 9.
00625  */
00626 static int
00627 lGetMask(llvm::Value *factor) {
00628     /* FIXME: This will break if we ever do 32-wide compilation, in which case
00629        it don't be possible to distinguish between -1 for "don't know" and
00630        "known and all bits on". */
00631     Assert(g->target.vectorWidth < 32);
00632 
00633 #ifdef LLVM_3_1svn
00634     llvm::ConstantDataVector *cv = llvm::dyn_cast<llvm::ConstantDataVector>(factor);
00635 #else
00636     llvm::ConstantVector *cv = llvm::dyn_cast<llvm::ConstantVector>(factor);
00637 #endif
00638     if (cv) {
00639         int mask = 0;
00640         llvm::SmallVector<llvm::Constant *, ISPC_MAX_NVEC> elements;
00641 #ifdef LLVM_3_1svn
00642         for (int i = 0; i < (int)cv->getNumElements(); ++i)
00643             elements.push_back(cv->getElementAsConstant(i));
00644 #else
00645         cv->getVectorElements(elements);
00646 #endif
00647 
00648         for (unsigned int i = 0; i < elements.size(); ++i) {
00649             llvm::APInt intMaskValue;
00650             // SSE has the "interesting" approach of encoding blending
00651             // masks as <n x float>.
00652             llvm::ConstantFP *cf = llvm::dyn_cast<llvm::ConstantFP>(elements[i]);
00653             if (cf) {
00654                 llvm::APFloat apf = cf->getValueAPF();
00655                 intMaskValue = apf.bitcastToAPInt();
00656             }
00657             else {
00658                 // Otherwise get it as an int
00659                 llvm::ConstantInt *ci = llvm::dyn_cast<llvm::ConstantInt>(elements[i]);
00660                 Assert(ci != NULL);  // vs return -1 if NULL?
00661                 intMaskValue = ci->getValue();
00662             }
00663             // Is the high-bit set?  If so, OR in the appropriate bit in
00664             // the result mask
00665             if (intMaskValue.countLeadingOnes() > 0)
00666                 mask |= (1 << i);
00667         }
00668         return mask;
00669     }
00670     else if (llvm::isa<llvm::ConstantAggregateZero>(factor))
00671         return 0;
00672     else {
00673 #if 0
00674         llvm::ConstantExpr *ce = llvm::dyn_cast<llvm::ConstantExpr>(factor);
00675         if (ce != NULL) {
00676             llvm::TargetMachine *targetMachine = g->target.GetTargetMachine();
00677             const llvm::TargetData *td = targetMachine->getTargetData();
00678             llvm::Constant *c = llvm::ConstantFoldConstantExpression(ce, td);
00679             c->dump();
00680             factor = c;
00681         }
00682         // else we should be able to handle it above...
00683         Assert(!llvm::isa<llvm::Constant>(factor));
00684 #endif
00685         return -1;
00686     }

00713 IntrinsicsOpt::runOnBasicBlock(llvm::BasicBlock &bb) {
00714 #if defined(LLVM_3_0) || defined(LLVM_3_0svn) || defined(LLVM_3_1svn)
00715     llvm::Function *avxMaskedLoad32 = 
00716         llvm::Intrinsic::getDeclaration(m->module, llvm::Intrinsic::x86_avx_maskload_ps_256);
00717     llvm::Function *avxMaskedLoad64 = 
00718         llvm::Intrinsic::getDeclaration(m->module, llvm::Intrinsic::x86_avx_maskload_pd_256);
00719     llvm::Function *avxMaskedStore32 = 
00720         llvm::Intrinsic::getDeclaration(m->module, llvm::Intrinsic::x86_avx_maskstore_ps_256);
00721     llvm::Function *avxMaskedStore64 = 
00722         llvm::Intrinsic::getDeclaration(m->module, llvm::Intrinsic::x86_avx_maskstore_pd_256);
00723     Assert(avxMaskedLoad32 != NULL && avxMaskedStore32 != NULL);
00724     Assert(avxMaskedLoad64 != NULL && avxMaskedStore64 != NULL);
00725 #endif
00726 
00727     bool modifiedAny = false;
00728  restart:
00729     for (llvm::BasicBlock::iterator iter = bb.begin(), e = bb.end(); iter != e; ++iter) {
00730         llvm::CallInst *callInst = llvm::dyn_cast<llvm::CallInst>(&*iter);
00731         if (callInst == NULL || callInst->getCalledFunction() == NULL)
00732             continue;
00733 
00734         BlendInstruction *blend = matchingBlendInstruction(callInst->getCalledFunction());
00735         if (blend != NULL) {
00736             llvm::Value *v[2] = { callInst->getArgOperand(blend->op0), 
00737                                   callInst->getArgOperand(blend->op1) };

00739 
00740             // If the values are the same, then no need to blend..
00741             if (v[0] == v[1]) {
00742                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(), 
00743                                            iter, v[0]);
00744                 modifiedAny = true;
00745                 goto restart;
00746             }
00747 
00748             // If one of the two is undefined, we're allowed to replace
00749             // with the value of the other.  (In other words, the only
00750             // valid case is that the blend factor ends up having a value
00751             // that only selects from the defined one of the two operands,
00752             // otherwise the result is undefined and any value is fine,
00753             // ergo the defined one is an acceptable result.)
00754             if (lIsUndef(v[0])) {
00755                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(), 
00756                                            iter, v[1]);
00757                 modifiedAny = true;
00758                 goto restart;
00759             }
00760             if (lIsUndef(v[1])) {
00761                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(), 
00762                                            iter, v[0]);
00763                 modifiedAny = true;
00764                 goto restart;
00765             }

00769             if (mask == 0)
00770                 // Mask all off -> replace with the first blend value
00771                 value = v[0];

00773                 // Mask all on -> replace with the second blend value
00774                 value = v[1];
00775 
00776             if (value != NULL) {
00777                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(), 
00778                                            iter, value);
00779                 modifiedAny = true;
00780                 goto restart;
00781             }
00782         }
00783         else if (matchesMaskInstruction(callInst->getCalledFunction())) {
00784             llvm::Value *factor = callInst->getArgOperand(0);
00785             int mask = lGetMask(factor);

00787                 // If the vector-valued mask has a known value, replace it
00788                 // with the corresponding integer mask from its elements
00789                 // high bits.
00790                 llvm::Value *value = LLVMInt32(mask);
00791                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(),
00792                                            iter, value);
00793                 modifiedAny = true;
00794                 goto restart;
00795             }
00796         }
00797 #if defined(LLVM_3_0) || defined(LLVM_3_0svn) || defined(LLVM_3_1svn)
00798         else if (callInst->getCalledFunction() == avxMaskedLoad32 ||
00799                  callInst->getCalledFunction() == avxMaskedLoad64) {
00800             llvm::Value *factor = callInst->getArgOperand(1);
00801             int mask = lGetMask(factor);
00802             if (mask == 0) {
00803                 // nothing being loaded, replace with undef value
00804                 llvm::Type *returnType = callInst->getType();
00805                 Assert(llvm::isa<llvm::VectorType>(returnType));
00806                 llvm::Value *undefValue = llvm::UndefValue::get(returnType);
00807                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(),
00808                                            iter, undefValue);
00809                 modifiedAny = true;
00810                 goto restart;
00811             }
00812             else if (mask == 0xff) {

00814                 llvm::Type *returnType = callInst->getType();

00816                 // cast the i8 * to the appropriate type
00817                 llvm::Value *castPtr = 
00818                     new llvm::BitCastInst(callInst->getArgOperand(0),
00819                                           llvm::PointerType::get(returnType, 0), 
00820                                           "ptr2vec", callInst);

00822                 int align = callInst->getCalledFunction() == avxMaskedLoad32 ? 4 : 8;
00823                 llvm::Instruction *loadInst = 
00824                     new llvm::LoadInst(castPtr, "load", false /* not volatile */,
00825                                        align, (llvm::Instruction *)NULL);
00826                 lCopyMetadata(loadInst, callInst);
00827                 llvm::ReplaceInstWithInst(callInst, loadInst);
00828                 modifiedAny = true;
00829                 goto restart;
00830             }
00831         }
00832         else if (callInst->getCalledFunction() == avxMaskedStore32 ||
00833                  callInst->getCalledFunction() == avxMaskedStore64) {
00834             // NOTE: mask is the 2nd parameter, not the 3rd one!!
00835             llvm::Value *factor = callInst->getArgOperand(1);
00836             int mask = lGetMask(factor);
00837             if (mask == 0) {
00838                 // nothing actually being stored, just remove the inst
00839                 callInst->eraseFromParent();
00840                 modifiedAny = true;
00841                 goto restart;
00842             }
00843             else if (mask == 0xff) {
00844                 // all lanes storing, so replace with a regular store
00845                 llvm::Value *rvalue = callInst->getArgOperand(2);
00846                 llvm::Type *storeType = rvalue->getType();
00847                 llvm::Value *castPtr = 
00848                     new llvm::BitCastInst(callInst->getArgOperand(0),
00849                                           llvm::PointerType::get(storeType, 0), 
00850                                           "ptr2vec", callInst);
00851                 lCopyMetadata(castPtr, callInst);
00852 
00853                 llvm::StoreInst *storeInst = 
00854                     new llvm::StoreInst(rvalue, castPtr, (llvm::Instruction *)NULL);
00855                 int align = callInst->getCalledFunction() == avxMaskedStore32 ? 4 : 8;
00856                 storeInst->setAlignment(align);
00857                 lCopyMetadata(storeInst, callInst);
00858                 llvm::ReplaceInstWithInst(callInst, storeInst);
00859 
00860                 modifiedAny = true;
00861                 goto restart;
00862             }
00863         }
00864 #endif
00865     }

01007 /** Check to make sure that this value is actually a pointer in the end.
01008     We need to make sure that given an expression like vec(offset) +
01009     ptr2int(ptr), lGetBasePointer() doesn't return vec(offset) for the base
01010     pointer such that we then treat ptr2int(ptr) as an offset.  This ends
01011     up being important so that we don't generate LLVM GEP instructions like
01012     "gep inttoptr 8, i64 %ptr", which in turn can lead to incorrect code
01013     since LLVM's pointer aliasing analysis assumes that operands after the
01014     first one to a GEP aren't pointers.
01015  */
01016 static llvm::Value *
01017 lCheckForActualPointer(llvm::Value *v) {
01018     if (v == NULL)
01019         return NULL;
01020     else if (llvm::isa<LLVM_TYPE_CONST llvm::PointerType>(v->getType()))
01021         return v;
01022     else if (llvm::isa<llvm::PtrToIntInst>(v))
01023         return v;
01024     else {
01025         llvm::ConstantExpr *uce = 
01026             llvm::dyn_cast<llvm::ConstantExpr>(v);
01027         if (uce != NULL &&
01028             uce->getOpcode() == llvm::Instruction::PtrToInt)
01029             return v;
01030         return NULL;
01031     }

01035 /** Given a llvm::Value representing a varying pointer, this function
01036     checks to see if all of the elements of the vector have the same value
01037     (i.e. there's a common base pointer).  If so, it returns the common
01038     pointer value; otherwise it returns NULL.
01039  */
01040 static llvm::Value *
01041 lGetBasePointer(llvm::Value *v) {
01042     llvm::InsertElementInst *ie = llvm::dyn_cast<llvm::InsertElementInst>(v);
01043     if (ie != NULL) {
01044         llvm::Value *elements[ISPC_MAX_NVEC];
01045         LLVMFlattenInsertChain(ie, g->target.vectorWidth, elements);
01046 
01047         // Make sure none of the elements is undefined.
01048         // TODO: it's probably ok to allow undefined elements and return
01049         // the base pointer if all of the other elements have the same
01050         // value.
01051         for (int i = 0; i < g->target.vectorWidth; ++i)
01052             if (elements[i] == NULL)
01053                 return NULL;
01054 
01055         // Do all of the elements have the same value?
01056         for (int i = 0; i < g->target.vectorWidth-1; ++i)
01057             if (elements[i] != elements[i+1])
01058                 return NULL;
01059 
01060         return lCheckForActualPointer(elements[0]);
01061     }
01062 
01063     // This case comes up with global/static arrays
01064     llvm::ConstantVector *cv = llvm::dyn_cast<llvm::ConstantVector>(v);
01065     if (cv != NULL)
01066         return lCheckForActualPointer(cv->getSplatValue());
01067 
01068     return NULL;
01069 }

01074     the offset; if so return the base and the offset. */

01076 lGetConstantAddExprBaseOffset(llvm::Constant *op0, llvm::Constant *op1,

01094     be determined to be indexing from a common uniform base pointer.  If
01095     so, the function returns the base pointer llvm::Value and initializes
01096     *offsets with an int vector of the per-lane offsets
01097  */
01098 static llvm::Value *
01099 lGetBasePtrAndOffsets(llvm::Value *ptrs, llvm::Value **offsets,
01100                       llvm::Instruction *insertBefore) {
01101     llvm::Value *base = lGetBasePointer(ptrs);
01102     if (base != NULL) {

01104         // actually all the same value.

01672             if (gsFuncs[i].func != NULL &&
01673                 callInst->getCalledFunction() == gsFuncs[i].func) {
01674                 info = &gsFuncs[i];
01675                 break;
01676             }
01677         if (info == NULL)
01678             continue;
01679 
01680         // Try to transform the array of pointers to a single base pointer
01681         // and an array of int32 offsets.  (All the hard work is done by
01682         // lGetBasePtrAndOffsets).
01683         llvm::Value *ptrs = callInst->getArgOperand(0);
01684         llvm::Value *offsetVector = NULL;
01685         llvm::Value *basePtr = lGetBasePtrAndOffsets(ptrs, &offsetVector, 
01686                                                      callInst);
01687 
01688         if (basePtr == NULL || offsetVector == NULL)
01689             // It's actually a fully general gather/scatter with a varying
01690             // set of base pointers, so leave it as is and continune onward
01691             // to the next instruction...
01692             continue;
01693 
01694         // Try to decompose the offset vector into a compile time constant
01695         // component and a varying component.  The constant component is
01696         // passed as a separate parameter to the gather/scatter functions,
01697         // which in turn allows their implementations to end up emitting
01698         // x86 instructions with constant offsets encoded in them.
01699         llvm::Value *constOffset, *variableOffset;
01700         lExtractConstantOffset(offsetVector, &constOffset, &variableOffset, 
01701                                callInst);
01702         if (constOffset == NULL)
01703             constOffset = lGetZeroOffsetVector(offsetVector);
01704         if (variableOffset == NULL)
01705             variableOffset = lGetZeroOffsetVector(offsetVector);
01706 
01707         // See if the varying component is scaled by 2, 4, or 8.  If so,
01708         // extract that scale factor and rewrite variableOffset to remove
01709         // it.  (This also is pulled out so that we can match the scales by
01710         // 2/4/8 offered by x86 addressing operators.)
01711         llvm::Value *offsetScale = lExtractOffsetVector248Scale(&variableOffset);
01712 
01713         // Cast the base pointer to a void *, since that's what the
01714         // __pseudo_*_base_offsets_* functions want.
01715         basePtr = new llvm::IntToPtrInst(basePtr, LLVMTypes::VoidPointerType,
01716                                          "base2void", callInst);
01717         lCopyMetadata(basePtr, callInst);
01718 
01719         llvm::Function *gatherScatterFunc = info->baseOffsetsFunc;
01720 
01721         if (g->opt.force32BitAddressing) {
01722             // If we're doing 32-bit addressing on a 64-bit target, here we
01723             // will see if we can call one of the 32-bit variants of the
01724             // pseudo gather/scatter functions.  Specifically, if the
01725             // offset vector turns out to be an i32 value that was sext'ed
01726             // to be i64 immediately before the scatter/gather, then we
01727             // walk past the sext to get the i32 offset values and then
01728             // call out to the corresponding 32-bit gather/scatter
01729             // function.
01730             llvm::SExtInst *sext = llvm::dyn_cast<llvm::SExtInst>(variableOffset);
01731             if (sext != NULL && 
01732                 sext->getOperand(0)->getType() == LLVMTypes::Int32VectorType) {
01733                 variableOffset = sext->getOperand(0);
01734                 gatherScatterFunc = info->baseOffsets32Func;
01735                 if (constOffset->getType() != LLVMTypes::Int32VectorType)
01736                     constOffset = 
01737                         new llvm::TruncInst(constOffset, LLVMTypes::Int32VectorType,
01738                                             "trunc_const_offset", callInst);
01739             }
01740         }
01741 
01742         if (info->isGather) {
01743             llvm::Value *mask = callInst->getArgOperand(1);
01744 
01745             // Generate a new function call to the next pseudo gather
01746             // base+offsets instruction.  Note that we're passing a NULL
01747             // llvm::Instruction to llvm::CallInst::Create; this means that
01748             // the instruction isn't inserted into a basic block and that
01749             // way we can then call ReplaceInstWithInst().
01750             llvm::Instruction *newCall = 
01751                 lCallInst(gatherScatterFunc, basePtr, variableOffset, offsetScale,
01752                           constOffset, mask, "newgather", NULL);
01753             lCopyMetadata(newCall, callInst);
01754             llvm::ReplaceInstWithInst(callInst, newCall);
01755         }
01756         else {
01757             llvm::Value *storeValue = callInst->getArgOperand(1);
01758             llvm::Value *mask = callInst->getArgOperand(2);
01759 
01760             // Generate a new function call to the next pseudo scatter
01761             // base+offsets instruction.  See above for why passing NULL
01762             // for the Instruction * is intended.
01763             llvm::Instruction *newCall = 
01764                 lCallInst(gatherScatterFunc, basePtr, variableOffset, offsetScale,
01765                           constOffset, storeValue, mask, "", NULL);
01766             lCopyMetadata(newCall, callInst);
01767             llvm::ReplaceInstWithInst(callInst, newCall);
01768         }
01769         modifiedAny = true;
01770         goto restart;
01771     }
01772 
01773     return modifiedAny;
01774 }

02021     pointing to stack-allocated memory.  It's conservative in that it
02022     should never return true for non-stack allocated memory, but may return
02023     false for memory that actually is stack allocated.  The basic strategy
02024     is to traverse through the operands and see if the pointer originally
02025     comes from an AllocaInst.
02026 */
02027 static bool
02028 lIsSafeToBlend(llvm::Value *lvalue) {
02029     llvm::BitCastInst *bc = llvm::dyn_cast<llvm::BitCastInst>(lvalue);
02030     if (bc != NULL)
02031         return lIsSafeToBlend(bc->getOperand(0));
02032     else {
02033         llvm::AllocaInst *ai = llvm::dyn_cast<llvm::AllocaInst>(lvalue);
02034         if (ai) {
02035             LLVM_TYPE_CONST llvm::Type *type = ai->getType();
02036             LLVM_TYPE_CONST llvm::PointerType *pt = 
02037                 llvm::dyn_cast<LLVM_TYPE_CONST llvm::PointerType>(type);
02038             assert(pt != NULL);
02039             type = pt->getElementType();
02040             LLVM_TYPE_CONST llvm::ArrayType *at;
02041             while ((at = llvm::dyn_cast<LLVM_TYPE_CONST llvm::ArrayType>(type))) {
02042                 type = at->getElementType();

02044             LLVM_TYPE_CONST llvm::VectorType *vt = 
02045                 llvm::dyn_cast<LLVM_TYPE_CONST llvm::VectorType>(type);
02046             return (vt != NULL && 
02047                     (int)vt->getNumElements() == g->target.vectorWidth);
02048         }
02049         else {
02050             llvm::GetElementPtrInst *gep = 
02051                 llvm::dyn_cast<llvm::GetElementPtrInst>(lvalue);
02052             if (gep != NULL)

02274 #endif
02275     if (cv != NULL)
02276         return lVectorIsLinearConstantInts(cv, vectorLength, stride);
02277 
02278     llvm::BinaryOperator *bop = llvm::dyn_cast<llvm::BinaryOperator>(v);
02279     if (bop != NULL) {
02280         // FIXME: is it right to pass the seenPhis to the all equal check as well??
02281         llvm::Value *op0 = bop->getOperand(0), *op1 = bop->getOperand(1);
02282 
02283         if (bop->getOpcode() == llvm::Instruction::Add)
02284             // There are two cases to check if we have an add:
02285             //
02286             // programIndex + unif -> ascending linear seqeuence
02287             // unif + programIndex -> ascending linear sequence
02288             return ((lVectorIsLinear(op0, vectorLength, stride, seenPhis) &&
02289                      LLVMVectorValuesAllEqual(op1, vectorLength, seenPhis)) ||
02290                     (lVectorIsLinear(op1, vectorLength, stride, seenPhis) &&
02291                      LLVMVectorValuesAllEqual(op0, vectorLength, seenPhis)));
02292         else if (bop->getOpcode() == llvm::Instruction::Sub)
02293             // For subtraction, we only match:
02294             //
02295             // programIndex - unif -> ascending linear seqeuence
02296             //
02297             // In the future, we could also look for:
02298             // unif - programIndex -> *descending* linear seqeuence
02299             // And generate code for that as a vector load + shuffle.
02300             return (lVectorIsLinear(bop->getOperand(0), vectorLength,
02301                                     stride, seenPhis) &&
02302                     LLVMVectorValuesAllEqual(bop->getOperand(1), vectorLength,
02303                                           seenPhis));
02304         else if (bop->getOpcode() == llvm::Instruction::Mul)
02305             // Multiplies are a bit trickier, so are handled in a separate
02306             // function.
02307             return (lCheckMulForLinear(op0, op1, vectorLength, stride, seenPhis) ||
02308                     lCheckMulForLinear(op1, op0, vectorLength, stride, seenPhis));
02309         else
02310             return false;
02311     }
02312 
02313     llvm::CastInst *ci = llvm::dyn_cast<llvm::CastInst>(v);
02314     if (ci != NULL)
02315         return lVectorIsLinear(ci->getOperand(0), vectorLength,
02316                                stride, seenPhis);
02317 
02318     if (llvm::isa<llvm::CallInst>(v) || llvm::isa<llvm::LoadInst>(v))
02319         return false;
02320 
02321     llvm::PHINode *phi = llvm::dyn_cast<llvm::PHINode>(v);
02322     if (phi != NULL) {
02323         for (unsigned int i = 0; i < seenPhis.size(); ++i)
02324             if (seenPhis[i] == phi)
02325                 return true;
02326 
02327         seenPhis.push_back(phi);
02328 
02329         unsigned int numIncoming = phi->getNumIncomingValues();
02330         // Check all of the incoming values: if all of them are all equal,
02331         // then we're good.
02332         for (unsigned int i = 0; i < numIncoming; ++i) {
02333             if (!lVectorIsLinear(phi->getIncomingValue(i), vectorLength, stride,
02334                                  seenPhis)) {
02335                 seenPhis.pop_back();
02336                 return false;
02337             }
02338         }
02339 
02340         seenPhis.pop_back();
02341         return true;
02342     }
02343 
02344     // TODO: is any reason to worry about these?
02345     if (llvm::isa<llvm::InsertElementInst>(v))
02346         return false;
02347 
02348     // TODO: we could also handle shuffles, but we haven't yet seen any
02349     // cases where doing so would detect cases where actually have a linear
02350     // vector.
02351     llvm::ShuffleVectorInst *shuffle = llvm::dyn_cast<llvm::ShuffleVectorInst>(v);
02352     if (shuffle != NULL)
02353         return false;
02354 
02355 #if 0
02356     fprintf(stderr, "linear check: ");
02357     v->dump();
02358     fprintf(stderr, "\n");
02359     llvm::Instruction *inst = llvm::dyn_cast<llvm::Instruction>(v);
02360     if (inst) {
02361         inst->getParent()->dump();
02362         fprintf(stderr, "\n");
02363         fprintf(stderr, "\n");
02364     }
02365 #endif
02366 
02367     return false;
02368 }

02425 bool
02426 GSToLoadStorePass::runOnBasicBlock(llvm::BasicBlock &bb) {
02427     GatherImpInfo gInfo[] = {
02428         GatherImpInfo("__pseudo_gather_base_offsets32_8", "__load_and_broadcast_8",
02429                       "__masked_load_8", 1),
02430         GatherImpInfo("__pseudo_gather_base_offsets32_16", "__load_and_broadcast_16",
02431                       "__masked_load_16", 2),
02432         GatherImpInfo("__pseudo_gather_base_offsets32_32", "__load_and_broadcast_32",
02433                       "__masked_load_32", 4),
02434         GatherImpInfo("__pseudo_gather_base_offsets32_64", "__load_and_broadcast_64",
02435                       "__masked_load_64", 8),
02436         GatherImpInfo("__pseudo_gather_base_offsets64_8", "__load_and_broadcast_8",
02437                       "__masked_load_8", 1),
02438         GatherImpInfo("__pseudo_gather_base_offsets64_16", "__load_and_broadcast_16",
02439                       "__masked_load_16", 2),
02440         GatherImpInfo("__pseudo_gather_base_offsets64_32", "__load_and_broadcast_32",
02441                       "__masked_load_32", 4),
02442         GatherImpInfo("__pseudo_gather_base_offsets64_64", "__load_and_broadcast_64",
02443                       "__masked_load_64", 8)
02444     };
02445     ScatterImpInfo sInfo[] = {
02446         ScatterImpInfo("__pseudo_scatter_base_offsets32_8",  "__pseudo_masked_store_8", 
02447                        LLVMTypes::Int8VectorPointerType, 1),
02448         ScatterImpInfo("__pseudo_scatter_base_offsets32_16", "__pseudo_masked_store_16",
02449                        LLVMTypes::Int16VectorPointerType, 2),
02450         ScatterImpInfo("__pseudo_scatter_base_offsets32_32", "__pseudo_masked_store_32",
02451                        LLVMTypes::Int32VectorPointerType, 4),
02452         ScatterImpInfo("__pseudo_scatter_base_offsets32_64", "__pseudo_masked_store_64",
02453                        LLVMTypes::Int64VectorPointerType, 8),
02454         ScatterImpInfo("__pseudo_scatter_base_offsets64_8",  "__pseudo_masked_store_8", 
02455                        LLVMTypes::Int8VectorPointerType, 1),
02456         ScatterImpInfo("__pseudo_scatter_base_offsets64_16", "__pseudo_masked_store_16",
02457                        LLVMTypes::Int16VectorPointerType, 2),
02458         ScatterImpInfo("__pseudo_scatter_base_offsets64_32", "__pseudo_masked_store_32",
02459                        LLVMTypes::Int32VectorPointerType, 4),
02460         ScatterImpInfo("__pseudo_scatter_base_offsets64_64", "__pseudo_masked_store_64",
02461                        LLVMTypes::Int64VectorPointerType, 8)
02462     };
02463 
02464     bool modifiedAny = false;
02465 
02466  restart:
02467     for (llvm::BasicBlock::iterator iter = bb.begin(), e = bb.end(); iter != e; ++iter) {
02468         // Iterate over all of the instructions and look for calls to
02469         // __pseudo_*_base_offsets_* calls.
02470         llvm::CallInst *callInst = llvm::dyn_cast<llvm::CallInst>(&*iter);
02471         if (callInst == NULL)
02472             continue;
02473 
02474         llvm::Function *calledFunc = callInst->getCalledFunction();
02475         if (calledFunc == NULL)
02476             continue;
02477 
02478         GatherImpInfo *gatherInfo = NULL;
02479         ScatterImpInfo *scatterInfo = NULL;
02480         for (unsigned int i = 0; i < sizeof(gInfo) / sizeof(gInfo[0]); ++i) {
02481             if (gInfo[i].pseudoFunc != NULL &&
02482                 calledFunc == gInfo[i].pseudoFunc) {
02483                 gatherInfo = &gInfo[i];
02484                 break;
02485             }
02486         }
02487         for (unsigned int i = 0; i < sizeof(sInfo) / sizeof(sInfo[0]); ++i) {
02488             if (sInfo[i].pseudoFunc != NULL &&
02489                 calledFunc == sInfo[i].pseudoFunc) {
02490                 scatterInfo = &sInfo[i];
02491                 break;
02492             }
02493         }
02494         if (gatherInfo == NULL && scatterInfo == NULL)
02495             continue;
02496 
02497         SourcePos pos;
02498         bool ok = lGetSourcePosFromMetadata(callInst, &pos);
02499         Assert(ok);     
02500 
02501         llvm::Value *base = callInst->getArgOperand(0);
02502         llvm::Value *varyingOffsets = callInst->getArgOperand(1);
02503         llvm::Value *offsetScale = callInst->getArgOperand(2);
02504         llvm::Value *constOffsets = callInst->getArgOperand(3);
02505         llvm::Value *storeValue = (scatterInfo != NULL) ? callInst->getArgOperand(4) : NULL;
02506         llvm::Value *mask = callInst->getArgOperand((gatherInfo != NULL) ? 4 : 5);
02507 
02508         // Compute the full offset vector: offsetScale * varyingOffsets + constOffsets
02509         llvm::ConstantInt *offsetScaleInt = 
02510             llvm::dyn_cast<llvm::ConstantInt>(offsetScale);
02511         Assert(offsetScaleInt != NULL);
02512         uint64_t scaleValue = offsetScaleInt->getZExtValue();
02513 
02514         std::vector<llvm::Constant *> scales;
02515         for (int i = 0; i < g->target.vectorWidth; ++i) {
02516             if (varyingOffsets->getType() == LLVMTypes::Int64VectorType)
02517                 scales.push_back(LLVMInt64(scaleValue));
02518             else
02519                 scales.push_back(LLVMInt32(scaleValue));
02520         }
02521         llvm::Constant *offsetScaleVec = llvm::ConstantVector::get(scales);
02522 
02523         llvm::Value *scaledVarying = 
02524             llvm::BinaryOperator::Create(llvm::Instruction::Mul, offsetScaleVec,
02525                                          varyingOffsets, "scaled_varying", callInst);
02526         llvm::Value *fullOffsets =
02527             llvm::BinaryOperator::Create(llvm::Instruction::Add, scaledVarying,
02528                                          constOffsets, "varying+const_offsets",
02529                                          callInst);
02530 
02531         std::vector<llvm::PHINode *> seenPhis;
02532         if (LLVMVectorValuesAllEqual(fullOffsets, g->target.vectorWidth, seenPhis)) {
02533             // If all the offsets are equal, then compute the single
02534             // pointer they all represent based on the first one of them
02535             // (arbitrarily).
02536             llvm::Value *ptr = lComputeCommonPointer(base, fullOffsets, callInst);
02537             lCopyMetadata(ptr, callInst);
02538 
02539             if (gatherInfo != NULL) {
02540                 // A gather with everyone going to the same location is
02541                 // handled as a scalar load and broadcast across the lanes.
02542                 // Note that we do still have to pass the mask to the
02543                 // __load_and_broadcast_* functions, since they shouldn't
02544                 // access memory if the mask is all off (the location may
02545                 // be invalid in that case).
02546                 Debug(pos, "Transformed gather to scalar load and broadcast!");
02547                 llvm::Instruction *newCall = 
02548                     lCallInst(gatherInfo->loadBroadcastFunc, ptr, mask, 
02549                               "load_braodcast");
02550                 lCopyMetadata(newCall, callInst);
02551                 llvm::ReplaceInstWithInst(callInst, newCall);
02552 
02553                 modifiedAny = true;
02554                 goto restart;
02555             }
02556             else {
02557                 // A scatter with everyone going to the same location is
02558                 // undefined (if there's more than one program instance in
02559                 // the gang).  Issue a warning.
02560                 if (g->target.vectorWidth > 1)
02561                     Warning(pos, "Undefined behavior: all program instances are "
02562                             "writing to the same location!");
02563 
02564                 // We could do something similar to the gather case, where
02565                 // we arbitrarily write one of the values, but we need to
02566                 // a) check to be sure the mask isn't all off and b) pick
02567                 // the value from an executing program instance in that
02568                 // case.  We'll just let a bunch of the program instances
02569                 // do redundant writes, since this isn't important to make
02570                 // fast anyway...
02571             }
02572         }
02573         else {
02574             int step = gatherInfo ? gatherInfo->align : scatterInfo->align;
02575 
02576             std::vector<llvm::PHINode *> seenPhis;
02577             if (step > 0 && lVectorIsLinear(fullOffsets, g->target.vectorWidth, 
02578                                             step, seenPhis)) {
02579                 // We have a linear sequence of memory locations being accessed
02580                 // starting with the location given by the offset from
02581                 // offsetElements[0], with stride of 4 or 8 bytes (for 32 bit
02582                 // and 64 bit gather/scatters, respectively.)
02583                 llvm::Value *ptr = lComputeCommonPointer(base, fullOffsets, callInst);
02584                 lCopyMetadata(ptr, callInst);
02585 
02586                 if (gatherInfo != NULL) {
02587                     Debug(pos, "Transformed gather to unaligned vector load!");
02588                     llvm::Instruction *newCall = 
02589                         lCallInst(gatherInfo->loadMaskedFunc, ptr, mask, "masked_load");
02590                     lCopyMetadata(newCall, callInst);
02591                     llvm::ReplaceInstWithInst(callInst, newCall);
02592                 }
02593                 else {
02594                     Debug(pos, "Transformed scatter to unaligned vector store!");
02595                     ptr = new llvm::BitCastInst(ptr, scatterInfo->vecPtrType, "ptrcast", 
02596                                                 callInst);
02597                     llvm::Instruction *newCall =
02598                         lCallInst(scatterInfo->maskedStoreFunc, ptr, storeValue, 
02599                                   mask, "");
02600                     lCopyMetadata(newCall, callInst);
02601                     llvm::ReplaceInstWithInst(callInst, newCall);
02602                 }
02603 
02604                 modifiedAny = true;
02605                 goto restart;
02606             }
02607         }
02608     }
02609 
02610     return modifiedAny;
02611 }

02752 /** LLVM IR implementations of target-specific functions may include calls
02753     to the functions "bool __is_compile_time_constant_*(...)"; these allow
02754     them to have specialied code paths for where the corresponding value is
02755     known at compile time.  For masks, for example, this allows them to not
02756     incur the cost of a MOVMSK call at runtime to compute its value in
02757     cases where the mask value isn't known until runtime.
02758 
02759     This pass resolves these calls into either 'true' or 'false' values so
02760     that later optimization passes can operate with these as constants.
02761 
02762     See stdlib.m4 for a number of uses of this idiom. 
02763  */
02764 
02765 class IsCompileTimeConstantPass : public llvm::BasicBlockPass {
02766 public:
02767     static char ID;
02768     IsCompileTimeConstantPass(bool last = false) : BasicBlockPass(ID) {
02769         isLastTry = last;
02770     }
02771 
02772     const char *getPassName() const { return "Resolve \"is compile time constant\""; }
02773     bool runOnBasicBlock(llvm::BasicBlock &BB);
02774 
02775     bool isLastTry;
02776 };
02777 
02778 char IsCompileTimeConstantPass::ID = 0;
02779 
02780 bool
02781 IsCompileTimeConstantPass::runOnBasicBlock(llvm::BasicBlock &bb) {
02782     llvm::Function *funcs[] = {
02783         m->module->getFunction("__is_compile_time_constant_mask"),
02784         m->module->getFunction("__is_compile_time_constant_uniform_int32"),
02785         m->module->getFunction("__is_compile_time_constant_varying_int32")
02786     };
02787 
02788     bool modifiedAny = false;
02789  restart:
02790     for (llvm::BasicBlock::iterator i = bb.begin(), e = bb.end(); i != e; ++i) {
02791         // Iterate through the instructions looking for calls to the
02792         // __is_compile_time_constant_*() functions
02793         llvm::CallInst *callInst = llvm::dyn_cast<llvm::CallInst>(&*i);
02794         if (callInst == NULL)
02795             continue;
02796 
02797         int j;
02798         int nFuncs = sizeof(funcs) / sizeof(funcs[0]);
02799         for (j = 0; j < nFuncs; ++j) {

02834             modifiedAny = true;
