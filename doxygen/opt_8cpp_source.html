00124 /** We have a protocol with the front-end LLVM IR code generation process
00125     that allows us to encode the source file position that corresponds with
00126     instructions.  (For example, this allows us to issue performance
00127     warnings related to things like scatter and gather after optimization
00128     has been performed, so that we aren't warning about scatters and
00129     gathers that have been improved to stores and loads by optimization
00130     passes.)  Note that this is slightly redundant with the source file
00131     position encoding generated for debugging symbols, though we don't
00132     always generate debugging information but we do always generate this
00133     position data.
00134 
00135     This function finds the SourcePos that the metadata in the instruction
00136     (if present) corresponds to.  See the implementation of
00137     FunctionEmitContext::addGSMetadata(), which encodes the source position during
00138     code generation.
00139 
00140     @param inst   Instruction to try to find the source position of
00141     @param pos    Output variable in which to store the position
00142     @returns      True if source file position metadata was present and *pos
00143                   has been set.  False otherwise.
00144 */
00145 static bool
00146 lGetSourcePosFromMetadata(const llvm::Instruction *inst, SourcePos *pos) {
00147     llvm::MDNode *filename = inst->getMetadata("filename");
00148     llvm::MDNode *first_line = inst->getMetadata("first_line");
00149     llvm::MDNode *first_column = inst->getMetadata("first_column");
00150     llvm::MDNode *last_line = inst->getMetadata("last_line");
00151     llvm::MDNode *last_column = inst->getMetadata("last_column");
00152     if (!filename || !first_line || !first_column || !last_line || !last_column)
00153         return false;
00154 
00155     // All of these asserts are things that FunctionEmitContext::addGSMetadata() is
00156     // expected to have done in its operation
00157     llvm::MDString *str = llvm::dyn_cast<llvm::MDString>(filename->getOperand(0));
00158     Assert(str);
00159     llvm::ConstantInt *first_lnum = 
00160         llvm::dyn_cast<llvm::ConstantInt>(first_line->getOperand(0));
00161     Assert(first_lnum);
00162     llvm::ConstantInt *first_colnum = 
00163         llvm::dyn_cast<llvm::ConstantInt>(first_column->getOperand(0));
00164     Assert(first_column);
00165     llvm::ConstantInt *last_lnum = 
00166         llvm::dyn_cast<llvm::ConstantInt>(last_line->getOperand(0));
00167     Assert(last_lnum);
00168     llvm::ConstantInt *last_colnum = 
00169         llvm::dyn_cast<llvm::ConstantInt>(last_column->getOperand(0));
00170     Assert(last_column);
00171 
00172     *pos = SourcePos(str->getString().data(), (int)first_lnum->getZExtValue(),
00173                      (int)first_colnum->getZExtValue(), (int)last_lnum->getZExtValue(),
00174                      (int)last_colnum->getZExtValue());

00544 /** Given an llvm::Value represinting a vector mask, see if the value is a
00545     constant.  If so, return the integer mask found by taking the high bits
00546     of the mask values in turn and concatenating them into a single integer.
00547     In other words, given the 4-wide mask: < 0xffffffff, 0, 0, 0xffffffff >, 
00548     we have 0b1001 = 9.
00549  */
00550 static int
00551 lGetMask(llvm::Value *factor) {
00552     /* FIXME: This will break if we ever do 32-wide compilation, in which case
00553        it don't be possible to distinguish between -1 for "don't know" and
00554        "known and all bits on". */
00555     Assert(g->target.vectorWidth < 32);
00556 
00557     llvm::ConstantVector *cv = llvm::dyn_cast<llvm::ConstantVector>(factor);
00558     if (cv) {
00559         int mask = 0;
00560         llvm::SmallVector<llvm::Constant *, ISPC_MAX_NVEC> elements;
00561         cv->getVectorElements(elements);
00562 
00563         for (unsigned int i = 0; i < elements.size(); ++i) {
00564             llvm::APInt intMaskValue;
00565             // SSE has the "interesting" approach of encoding blending
00566             // masks as <n x float>.
00567             llvm::ConstantFP *cf = llvm::dyn_cast<llvm::ConstantFP>(elements[i]);
00568             if (cf) {
00569                 llvm::APFloat apf = cf->getValueAPF();
00570                 intMaskValue = apf.bitcastToAPInt();
00571             }
00572             else {
00573                 // Otherwise get it as an int
00574                 llvm::ConstantInt *ci = llvm::dyn_cast<llvm::ConstantInt>(elements[i]);
00575                 Assert(ci != NULL);  // vs return -1 if NULL?
00576                 intMaskValue = ci->getValue();
00577             }
00578             // Is the high-bit set?  If so, OR in the appropriate bit in
00579             // the result mask
00580             if (intMaskValue.countLeadingOnes() > 0)
00581                 mask |= (1 << i);
00582         }
00583         return mask;
00584     }
00585     else if (llvm::isa<llvm::ConstantAggregateZero>(factor))
00586         return 0;
00587     else {
00588 #if 0
00589         llvm::ConstantExpr *ce = llvm::dyn_cast<llvm::ConstantExpr>(factor);
00590         if (ce != NULL) {
00591             llvm::TargetMachine *targetMachine = g->target.GetTargetMachine();

00594             c->dump();

00596         }
00597         // else we should be able to handle it above...
00598         Assert(!llvm::isa<llvm::Constant>(factor));
00599 #endif
00600         return -1;

00642     bool modifiedAny = false;

00644     for (llvm::BasicBlock::iterator iter = bb.begin(), e = bb.end(); iter != e; ++iter) {
00645         llvm::CallInst *callInst = llvm::dyn_cast<llvm::CallInst>(&*iter);
00646         if (callInst == NULL || callInst->getCalledFunction() == NULL)
00647             continue;
00648 
00649         BlendInstruction *blend = matchingBlendInstruction(callInst->getCalledFunction());
00650         if (blend != NULL) {

00665             // valid case is that the blend factor ends up having a value
00666             // that only selects from the defined one of the two operands,
00667             // otherwise the result is undefined and any value is fine,
00668             // ergo the defined one is an acceptable result.)
00669             if (lIsUndef(v[0])) {
00670                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(), 
00671                                            iter, v[1]);

00689                 value = v[1];
00690 
00691             if (value != NULL) {
00692                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(), 
00693                                            iter, value);
00694                 modifiedAny = true;

00699             llvm::Value *factor = callInst->getArgOperand(0);

00702                 // If the vector-valued mask has a known value, replace it
00703                 // with the corresponding integer mask from its elements
00704                 // high bits.
00705                 llvm::Value *value = LLVMInt32(mask);
00706                 llvm::ReplaceInstWithValue(iter->getParent()->getInstList(),
00707                                            iter, value);
00708                 modifiedAny = true;
00709                 goto restart;
00710             }
00711         }
00712 #if defined(LLVM_3_0) || defined(LLVM_3_0svn) || defined(LLVM_3_1svn)
00713         else if (callInst->getCalledFunction() == avxMaskedLoad32 ||
00714                  callInst->getCalledFunction() == avxMaskedLoad64) {

00735                                           "ptr2vec", callInst);

00737                 int align = callInst->getCalledFunction() == avxMaskedLoad32 ? 4 : 8;
00738                 llvm::Instruction *loadInst = 
00739                     new llvm::LoadInst(castPtr, "load", false /* not volatile */,
00740                                        align, (llvm::Instruction *)NULL);
00741                 lCopyMetadata(loadInst, callInst);
00742                 llvm::ReplaceInstWithInst(callInst, loadInst);
00743                 modifiedAny = true;
00744                 goto restart;
00745             }
00746         }
00747         else if (callInst->getCalledFunction() == avxMaskedStore32 ||
00748                  callInst->getCalledFunction() == avxMaskedStore64) {
00749             // NOTE: mask is the 2nd parameter, not the 3rd one!!
00750             llvm::Value *factor = callInst->getArgOperand(1);
00751             int mask = lGetMask(factor);
00752             if (mask == 0) {
00753                 // nothing actually being stored, just remove the inst
00754                 callInst->eraseFromParent();
00755                 modifiedAny = true;
00756                 goto restart;
00757             }
00758             else if (mask == 0xff) {
00759                 // all lanes storing, so replace with a regular store
00760                 llvm::Value *rvalue = callInst->getArgOperand(2);
00761                 llvm::Type *storeType = rvalue->getType();
00762                 llvm::Value *castPtr = 
00763                     new llvm::BitCastInst(callInst->getArgOperand(0),
00764                                           llvm::PointerType::get(storeType, 0), 
00765                                           "ptr2vec", callInst);
00766                 lCopyMetadata(castPtr, callInst);
00767 
00768                 llvm::StoreInst *storeInst = 
00769                     new llvm::StoreInst(rvalue, castPtr, (llvm::Instruction *)NULL);
00770                 int align = callInst->getCalledFunction() == avxMaskedStore32 ? 4 : 8;
00771                 storeInst->setAlignment(align);
00772                 lCopyMetadata(storeInst, callInst);
00773                 llvm::ReplaceInstWithInst(callInst, storeInst);
00774 
00775                 modifiedAny = true;
00776                 goto restart;
00777             }
00778         }
00779 #endif
00780     }
00781     return modifiedAny;
00782 }

00924 /** Check to make sure that this value is actually a pointer in the end.
00925     We need to make sure that given an expression like vec(offset) +
00926     ptr2int(ptr), lGetBasePointer() doesn't return vec(offset) for the base
00927     pointer such that we then treat ptr2int(ptr) as an offset.  This ends
00928     up being important so that we don't generate LLVM GEP instructions like
00929     "gep inttoptr 8, i64 %ptr", which in turn can lead to incorrect code

00935     if (v == NULL)
00936         return NULL;

00938         return v;
00939     else if (llvm::isa<llvm::PtrToIntInst>(v))
00940         return v;
00941     else {
00942         llvm::ConstantExpr *uce = 
00943             llvm::dyn_cast<llvm::ConstantExpr>(v);
00944         if (uce != NULL &&
00945             uce->getOpcode() == llvm::Instruction::PtrToInt)
00946             return v;
00947         return NULL;
00948     }
00949 }

00952 /** Given a llvm::Value representing a varying pointer, this function
00953     checks to see if all of the elements of the vector have the same value
00954     (i.e. there's a common base pointer).  If so, it returns the common
00955     pointer value; otherwise it returns NULL.
00956  */
00957 static llvm::Value *
00958 lGetBasePointer(llvm::Value *v) {
00959     llvm::InsertElementInst *ie = llvm::dyn_cast<llvm::InsertElementInst>(v);
00960     if (ie != NULL) {

00962         LLVMFlattenInsertChain(ie, g->target.vectorWidth, elements);

00965         // TODO: it's probably ok to allow undefined elements and return
00966         // the base pointer if all of the other elements have the same
00967         // value.
00968         for (int i = 0; i < g->target.vectorWidth; ++i)
00969             if (elements[i] == NULL)
00970                 return NULL;
00971 
00972         // Do all of the elements have the same value?
00973         for (int i = 0; i < g->target.vectorWidth-1; ++i)
00974             if (elements[i] != elements[i+1])
00975                 return NULL;
00976 
00977         return lCheckForActualPointer(elements[0]);
00978     }
00979 
00980     // This case comes up with global/static arrays
00981     llvm::ConstantVector *cv = llvm::dyn_cast<llvm::ConstantVector>(v);
00982     if (cv != NULL)
00983         return lCheckForActualPointer(cv->getSplatValue());

00985     return NULL;

00989 /** Given the two operands to a constant add expression, see if we have the
00990     form "base pointer + offset", whee op0 is the base pointer and op1 is
00991     the offset; if so return the base and the offset. */
00992 static llvm::Constant *
00993 lGetConstantAddExprBaseOffset(llvm::Constant *op0, llvm::Constant *op1, 
00994                               llvm::Constant **delta) {
00995     llvm::ConstantExpr *op = llvm::dyn_cast<llvm::ConstantExpr>(op0);
00996     if (op == NULL || op->getOpcode() != llvm::Instruction::PtrToInt)
00997         // the first operand isn't a pointer
00998         return NULL;
00999 
01000     llvm::ConstantInt *opDelta = llvm::dyn_cast<llvm::ConstantInt>(op1);
01001     if (opDelta == NULL)
01002         // the second operand isn't an integer operand
01003         return NULL;
01004 
01005     *delta = opDelta;
01006     return op0;
01007 }

01010 /** Given a varying pointer in ptrs, this function checks to see if it can
01011     be determined to be indexing from a common uniform base pointer.  If
01012     so, the function returns the base pointer llvm::Value and initializes
01013     *offsets with an int vector of the per-lane offsets
01014  */
01015 static llvm::Value *
01016 lGetBasePtrAndOffsets(llvm::Value *ptrs, llvm::Value **offsets,
01017                       llvm::Instruction *insertBefore) {
01018     llvm::Value *base = lGetBasePointer(ptrs);
01019     if (base != NULL) {
01020         // We have a straight up varying pointer with no indexing that's
01021         // actually all the same value.
01022         if (g->target.is32Bit)
01023             *offsets = LLVMInt32Vector(0);
01024         else
01025             *offsets = LLVMInt64Vector((int64_t)0);
01026         return base;
01027     }
01028 
01029     llvm::BinaryOperator *bop = llvm::dyn_cast<llvm::BinaryOperator>(ptrs);
01030     if (bop != NULL && bop->getOpcode() == llvm::Instruction::Add) {
01031         // If we have a common pointer plus something, then we're also
01032         // good.
01033         if ((base = lGetBasePtrAndOffsets(bop->getOperand(0), 
01034                                           offsets, insertBefore)) != NULL) {
01035             *offsets = 
01036                 llvm::BinaryOperator::Create(llvm::Instruction::Add, *offsets,
01037                                              bop->getOperand(1), "new_offsets",
01038                                              insertBefore);
01039             return base;
01040         }
01041         else if ((base = lGetBasePtrAndOffsets(bop->getOperand(1), 
01042                                                offsets, insertBefore)) != NULL) {
01043             *offsets = 
01044                 llvm::BinaryOperator::Create(llvm::Instruction::Add, *offsets,
01045                                              bop->getOperand(0), "new_offsets",
01046                                              insertBefore);
01047             return base;
01048         }
01049     }
01050 
01051     llvm::ConstantVector *cv = llvm::dyn_cast<llvm::ConstantVector>(ptrs);
01052     if (cv != NULL) {
01053         // Indexing into global arrays can lead to this form, with
01054         // ConstantVectors..
01055         llvm::SmallVector<llvm::Constant *, ISPC_MAX_NVEC> elements;
01056         cv->getVectorElements(elements);
01057 
01058         llvm::Constant *delta[ISPC_MAX_NVEC];
01059         for (unsigned int i = 0; i < elements.size(); ++i) {
01060             // For each element, try to decompose it into either a straight
01061             // up base pointer, or a base pointer plus an integer value.
01062             llvm::ConstantExpr *ce = llvm::dyn_cast<llvm::ConstantExpr>(elements[i]);
01063             if (ce == NULL)
01064                 return NULL;
01065 
01066             delta[i] = NULL;
01067             llvm::Value *elementBase = NULL; // base pointer for this element
01068             if (ce->getOpcode() == llvm::Instruction::PtrToInt) {
01069                 // If the element is just a ptr to int instruction, treat
01070                 // it as having an offset of zero
01071                 elementBase = ce;
01072                 delta[i] = g->target.is32Bit ? LLVMInt32(0) : LLVMInt64(0);
01073             }
01074             else if (ce->getOpcode() == llvm::Instruction::Add) {
01075                 // Try both orderings of the operands to see if we can get
01076                 // a pointer+offset out of them.
01077                 elementBase =
01078                     lGetConstantAddExprBaseOffset(ce->getOperand(0), 
01079                                                   ce->getOperand(1),
01080                                                   &delta[i]);
01081                 if (elementBase == NULL)
01082                     elementBase = 
01083                         lGetConstantAddExprBaseOffset(ce->getOperand(1), 
01084                                                       ce->getOperand(0),
01085                                                       &delta[i]);
01086             }
01087 
01088             // We weren't able to find a base pointer in the above.  (We
01089             // don't expect this to happen; if it does, it may be necessary
01090             // to handle more cases in the decomposition above.)
01091             if (elementBase == NULL)
01092                 return NULL;
01093 
01094             Assert(delta[i] != NULL);
01095             if (base == NULL)
01096                 // The first time we've found a base pointer
01097                 base = elementBase;
01098             else if (base != elementBase)
01099                 // Different program instances have different base
01100                 // pointers, so no luck.
01101                 return NULL;
01102         }
01103 
01104         Assert(base != NULL);
01105 #ifdef LLVM_2_9
01106         *offsets = llvm::ConstantVector::get(delta);
01107 #else
01108         llvm::ArrayRef<llvm::Constant *> deltas(&delta[0], 
01109                                                 &delta[elements.size()]);
01110         *offsets = llvm::ConstantVector::get(deltas);
01111 #endif
01112         return base;
01113     }
01114 
01115     return NULL;
01116 }

01124     llvm::ConstantVector *cvec = llvm::dyn_cast<llvm::ConstantVector>(v);
01125     if (cvec == NULL)

01133         llvm::dyn_cast<llvm::ConstantInt>(splatConst);
01134     if (ci == NULL)

01146 /** Given a vector of integer offsets to a base pointer being used for a
01147     gather or a scatter, see if its root operation is a multiply by a
01148     vector of some value by all 2s/4s/8s.  If not, return NULL.
01149 
01150     If it is return an i32 value of 2, 4, 8 from the function and modify
01151     *vec so that it points to the operand that is being multiplied by
01152     2/4/8.
01153 
01154     We go through all this trouble so that we can pass the i32 scale factor
01155     to the {gather,scatter}_base_offsets function as a separate scale
01156     factor for the offsets.  This in turn is used in a way so that the LLVM
01157     x86 code generator matches it to apply x86's free scale by 2x, 4x, or
01158     8x to one of two registers being added together for an addressing
01159     calculation.
01160  */
01161 static llvm::Value *
01162 lExtractOffsetVector248Scale(llvm::Value **vec) {
01163     llvm::SExtInst *sext = llvm::dyn_cast<llvm::SExtInst>(*vec);
01164     if (sext != NULL) {
01165         llvm::Value *sextOp = sext->getOperand(0);
01166         // Check the sext target.
01167         llvm::Value *scale = lExtractOffsetVector248Scale(&sextOp);
01168         if (scale == NULL)
01169             return NULL;
01170 
01171         // make a new sext instruction so that we end up with the right
01172         // type
01173         *vec = new llvm::SExtInst(sextOp, sext->getType(), "offset_sext", sext);
01174         return scale;
01175     }
01176 
01177 
01178     // If we don't have a binary operator, then just give up
01179     llvm::BinaryOperator *bop = llvm::dyn_cast<llvm::BinaryOperator>(*vec);
01180     if (bop == NULL)
01181         return LLVMInt32(1);
01182 
01183     llvm::Value *op0 = bop->getOperand(0), *op1 = bop->getOperand(1);
01184     if (bop->getOpcode() == llvm::Instruction::Add) {
01185         if (llvm::isa<llvm::ConstantAggregateZero>(op0)) {
01186             *vec = op1;
01187             return lExtractOffsetVector248Scale(vec);
01188         }
01189         else if (llvm::isa<llvm::ConstantAggregateZero>(op1)) {
01190             *vec = op0;
01191             return lExtractOffsetVector248Scale(vec);
01192         }
01193         else {
01194             llvm::Value *s0 = lExtractOffsetVector248Scale(&op0);
01195             llvm::Value *s1 = lExtractOffsetVector248Scale(&op1);
01196             if (s0 == s1) {
01197                 *vec = llvm::BinaryOperator::Create(llvm::Instruction::Add,
01198                                                     op0, op1, "new_add", bop);
01199                 return s0;
01200             }
01201             else
01202                 return LLVMInt32(1);
01203         }
01204     }
01205     else if (bop->getOpcode() == llvm::Instruction::Mul) {
01206         // Check each operand for being one of the scale factors we care about.
01207         int splat;
01208         if (lIs248Splat(op0, &splat)) {
01209             *vec = op1;
01210             return LLVMInt32(splat);
01211         }
01212         else if (lIs248Splat(op1, &splat)) {
01213             *vec = op0;
01214             return LLVMInt32(splat);
01215         }
01216         else
01217             return LLVMInt32(1);
01218     }
01219     else
01220         return LLVMInt32(1);

01683         llvm::Value *mask  = callInst->getArgOperand(1);
01684         int allOnMask = (1 << g->target.vectorWidth) - 1;
01685 
01686         int maskAsInt = lGetMask(mask);
01687         if (maskAsInt == 0) {
01688             // Zero mask - no-op, so replace the load with an undef value
01689             llvm::ReplaceInstWithValue(iter->getParent()->getInstList(),
01690                                        iter, llvm::UndefValue::get(callInst->getType()));
01691             modifiedAny = true;
01692             goto restart;
01693         }
01694         else if (maskAsInt == allOnMask) {
01695             // The mask is all on, so turn this into a regular load
01696             LLVM_TYPE_CONST llvm::Type *ptrType = 
01697                 llvm::PointerType::get(callInst->getType(), 0);
01698             ptr = new llvm::BitCastInst(ptr, ptrType, "ptr_cast_for_load", 
01699                                         callInst);
01700             llvm::Instruction *load = 
01701                 new llvm::LoadInst(ptr, callInst->getName(), false /* not volatile */,
01702                                    info->align, (llvm::Instruction *)NULL);
01703             lCopyMetadata(load, callInst);
01704             llvm::ReplaceInstWithInst(callInst, load);
01705             modifiedAny = true;
01706             goto restart;
01707         }
01708     }
01709     return modifiedAny;

01742 /** This routine attempts to determine if the given pointer in lvalue is
01743     pointing to stack-allocated memory.  It's conservative in that it
01744     should never return true for non-stack allocated memory, but may return
01745     false for memory that actually is stack allocated.  The basic strategy
01746     is to traverse through the operands and see if the pointer originally
01747     comes from an AllocaInst.
01748 */
01749 static bool
01750 lIsSafeToBlend(llvm::Value *lvalue) {
01751     llvm::BitCastInst *bc = llvm::dyn_cast<llvm::BitCastInst>(lvalue);
01752     if (bc != NULL)
01753         return lIsSafeToBlend(bc->getOperand(0));
01754     else {
01755         llvm::AllocaInst *ai = llvm::dyn_cast<llvm::AllocaInst>(lvalue);
01756         if (ai) {
01757             LLVM_TYPE_CONST llvm::Type *type = ai->getType();
01758             LLVM_TYPE_CONST llvm::PointerType *pt = 
01759                 llvm::dyn_cast<LLVM_TYPE_CONST llvm::PointerType>(type);
01760             assert(pt != NULL);
01761             type = pt->getElementType();
01762             LLVM_TYPE_CONST llvm::ArrayType *at;
01763             while ((at = llvm::dyn_cast<LLVM_TYPE_CONST llvm::ArrayType>(type))) {
01764                 type = at->getElementType();
01765             }
01766             LLVM_TYPE_CONST llvm::VectorType *vt = 
01767                 llvm::dyn_cast<LLVM_TYPE_CONST llvm::VectorType>(type);
01768             return (vt != NULL && 
01769                     (int)vt->getNumElements() == g->target.vectorWidth);
01770         }
01771         else {
01772             llvm::GetElementPtrInst *gep = 
01773                 llvm::dyn_cast<llvm::GetElementPtrInst>(lvalue);
01774             if (gep != NULL)
01775                 return lIsSafeToBlend(gep->getOperand(0));
01776             else
01777                 return false;
01778         }
01779     }
01780 }

01901 /** Given a vector of compile-time constant integer values, test to see if
01902     they are a linear sequence of constant integers starting from an
01903     arbirary value but then having a step of value "stride" between
01904     elements.
01905  */
01906 static bool
01907 lVectorIsLinearConstantInts(llvm::ConstantVector *cv, int vectorLength, 
01908                             int stride) {
01909     // Flatten the vector out into the elements array
01910     llvm::SmallVector<llvm::Constant *, ISPC_MAX_NVEC> elements;
01911     cv->getVectorElements(elements);
01912     Assert((int)elements.size() == vectorLength);
01913 
01914     llvm::ConstantInt *ci = llvm::dyn_cast<llvm::ConstantInt>(elements[0]);
01915     if (ci == NULL)
01916         // Not a vector of integers
01917         return false;
01918 
01919     int64_t prevVal = ci->getSExtValue();
01920 
01921     // For each element in the array, see if it is both a ConstantInt and
01922     // if the difference between it and the value of the previous element
01923     // is stride.  If not, fail.
01924     for (int i = 1; i < vectorLength; ++i) {
01925         ci = llvm::dyn_cast<llvm::ConstantInt>(elements[i]);
01926         if (ci == NULL) 
01927             return false;
01928 
01929         int64_t nextVal = ci->getSExtValue();
01930         if (prevVal + stride != nextVal)
01931             return false;
01932 
01933         prevVal = nextVal;
01934     }
01935     return true;
01936 }

01942 /** Checks to see if (op0 * op1) is a linear vector where the result is a
01943     vector with values that increase by stride.
01944  */
01945 static bool
01946 lCheckMulForLinear(llvm::Value *op0, llvm::Value *op1, int vectorLength, 
01947                    int stride, std::vector<llvm::PHINode *> &seenPhis) {
01948     // Is the first operand a constant integer value splatted across all of
01949     // the lanes?
01950     llvm::ConstantVector *cv = llvm::dyn_cast<llvm::ConstantVector>(op0);
01951     if (cv == NULL)
01952         return false;
01953     llvm::ConstantInt *splat = 
01954         llvm::dyn_cast<llvm::ConstantInt>(cv->getSplatValue());
01955     if (splat == NULL)
01956         return false;
01957 
01958     // If the splat value doesn't evenly divide the stride we're looking
01959     // for, there's no way that we can get the linear sequence we're
01960     // looking or.
01961     int64_t splatVal = splat->getSExtValue();
01962     if (splatVal == 0 || splatVal > stride || (stride % splatVal) != 0)
01963         return false;
01964 
01965     // Check to see if the other operand is a linear vector with stride
01966     // given by stride/splatVal.
01967     return lVectorIsLinear(op1, vectorLength, (int)(stride / splatVal), 
01968                            seenPhis);
01969 }

01972 /** Given vector of integer-typed values, see if the elements of the array
01973     have a step of 'stride' between their values.  This function tries to
01974     handle as many possibilities as possible, including things like all
01975     elements equal to some non-constant value plus an integer offset, etc.
01976 */
01977 static bool
01978 lVectorIsLinear(llvm::Value *v, int vectorLength, int stride,
01979                 std::vector<llvm::PHINode *> &seenPhis) {
01980     // First try the easy case: if the values are all just constant
01981     // integers and have the expected stride between them, then we're done.
01982     llvm::ConstantVector *cv = llvm::dyn_cast<llvm::ConstantVector>(v);
01983     if (cv != NULL)
01984         return lVectorIsLinearConstantInts(cv, vectorLength, stride);
01985 
01986     llvm::BinaryOperator *bop = llvm::dyn_cast<llvm::BinaryOperator>(v);
01987     if (bop != NULL) {
01988         // FIXME: is it right to pass the seenPhis to the all equal check as well??
01989         llvm::Value *op0 = bop->getOperand(0), *op1 = bop->getOperand(1);
01990 
01991         if (bop->getOpcode() == llvm::Instruction::Add)
01992             // There are two cases to check if we have an add:
01993             //
01994             // programIndex + unif -> ascending linear seqeuence
01995             // unif + programIndex -> ascending linear sequence
01996             return ((lVectorIsLinear(op0, vectorLength, stride, seenPhis) &&
01997                      LLVMVectorValuesAllEqual(op1, vectorLength, seenPhis)) ||
01998                     (lVectorIsLinear(op1, vectorLength, stride, seenPhis) &&
01999                      LLVMVectorValuesAllEqual(op0, vectorLength, seenPhis)));
02000         else if (bop->getOpcode() == llvm::Instruction::Sub)
02001             // For subtraction, we only match:
02002             //
02003             // programIndex - unif -> ascending linear seqeuence
02004             //
02005             // In the future, we could also look for:
02006             // unif - programIndex -> *descending* linear seqeuence
02007             // And generate code for that as a vector load + shuffle.
02008             return (lVectorIsLinear(bop->getOperand(0), vectorLength,
02009                                     stride, seenPhis) &&
02010                     LLVMVectorValuesAllEqual(bop->getOperand(1), vectorLength,
02011                                           seenPhis));
02012         else if (bop->getOpcode() == llvm::Instruction::Mul)
02013             // Multiplies are a bit trickier, so are handled in a separate
02014             // function.
02015             return (lCheckMulForLinear(op0, op1, vectorLength, stride, seenPhis) ||
02016                     lCheckMulForLinear(op1, op0, vectorLength, stride, seenPhis));
02017         else
02018             return false;
02019     }
02020 
02021     llvm::CastInst *ci = llvm::dyn_cast<llvm::CastInst>(v);
02022     if (ci != NULL)
02023         return lVectorIsLinear(ci->getOperand(0), vectorLength,
02024                                stride, seenPhis);
02025 
02026     if (llvm::isa<llvm::CallInst>(v) || llvm::isa<llvm::LoadInst>(v))
02027         return false;
02028 
02029     llvm::PHINode *phi = llvm::dyn_cast<llvm::PHINode>(v);
02030     if (phi != NULL) {
02031         for (unsigned int i = 0; i < seenPhis.size(); ++i)
02032             if (seenPhis[i] == phi)
02033                 return true;
02034 
02035         seenPhis.push_back(phi);
02036 
02037         unsigned int numIncoming = phi->getNumIncomingValues();
02038         // Check all of the incoming values: if all of them are all equal,
02039         // then we're good.
02040         for (unsigned int i = 0; i < numIncoming; ++i) {
02041             if (!lVectorIsLinear(phi->getIncomingValue(i), vectorLength, stride,
02042                                  seenPhis)) {
02043                 seenPhis.pop_back();
02044                 return false;
02045             }
02046         }
02047 
02048         seenPhis.pop_back();
02049         return true;
02050     }
02051 
02052     // TODO: is any reason to worry about these?
02053     if (llvm::isa<llvm::InsertElementInst>(v))
02054         return false;
02055 
02056     // TODO: we could also handle shuffles, but we haven't yet seen any
02057     // cases where doing so would detect cases where actually have a linear
02058     // vector.
02059     llvm::ShuffleVectorInst *shuffle = llvm::dyn_cast<llvm::ShuffleVectorInst>(v);
02060     if (shuffle != NULL)
02061         return false;
02062 
02063 #if 0
02064     fprintf(stderr, "linear check: ");
02065     v->dump();
02066     fprintf(stderr, "\n");
02067     llvm::Instruction *inst = llvm::dyn_cast<llvm::Instruction>(v);
02068     if (inst) {
02069         inst->getParent()->dump();
02070         fprintf(stderr, "\n");
02071         fprintf(stderr, "\n");
02072     }
02073 #endif
02074 
02075     return false;
02076 }
